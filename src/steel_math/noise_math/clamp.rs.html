5/// Clamped linear interpolation.
6///
7/// Clamps the interpolation factor to [0, 1] before interpolating.
8///
9/// Java reference: `Mth.clampedLerp(double, double, double)`.
10/// Note: Vanilla's parameter order is `(factor, min, max)`, ours is `(min, max, factor)`.
11#[inline]
12#[must_use]
13pub fn clamped_lerp(min: f64, max: f64, factor: f64) -> f64 {
14    if factor < 0.0 {
15        min
16    } else if factor > 1.0 {
17        max
18    } else {
19        lerp(factor, min, max)
20    }
21}

23/// Clamped lerp for N lanes.
24#[inline]
25#[must_use]
26pub fn clamped_lerp_simd<const N: usize>(
27    min: Simd<f64, N>,
28    max: Simd<f64, N>,
29    factor: Simd<f64, N>,
30) -> Simd<f64, N> {
31    let zero = Simd::splat(0.0);
32    let one = Simd::splat(1.0);
33    let below = factor.simd_lt(zero);
34    let above = factor.simd_gt(one);
35
36    // lerp result for the middle case
37    let lerped = min + factor * (max - min);
38
39    // Select: below zero → min, above one → max, otherwise → lerped
40    let result = below.select(min, lerped);
41    above.select(max, result)
42}

44/// Clamp a value to the range [min, max].
45///
46/// Java reference: `Mth.clamp(double, double, double)`
47#[inline]
48#[must_use]
49pub fn clamp(value: f64, min: f64, max: f64) -> f64 {
50    if value < min {
51        min
52    } else if value > max {
53        max
54    } else {
55        value
56    }
57}
