1//! Port of Mojang `net.minecraft.util.Mth`'s float trig lookup tables.
2//!
3//! Carvers, structures, entity physics, and a lot of other gameplay code call
4//! `Mth.sin(float)` / `Mth.cos(float)`, not `Math.sin(double)`. Matching
5//! block-level determinism with vanilla requires using the same 65536-entry
6//! `float` table instead of the precise FPU sine.
7//!
8//! # Vanilla reference
9//! ```java
10//! private static final float[] SIN = new float[65536];
11//! static {
12//!     for (int i = 0; i < 65536; i++)
13//!         SIN[i] = (float) Math.sin(i * (Math.PI * 2.0 / 65536.0));
14//! }
15//! public static float sin(double i)  { return SIN[(int)((long)(i * 10430.378350470453) & 65535L)]; }
16//! public static float cos(double i)  { return SIN[(int)((long)(i * 10430.378350470453 + 16384.0) & 65535L)]; }
17//! ```
18//! Note that the multiplier `10430.378350470453` is exactly
19//! `65536 / (2π)` rounded to a double, and the table is indexed by the
20//! low 16 bits of the scaled angle.
21
22use std::sync::LazyLock;
23
24/// `65536 / (2π)` — Mojang's stored constant.
25const INDEX_SCALE: f64 = 10_430.378_350_470_453;
26const TABLE_LEN: usize = 65_536;
27const TABLE_MASK: i64 = 0xFFFF;
28
29/// The `Mth.SIN` table (one entry per 2π / 65536 radians, float-valued).
30static SIN_TABLE: LazyLock<Box<[f32; TABLE_LEN]>> = LazyLock::new(|| {
31    // Box on the heap to keep the binary's .bss section small.
32    let mut table: Box<[f32; TABLE_LEN]> = vec![0.0_f32; TABLE_LEN]
33        .into_boxed_slice()
34        .try_into()
35        .expect("65536-element vec");
36    for (i, slot) in table.iter_mut().enumerate() {
37        *slot = (i as f64 / INDEX_SCALE).sin() as f32;
38    }
39    table
40});
41
42/// Vanilla `Mth.sin(double)` — returns the float from the 65536-entry sine
43/// table indexed by the angle's low-16-bit bucket.
44#[inline]
45#[must_use]
46pub fn sin(angle: f64) -> f32 {
47    let idx = (((angle * INDEX_SCALE) as i64) & TABLE_MASK) as usize;
48    SIN_TABLE[idx]
49}
50
51/// Vanilla `Mth.cos(double)` — like [`sin`] but phase-shifted by +π/2 using
52/// the same table.
53#[inline]
54#[must_use]
55pub fn cos(angle: f64) -> f32 {
56    let idx = (((angle * INDEX_SCALE + 16_384.0) as i64) & TABLE_MASK) as usize;
57    SIN_TABLE[idx]
58}
59
60#[cfg(test)]
61#[expect(
62    clippy::float_cmp,
63    reason = "tests compare exact f32 table entries, not approximations"
64)]
65mod test {
66    use super::*;
67
68    #[test]
69    fn sin_at_zero_is_zero() {
70        assert_eq!(sin(0.0), 0.0);
71        assert_eq!(cos(0.0), 1.0);
72    }
73
74    #[test]
75    fn sin_matches_table_entries() {
76        // Index 0 is sin(0), should be 0.
77        // Index 16384 is sin(π/2), should be 1.
78        // Verify a few well-known table entries.
79        for &i in &[0usize, 1000, 16_384, 32_768, 49_152] {
80            let expected = (i as f64 / INDEX_SCALE).sin() as f32;
81            assert_eq!(SIN_TABLE[i], expected);
82        }
83    }
84
85    #[test]
86    fn cos_is_sin_shifted() {
87        // cos(x) == sin(x + π/2) via the same table.
88        let angle = 0.7;
89        let expected = {
90            let idx = (((angle * INDEX_SCALE + 16_384.0) as i64) & TABLE_MASK) as usize;
91            SIN_TABLE[idx]
92        };
93        assert_eq!(cos(angle), expected);
94    }
95}
