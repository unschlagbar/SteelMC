2//!
3//! This module handles loading the server configuration from disk.
4//! The config is loaded once at startup, split into creation-time values
5//! (consumed by the server constructor) and a `RuntimeConfig` (stored on `Server`).

21/// Top-level TOML deserialization target — used once at startup, not stored globally.
22#[derive(Debug, Clone, Deserialize)]
23#[serde(deny_unknown_fields)]
24pub struct SteelConfig {
25    /// The full server configuration (`[server]` section)
26    pub server: ServerConfig,
27    /// Logging configuration (`[log]` section)
28    pub log: Option<LogConfig>,
29    /// World and domain configuration from `worlds.toml`.
30    #[serde(skip, default = "empty_worlds_config")]
31    pub worlds: WorldsConfig,
32}

62/// The full server configuration as deserialized from TOML.
63///
64/// Contains both creation-time values (seed, world generator, storage)
65/// and runtime values that get moved into `RuntimeConfig`.
66#[derive(Debug, Clone, Deserialize)]
67#[serde(deny_unknown_fields)]
68pub struct ServerConfig {
69    /// The port the server will listen on.
70    pub server_port: u16,
71    /// The maximum number of players that can be on the server at once.
72    pub max_players: u32,
73    /// Allow `view_distance` above vanilla's 32-chunk cap.
74    #[serde(default)]
75    pub allow_extended_view_distance: bool,
76    /// The view distance of the server.
77    pub view_distance: u8,
78    /// The simulation distance of the server.
79    pub simulation_distance: u8,
80    /// Whether the server is in online mode.
81    pub online_mode: bool,
82    /// Optional authentication endpoint for online-mode `hasJoined` checks.
83    pub auth_server: Option<String>,
84    /// Whether the server should use encryption.
85    pub encryption: bool,
86    /// Whether vanilla floating/flying movement checks permit unauthorized flight.
87    #[serde(default)]
88    pub allow_flight: bool,
89    /// The message of the day.
90    pub motd: String,
91    /// Whether to use a favicon.
92    pub use_favicon: bool,
93    /// The path to the favicon.
94    pub favicon: String,
95    /// Whether to enforce secure chat.
96    pub enforce_secure_chat: bool,
97    /// Vanilla chat spam threshold window in seconds
98    #[serde(default = "default_spam_threshold_seconds")]
99    pub chat_spam_threshold_seconds: i32,
100    /// Vanilla command spam threshold window in seconds
101    #[serde(default = "default_spam_threshold_seconds")]
102    pub command_spam_threshold_seconds: i32,
103    /// The compression settings for the server.
104    pub compression: Option<CompressionInfo>,
105    /// All settings and configurations for server links.
106    pub server_links: Option<ServerLinks>,
107    /// Thread counts for server thread pools.
108    #[serde(default)]
109    pub threads: ThreadConfig,
110}

123            allow_flight: self.allow_flight,
124            motd: self.motd,
125            use_favicon: self.use_favicon,

128            chat_spam_threshold_seconds: self.chat_spam_threshold_seconds,
129            command_spam_threshold_seconds: self.command_spam_threshold_seconds,
130            compression: self.compression,
131            server_links: self.server_links,
132            chunk_generation_threads: self.threads.chunk_generation,
133        }
134    }

151/// Logging configuration
152#[derive(Debug, Clone, Deserialize)]
153#[serde(deny_unknown_fields)]
154pub struct LogConfig {
155    /// Path where store the log files and history
156    #[serde(default = "default_log_path")]
157    pub log_path: String,
158    /// The level of information the logger will show
159    #[serde(default)]
160    pub log_level: LogLevel,
161    /// Time display format: "none", "date" (HH:MM:SS:mmm), or "uptime" (seconds since start)
162    #[serde(default)]
163    pub time: LogTimeFormat,
164    /// Whether the `module_path` of the log should be displayed
165    #[serde(default)]
166    pub module_path: bool,
167    /// Whether the extra data of the log should be displayed
168    #[serde(default)]
169    pub extra: bool,
170    /// Whether the log should be written into a file
171    #[serde(default = "default_log_file")]
172    pub log_file: bool,
173    /// Time between log file rotations
174    #[serde(default)]
175    pub rotation_time: RotationTimeFormat,
176    /// Amount of console commands saved
177    #[serde(default = "default_max_history")]
178    pub max_history: usize,
179}

181/// Time format for log entries
182#[derive(Debug, Clone, Copy, Default, Deserialize, PartialEq, Eq)]
183#[serde(rename_all = "lowercase")]
184pub enum LogTimeFormat {
185    /// No time displayed
186    None,
187    /// Current time (HH:MM:SS:mmm)
188    #[default]
189    Date,
190    /// Seconds since server start
191    Uptime,
192}
193
194/// Time for log files rotation
195#[derive(Debug, Clone, Copy, Default, Deserialize, PartialEq, Eq)]
196#[serde(rename_all = "lowercase")]
197pub enum RotationTimeFormat {
198    /// No rotation
199    None,
200    /// Rotate hourly
201    Hourly,
202    /// Rotate daily
203    #[default]
204    Daily,
205    /// Rotate weekly
206    Weekly,
207    /// Rotate monthly
208    Monthly,
209}

211/// The level of information the logger will show
212#[derive(Debug, Clone, Copy, Default, Deserialize, PartialEq, Eq)]
213#[serde(rename_all = "lowercase")]
214pub enum LogLevel {
215    /// Only error logs
216    Error,
217    /// Error and warn logs
218    Warn,
219    /// All standard logs
220    #[default]
221    Info,
222    /// Standard + Debug info enabled
223    Debug,
224    /// All logs are shown
225    Trace,
226}
227impl LogLevel {
228    /// Converts the log level in it's respective logging directive
229    #[must_use]
230    pub fn to_directive(self) -> Directive {
231        match self {
232            LogLevel::Error => LevelFilter::ERROR.into(),
233            LogLevel::Warn => LevelFilter::WARN.into(),
234            LogLevel::Info => LevelFilter::INFO.into(),
235            LogLevel::Debug => LevelFilter::DEBUG.into(),
236            LogLevel::Trace => LevelFilter::TRACE.into(),
237        }
238    }
239}

241/// Loads the server configuration from the given path, or creates it if it doesn't exist.
242///
243pub fn load_or_create(path: &Path) -> Result<SteelConfig, String> {
244    let mut config = if path.exists() {
245        let config_str = fs::read_to_string(path)
246            .map_err(|e| format!("failed to read config file {}: {e}", path.display()))?;
247        let config: SteelConfig = toml::from_str(config_str.as_str())
248            .map_err(|e| format!("failed to parse config: {e}"))?;
249        validate(&config.server).map_err(|e| format!("failed to validate config: {e}"))?;
250        config
251    } else {
252        let parent = path
253            .parent()
254            .ok_or_else(|| format!("failed to get config directory for {}", path.display()))?;
255        fs::create_dir_all(parent).map_err(|e| {
256            format!(
257                "failed to create config directory {}: {e}",
258                parent.display()
259            )
260        })?;
261        fs::write(path, DEFAULT_CONFIG)
262            .map_err(|e| format!("failed to write config file {}: {e}", path.display()))?;
263        let config: SteelConfig = toml::from_str(DEFAULT_CONFIG)
264            .map_err(|e| format!("failed to parse default config: {e}"))?;
265        validate(&config.server).map_err(|e| format!("failed to validate default config: {e}"))?;
266        config
267    };
268
269    let worlds_path = path
270        .parent()
271        .ok_or_else(|| format!("failed to get config directory for {}", path.display()))?
272        .join("worlds.toml");
273    config.worlds = load_or_create_worlds(&worlds_path)?;
274
275    // If icon file doesnt exist, write it
276    #[cfg(feature = "stand-alone")]
277    if config.server.use_favicon && !Path::new(&config.server.favicon).exists() {
278        fs::write(Path::new(&config.server.favicon), DEFAULT_FAVICON).map_err(|e| {
279            format!(
280                "failed to write favicon file {}: {e}",
281                config.server.favicon
282            )
283        })?;
284    }
285
286    Ok(config)
287}

289fn load_or_create_worlds(path: &Path) -> Result<WorldsConfig, String> {
290    if path.exists() {
291        let worlds_str = fs::read_to_string(path)
292            .map_err(|e| format!("failed to read worlds config file {}: {e}", path.display()))?;
293        toml::from_str(worlds_str.as_str())
294            .map_err(|e| format!("failed to parse worlds config {}: {e}", path.display()))
295    } else {
296        fs::write(path, DEFAULT_WORLDS)
297            .map_err(|e| format!("failed to write worlds config file {}: {e}", path.display()))?;
298        toml::from_str(DEFAULT_WORLDS)
299            .map_err(|e| format!("failed to parse default worlds config: {e}"))
300    }
301}

303/// Validates the server configuration.
304///
305/// # Errors
306/// This function will return an error if the configuration is invalid.
307fn validate(config: &ServerConfig) -> Result<(), &'static str> {
308    if !config.allow_extended_view_distance && !(1..=32).contains(&config.view_distance) {
309        return Err("View distance must in range 1..32");
310    }
311    if config.allow_extended_view_distance && !(1..=127).contains(&config.view_distance) {
312        return Err("View distance must in range 1..127");
313    }
314    if let Some(auth_server) = &config.auth_server {
315        let Ok(url) = Url::parse(auth_server) else {
316            return Err("auth_server must be an absolute URL");
317        };
318        if !matches!(url.scheme(), "http" | "https") {
319            return Err("auth_server must use http or https");
320        }
321    }
322    if config.simulation_distance > config.view_distance {
323        return Err("Simulation distance must be less than or equal to view distance");
324    }
325    if let Some(compression) = config.compression {
326        if compression.threshold.get() < 256 {
327            return Err("Compression threshold must be greater than or equal to 256");
328        }
329        if !(1..=9).contains(&compression.level) {
330            return Err("Compression level must be between 1 and 9");
331        }
332    }
333    if config.enforce_secure_chat {
334        if !config.online_mode {
335            return Err("online_mode must be true when enforce_secure_chat is enabled");
336        }
337        if !config.encryption {
338            return Err("encryption must be true when enforce_secure_chat is enabled");
339        }
340    }
341    Ok(())
342}
