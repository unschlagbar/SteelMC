11pub struct LogFile {
12    file: Option<File>,
13    date: DateTime<Utc>,
14    base_path: PathBuf,
15    rotation_time: RotationTimeFormat,
16}
17impl LogFile {
18    pub fn new(
19        base_path: PathBuf,
20        rotation_time: RotationTimeFormat,
21        enabled: bool,
22    ) -> Result<Self> {
23        let date = Utc::now();
24
25        let file = if enabled {
26            Self::archive_existing_latest(&base_path)?;
27            Some(Self::open_latest(&base_path)?)
28        } else {
29            None
30        };
31        Ok(Self {
32            file,
33            date,
34            base_path,
35            rotation_time,
36        })
37    }
38
39    fn latest_filename(base_path: &Path) -> PathBuf {
40        base_path.join("latest.log")
41    }
42
43    fn open_latest(base_path: &Path) -> Result<File> {
44        OpenOptions::new()
45            .create(true)
46            .write(true)
47            .truncate(true)
48            .open(Self::latest_filename(base_path))
49    }
50
51    fn archive_filename(base_path: &Path, date: DateTime<Local>) -> PathBuf {
52        let base_name = date.format("%Y-%m-%d-%H%M%S");
53        let mut path = base_path.join(format!("{base_name}.log"));
54        let mut suffix = 1;
55        while path.exists() {
56            path = base_path.join(format!("{base_name}-{suffix}.log"));
57            suffix += 1;
58        }
59        path
60    }
61
62    fn archive_existing_latest(base_path: &Path) -> Result<()> {
63        let latest = Self::latest_filename(base_path);
64        let Ok(metadata) = fs::metadata(&latest) else {
65            return Ok(());
66        };
67        if metadata.len() == 0 {
68            return Ok(());
69        }
70
71        let modified = metadata.modified()?;
72        let archive = Self::archive_filename(base_path, modified.into());
73        fs::rename(latest, archive)
74    }
75
76    fn rotate_latest(&mut self, now: DateTime<Utc>) -> Result<()> {
77        if let Some(mut file) = self.file.take() {
78            file.flush()?;
79        }
80
81        let latest = Self::latest_filename(&self.base_path);
82        if fs::metadata(&latest).is_ok_and(|metadata| metadata.len() > 0) {
83            let archive = Self::archive_filename(&self.base_path, DateTime::<Local>::from(now));
84            fs::rename(latest, archive)?;
85        }
86
87        self.date = now;
88        self.file = Some(Self::open_latest(&self.base_path)?);
89        Ok(())
90    }
91
92    fn check_time(&self, now: DateTime<Utc>) -> bool {
93        match self.rotation_time {
94            RotationTimeFormat::None => false,
95            RotationTimeFormat::Hourly => {
96                self.date.hour() != now.hour()
97                    || self.date.day() != now.day()
98                    || self.date.month() != now.month()
99                    || self.date.year() != now.year()
100            }
101            RotationTimeFormat::Daily => {
102                self.date.day() != now.day()
103                    || self.date.month() != now.month()
104                    || self.date.year() != now.year()
105            }
106            RotationTimeFormat::Weekly => {
107                now.signed_duration_since(self.date) >= chrono::TimeDelta::weeks(1)
108            }
109            RotationTimeFormat::Monthly => {
110                self.date.month() != now.month() || self.date.year() != now.year()
111            }
112        }
113    }
114
115    pub fn disable(&mut self) {
116        self.file = None;
117    }
118}
119
120impl Write for LogFile {
121    fn write(&mut self, buf: &[u8]) -> Result<usize> {
122        let now = Utc::now();
123        if self.check_time(now) {
124            self.rotate_latest(now)?;
125        }
126        let Some(file) = self.file.as_mut() else {
127            return Ok(buf.len());
128        };
129        file.write(buf)
130    }
131
132    fn flush(&mut self) -> Result<()> {
133        let Some(ref mut file) = self.file else {
134            return Ok(());
135        };
136        file.flush()
137    }
138}

161    #[test]
162    fn weekly_rotation_uses_elapsed_time_without_day_underflow() {
163        let date = Utc
164            .with_ymd_and_hms(2026, 1, 31, 0, 0, 0)
165            .single()
166            .expect("valid date");
167        let log_file = LogFile {
168            file: None,
169            date,
170            base_path: PathBuf::new(),
171            rotation_time: RotationTimeFormat::Weekly,
172        };
173
174        let next_day = Utc
175            .with_ymd_and_hms(2026, 2, 1, 0, 0, 0)
176            .single()
177            .expect("valid date");
178        let next_week = Utc
179            .with_ymd_and_hms(2026, 2, 7, 0, 0, 0)
180            .single()
181            .expect("valid date");
182
183        assert!(!log_file.check_time(next_day));
184        assert!(log_file.check_time(next_week));
185    }

187    #[test]
188    fn archive_filename_uses_collision_suffixes() {
189        let path = test_log_dir("collision");
190        fs::create_dir_all(&path).expect("test dir should be created");
191
192        let date = Local
193            .with_ymd_and_hms(2026, 6, 18, 14, 22, 3)
194            .single()
195            .expect("valid date");
196        fs::write(path.join("2026-06-18-142203.log"), "first")
197            .expect("archive placeholder should be written");
198        fs::write(path.join("2026-06-18-142203-1.log"), "second")
199            .expect("archive placeholder should be written");
200
201        let archive = LogFile::archive_filename(&path, date);
202
203        assert_eq!(
204            archive.file_name().and_then(|name| name.to_str()),
205            Some("2026-06-18-142203-2.log")
206        );
207        fs::remove_dir_all(path).expect("test dir should be removed");
208    }

210    #[test]
211    fn startup_archives_existing_non_empty_latest() {
212        let path = test_log_dir("startup");
213        fs::create_dir_all(&path).expect("test dir should be created");
214        let latest = path.join("latest.log");
215        fs::write(&latest, "previous run").expect("latest should be written");
216
217        LogFile::new(path.clone(), RotationTimeFormat::Daily, true).expect("log file should open");
218
219        assert!(latest.exists());
220        assert_eq!(
221            fs::read_to_string(&latest).expect("latest should be readable"),
222            ""
223        );
224        let archives = fs::read_dir(&path)
225            .expect("test dir should be readable")
226            .filter_map(StdResult::ok)
227            .filter(|entry| entry.file_name() != "latest.log")
228            .collect::<Vec<_>>();

230        assert_eq!(
231            fs::read_to_string(archives[0].path()).expect("archive should be readable"),
