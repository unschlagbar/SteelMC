16pub struct Completer {
17    pub enabled: bool,
18    pub error: bool,
19    pub selected: usize,
20    pub completed: String,
21    pub suggestions: Vec<String>,
22}
23impl Completer {
24    pub const fn new() -> Self {
25        Completer {
26            enabled: false,
27            error: false,
28            selected: 0,
29            completed: String::new(),
30            suggestions: vec![],
31        }
32    }
33}
34/// Modify suggestions
35impl Completer {
36    pub fn update(&mut self, out: &mut Output, pos: usize) {
37        let char_start = if pos == 0 {
38            0
39        } else {
40            let (start, size) = out.char_pos(pos.saturating_sub(1));
41            start + size
42        };
43        // Gets the right chars
44        let command = &out.text[..char_start];
45
46        let Some(server) = SERVER.get() else {
47            self.completed = String::new();
48            self.selected = 0;
49            self.error = true;
50            return;
51        };
52        // Gets the suggested commands
53        self.suggestions = server
54            .command_dispatcher
55            .read()
56            .handle_suggestions(CommandSender::Console, command, server.clone())
57            .0
58            .into_iter()
59            .map(|suggestion| suggestion.text)
60            .collect();
61        if self.suggestions.is_empty() {
62            self.completed = String::new();
63            self.selected = 0;
64            self.error = true;
65        } else {
66            self.selected = self.selected.min(self.suggestions.len() - 1);
67            self.error = false;
68        }
69    }
70    pub fn rewrite(&mut self, out: &mut Output, dir: Move) -> Result<()> {
71        out.cursor_to_relative(out.pos)?;
72        if out.is_at_end() {
73            write!(out, "{}", Clear(ClearType::UntilNewLine))?;
74        }
75        write!(out, "{SavePosition}\r\n")?;
76        write!(out, "{}", Clear(ClearType::FromCursorDown))?;
77        if self.suggestions.is_empty() {
78            write!(out, "{RestorePosition}")?;
79            out.flush()?;
80            return Ok(());
81        }
82
83        // Updates completion position
84        let len = self.suggestions.len();
85        match dir {
86            Move::Up => self.selected = (self.selected + len - 1) % len,
87            Move::Down => self.selected = (self.selected + 1) % len,
88            Move::None => (),
89        }
90
91        // Updates the screen width
92        let width = (super::terminal_width() / 20).max(1);
93        let completion_height = 3.min(terminal_height().saturating_sub(4));
94        let grid_size = width * completion_height;
95        if grid_size == 0 {
96            write!(out, "{RestorePosition}")?;
97            out.flush()?;
98            return Ok(());
99        }
100        let start = (self.selected.checked_div(grid_size).unwrap_or(0)) * grid_size;
101        for h in 0..completion_height {
102            write!(out, "\r")?;
103            for w in 0..width {
104                let pos = start + w * completion_height + h;
105                if pos >= self.suggestions.len() {
106                    break;
107                }
108
109                let color = if pos == self.selected {
110                    Yellow
111                } else {
112                    DarkGrey
113                };
114
115                write!(
116                    out,
117                    "{}{:<20}{}",
118                    SetForegroundColor(color),
119                    if self.suggestions[pos].len() > 20 {
120                        format!("{}...", &self.suggestions[pos][..17])
121                    } else {
122                        self.suggestions[pos].clone()
123                    },
124                    ResetColor
125                )?;
126            }
127            if h + 1 < completion_height {
128                writeln!(out)?;
129            }
130        }
131        write!(out, "{RestorePosition}")?;
132
133        let char_pos = if out.is_at_start() {
134            0
135        } else {
136            let (pos, char) = out.char_pos(out.pos.saturating_sub(1));
137            pos + char
138        };
139        let text = if let Some(text) = out.text[..char_pos].split_whitespace().last()
140            && let Some(striped) = self.suggestions[self.selected].strip_prefix(text)
141        {
142            striped
143        } else {
144            &self.suggestions[self.selected]
145        };
146        self.completed = text.to_string();
147        out.flush()?;
148
149        if !out.is_at_end() {
150            return Ok(());
151        }
152        write!(
153            out,
154            "{SavePosition}{}{}{RestorePosition}",
155            SetForegroundColor(DarkGrey),
156            self.completed
157        )?;
158        out.flush()
159    }
160}
