1use std::io::{Result, Write};
2use steel_macros::ClientPacket;
3use steel_registry::RegistryExt;
4use steel_registry::packets::play::C_SET_TIME;
5use steel_utils::codec::{VarInt, VarLong};
6use steel_utils::serial::WriteTo;
7
8#[derive(ClientPacket, Clone, Debug)]
9#[packet_id(Play = C_SET_TIME)]
10pub struct CSetTime {
11    pub game_time: i64,
12    /// (clock_registry_id, total_ticks, partial_tick, rate)
13    pub clock_updates: Vec<(i32, i64, f32, f32)>,
14}
15
16impl WriteTo for CSetTime {
17    fn write(&self, writer: &mut impl Write) -> Result<()> {
18        self.game_time.write(writer)?;
19        VarInt(self.clock_updates.len() as i32).write(writer)?;
20        for &(clock_id, total_ticks, partial_tick, rate) in &self.clock_updates {
21            VarInt(clock_id).write(writer)?;
22            VarLong(total_ticks).write(writer)?;
23            partial_tick.write(writer)?;
24            rate.write(writer)?;
25        }
26        Ok(())
27    }
28}
29
30impl CSetTime {
31    #[must_use]
32    pub fn new(game_time: i64, day_time: i64, partial_tick: f32, rate: f32) -> Self {
33        use steel_registry::{REGISTRY, vanilla_world_clocks};
34        let clock_id = REGISTRY
35            .world_clocks
36            .id_from_key(&vanilla_world_clocks::OVERWORLD.key)
37            .unwrap() as i32;
38        Self {
39            game_time,
40            clock_updates: vec![(clock_id, day_time, partial_tick, rate)],
41        }
42    }
43}
