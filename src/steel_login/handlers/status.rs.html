1//! Status state packet handlers (server list ping).
2
3use steel_core::config::RuntimeConfig;
4use steel_protocol::packets::{
5    common::{CPongResponse, SPingRequest},
6    status::{CStatusResponse, Players, Sample, Status, Version},
7};
8use steel_registry::packets::CURRENT_MC_PROTOCOL;
9use steel_utils::MC_VERSION;
10
11use crate::tcp_client::JavaTcpClient;
12
13impl JavaTcpClient {
14    /// Handles a status request from the client.
15    pub async fn handle_status_request(&self) {
16        let res_packet = CStatusResponse::new(Status {
17            description: self.server.config.motd.clone(),
18            players: Some(Players {
19                max: self.server.config.max_players.cast_signed(),
20                online: self.server.player_count() as i32,
21                sample: self
22                    .server
23                    .player_sample()
24                    .into_iter()
25                    .map(|(name, id)| Sample { name, id })
26                    .collect(),
27            }),
28            enforce_secure_chat: self.server.config.enforce_secure_chat,
29            favicon: load_favicon(&self.server.config),
30            version: Some(Version {
31                name: MC_VERSION,
32                protocol: CURRENT_MC_PROTOCOL,
33            }),
34        });
35        self.send_bare_packet_now(res_packet).await;
36    }
37
38    /// Handles a ping request from the client.
39    pub async fn handle_ping_request(&self, packet: SPingRequest) {
40        self.send_bare_packet_now(CPongResponse::new(packet.time))
41            .await;
42        self.close();
43    }
44}
45
46/// Loads the favicon from config.
47fn load_favicon(config: &RuntimeConfig) -> Option<String> {
48    use base64::{Engine, prelude::BASE64_STANDARD};
49    use std::fs;
50    use std::path::Path;
51
52    const ICON_PREFIX: &str = "data:image/png;base64,";
53
54    if !config.use_favicon {
55        return None;
56    }
57
58    let path = Path::new(&config.favicon);
59    let Ok(icon) = fs::read(path) else {
60        return None;
61    };
62
63    let cap = ICON_PREFIX.len() + icon.len().div_ceil(3) * 4;
64    let mut base64 = String::with_capacity(cap);
65    base64 += ICON_PREFIX;
66    BASE64_STANDARD.encode_string(icon, &mut base64);
67    Some(base64)
68}
