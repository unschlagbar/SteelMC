23    /// Handles a custom payload packet during the configuration state.

25    pub fn handle_config_custom_payload(&self, packet: SCustomPayload) {
26        log::debug!("Custom payload packet: {packet:?}");
27    }

29    /// Handles the client information packet during the configuration state.
30    pub async fn handle_client_information(&self, packet: SClientInformation) {
31        log::debug!("Client information packet: {packet:?}");
32
33        // Convert packet to our ClientInformation struct and store it
34        let info = ClientInformation {
35            language: packet.language,
36            view_distance: packet
37                .view_distance
38                .clamp(2, i32::from(self.server.config.view_distance).max(2))
39                as u8,
40            chat_visibility: packet.chat_visibility,
41            chat_colors: packet.chat_colors,
42            model_customization: packet.model_customization,
43            main_hand: packet.main_hand,
44            text_filtering_enabled: packet.text_filtering_enabled,
45            allows_listing: packet.allows_listing,
46            particle_status: packet.particle_status,
47        };
48
49        *self.client_information.lock().await = info;
50    }
51
52    /// Starts the configuration process by sending initial packets.
53    pub async fn start_configuration(&self) {
54        self.send_bare_packet_now(CCustomPayload::new(
55            Identifier::vanilla_static("brand"),
56            Box::new(BRAND_PAYLOAD),
57        ))
58        .await;
59
60        // Send server links if enabled and configured
61        if let Some(server_links) = self.server.config.server_links_packet() {
62            self.send_bare_packet_now(server_links).await;
63        }
64
65        self.send_bare_packet_now(CSelectKnownPacks::new(vec![KnownPack::new(
66            "minecraft".to_string(),
67            "core".to_string(),
68            steel_utils::MC_VERSION.to_string(),
69        )]))
70        .await;
71    }
72
73    /// Handles the select known packs packet during the configuration state.
74    pub async fn handle_select_known_packs(&self, packet: SSelectKnownPacks) {
75        log::debug!("Select known packs packet: {packet:?}");
76
77        let registry_cache = self.server.registry_cache.registry_packets.clone();
78        for encoded_packet in registry_cache.iter() {
79            self.send_packet_now(encoded_packet).await;
80        }
81
82        // Send the packet for tags
83        self.send_packet_now(&self.server.registry_cache.tags_packet)
84            .await;
85
86        // Finish configuration with CFinishConfigurationPacket
87        self.send_bare_packet_now(CFinishConfiguration {}).await;
88    }

90    /// Finishes the configuration process and transitions to the play state.
91    ///
92    /// # Panics
93    /// This function will panic if the game profile is empty, should be impossible at this point.
94    pub(crate) async fn finish_configuration(&self) -> ConnectionAction {
95        self.protocol.store(ConnectionProtocol::Play);
96
97        let gameprofile = self
98            .gameprofile
99            .lock()
100            .await
101            .clone()
102            .expect("Game profile is empty");
103
104        let client_info = self.client_information.lock().await.clone();
105
106        let world = self.server.overworld().clone();
107        let entity_id = next_entity_id();
108
109        let player = Arc::new_cyclic(|player_weak| {
110            let java_connection = JavaConnection::new(
111                self.outgoing_queue.clone(),
112                self.cancel_token.clone(),
113                self.compression.load(),
114                self.network_writer.clone(),
115                self.id,
116                player_weak.clone(),
117            );
118            let connection = Arc::new(PlayerConnection::Java(java_connection));
119
120            Player::new(
121                gameprofile,
122                connection,
123                world,
124                Arc::downgrade(&self.server),
125                self.server.config.clone(),
126                entity_id,
127                player_weak,
128                client_info,
129            )
130        });
131
132        let connection = Arc::clone(&player.connection);
133        if self
134            .connection_updates
135            .send(ConnectionUpdate::Upgrade(Arc::clone(&connection)))
136            .is_err()
137        {
138            self.kick("Failed to update connection state".into()).await;
139            return ConnectionAction::none();
140        }
141
142        tokio::select! {
143            () = self.connection_updated.notified() => {}
144            () = self.cancel_token.cancelled() => return ConnectionAction::none(),
145        }
146        self.server.queue_player_join(player);
147
148        ConnectionAction::upgrade(connection)
149    }
