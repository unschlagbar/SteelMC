45fn empty_postprocessing(height: i32) -> Box<[Vec<u16>]> {
46    let section_count = (height / 16) as usize;
47    (0..section_count).map(|_| Vec::new()).collect()
48}

97    /// Ticks this chunk, processing scheduled and random block ticks.
98    ///
99    /// For each section that contains randomly-ticking blocks, selects
100    /// `random_tick_speed` random blocks and calls their `random_tick` behavior.
101    /// # Arguments
102    /// * `random_tick_speed` - Number of random blocks to tick per section per tick.
103    ///   This is controlled by the `randomTickSpeed` game rule.
104    /// * `tick_count` - Current server tick count (for entity sync timing).
105    ///
106    /// # Panics
107    /// Panics if the block behavior registry has not been initialized.
108    pub fn tick(
109        &self,
110        random_tick_speed: u32,
111        _tick_count: i32,
112        ready_block_ticks: &mut Vec<BlockTick>,
113        ready_fluid_ticks: &mut Vec<FluidTick>,
114    ) {
115        self.drain_ready_scheduled_ticks(ready_block_ticks, ready_fluid_ticks);
116        self.tick_random_blocks(random_tick_speed);
117    }

119    /// Drains ready scheduled block and fluid ticks into the provided buffers.
120    pub fn drain_ready_scheduled_ticks(
121        &self,
122        ready_block_ticks: &mut Vec<BlockTick>,
123        ready_fluid_ticks: &mut Vec<FluidTick>,
124    ) {
125        ready_block_ticks.extend(self.block_ticks.lock().drain_ready());
126        ready_fluid_ticks.extend(self.fluid_ticks.lock().drain_ready());
127    }

129    /// Runs vanilla random block ticks for this chunk.
130    pub fn tick_random_blocks(&self, random_tick_speed: u32) {
131        if random_tick_speed == 0 {
132            return;
133        }
134
135        for (section_index, section) in self.sections.sections.iter().enumerate() {
136            // Skip sections with no randomly-ticking blocks (lock-free check)
137            if !section.is_randomly_ticking() {
138                continue;
139            }
140
141            let Some(world) = self.get_level() else {
142                return;
143            };
144
145            let block_behaviors = &*BLOCK_BEHAVIORS;
146            let mut rng = rand::rng();
147            let chunk_base_x = self.pos.0.x * 16;
148            let chunk_base_z = self.pos.0.y * 16;
149
150            let section_base_y = self.min_y + (section_index as i32 * 16);
151
152            // Collect blocks to tick while holding the read lock, then release it
153            // before calling random_tick to avoid deadlock (random_tick may call set_block)
154            let blocks_to_tick: Vec<(BlockStateId, BlockPos)> = {
155                let section_guard = section.read();
156
157                let mut blocks = Vec::with_capacity(random_tick_speed as usize);
158
159                for _ in 0..random_tick_speed {
160                    let local_x = rng.random_range(0..16);
161                    let local_y = rng.random_range(0..16);
162                    let local_z = rng.random_range(0..16);
163
164                    let state = section_guard.states.get(local_x, local_y, local_z);
165
166                    if block_behaviors
167                        .get_behavior(state.get_block())
168                        .is_randomly_ticking(state)
169                    {
170                        let pos = BlockPos::new(
171                            chunk_base_x + local_x as i32,
172                            section_base_y + local_y as i32,
173                            chunk_base_z + local_z as i32,
174                        );
175                        blocks.push((state, pos));
176                    }
177                }
178
179                blocks
180            }; // section_guard dropped here
181
182            // Now process the collected blocks without holding any lock
183            for (state, pos) in blocks_to_tick {
184                let behavior = block_behaviors.get_behavior(state.get_block());
185                behavior.random_tick(state, &world, pos);
186            }
187        }
188    }

190    /// Creates a new `LevelChunk` from a `ProtoChunk`.
191    ///
192    /// Transfers final heightmaps from the proto chunk if available.
193    /// Recalculates section block counts for random tick optimization.
194    ///
195    /// # Arguments
196    /// * `proto_chunk` - The proto chunk to convert
197    /// * `min_y` - The minimum Y coordinate of the world
198    /// * `height` - The total height of the world
199    /// * `level` - Weak reference to the world (mirrors Java's `LevelChunk.level`)
200    ///
201    /// # Panics
202    /// Panics if the block behavior registry has not been initialized.
203    #[must_use]
204    pub fn from_proto(
205        proto_chunk: ProtoChunk,
206        min_y: i32,
207        height: i32,
208        level: Weak<World>,
209    ) -> LevelChunkPromotion {
210        // Ensure full chunks always have populated final heightmaps. Some stages
211        // may not touch blocks (carvers are currently empty), so lazy final
212        // heightmaps are not guaranteed to exist before promotion.
213        let mut proto_heightmaps = proto_chunk.heightmaps.into_inner();
214        proto_heightmaps.prime_from_sections(
215            HeightmapType::final_types(),
216            min_y,
217            height,
218            &proto_chunk.sections.sections,
219        );
220        let chunk_heightmaps = ChunkHeightmaps::from_proto(&mut proto_heightmaps, min_y, height);
221
222        // Recalculate section counts for random tick optimization
223        for section in &proto_chunk.sections.sections {
224            section.write().recalculate_counts();
225        }
226
227        let structure_starts = proto_chunk.structure_starts.into_inner();
228        let structure_references = proto_chunk.structure_references.into_inner();
229        let postprocessing = proto_chunk.postprocessing.into_inner();
230        let block_ticks = proto_chunk.block_ticks.into_inner();
231        let fluid_ticks = proto_chunk.fluid_ticks.into_inner();
232        let block_entities = proto_chunk.block_entities;
233        let pending_entities = proto_chunk.entities.get_all();
234        let sky_light_sources = proto_chunk.sky_light_sources.into_inner();
235        let mut light = proto_chunk.light.into_inner();
236        if let Err(error) = light.refresh_emptiness_maps_from_sections(&proto_chunk.sections) {
237            panic!("invalid proto chunk light emptiness map length: {error:?}");
238        }
239
240        Self::populate_poi(&level, &proto_chunk.sections, proto_chunk.pos, min_y);
241
242        let chunk = Self {
243            sections: proto_chunk.sections,
244            pos: proto_chunk.pos,
245            dirty: AtomicBool::new(proto_chunk.dirty.load(Ordering::Acquire)),
246            heightmaps: SyncRwLock::new(chunk_heightmaps),
247            min_y,
248            height,
249            level,
250            block_entities,
251            block_ticks: SyncMutex::new(block_ticks),
252            fluid_ticks: SyncMutex::new(fluid_ticks),
253            structure_starts: SyncRwLock::new(structure_starts),
254            structure_references: SyncRwLock::new(structure_references),
255            postprocessing: SyncMutex::new(postprocessing),
256            sky_light_sources: SyncRwLock::new(sky_light_sources),
257            light: SyncRwLock::new(light),
258        };
259        LevelChunkPromotion {
260            chunk,
261            pending_entities,
262        }
263    }

265    /// Creates a new `LevelChunk` that was loaded from disk (not dirty).
266    ///
267    /// Recalculates section block counts for random tick optimization.
268    ///
269    /// # Arguments
270    /// * `sections` - The chunk sections
271    /// * `pos` - The chunk position
272    /// * `min_y` - The minimum Y coordinate of the world
273    /// * `height` - The total height of the world
274    /// * `level` - Weak reference to the world (mirrors Java's `LevelChunk.level`)
275    /// * `block_ticks` - Scheduled block ticks loaded from disk
276    /// * `fluid_ticks` - Scheduled fluid ticks loaded from disk
277    /// * `heightmaps` - Heightmaps loaded from disk
278    /// * `light` - Chunk-owned light data loaded from disk
279    ///
280    /// # Panics
281    /// Panics if the block behavior registry has not been initialized.
282    #[must_use]
283    #[expect(
284        clippy::too_many_arguments,
285        reason = "all parameters are required to fully restore a chunk from disk"
286    )]
287    pub fn from_disk(
288        sections: Sections,
289        pos: ChunkPos,
290        min_y: i32,
291        height: i32,
292        level: Weak<World>,
293        block_ticks: BlockTickList,
294        fluid_ticks: FluidTickList,
295        heightmaps: ChunkHeightmaps,
296        structure_starts: StructureStartMap,
297        structure_references: StructureReferenceMap,
298        mut light: ChunkLightData,
299    ) -> Self {
300        // Recalculate section counts for random tick optimization
301        for section in &sections.sections {
302            section.write().recalculate_counts();
303        }
304        if let Err(error) = light.refresh_emptiness_maps_from_sections(&sections) {
305            panic!("invalid loaded chunk light emptiness map length: {error:?}");
306        }
307        let sky_light_sources = {
308            let mut sources = ChunkSkyLightSources::for_valid_world_height(min_y, height);
309            sources.fill_from_sections(&sections);
310            sources
311        };
312
313        Self::populate_poi(&level, &sections, pos, min_y);
314
315        Self {
316            sections,
317            pos,
318            dirty: AtomicBool::new(false),
319            heightmaps: SyncRwLock::new(heightmaps),
320            min_y,
321            height,
322            level,
323            block_entities: BlockEntityStorage::new(),
324            block_ticks: SyncMutex::new(block_ticks),
325            fluid_ticks: SyncMutex::new(fluid_ticks),
326            structure_starts: SyncRwLock::new(structure_starts),
327            structure_references: SyncRwLock::new(structure_references),
328            postprocessing: SyncMutex::new(empty_postprocessing(height)),
329            sky_light_sources: SyncRwLock::new(sky_light_sources),
330            light: SyncRwLock::new(light),
331        }
332    }

359    /// Drains the vanilla proto postprocessing offsets carried through promotion.
360    pub(crate) fn take_postprocessing(&self) -> Option<Box<[Vec<u16>]>> {
361        let mut postprocessing = self.postprocessing.lock();
362        if postprocessing.iter().all(Vec::is_empty) {
363            return None;
364        }
365
366        Some(mem::replace(
367            &mut *postprocessing,
368            empty_postprocessing(self.height),
369        ))
370    }

372    /// Runs vanilla proto postprocessing after this chunk has been promoted to full.
373    pub(crate) fn post_process_generation(
374        world: &Arc<World>,
375        chunk_pos: ChunkPos,
376        min_y: i32,
377        postprocessing: Box<[Vec<u16>]>,
378    ) {
379        for (section_index, packed_offsets) in postprocessing.into_vec().into_iter().enumerate() {
380            if packed_offsets.is_empty() {
381                continue;
382            }
383
384            let section_y = Self::section_y_from_section_index(min_y, section_index);
385            for packed in packed_offsets {
386                let pos = ProtoChunk::unpack_postprocessing_offset(packed, section_y, chunk_pos);
387                let state = world.get_postprocessing_block_state(pos);
388                let fluid_state = state.get_fluid_state();
389
390                if !fluid_state.is_empty() {
391                    FLUID_BEHAVIORS
392                        .get_behavior(fluid_state.fluid_id)
393                        .tick(world, pos);
394                }
395
396                if state.get_block().config.liquid {
397                    BLOCK_BEHAVIORS
398                        .get_behavior(state.get_block())
399                        .tick(state, world, pos);
400                } else {
401                    let new_state = Self::update_from_neighbor_shapes(world, state, pos);
402                    if new_state != state {
403                        let flags = UpdateFlags::UPDATE_INVISIBLE
404                            | UpdateFlags::UPDATE_KNOWN_SHAPE
405                            | UpdateFlags::UPDATE_SKIP_BLOCK_ENTITY_SIDEEFFECTS;
406                        world.set_block(pos, new_state, flags);
407                    }
408                }
409            }
410        }
411    }

413    fn update_from_neighbor_shapes(
414        world: &Arc<World>,
415        state: BlockStateId,
416        pos: BlockPos,
417    ) -> BlockStateId {
418        let mut updated = state;
419        for direction in Direction::UPDATE_SHAPE_ORDER {
420            let neighbor_pos = pos.relative(direction);
421            let neighbor_state = world.get_postprocessing_block_state(neighbor_pos);
422            let behavior = BLOCK_BEHAVIORS.get_behavior(updated.get_block());
423            updated =
424                behavior.update_shape(updated, world, pos, direction, neighbor_pos, neighbor_state);
425        }
426        updated
427    }

429    /// Scans chunk sections for POI block states and populates world POI storage.
430    fn populate_poi(level: &Weak<World>, sections: &Sections, pos: ChunkPos, min_y: i32) {
431        let Some(world) = level.upgrade() else {
432            return;
433        };
434
435        // Palette pre-check WITHOUT the global POI lock: collect only the
436        // sections that actually contain POI blocks. The vast majority of
437        // worldgen chunks have none, so they never touch the (heavily
438        // contended) `poi_storage` mutex and never do a per-block scan.
439        // `Vec::new()` doesn't allocate until the first push, so the common
440        // empty case is allocation-free.
441        let mut poi_sections: Vec<(usize, SectionPos)> = Vec::new();
442        for (i, section) in sections.sections.iter().enumerate() {
443            if section.read().contains_poi() {
444                let section_y = min_y / 16 + i as i32;
445                poi_sections.push((i, SectionPos::new(pos.0.x, section_y, pos.0.y)));
446            }
447        }
448        if poi_sections.is_empty() {
449            return;
450        }
451
452        let mut poi_storage = world.poi_storage.lock();
453        for (i, section_pos) in poi_sections {
454            let guard = sections.sections[i].read();
455            poi_storage.scan_and_populate(&guard, section_pos);
456        }
457    }

551    /// Ticks all ticking block entities in this chunk.
552    ///
553    /// Called each game tick for chunks that are in ticking range.
554    pub fn tick_block_entities(&self) {
555        let Some(world) = self.get_level() else {
556            return;
557        };
558
559        // Get entities to tick (already filters out removed)
560        let entities = self.block_entities.get_tickers();
561
562        // Tick each entity
563        for entity in entities {
564            let action = {
565                let mut guard = entity.lock();
566                if guard.is_removed() {
567                    continue;
568                }
569                guard.tick(&world)
570            };
571
572            if let Some(action) = action {
573                match action {
574                    BlockEntityTickAction::SetBlock {
575                        pos,
576                        state,
577                        flags,
578                        game_event,
579                    } => {
580                        world.set_block(pos, state, flags);
581                        if let Some((event, event_state)) = game_event {
582                            world.game_event(
583                                event,
584                                pos,
585                                &GameEventContext::new(None, Some(event_state)),
586                            );
587                        }
588                    }
589                }
590            }
591        }
592
593        // Clean up removed entities from the ticking list
594        self.block_entities.cleanup_tickers();
595    }

597    /// Sets a block state at the given position.
598    ///
599    /// Returns the old block state, or `None` if nothing changed.
600    ///
601    /// # Arguments
602    /// * `pos` - The absolute block position
603    /// * `state` - The new block state to set
604    /// * `flags` - Update flags controlling behavior
605    ///
606    /// # Panics
607    ///
608    /// Panics if the behavior registry has not been initialized.
609    #[must_use]
610    #[expect(
611        clippy::too_many_lines,
612        reason = "block mutation keeps vanilla side effects in one ordered transaction"
613    )]
614    pub fn set_block_state(
615        &self,
616        pos: BlockPos,
617        state: BlockStateId,
618        flags: UpdateFlags,
619    ) -> Option<BlockStateId> {
620        let y = pos.0.y;
621
622        if y < self.min_y || y >= self.min_y + self.height {
623            return None;
624        }
625
626        let section_index = self.get_section_index(y);
627
628        if section_index >= self.sections.sections.len() {
629            return None;
630        }
631
632        let section = &self.sections.sections[section_index];
633
634        let local_x = (pos.0.x & 15) as usize;
635        let local_y = (y & 15) as usize;
636        let local_z = (pos.0.z & 15) as usize;
637
638        let (old_state, was_empty, is_empty) = {
639            let mut section_guard = section.write();
640            let was_empty = section_guard.is_empty();
641            let old_state = section_guard.set_block_state(local_x, local_y, local_z, state);
642            let is_empty = section_guard.is_empty();
643            (old_state, was_empty, is_empty)
644        };
645
646        if old_state == state {
647            return None;
648        }
649
650        let min_y = self.min_y;
651        let sections = &self.sections;
652        self.heightmaps
653            .write()
654            .update(local_x, y, local_z, state, |lx, scan_y, lz| {
655                let scan_section_index = ((scan_y - min_y) / 16) as usize;
656                let scan_local_y = ((scan_y - min_y) % 16) as usize;
657                sections.sections[scan_section_index]
658                    .read()
659                    .states
660                    .get(lx, scan_local_y, lz)
661            });
662
663        let old_block = old_state.get_block();
664        let new_block = state.get_block();
665
666        let empty_section_change = if was_empty == is_empty {
667            None
668        } else {
669            self.update_light_section_emptiness(y, is_empty);
670            Some(LightSectionEmptinessChange {
671                section_pos: SectionPos::new(
672                    self.pos.0.x,
673                    SectionPos::block_to_section_coord(y),
674                    self.pos.0.y,
675                ),
676                empty: is_empty,
677            })
678        };
679
680        let light_properties_changed = has_different_light_properties(old_state, state);
681        if light_properties_changed {
682            self.update_sky_light_sources(local_x, y, local_z);
683        }
684
685        // Re-read the block to verify it wasn't changed concurrently
686        let current_block = section
687            .read()
688            .states
689            .get(local_x, local_y, local_z)
690            .get_block();
691        if current_block != new_block {
692            return None;
693        }
694
695        if let Some(level) = self.get_level() {
696            if light_properties_changed || empty_section_change.is_some() {
697                level.queue_light_change_after_block_set(
698                    pos,
699                    old_state,
700                    state,
701                    empty_section_change,
702                );
703            }
704
705            // Update POI storage when block states change
706            level
707                .poi_storage
708                .lock()
709                .on_block_state_change(pos, old_state, state);
710            let block_changed = old_block != new_block;
711            let moved_by_piston = flags.contains(UpdateFlags::UPDATE_MOVE_BY_PISTON);
712            let side_effects = !flags.contains(UpdateFlags::UPDATE_SKIP_BLOCK_ENTITY_SIDEEFFECTS);
713
714            let block_behaviors = &*BLOCK_BEHAVIORS;
715            let old_behavior = block_behaviors.get_behavior(old_block);
716            let new_behavior = block_behaviors.get_behavior(new_block);
717
718            // Block entity removal when block type changes
719            if block_changed && old_behavior.has_block_entity() {
720                let should_keep = new_behavior.should_keep_block_entity(old_state, state);
721                if !should_keep {
722                    if side_effects && let Some(block_entity) = self.get_block_entity(pos) {
723                        block_entity.lock().pre_remove_side_effects(pos, old_state);
724                    }
725                    self.remove_block_entity(pos);
726                }
727            }
728
729            // Notify neighbors that we were removed (for rails, etc.)
730            if block_changed && (flags.contains(UpdateFlags::UPDATE_NEIGHBORS) || moved_by_piston) {
731                old_behavior.affect_neighbors_after_removal(
732                    old_state,
733                    &level,
734                    pos,
735                    moved_by_piston,
736                );
737            }
738
739            // Call on_place for the new block
740            if !flags.contains(UpdateFlags::UPDATE_SKIP_ON_PLACE) {
741                new_behavior.on_place(state, &level, pos, old_state, moved_by_piston);
742            }
743
744            // Block entity creation after on_place
745            if new_behavior.has_block_entity() {
746                if let Some(existing) = self.get_block_entity(pos) {
747                    // Update existing block entity's state
748                    existing.lock().set_block_state(state);
749                    self.update_block_entity_ticker(&existing);
750                } else {
751                    // Create new block entity
752                    if let Some(entity) =
753                        new_behavior.new_block_entity(self.level.clone(), pos, state)
754                    {
755                        self.add_and_register_block_entity(entity);
756                    }
757                }
758            }
759        }
760
761        self.mark_unsaved();
762        Some(old_state)
763    }

765    fn update_light_section_emptiness(&self, y: i32, is_empty: bool) {
766        let section_y = SectionPos::block_to_section_coord(y);
767        self.light.write().set_section_empty(section_y, is_empty);
768    }

770    fn update_sky_light_sources(&self, local_x: usize, y: i32, local_z: usize) {
771        let chunk_min_x = self.pos.0.x * 16;
772        let chunk_min_z = self.pos.0.y * 16;
773        self.sky_light_sources
774            .write()
775            .update(local_x, y, local_z, |scan_x, scan_y, scan_z| {
776                self.get_block_state(BlockPos::new(
777                    chunk_min_x + scan_x as i32,
778                    scan_y,
779                    chunk_min_z + scan_z as i32,
780                ))
781            });
782    }

794    /// Gets a block state at the given position.
795    #[must_use]
796    pub fn get_block_state(&self, pos: BlockPos) -> BlockStateId {
797        let y = pos.0.y;
798        let section_index = self.get_section_index(y);
799
800        // Bounds check - return air if out of range
801        if section_index >= self.sections.sections.len() {
802            return REGISTRY.blocks.get_base_state_id(&vanilla_blocks::VOID_AIR);
803        }
804
805        let section = &self.sections.sections[section_index];
806        let section_guard = section.read();
807
808        if section_guard.is_empty() {
809            return REGISTRY.blocks.get_base_state_id(&vanilla_blocks::VOID_AIR);
810        }
811
812        let local_x = (pos.0.x & 15) as usize;
813        let local_y = (y & 15) as usize;
814        let local_z = (pos.0.z & 15) as usize;
815
816        section_guard.states.get(local_x, local_y, local_z)
817    }

819    /// Extracts the chunk data for sending to the client.
820    #[must_use]
821    pub fn extract_chunk_data(&self) -> ChunkPacketData {
822        let data = Vec::new();
823
824        let mut cursor = Cursor::new(data);
825        self.sections.sections.iter().for_each(|section| {
826            section.read().write(&mut cursor);
827        });
828
829        let heightmaps_guard = self.heightmaps.read();
830
831        // Collect block entity data for client sync
832        let block_entities: Vec<BlockEntityInfo> = self
833            .block_entities
834            .get_all()
835            .iter()
836            .map(|entity| {
837                let guard = entity.lock();
838                let pos = guard.get_block_pos();
839                let type_id = guard.get_type().id() as i32;
840                let update_tag = guard.get_update_tag();
841
842                BlockEntityInfo {
843                    packed_xz: PackedChunkLocalXZ::from_block_pos(pos),
844                    y: pos.0.y as i16,
845                    type_id,
846                    data: update_tag.into(),
847                }
848            })
849            .collect();
850
851        ChunkPacketData {
852            heightmaps: Heightmaps {
853                heightmaps: vec![
854                    (
855                        ProtocolHeightmapType::WorldSurface,
856                        heightmaps_guard
857                            .get(HeightmapType::WorldSurface)
858                            .get_raw_data(),
859                    ),
860                    (
861                        ProtocolHeightmapType::MotionBlocking,
862                        heightmaps_guard
863                            .get(HeightmapType::MotionBlocking)
864                            .get_raw_data(),
865                    ),
866                    (
867                        ProtocolHeightmapType::MotionBlockingNoLeaves,
868                        heightmaps_guard
869                            .get(HeightmapType::MotionBlockingNoLeaves)
870                            .get_raw_data(),
871                    ),
872                ],
873            },
874            data: cursor.into_inner(),
875            block_entities,
876        }
877    }

891    use steel_registry::test_support::init_test_registry;

895    use crate::behavior::init_behaviors;

902    #[test]
903    fn extract_light_data_uses_chunk_owned_light_and_skylight_flag() {
904        init_test_registry();
905        init_behaviors();
906        let proto = ProtoChunk::new(
907            Sections::from_owned(vec![ChunkSection::new_empty()].into_boxed_slice()),
908            ChunkPos::new(0, 0),
909            0,
910            16,
911            Weak::new(),
912        );
913        let chunk = LevelChunk::from_proto(proto, 0, 16, Weak::new()).chunk;
914
915        {
916            let mut light = chunk.light.write();
917            let Some(sky_section) = light.sky.section_mut(0) else {
918                panic!("single-section light range should contain section 0");
919            };
920            *sky_section = LightSection::visible(LightSectionData::homogeneous(15));
921
922            let Some(block_section) = light.block.section_mut(0) else {
923                panic!("single-section light range should contain section 0");
924            };
925            let mut block_data = LightSectionData::homogeneous(0);
926            block_data.set(1, 2, 3, 12);
927            *block_section = LightSection::visible(block_data);
928        }
929
930        let with_sky = chunk.extract_light_data(true);
931        assert_eq!(with_sky.sky_y_mask.0[0] & 0b10, 0b10);
932        assert_eq!(with_sky.block_y_mask.0[0] & 0b10, 0b10);
933        assert_eq!(with_sky.sky_updates.len(), 1);
934        assert_eq!(with_sky.block_updates.len(), 1);
935        assert!(with_sky.sky_updates[0].iter().all(|byte| *byte == 0xff));
936
937        let without_sky = chunk.extract_light_data(false);
938        assert_eq!(without_sky.sky_y_mask.0[0], 0);
939        assert!(without_sky.sky_updates.is_empty());
940        assert_eq!(without_sky.block_y_mask.0[0] & 0b10, 0b10);
941        assert_eq!(without_sky.block_updates.len(), 1);
942    }
