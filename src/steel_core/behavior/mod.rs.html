1//! Block and item behavior system.
2//!
3//! This module contains the behavior traits and registries that define how
4//! blocks and items behave dynamically. This is separate from the static data
5//! in steel-registry to maintain a clean separation between constant data and
6//! functional/dynamic behavior.
7//!
8//! # Architecture
9//!
10//! After the main registry (`steel-registry`) is frozen, behavior registries
11//! are created:
12//! - `BlockBehaviorRegistry` - assigns default or custom behaviors to each block
13//! - `ItemBehaviorRegistry` - assigns default or custom behaviors to each item
14//!
15//! # Usage
16//!
17//! ```ignore
18//! use steel_core::behavior::{init_behaviors, BLOCK_BEHAVIORS, ITEM_BEHAVIORS};
19//!
20//! // After registry is frozen, call once at startup:
21//! init_behaviors();
22//!
23//! // Then access behaviors via the global registries:
24//! let behavior = BLOCK_BEHAVIORS.get_behavior(block);
25//! ```

91/// Wrapper for the global block behavior registry that implements `Deref`.
92pub struct BlockBehaviorLock(OnceLock<BlockBehaviorRegistry>);
93
94impl Deref for BlockBehaviorLock {
95    type Target = BlockBehaviorRegistry;
96
97    fn deref(&self) -> &Self::Target {
98        self.0.get().expect("Block behaviors not initialized")
99    }
100}

113/// Extension trait for `BlockStateId` that provides access to behavior-dependent methods.
114///
115/// This is separate from `BlockStateExt` (in steel-registry) because these methods
116/// require access to the behavior registry which lives in steel-core.
117pub trait BlockStateBehaviorExt {
118    /// Returns the fluid state for this block state.
119    ///
120    /// Delegates to the block's `BlockBehavior::get_fluid_state` implementation.
121    fn get_fluid_state(&self) -> FluidState;
122
123    /// Returns whether this block state is pathfindable for the supplied vanilla computation type.
124    fn is_pathfindable(&self, computation_type: PathComputationType) -> bool;
125}
126
127impl BlockStateBehaviorExt for BlockStateId {
128    fn get_fluid_state(&self) -> FluidState {
129        let block = self.get_block();
130        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
131        behavior.get_fluid_state(*self)
132    }
133
134    fn is_pathfindable(&self, computation_type: PathComputationType) -> bool {
135        let block = self.get_block();
136        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
137        behavior.is_pathfindable(*self, computation_type)
138    }
139}

141/// Global block behavior registry.
142///
143/// Access behaviors directly via deref: `BLOCK_BEHAVIORS.get_behavior(block)`
144pub static BLOCK_BEHAVIORS: BlockBehaviorLock = BlockBehaviorLock(OnceLock::new());

146/// Global item behavior registry.
147///
148/// Access behaviors directly via deref: `ITEM_BEHAVIORS.get_behavior(item)`
149pub static ITEM_BEHAVIORS: ItemBehaviorLock = ItemBehaviorLock(OnceLock::new());

151/// Initializes the global behavior registries.
152///
153/// This should be called after the main registry is frozen. Repeated calls are a no-op.
154pub fn init_behaviors() {
155    BLOCK_BEHAVIORS.0.get_or_init(|| {
156        let mut block_behaviors = BlockBehaviorRegistry::new();
157        register_block_behaviors(&mut block_behaviors);
158        block_behaviors
159    });
160
161    FLUID_BEHAVIORS.0.get_or_init(|| {
162        let mut fluid_behaviors = FluidBehaviorRegistry::new();
163
164        // Water: WaterFluid implements FluidBehavior directly
165        let water_behavior: Box<dyn FluidBehavior> = Box::new(WaterFluid);
166        // Both WATER and FLOWING_WATER share the same behavior
167        fluid_behaviors.set_behavior(&vanilla_fluids::WATER, water_behavior);
168        fluid_behaviors.set_behavior(&vanilla_fluids::FLOWING_WATER, Box::new(WaterFluid));
169
170        // Lava: LavaFluid implements FluidBehavior directly
171        let lava_behavior: Box<dyn FluidBehavior> = Box::new(LavaFluid);
172        fluid_behaviors.set_behavior(&vanilla_fluids::LAVA, lava_behavior);
173        fluid_behaviors.set_behavior(&vanilla_fluids::FLOWING_LAVA, Box::new(LavaFluid));
174
175        fluid_behaviors
176    });
177
178    ITEM_BEHAVIORS.0.get_or_init(|| {
179        let mut item_behaviors = ItemBehaviorRegistry::new();
180        register_item_behaviors(&mut item_behaviors);
181        item_behaviors
182    });
