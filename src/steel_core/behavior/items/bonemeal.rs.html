18/// Behavior for the Bonemeal item.
19#[item_behavior]
20pub struct BoneMealItem;
21
22impl BoneMealItem {
23    fn grow(world: &Arc<World>, pos: BlockPos) -> bool {
24        let state = world.get_block_state(pos);
25        let Some(behavior) = BLOCK_BEHAVIORS.get_behavior_for_state(state) else {
26            return false;
27        };
28        if let Some(bonemealable) = behavior.as_bonemealable() {
29            if !bonemealable.is_valid_bonemeal_target(state, world.as_ref(), pos) {
30                return false;
31            }
32
33            let mut rng = rand::rng();
34            if bonemealable.is_bonemeal_success(state, world, &mut rng, pos) {
35                bonemealable.perform_bonemeal(state, world, &mut rng, pos);
36            }
37
38            return true;
39        }
40        false
41    }
42
43    fn grow_water_plant(world: &Arc<World>, pos: BlockPos, _clicked_face: Direction) -> bool {
44        let state = world.get_block_state(pos);
45        if state.get_block() != &vanilla_blocks::WATER || state.get_fluid_state().amount != 8 {
46            return false;
47        }
48
49        let Some(bonemealable) = BLOCK_BEHAVIORS
50            .get_behavior(&vanilla_blocks::SEAGRASS)
51            .as_bonemealable()
52        else {
53            return false;
54        };
55
56        let mut rng = rand::rng();
57
58        'outer: for i in 0..128 {
59            let mut new_pos = pos;
60            let new_state = vanilla_blocks::SEAGRASS.default_state();
61
62            for _ in 0..(i / 16) {
63                new_pos = new_pos.offset(
64                    rng.random_range(0i32..3) - 1,
65                    (rng.random_range(0i32..3) - 1) * rng.random_range(0i32..3) / 2,
66                    rng.random_range(0i32..3) - 1,
67                );
68
69                if is_offset_shape_full_block(
70                    world
71                        .get_block_state(new_pos)
72                        .get_collision_shape_at(new_pos),
73                ) {
74                    continue 'outer;
75                }
76            }
77
78            // TODO: implement coral and underwater bonemeal tag selection.
79
80            let Some(behavior) = BLOCK_BEHAVIORS.get_behavior_for_state(new_state) else {
81                return false;
82            };
83
84            if behavior.can_survive(new_state, world, new_pos) {
85                let current_state = world.get_block_state(new_pos);
86                if current_state.get_block() == &vanilla_blocks::WATER
87                    && current_state.get_fluid_state().amount == 8
88                {
89                    world.set_block(new_pos, new_state, UpdateFlags::UPDATE_ALL);
90                } else if current_state.get_block() == &vanilla_blocks::SEAGRASS
91                    && bonemealable.is_valid_bonemeal_target(current_state, world.as_ref(), new_pos)
92                    && rng.random_range(0..10) == 0
93                {
94                    bonemealable.perform_bonemeal(current_state, world, &mut rng, new_pos);
95                }
96            }
97        }
98
99        true
100    }
101}

103impl ItemBehavior for BoneMealItem {
104    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
105        if Self::grow(context.world, context.hit_result.block_pos) {
106            context.inv.with_item(|item| item.shrink(1));
107            // TODO: particles
108            return InteractionResult::Success;
109        }
110        let state = context.world.get_block_state(context.hit_result.block_pos);
111        let is_clicked_face_sturdy =
112            state.is_face_sturdy_at(context.hit_result.block_pos, context.hit_result.direction);
113        if is_clicked_face_sturdy
114            && Self::grow_water_plant(
115                context.world,
116                context
117                    .hit_result
118                    .block_pos
119                    .relative(context.hit_result.direction),
120                context.hit_result.direction,
121            )
122        {
123            context.inv.with_item(|item| item.shrink(1));
124            return InteractionResult::Success;
125        }
126        InteractionResult::Pass
127    }
