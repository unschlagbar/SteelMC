1//! Default item behavior implementation.
2
3use crate::behavior::{InteractionResult, ItemBehavior, UseItemContext};
4use crate::entity::Entity;
5use crate::player::player_inventory::EquipmentSwapResult;
6
7/// Default item behavior - does nothing special.
8pub struct DefaultItemBehavior;
9
10impl ItemBehavior for DefaultItemBehavior {
11    fn use_item(&self, context: &mut UseItemContext) -> InteractionResult {
12        let Some(equippable) = context.inv.with_item(|item| item.get_equippable().cloned()) else {
13            return InteractionResult::Pass;
14        };
15
16        if !equippable.swappable || !equippable.can_be_equipped_by(context.player.entity_type()) {
17            return InteractionResult::Pass;
18        }
19
20        let slot = equippable.slot;
21        let result = context.inv.with_inventory(|inventory| {
22            inventory.try_swap_with_equipment_slot(
23                context.hand,
24                slot,
25                context.player.has_infinite_materials(),
26            )
27        });
28
29        match result {
30            EquipmentSwapResult::Success(overflow) => {
31                if !overflow.is_empty() {
32                    context.player.drop_item(overflow, false, false);
33                }
34                InteractionResult::Success
35            }
36            EquipmentSwapResult::Fail => InteractionResult::Fail,
37        }
38    }
39}
