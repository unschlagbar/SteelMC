20const MAX_DEPTH: i32 = 6;
21const MAX_COUNT: i32 = 64;
22
23#[block_behavior]
24/// Sponge behavior
25pub struct SpongeBlock {
26    block: BlockRef,
27}
28
29impl SpongeBlock {
30    /// Creates a new sponge block
31    #[must_use]
32    pub const fn new(block: BlockRef) -> Self {
33        Self { block }
34    }
35
36    fn try_absorb_water(world: &Arc<World>, pos: BlockPos) {
37        if !Self::remove_water_breadth_first_search(world, pos) {
38            return;
39        }
40
41        world.set_block(
42            pos,
43            vanilla_blocks::WET_SPONGE.default_state(),
44            UpdateFlags::UPDATE_CLIENTS,
45        );
46        world.play_sound(
47            &sound_events::BLOCK_SPONGE_ABSORB,
48            SoundSource::Blocks,
49            pos,
50            1.0,
51            1.0,
52            None,
53        );
54    }
55
56    fn remove_water_breadth_first_search(world: &Arc<World>, start_pos: BlockPos) -> bool {
57        BlockPos::breadth_first_traversal(
58            start_pos,
59            MAX_DEPTH,
60            MAX_COUNT + 1,
61            |pos, consumer| {
62                for direction in Direction::ALL {
63                    consumer(pos.relative(direction));
64                }
65            },
66            |pos| {
67                if pos == start_pos || Self::remove_water_at(world, pos) {
68                    TraversalNodeStatus::Accept
69                } else {
70                    TraversalNodeStatus::Skip
71                }
72            },
73        ) > 1
74    }
75
76    fn remove_water_at(world: &Arc<World>, pos: BlockPos) -> bool {
77        let state = world.get_block_state(pos);
78        let fluid_state = state.get_fluid_state();
79        if !fluid_state
80            .fluid_id
81            .has_tag(&vanilla_fluid_tags::FluidTag::WATER)
82        {
83            return false;
84        }
85
86        let behavior = BLOCK_BEHAVIORS.get_behavior(state.get_block());
87        if behavior.pickup_block(world, pos, state, None).is_some() {
88            return true;
89        }
90
91        if pickup_waterlogged_block(behavior, world, pos, state, None).is_some() {
92            return true;
93        }
94
95        if state.get_block() == &vanilla_blocks::WATER {
96            return world.set_block(
97                pos,
98                vanilla_blocks::AIR.default_state(),
99                UpdateFlags::UPDATE_ALL,
100            );
101        }
102
103        if !Self::is_absorbable_water_plant(state) {
104            return false;
105        }
106
107        world.drop_resources(state, pos);
108        world.set_block(
109            pos,
110            vanilla_blocks::AIR.default_state(),
111            UpdateFlags::UPDATE_ALL,
112        )
113    }
114
115    fn is_absorbable_water_plant(state: BlockStateId) -> bool {
116        let block = state.get_block();
117        block == &vanilla_blocks::KELP
118            || block == &vanilla_blocks::KELP_PLANT
119            || block == &vanilla_blocks::SEAGRASS
120            || block == &vanilla_blocks::TALL_SEAGRASS
121    }
122}
123
124impl BlockBehavior for SpongeBlock {
125    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
126        Some(self.block.default_state())
127    }
128
129    fn on_place(
130        &self,
131        state: BlockStateId,
132        world: &Arc<World>,
133        pos: BlockPos,
134        old_state: BlockStateId,
135        _moved_by_piston: bool,
136    ) {
137        if old_state.get_block() != state.get_block() {
138            Self::try_absorb_water(world, pos);
139        }
140    }
141
142    fn handle_neighbor_changed(
143        &self,
144        _state: BlockStateId,
145        world: &Arc<World>,
146        pos: BlockPos,
147        _source_block: BlockRef,
148        _moved_by_piston: bool,
149    ) {
150        Self::try_absorb_water(world, pos);
151    }
152}
