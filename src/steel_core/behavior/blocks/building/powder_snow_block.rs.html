21const IN_BLOCK_SPEED_MULTIPLIER: DVec3 = DVec3::new(0.9, 1.5, 0.9);
22const NUM_BLOCKS_TO_FALL_INTO_BLOCK: f64 = 2.5;
23const MIN_FALL_DISTANCE_FOR_SOUND: f64 = 4.0;
24const MIN_FALL_DISTANCE_FOR_BIG_SOUND: f64 = 7.0;
25const FALLING_COLLISION_BOXES: &[BlockLocalAabb] =
26    &[BlockLocalAabb::new(0.0, 0.0, 0.0, 1.0, 0.9, 1.0)];
27const FALLING_COLLISION_SHAPE: VoxelShape = VoxelShape::from_boxes(FALLING_COLLISION_BOXES);
28
29/// Behavior for powder snow blocks.
30///
31/// Vanilla handles several powder-snow collision variants in one class; Steel
32/// keeps those pieces here so the movement pipeline has a single block-behavior
33/// entry point for powder snow.
34#[block_behavior]
35pub struct PowderSnowBlock {
36    block: BlockRef,
37}
38
39impl PowderSnowBlock {
40    /// Creates a powder snow block behavior.
41    #[must_use]
42    pub const fn new(block: BlockRef) -> Self {
43        Self { block }
44    }
45
46    #[must_use]
47    fn fall_sound(context: EntityFallOnContext<'_>) -> Option<SoundEventRef> {
48        if context.fall_distance < MIN_FALL_DISTANCE_FOR_SOUND || !context.entity.is_living_entity {
49            return None;
50        }
51
52        let (small, big) = context.entity.fall_sounds;
53        Some(if context.fall_distance < MIN_FALL_DISTANCE_FOR_BIG_SOUND {
54            small
55        } else {
56            big
57        })
58    }
59}
60
61impl BlockBehavior for PowderSnowBlock {
62    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
63        Some(self.block.default_state())
64    }
65
66    fn is_pathfindable(
67        &self,
68        _state: BlockStateId,
69        _computation_type: PathComputationType,
70    ) -> bool {
71        true
72    }
73
74    fn fall_on(
75        &self,
76        _state: BlockStateId,
77        _world: &Arc<World>,
78        _pos: BlockPos,
79        context: EntityFallOnContext<'_>,
80    ) -> Option<EntityFallDamage> {
81        if let Some(sound) = Self::fall_sound(context)
82            && let Some(entity) = context.source_entity()
83        {
84            entity.play_sound(sound, 1.0, 1.0);
85        }
86
87        None
88    }
89
90    fn get_entity_inside_collision_shape(
91        &self,
92        state: BlockStateId,
93        world: &dyn LevelReader,
94        pos: BlockPos,
95        entity: &dyn Entity,
96    ) -> VoxelShape {
97        let collision_shape = self.get_collision_shape(
98            state,
99            world,
100            pos,
101            BlockCollisionContext::entity(entity.position().y, entity.is_descending())
102                .with_fall_distance(entity.fall_distance())
103                .with_can_walk_on_powder_snow(entity.can_walk_on_powder_snow())
104                .with_falling_block(entity.entity_type() == &vanilla_entities::FALLING_BLOCK),
105        );
106        if collision_shape.is_empty() {
107            self.default_get_entity_inside_collision_shape(state, world, pos, entity)
108        } else {
109            collision_shape
110        }
111    }
112
113    fn get_collision_shape(
114        &self,
115        state: BlockStateId,
116        world: &dyn LevelReader,
117        pos: BlockPos,
118        context: BlockCollisionContext,
119    ) -> VoxelShape {
120        if context.is_placement() {
121            return VoxelShape::EMPTY;
122        }
123        if context.fall_distance() > NUM_BLOCKS_TO_FALL_INTO_BLOCK {
124            return FALLING_COLLISION_SHAPE;
125        }
126        if context.is_falling_block()
127            || (context.can_walk_on_powder_snow()
128                && context.is_above(VoxelShape::FULL_BLOCK, pos, false)
129                && !context.is_descending())
130        {
131            return self.default_get_collision_shape(state, world, pos, context);
132        }
133
134        VoxelShape::EMPTY
135    }
136
137    fn entity_inside(
138        &self,
139        state: BlockStateId,
140        world: &Arc<World>,
141        pos: BlockPos,
142        entity: &dyn Entity,
143        effect_collector: &mut InsideBlockEffectCollector,
144        _is_precise: bool,
145    ) {
146        if !entity.is_living_entity() || entity.in_block_state(world).get_block() == self.block {
147            entity.make_stuck_in_block(state, IN_BLOCK_SPEED_MULTIPLIER);
148        }
149
150        let world = Arc::clone(world);
151        effect_collector.run_before(
152            InsideBlockEffectType::Extinguish,
153            Box::new(move |entity| {
154                if !entity.is_on_fire() {
155                    return;
156                }
157
158                let mob_griefing = world.get_game_rule(&vanilla_game_rules::MOB_GRIEFING)
159                    == GameRuleValue::Bool(true);
160                if (mob_griefing || entity.entity_type() == &vanilla_entities::PLAYER)
161                    && entity.may_interact(world.as_ref(), pos)
162                {
163                    world.destroy_block(pos, false);
164                }
165            }),
166        );
167        effect_collector.apply(InsideBlockEffectType::Freeze);
168        effect_collector.apply(InsideBlockEffectType::Extinguish);
169    }
170}

226    #[test]
227    fn powder_snow_fall_sound_uses_vanilla_living_thresholds() {
228        test_support::init_test_registry();
229        assert!(PowderSnowBlock::fall_sound(fall_context(3.99, true)).is_none());
230        assert_eq!(
231            PowderSnowBlock::fall_sound(fall_context(4.0, true)),
232            Some(&sound_events::ENTITY_PLAYER_SMALL_FALL)

234        assert_eq!(
235            PowderSnowBlock::fall_sound(fall_context(7.0, true)),
236            Some(&sound_events::ENTITY_PLAYER_BIG_FALL)
