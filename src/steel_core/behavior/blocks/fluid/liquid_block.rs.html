32/// Behavior for liquid blocks (water and lava).
33///
34/// Liquid blocks have a LEVEL property (0-15) that determines the fluid state:
35/// - LEVEL 0 = source block (full fluid)
36/// - LEVEL 1-7 = flowing fluid with decreasing height
37/// - LEVEL 8-15 = falling fluid
38#[block_behavior]
39pub struct LiquidBlock {
40    block: BlockRef,
41    #[json_arg(vanilla_fluids, ref)]
42    fluid: FluidRef,
43}
44
45impl LiquidBlock {
46    /// Creates a new liquid block behavior.
47    #[must_use]
48    pub const fn new(block: BlockRef, fluid: FluidRef) -> Self {
49        Self { block, fluid }
50    }
51
52    /// Checks if this liquid should spread and handles lava-water interactions.
53    /// Based on vanilla's `LiquidBlock.shouldSpreadLiquid()`.
54    ///
55    /// Returns `true` if the liquid should spread (schedule tick),
56    /// Returns `false` if the liquid was converted to a block (obsidian/cobblestone/basalt).
57    fn should_spread_liquid(&self, world: &Arc<World>, pos: BlockPos) -> bool {
58        // Only lava has special interactions with water and blue ice
59        if !is_lava_fluid(self.fluid) {
60            return true;
61        }
62        // Check if there's soul soil below (for basalt generation)
63        let below_pos = pos.offset(0, -1, 0);
64        let below_state = world.get_block_state(below_pos);
65        let has_soul_soil_below = below_state.get_block() == &vanilla_blocks::SOUL_SOIL;
66
67        // Get fluid state to check if this is a source
68        let fluid_state = world.get_block_state(pos).get_fluid_state();
69
70        for direction in Direction::FLOW_NEIGHBOR_CHECK {
71            let neighbor_pos = direction.relative(pos);
72            let neighbor_fluid = world.get_block_state(neighbor_pos).get_fluid_state();
73
74            // Check for water (including flowing_water and waterlogged blocks)
75            // Using fluid tag check to support modded fluids registered in the water tag
76            if neighbor_fluid.is_water() {
77                // Lava + Water = Obsidian (if source) or Cobblestone (if flowing)
78                let new_block = if fluid_state.is_source() {
79                    &vanilla_blocks::OBSIDIAN
80                } else {
81                    &vanilla_blocks::COBBLESTONE
82                };
83
84                let new_state = REGISTRY.blocks.get_default_state_id(new_block);
85                world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL);
86                world.level_event(level_events::LAVA_FIZZ, pos, 0, None);
87                return false; // Don't schedule fluid tick - block was converted
88            }
89
90            // Check for basalt generation: soul soil below + blue ice adjacent
91            if has_soul_soil_below {
92                let neighbor_state = world.get_block_state(neighbor_pos);
93                if neighbor_state.get_block() == &vanilla_blocks::BLUE_ICE {
94                    let new_state = REGISTRY
95                        .blocks
96                        .get_default_state_id(&vanilla_blocks::BASALT);
97                    world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL);
98                    world.level_event(level_events::LAVA_FIZZ, pos, 0, None);
99                    return false; // Don't schedule fluid tick - block was converted
100                }
101            }
102        }
103
104        true // No interaction occurred, proceed with normal fluid tick
105    }
106}

108impl BlockBehavior for LiquidBlock {
109    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
110        Some(self.block.default_state())
111    }
112
113    fn get_fluid_state(&self, state: BlockStateId) -> FluidState {
114        let level = state.get_value(&BlockStateProperties::LEVEL);
115        FluidState::from_block_level(self.fluid, level)
116    }
117
118    fn is_pathfindable(
119        &self,
120        _state: BlockStateId,
121        _computation_type: PathComputationType,
122    ) -> bool {
123        !is_lava_fluid(self.fluid)
124    }
125
126    /// Called when the block is placed.
127    fn on_place(
128        &self,
129        _state: BlockStateId,
130        world: &Arc<World>,
131        pos: BlockPos,
132        _old_state: BlockStateId,
133        _moved_by_piston: bool,
134    ) {
135        if self.should_spread_liquid(world, pos) {
136            let delay = FLUID_BEHAVIORS.get_behavior(self.fluid).tick_delay(world);
137            world.schedule_fluid_tick_default(pos, self.fluid, delay);
138        }
139    }
140
141    /// Called when a neighboring block changes.
142    fn handle_neighbor_changed(
143        &self,
144        _state: BlockStateId,
145        world: &Arc<World>,
146        pos: BlockPos,
147        _source_block: BlockRef,
148        _moved_by_piston: bool,
149    ) {
150        if self.should_spread_liquid(world, pos) {
151            let delay = FLUID_BEHAVIORS.get_behavior(self.fluid).tick_delay(world);
152            world.schedule_fluid_tick_default(pos, self.fluid, delay);
153        }
154    }
155
156    /// Called when a neighbor's shape changes.
157    ///
158    /// Vanilla parity: `LiquidBlock.updateShape` schedules a tick only when
159    /// either the current block or the neighbor contains a source fluid.
160    fn update_shape(
161        &self,
162        state: BlockStateId,
163        world: &dyn ScheduledTickAccess,
164        pos: BlockPos,
165        _direction: Direction,
166        _neighbor_pos: BlockPos,
167        neighbor_state: BlockStateId,
168    ) -> BlockStateId {
169        let fluid_state =
170            FluidState::from_block_level(self.fluid, state.get_value(&BlockStateProperties::LEVEL));
171        let neighbor_fluid = neighbor_state.get_fluid_state();
172
173        if fluid_state.is_source() || neighbor_fluid.is_source() {
174            let delay = world.fluid_tick_delay(self.fluid);
175            world.schedule_fluid_tick_default(pos, self.fluid, delay);
176        }
177
178        state
179    }
180
181    /// Vanilla parity: `LiquidBlock.isRandomlyTicking` delegates to the fluid.
182    fn is_randomly_ticking(&self, _state: BlockStateId) -> bool {
183        FLUID_BEHAVIORS
184            .get_behavior(self.fluid)
185            .is_randomly_ticking()
186    }
187
188    /// Vanilla parity: `LiquidBlock.randomTick` delegates to the fluid.
189    fn random_tick(&self, _state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
190        FLUID_BEHAVIORS
191            .get_behavior(self.fluid)
192            .random_tick(world, pos);
193    }
194
195    fn pickup_block(
196        &self,
197        world: &Arc<World>,
198        pos: BlockPos,
199        state: BlockStateId,
200        _player: Option<&Player>,
201    ) -> Option<PickupResult> {
202        if state.try_get_value(&BlockStateProperties::LEVEL) != Some(0) {
203            return None;
204        }
205
206        let air = REGISTRY.blocks.get_default_state_id(&vanilla_blocks::AIR);
207        world.set_block(pos, air, UpdateFlags::UPDATE_ALL_IMMEDIATE);
208
209        let bucket = if is_water_fluid(self.fluid) {
210            &vanilla_items::ITEMS.water_bucket
211        } else {
212            &vanilla_items::ITEMS.lava_bucket
213        };
214
215        let sound = if is_water_fluid(self.fluid) {
216            &sound_events::ITEM_BUCKET_FILL
217        } else {
218            &sound_events::ITEM_BUCKET_FILL_LAVA
219        };
220
221        Some(PickupResult {
222            filled_bucket: bucket,
223            sound: Some(sound),
224        })
225    }
