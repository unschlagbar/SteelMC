13/// Vanilla `VineBlock` survival and neighbor shape updates.
14// TODO: Implement placement and random tick spread.
15#[block_behavior]
16pub struct VineBlock {
17    block: BlockRef,
18}
19
20impl VineBlock {
21    /// Creates a new vine block behavior.
22    #[must_use]
23    pub const fn new(block: BlockRef) -> Self {
24        Self { block }
25    }
26
27    fn has_faces(state: BlockStateId) -> bool {
28        VINE_FACE_DIRECTIONS
29            .into_iter()
30            .any(|direction| state.get_value(face_property(direction)))
31    }
32
33    fn can_support_at_face(
34        &self,
35        world: &dyn LevelReader,
36        pos: BlockPos,
37        direction: Direction,
38    ) -> bool {
39        if direction == Direction::Down {
40            return false;
41        }
42
43        if can_attach_to_multiface(world, pos.relative(direction), direction) {
44            return true;
45        }
46
47        if !direction.is_horizontal() {
48            return false;
49        }
50
51        let property = face_property(direction);
52        let above = world.get_block_state(pos.above());
53        above.get_block() == self.block && above.get_value(property)
54    }
55
56    fn updated_state(
57        &self,
58        mut state: BlockStateId,
59        world: &dyn LevelReader,
60        pos: BlockPos,
61    ) -> BlockStateId {
62        let above_pos = pos.above();
63        if state.get_value(&BlockStateProperties::UP) {
64            state = state.set_value(
65                &BlockStateProperties::UP,
66                can_attach_to_multiface(world, above_pos, Direction::Down),
67            );
68        }
69
70        let mut above_state: Option<BlockStateId> = None;
71        for direction in VINE_HORIZONTAL_DIRECTIONS {
72            let property = face_property(direction);
73            if !state.get_value(property) {
74                continue;
75            }
76
77            let mut can_support = self.can_support_at_face(world, pos, direction);
78            if !can_support {
79                let above = *above_state.get_or_insert_with(|| world.get_block_state(above_pos));
80                can_support = above.get_block() == self.block && above.get_value(property);
81            }
82
83            state = state.set_value(property, can_support);
84        }
85
86        state
87    }
88}

90const VINE_FACE_DIRECTIONS: [Direction; 5] = [
91    Direction::Up,
92    Direction::North,
93    Direction::East,
94    Direction::South,
95    Direction::West,
96];
97
98const VINE_HORIZONTAL_DIRECTIONS: [Direction; 4] = [
99    Direction::North,
100    Direction::East,
101    Direction::South,
102    Direction::West,
103];
104
105/// Vanilla `VineBlock.getPropertyForFace`; vines have no `Down` face property.
106fn face_property(direction: Direction) -> &'static BoolProperty {
107    match direction {
108        Direction::Up => &BlockStateProperties::UP,
109        Direction::North => &BlockStateProperties::NORTH,
110        Direction::East => &BlockStateProperties::EAST,
111        Direction::South => &BlockStateProperties::SOUTH,
112        Direction::West => &BlockStateProperties::WEST,
113        Direction::Down => unreachable!("vine has no DOWN face property"),
114    }
115}

117impl BlockBehavior for VineBlock {
118    fn can_survive(&self, state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
119        Self::has_faces(self.updated_state(state, world, pos))
120    }
121
122    fn update_shape(
123        &self,
124        state: BlockStateId,
125        world: &dyn ScheduledTickAccess,
126        pos: BlockPos,
127        direction: Direction,
128        _neighbor_pos: BlockPos,
129        _neighbor_state: BlockStateId,
130    ) -> BlockStateId {
131        if direction == Direction::Down {
132            return state;
133        }
134
135        let updated = self.updated_state(state, world, pos);
136        if Self::has_faces(updated) {
137            updated
138        } else {
139            vanilla_blocks::AIR.default_state()
140        }
141    }
142
143    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
144        // TODO: Vanilla picks a face based on nearest looking direction and
145        // supports replacing an existing vine. Placeholder: default state if it
146        // survives.
147        default_surviving_state(self.block, self, context)
148    }
149}
