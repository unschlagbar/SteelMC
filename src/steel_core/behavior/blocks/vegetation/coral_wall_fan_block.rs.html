12/// Vanilla `CoralWallFanBlock` survival (live coral wall fans).
13///
14/// Inherits `canSurvive` from `BaseCoralWallFanBlock`. Death tick is left as a
15/// TODO.
16// TODO: Implement death tick and scheduled tick on placement.
17#[block_behavior]
18pub struct CoralWallFanBlock {
19    block: BlockRef,
20}
21
22impl CoralWallFanBlock {
23    /// Creates a new live coral wall fan block behavior.
24    #[must_use]
25    pub const fn new(block: BlockRef) -> Self {
26        Self { block }
27    }
28}
29
30impl BlockBehavior for CoralWallFanBlock {
31    fn can_survive(&self, state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
32        let facing = state.get_value(&BlockStateProperties::HORIZONTAL_FACING);
33        coral_wall_fan_can_survive(world, pos, facing)
34    }
35
36    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
37        let facing = context.horizontal_direction.opposite();
38        let state = self
39            .block
40            .default_state()
41            .set_value(&BlockStateProperties::HORIZONTAL_FACING, facing)
42            .set_value(
43                &BlockStateProperties::WATERLOGGED,
44                context.is_water_source(),
45            );
46        self.can_survive(state, context.world, context.relative_pos)
47            .then_some(state)
48    }
49}
