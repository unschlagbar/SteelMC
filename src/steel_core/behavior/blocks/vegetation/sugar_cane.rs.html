1//! Sugar Cane block behavior.
2//!
3//! Sugar cane grows up to 3 blocks tall via random ticks. It requires water adjacent
4//! to the block it is planted on (or frosted ice).
5

21/// Maximum sugar cane stack height (vanilla: 3 blocks).
22const MAX_SUGAR_CANE_HEIGHT: i32 = 3;
23
24/// Behavior for sugar cane blocks.
25#[block_behavior]
26pub struct SugarCaneBlock {
27    block: BlockRef,
28}
29
30impl SugarCaneBlock {
31    /// Creates a new sugar cane block behavior.
32    #[must_use]
33    pub const fn new(block: BlockRef) -> Self {
34        Self { block }
35    }
36}
37
38impl BlockBehavior for SugarCaneBlock {
39    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
40        let pos = context.relative_pos;
41        if self.can_survive(
42            vanilla_blocks::SUGAR_CANE.default_state(), // state argument is unused
43            context.world,
44            pos,
45        ) {
46            Some(self.block.default_state())
47        } else {
48            None
49        }
50    }
51
52    /// Called when this block is placed.
53    fn on_place(
54        &self,
55        state: BlockStateId,
56        world: &Arc<World>,
57        pos: BlockPos,
58        old_state: BlockStateId,
59        _moved_by_piston: bool,
60    ) {
61        if state.get_block() == old_state.get_block() {
62            return;
63        }
64
65        if !self.can_survive(state, world, pos) {
66            world.schedule_block_tick_default(pos, state.get_block(), 1);
67        }
68    }
69
70    fn tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
71        if !self.can_survive(state, world, pos) {
72            world.destroy_block(pos, true);
73        }
74    }
75
76    fn is_randomly_ticking(&self, _state: BlockStateId) -> bool {
77        true
78    }
79
80    fn random_tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
81        let above_pos = pos.above();
82
83        if !world.get_block_state(above_pos).is_air() {
84            return;
85        }
86
87        let mut height = 1i32;
88        while world.get_block_state(pos.below_n(height)).get_block() == self.block {
89            height += 1;
90        }
91
92        if height >= MAX_SUGAR_CANE_HEIGHT {
93            return;
94        }
95
96        let age = state.get_value(&BlockStateProperties::AGE_15);
97
98        if age == 15 {
99            world.set_block(
100                above_pos,
101                self.block.default_state(),
102                UpdateFlags::UPDATE_ALL,
103            );
104            let new_state = state.set_value(&BlockStateProperties::AGE_15, 0);
105            world.set_block(pos, new_state, UpdateFlags::UPDATE_CLIENTS);
106        } else {
107            let new_state = state.set_value(&BlockStateProperties::AGE_15, age + 1);
108            world.set_block(pos, new_state, UpdateFlags::UPDATE_CLIENTS);
109        }
110    }
111
112    fn update_shape(
113        &self,
114        state: BlockStateId,
115        world: &dyn ScheduledTickAccess,
116        pos: BlockPos,
117        _direction: Direction,
118        _neighbor_pos: BlockPos,
119        _neighbor_state: BlockStateId,
120    ) -> BlockStateId {
121        if !self.can_survive(state, world, pos) {
122            world.schedule_block_tick_default(pos, self.block, 1);
123        }
124        state
125    }
126
127    fn can_survive(&self, _state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
128        let below_pos = pos.below();
129        let below_state = world.get_block_state(below_pos);
130        let below_block = below_state.get_block();
131
132        if below_block == self.block {
133            return true;
134        }
135
136        let is_valid_ground = below_block.has_tag(&BlockTag::SUPPORTS_SUGAR_CANE);
137
138        if !is_valid_ground {
139            return false;
140        }
141
142        for dir in [
143            Direction::North,
144            Direction::South,
145            Direction::East,
146            Direction::West,
147        ] {
148            let neighbor_pos = dir.relative(below_pos);
149            let neighbor_state = world.get_block_state(neighbor_pos);
150
151            if neighbor_state
152                .get_block()
153                .has_tag(&BlockTag::SUPPORTS_SUGAR_CANE_ADJACENTLY)
154                || neighbor_state
155                    .get_fluid_state()
156                    .fluid_id
157                    .has_tag(&vanilla_fluid_tags::FluidTag::SUPPORTS_SUGAR_CANE_ADJACENTLY)
158            {
159                return true;
160            }
161        }
162
163        false
164    }
165}

190            vanilla_blocks::AIR.default_state()
191        }
192
193        fn raw_brightness(&self, _pos: BlockPos, _sky_darkening: u8) -> u8 {
194            0
195        }

198            -64
199        }
200
201        fn height(&self) -> i32 {
202            384
203        }
204    }

206    impl ScheduledTickAccess for EmptyLevel {
207        fn fluid_tick_delay(&self, _fluid: FluidRef) -> i32 {
208            5
209        }
210
211        fn schedule_block_tick_default(
212            &self,
213            _pos: BlockPos,
214            block: BlockRef,
215            _delay: i32,
216        ) -> bool {
217            let is_sugar_cane = block == &vanilla_blocks::SUGAR_CANE;
218            self.scheduled_block_tick.set(is_sugar_cane);
219            is_sugar_cane
220        }

223            &self,
224            _pos: BlockPos,
225            _fluid: FluidRef,
226            _delay: i32,

229        }
230    }

232    #[test]
233    fn sugar_cane_update_shape_schedules_break_tick_when_unsupported() {
234        init_test_registry();
235        let behavior = SugarCaneBlock::new(&vanilla_blocks::SUGAR_CANE);
236        let level = EmptyLevel::new();
237        let state = vanilla_blocks::SUGAR_CANE.default_state();
238
239        let updated = behavior.update_shape(
240            state,
241            &level,
242            BlockPos::ZERO,
243            Direction::Down,
244            BlockPos::ZERO.below(),
245            vanilla_blocks::AIR.default_state(),
246        );
247
248        assert_eq!(updated, state);
249        assert!(level.scheduled_block_tick.get());
