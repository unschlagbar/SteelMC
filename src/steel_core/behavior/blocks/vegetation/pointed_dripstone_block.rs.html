25/// Vanilla `PointedDripstoneBlock` survival and thickness updates.
26///
27/// Survival mirrors vanilla's `isValidPointedDripstonePlacement`: the block
28/// opposite the tip direction must be face-sturdy on the face pointing toward
29/// us, or be another pointed dripstone with the same `vertical_direction`.
30// TODO: Implement falling stalactites after falling block entities exist.
31// TODO: Implement trident projectile breakage and fluid transfer after
32// projectile and cauldron drip-fill foundations exist.
33#[block_behavior]
34pub struct PointedDripstoneBlock {
35    block: BlockRef,
36}
37
38impl PointedDripstoneBlock {
39    /// Creates a new pointed dripstone block behavior.
40    #[must_use]
41    pub const fn new(block: BlockRef) -> Self {
42        Self { block }
43    }
44
45    #[must_use]
46    fn fall_damage_for_state(state: BlockStateId, fall_distance: f64) -> Option<EntityFallDamage> {
47        if state.get_value(&BlockStateProperties::VERTICAL_DIRECTION) != Direction::Up
48            || state.get_value(&BlockStateProperties::SPELEOTHEM_THICKNESS)
49                != SpeleothemThickness::Tip
50        {
51            return None;
52        }
53
54        Some(EntityFallDamage::new(
55            fall_distance + 2.5,
56            2.0,
57            DamageSource::environment(&vanilla_damage_types::STALAGMITE),
58        ))
59    }
60
61    const fn speleothem(&self) -> SpeleothemBlockBehavior {
62        SpeleothemBlockBehavior {
63            block: self.block,
64            kind: SpeleothemKind::PointedDripstone,
65        }
66    }
67}

171struct SpeleothemBlockBehavior {
172    block: BlockRef,
173    kind: SpeleothemKind,
174}
175
176#[derive(Clone, Copy)]
177enum SpeleothemKind {
178    PointedDripstone,
179    Sulfur,
180}
181
182const GROWTH_PROBABILITY_PER_RANDOM_TICK: f32 = 0.011_377_778;
183const MAX_GROWTH_LENGTH: i32 = 7;
184const MAX_STALAGMITE_SEARCH_RANGE_WHEN_GROWING: i32 = 10;
185const DRIP_THROUGH_COLUMN_BOXES: &[BlockLocalAabb] =
186    &[BlockLocalAabb::new(0.375, 0.0, 0.375, 0.625, 1.0, 0.625)];
187const REQUIRED_SPACE_TO_DRIP_THROUGH_NON_SOLID_BLOCK: VoxelShape =
188    VoxelShape::from_boxes(DRIP_THROUGH_COLUMN_BOXES);
189
190impl SpeleothemBlockBehavior {
191    fn state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
192        let default_tip_direction = context.get_nearest_looking_vertical_direction().opposite();
193        let tip_direction = self.calculate_tip_direction(
194            context.world.as_ref(),
195            context.relative_pos,
196            default_tip_direction,
197        )?;
198        let merge_opposing_tips = !context.is_secondary_use_active;
199        let thickness = self.calculate_thickness(
200            context.world.as_ref(),
201            context.relative_pos,
202            tip_direction,
203            merge_opposing_tips,
204        );
205        let state = self
206            .block
207            .default_state()
208            .set_value(&BlockStateProperties::VERTICAL_DIRECTION, tip_direction)
209            .set_value(
210                &BlockStateProperties::WATERLOGGED,
211                context.is_water_source(),
212            );
213
214        Some(Self::with_thickness(state, thickness))
215    }
216
217    fn can_survive(&self, state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
218        let tip_direction = state.get_value(&BlockStateProperties::VERTICAL_DIRECTION);
219        let behind_pos = pos.relative(tip_direction.opposite());
220        let behind_state = world.get_block_state(behind_pos);
221
222        behind_state.is_face_sturdy_at(behind_pos, tip_direction)
223            || (Self::is_speleothem_with_direction(behind_state, tip_direction)
224                && behind_state.get_block() == self.block)
225    }
226
227    fn update_shape(
228        &self,
229        state: BlockStateId,
230        world: &dyn ScheduledTickAccess,
231        pos: BlockPos,
232        direction: Direction,
233        _neighbor_pos: BlockPos,
234        _neighbor_state: BlockStateId,
235    ) -> BlockStateId {
236        if state.get_value(&BlockStateProperties::WATERLOGGED) {
237            let delay = world.fluid_tick_delay(&vanilla_fluids::WATER);
238            let _ = world.schedule_fluid_tick_default(pos, &vanilla_fluids::WATER, delay);
239        }
240
241        if direction != Direction::Up && direction != Direction::Down {
242            return state;
243        }
244
245        let tip_direction = state.get_value(&BlockStateProperties::VERTICAL_DIRECTION);
246        if tip_direction == Direction::Down && world.has_scheduled_block_tick(pos, self.block) {
247            return state;
248        }
249
250        if direction == tip_direction.opposite() && !self.can_survive(state, world, pos) {
251            let delay = if tip_direction == Direction::Down {
252                2
253            } else {
254                1
255            };
256            let _ = world.schedule_block_tick_default(pos, self.block, delay);
257            return state;
258        }
259
260        let merge_opposing_tips = Self::thickness(state) == SpeleothemThickness::TipMerge;
261        let thickness = self.calculate_thickness(world, pos, tip_direction, merge_opposing_tips);
262        Self::with_thickness(state, thickness)
263    }
264
265    fn tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
266        if Self::is_stalagmite(state) && !self.can_survive(state, world.as_ref(), pos) {
267            world.destroy_block(pos, true);
268        }
269    }
270
271    fn calculate_tip_direction(
272        &self,
273        world: &dyn LevelReader,
274        pos: BlockPos,
275        default_tip_direction: Direction,
276    ) -> Option<Direction> {
277        let default_state = self.block.default_state().set_value(
278            &BlockStateProperties::VERTICAL_DIRECTION,
279            default_tip_direction,
280        );
281        if self.can_survive(default_state, world, pos) {
282            return Some(default_tip_direction);
283        }
284
285        let opposite_tip_direction = default_tip_direction.opposite();
286        let opposite_state = self.block.default_state().set_value(
287            &BlockStateProperties::VERTICAL_DIRECTION,
288            opposite_tip_direction,
289        );
290        self.can_survive(opposite_state, world, pos)
291            .then_some(opposite_tip_direction)
292    }
293
294    fn calculate_thickness(
295        &self,
296        world: &dyn LevelReader,
297        pos: BlockPos,
298        tip_direction: Direction,
299        merge_opposing_tips: bool,
300    ) -> SpeleothemThickness {
301        let base_direction = tip_direction.opposite();
302        let in_front_state = world.get_block_state(pos.relative(tip_direction));
303        if Self::is_speleothem_with_direction(in_front_state, base_direction)
304            && in_front_state.get_block() == self.block
305        {
306            if merge_opposing_tips
307                || Self::thickness(in_front_state) == SpeleothemThickness::TipMerge
308            {
309                return SpeleothemThickness::TipMerge;
310            }
311            return SpeleothemThickness::Tip;
312        }
313
314        if !Self::is_speleothem_with_direction(in_front_state, tip_direction) {
315            return SpeleothemThickness::Tip;
316        }
317
318        let in_front_thickness = Self::thickness(in_front_state);
319        if matches!(
320            in_front_thickness,
321            SpeleothemThickness::Tip | SpeleothemThickness::TipMerge
322        ) {
323            return SpeleothemThickness::Frustum;
324        }
325
326        let behind_state = world.get_block_state(pos.relative(base_direction));
327        if !Self::is_speleothem_with_direction(behind_state, tip_direction) {
328            return SpeleothemThickness::Base;
329        }
330        SpeleothemThickness::Middle
331    }
332
333    fn is_speleothem_with_direction(state: BlockStateId, tip_direction: Direction) -> bool {
334        state.get_block().has_tag(&BlockTag::SPELEOTHEMS)
335            && state.get_value(&BlockStateProperties::VERTICAL_DIRECTION) == tip_direction
336    }
337
338    fn is_stalagmite(state: BlockStateId) -> bool {
339        Self::is_speleothem_with_direction(state, Direction::Up)
340    }
341
342    fn is_stalactite(state: BlockStateId) -> bool {
343        Self::is_speleothem_with_direction(state, Direction::Down)
344    }
345
346    fn is_stalactite_start_pos(
347        &self,
348        state: BlockStateId,
349        world: &dyn LevelReader,
350        pos: BlockPos,
351    ) -> bool {
352        Self::is_stalactite(state) && world.get_block_state(pos.above()).get_block() != self.block
353    }
354
355    fn random_tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
356        let mut rng = rand::rng();
357        if matches!(self.kind, SpeleothemKind::PointedDripstone) {
358            let _fluid_transfer_random_value = rng.random::<f32>();
359        }
360
361        if rng.random::<f32>() < GROWTH_PROBABILITY_PER_RANDOM_TICK
362            && self.is_stalactite_start_pos(state, world.as_ref(), pos)
363        {
364            self.grow_stalactite_or_stalagmite_if_possible(state, world, pos, &mut rng);
365        }
366    }
367
368    fn grow_stalactite_or_stalagmite_if_possible<R: RngExt + ?Sized>(
369        &self,
370        stalactite_start_state: BlockStateId,
371        world: &Arc<World>,
372        stalactite_start_pos: BlockPos,
373        rng: &mut R,
374    ) {
375        if !self.can_grow(world.as_ref(), stalactite_start_pos) {
376            return;
377        }
378
379        let Some(stalactite_tip_pos) = self.find_tip(
380            stalactite_start_state,
381            world.as_ref(),
382            stalactite_start_pos,
383            MAX_GROWTH_LENGTH,
384            false,
385        ) else {
386            return;
387        };
388
389        let stalactite_tip_state = world.get_block_state(stalactite_tip_pos);
390        if !Self::is_free_hanging_stalactite(stalactite_tip_state)
391            || !self.can_tip_grow(stalactite_tip_state, world, stalactite_tip_pos)
392        {
393            return;
394        }
395
396        if rng.random::<bool>() {
397            self.grow(world, stalactite_tip_pos, Direction::Down);
398        } else {
399            self.grow_stalagmite_below(world, stalactite_tip_pos);
400        }
401    }
402
403    fn can_grow(&self, world: &dyn LevelReader, pos: BlockPos) -> bool {
404        if world.get_block_state(pos.above()).get_block() != self.block_to_grow_on() {
405            return false;
406        }
407
408        if !matches!(self.kind, SpeleothemKind::PointedDripstone) {
409            return true;
410        }
411
412        let fluid_state = world.get_block_state(pos.above_n(2)).get_fluid_state();
413        fluid_state.is_water() && fluid_state.is_source()
414    }
415
416    fn block_to_grow_on(&self) -> BlockRef {
417        match self.kind {
418            SpeleothemKind::PointedDripstone => &vanilla_blocks::DRIPSTONE_BLOCK,
419            SpeleothemKind::Sulfur => &vanilla_blocks::SULFUR,
420        }
421    }
422
423    fn find_tip(
424        &self,
425        speleothem_state: BlockStateId,
426        world: &dyn LevelReader,
427        speleothem_pos: BlockPos,
428        max_search_length: i32,
429        include_merged_tip: bool,
430    ) -> Option<BlockPos> {
431        if Self::is_tip(speleothem_state, include_merged_tip) {
432            return Some(speleothem_pos);
433        }
434
435        let search_direction =
436            speleothem_state.get_value(&BlockStateProperties::VERTICAL_DIRECTION);
437        let mut current_pos = speleothem_pos;
438        for _ in 1..max_search_length {
439            current_pos = current_pos.relative(search_direction);
440            let state = world.get_block_state(current_pos);
441            if Self::is_tip(state, include_merged_tip) {
442                return Some(current_pos);
443            }
444
445            if world.is_outside_build_height(current_pos.y())
446                || state.get_block() != self.block
447                || state.get_value(&BlockStateProperties::VERTICAL_DIRECTION) != search_direction
448            {
449                return None;
450            }
451        }
452
453        None
454    }
455
456    fn is_tip(state: BlockStateId, include_merged_tip: bool) -> bool {
457        if !state.get_block().has_tag(&BlockTag::SPELEOTHEMS) {
458            return false;
459        }
460
461        let thickness = Self::thickness(state);
462        thickness == SpeleothemThickness::Tip
463            || (include_merged_tip && thickness == SpeleothemThickness::TipMerge)
464    }
465
466    fn is_free_hanging_stalactite(state: BlockStateId) -> bool {
467        Self::is_stalactite(state)
468            && Self::thickness(state) == SpeleothemThickness::Tip
469            && !state.get_value(&BlockStateProperties::WATERLOGGED)
470    }
471
472    fn can_tip_grow(&self, tip_state: BlockStateId, world: &Arc<World>, tip_pos: BlockPos) -> bool {
473        let grow_direction = tip_state.get_value(&BlockStateProperties::VERTICAL_DIRECTION);
474        let grow_pos = tip_pos.relative(grow_direction);
475        let state_at_grow_pos = world.get_block_state(grow_pos);
476        if !state_at_grow_pos.get_fluid_state().is_empty() {
477            return false;
478        }
479
480        state_at_grow_pos.is_air()
481            || self.is_unmerged_tip_with_direction(state_at_grow_pos, grow_direction.opposite())
482    }
483
484    fn is_unmerged_tip_with_direction(
485        &self,
486        state: BlockStateId,
487        tip_direction: Direction,
488    ) -> bool {
489        Self::is_tip(state, false)
490            && state.get_block() == self.block
491            && state.get_value(&BlockStateProperties::VERTICAL_DIRECTION) == tip_direction
492    }
493
494    fn grow(&self, world: &Arc<World>, grow_from_pos: BlockPos, grow_to_direction: Direction) {
495        let target_pos = grow_from_pos.relative(grow_to_direction);
496        let existing_state_at_target_pos = world.get_block_state(target_pos);
497        if self.is_unmerged_tip_with_direction(
498            existing_state_at_target_pos,
499            grow_to_direction.opposite(),
500        ) {
501            self.create_merged_tips(existing_state_at_target_pos, world, target_pos);
502            return;
503        }
504
505        if existing_state_at_target_pos.is_air()
506            || existing_state_at_target_pos.get_block() == &vanilla_blocks::WATER
507        {
508            self.create_speleothem(
509                world,
510                target_pos,
511                grow_to_direction,
512                SpeleothemThickness::Tip,
513            );
514        }
515    }
516
517    fn create_speleothem(
518        &self,
519        world: &Arc<World>,
520        pos: BlockPos,
521        direction: Direction,
522        thickness: SpeleothemThickness,
523    ) {
524        let waterlogged = world.get_block_state(pos).get_fluid_state().is_water();
525        let state = self
526            .block
527            .default_state()
528            .set_value(&BlockStateProperties::VERTICAL_DIRECTION, direction)
529            .set_value(&BlockStateProperties::SPELEOTHEM_THICKNESS, thickness)
530            .set_value(&BlockStateProperties::WATERLOGGED, waterlogged);
531        world.set_block(pos, state, UpdateFlags::UPDATE_ALL);
532    }
533
534    fn create_merged_tips(&self, tip_state: BlockStateId, world: &Arc<World>, tip_pos: BlockPos) {
535        let (stalactite_pos, stalagmite_pos) =
536            if tip_state.get_value(&BlockStateProperties::VERTICAL_DIRECTION) == Direction::Up {
537                (tip_pos.above(), tip_pos)
538            } else {
539                (tip_pos, tip_pos.below())
540            };
541
542        self.create_speleothem(
543            world,
544            stalactite_pos,
545            Direction::Down,
546            SpeleothemThickness::TipMerge,
547        );
548        self.create_speleothem(
549            world,
550            stalagmite_pos,
551            Direction::Up,
552            SpeleothemThickness::TipMerge,
553        );
554    }
555
556    fn grow_stalagmite_below(&self, world: &Arc<World>, pos_above_stalagmite: BlockPos) {
557        let mut pos = pos_above_stalagmite;
558        for _ in 0..MAX_STALAGMITE_SEARCH_RANGE_WHEN_GROWING {
559            pos = pos.below();
560            let state = world.get_block_state(pos);
561            if !state.get_fluid_state().is_empty() {
562                return;
563            }
564
565            if self.is_unmerged_tip_with_direction(state, Direction::Up)
566                && self.can_tip_grow(state, world, pos)
567            {
568                self.grow(world, pos, Direction::Up);
569                return;
570            }
571
572            let placement_state = self
573                .block
574                .default_state()
575                .set_value(&BlockStateProperties::VERTICAL_DIRECTION, Direction::Up);
576            if self.can_survive(placement_state, world.as_ref(), pos)
577                && !Self::is_water_at(world, pos.below())
578            {
579                self.grow(world, pos.below(), Direction::Up);
580                return;
581            }
582
583            if self.blocks_stalagmite_scan(world.as_ref(), pos, state) {
584                return;
585            }
586        }
587    }
588
589    fn is_water_at(world: &Arc<World>, pos: BlockPos) -> bool {
590        world.get_block_state(pos).get_fluid_state().is_water()
591    }
592
593    fn blocks_stalagmite_scan(
594        &self,
595        world: &dyn LevelReader,
596        pos: BlockPos,
597        state: BlockStateId,
598    ) -> bool {
599        match self.kind {
600            SpeleothemKind::PointedDripstone => !Self::can_drip_through(world, pos, state),
601            SpeleothemKind::Sulfur => false,
602        }
603    }
604
605    fn can_drip_through(world: &dyn LevelReader, pos: BlockPos, state: BlockStateId) -> bool {
606        if state.is_air() {
607            return true;
608        }
609
610        if state.is_solid_render() || !state.get_fluid_state().is_empty() {
611            return false;
612        }
613
614        let collision_shape = BLOCK_BEHAVIORS
615            .get_behavior(state.get_block())
616            .get_collision_shape(state, world, pos, BlockCollisionContext::empty());
617        !join_is_not_empty(
618            REQUIRED_SPACE_TO_DRIP_THROUGH_NON_SOLID_BLOCK,
619            collision_shape,
620            BooleanOp::And,
621        )
622    }
623
624    fn thickness(state: BlockStateId) -> SpeleothemThickness {
625        state.get_value(&BlockStateProperties::SPELEOTHEM_THICKNESS)
626    }
627
628    fn with_thickness(state: BlockStateId, thickness: SpeleothemThickness) -> BlockStateId {
629        state.set_value(&BlockStateProperties::SPELEOTHEM_THICKNESS, thickness)
630    }
631}
