1//! Cactus block behavior.
2//!
3//! Cactus grows up to 3 blocks tall via random ticks. It requires sand below
4//! and breaks if any solid block or lava is adjacent horizontally.

26/// Maximum cactus stack height (vanilla: 3 blocks).
27const MAX_CACTUS_HEIGHT: u32 = 3;
28
29/// Age at which cactus can attempt to grow a flower (vanilla 1.21+).
30const CACTUS_FLOWER_AGE: u8 = 8;
31
32/// Chance for small cactus (< 3 blocks) to spawn flower.
33const FLOWER_CHANCE_SMALL: f64 = 0.1;
34
35/// Chance for tall cactus (>= 3 blocks) to spawn flower.
36const FLOWER_CHANCE_TALL: f64 = 0.25;
37
38/// Behavior for cactus blocks.
39///
40/// Cactus has special requirements:
41/// - Must be placed on sand, red sand, or another cactus
42/// - Cannot have solid blocks adjacent horizontally
43/// - Grows up to 3 blocks tall via random ticks
44/// - Damages entities that touch it (1 HP per tick)
45#[block_behavior]
46pub struct CactusBlock {
47    block: BlockRef,
48}
49
50impl CactusBlock {
51    /// Creates a new cactus block behavior.
52    #[must_use]
53    pub const fn new(block: BlockRef) -> Self {
54        Self { block }
55    }
56}
57
58impl BlockBehavior for CactusBlock {
59    /// Checks if cactus can survive at the given position.
60    ///
61    /// Survival requirements:
62    /// 1. No solid blocks on horizontal neighbors
63    /// 2. No lava on horizontal neighbors
64    /// 3. Block below must be `CACTUS`, `SAND`, or `RED_SAND`
65    /// 4. Block above must not be liquid
66    fn can_survive(&self, _state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
67        // Check horizontal neighbors - no solid blocks or lava
68        for dir in [
69            Direction::North,
70            Direction::South,
71            Direction::East,
72            Direction::West,
73        ] {
74            let neighbor_pos = dir.relative(pos);
75            let neighbor = world.get_block_state(neighbor_pos);
76            if neighbor.is_solid() {
77                return false;
78            }
79
80            let fluid = neighbor.get_fluid_state();
81            if fluid.fluid_id.has_tag(&vanilla_fluid_tags::FluidTag::LAVA) {
82                return false;
83            }
84        }
85
86        // Block below must be CACTUS or SAND variant
87        let below_pos = pos.below();
88        let below = world.get_block_state(below_pos);
89        let below_block = below.get_block();
90
91        let valid_below = below_block == &vanilla_blocks::CACTUS
92            || below_block.has_tag(&BlockTag::SUPPORTS_CACTUS);
93
94        if !valid_below {
95            return false;
96        }
97
98        // Block above must not be liquid
99        let above = world.get_block_state(pos.above());
100
101        if !above.get_fluid_state().is_empty() {
102            return false;
103        }
104
105        true
106    }
107
108    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
109        let default_state = self.block.default_state();
110        if self.can_survive(default_state, context.world, context.relative_pos) {
111            Some(default_state)
112        } else {
113            None
114        }
115    }
116
117    fn tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
118        if !self.can_survive(state, world, pos) {
119            world.destroy_block(pos, true);
120        }
121    }
122
123    fn is_randomly_ticking(&self, _state: BlockStateId) -> bool {
124        true
125    }
126
127    fn random_tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
128        let above_pos = pos.above();
129
130        if !world.get_block_state(above_pos).is_air() {
131            return;
132        }
133
134        // Count cactus blocks below
135        let mut height = 1u32;
136        let age = state.get_value(&BlockStateProperties::AGE_15);
137
138        while world
139            .get_block_state(pos.offset(0, -(height as i32), 0))
140            .get_block()
141            == &vanilla_blocks::CACTUS
142        {
143            height += 1;
144            if height == MAX_CACTUS_HEIGHT && age == 15 {
145                return;
146            }
147        }
148
149        // At age 8, chance to grow a cactus flower above
150        if age == CACTUS_FLOWER_AGE
151            && self.can_survive(self.block.default_state(), world, above_pos)
152        {
153            let chance = if height >= MAX_CACTUS_HEIGHT {
154                FLOWER_CHANCE_TALL
155            } else {
156                FLOWER_CHANCE_SMALL
157            };
158            if rand::random::<f64>() <= chance {
159                world.set_block(
160                    above_pos,
161                    vanilla_blocks::CACTUS_FLOWER.default_state(),
162                    UpdateFlags::UPDATE_ALL,
163                );
164            }
165        } else if age == 15 && height < MAX_CACTUS_HEIGHT {
166            world.set_block(
167                above_pos,
168                vanilla_blocks::CACTUS.default_state(),
169                UpdateFlags::UPDATE_ALL,
170            );
171            let new_state = state.set_value(&BlockStateProperties::AGE_15, 0);
172            world.set_block(pos, new_state, UpdateFlags::UPDATE_NONE);
173            world.neighbor_changed(above_pos, &vanilla_blocks::CACTUS, false);
174        }
175
176        if age < 15 {
177            let new_state = state.set_value(&BlockStateProperties::AGE_15, age + 1);
178            world.set_block(pos, new_state, UpdateFlags::UPDATE_NONE);
179        }
180    }
181
182    fn update_shape(
183        &self,
184        state: BlockStateId,
185        world: &dyn ScheduledTickAccess,
186        pos: BlockPos,
187        _direction: Direction,
188        _neighbor_pos: BlockPos,
189        _neighbor_state: BlockStateId,
190    ) -> BlockStateId {
191        // Vanilla: only schedule a tick if the cactus can't survive
192        if !self.can_survive(state, world, pos) {
193            world.schedule_block_tick_default(pos, self.block, 1);
194        }
195        state
196    }
197
198    fn entity_inside(
199        &self,
200        _state: BlockStateId,
201        _world: &Arc<World>,
202        _pos: BlockPos,
203        entity: &dyn Entity,
204        _effect_collector: &mut InsideBlockEffectCollector,
205        _is_precise: bool,
206    ) {
207        entity.hurt(
208            &DamageSource::environment(&vanilla_damage_types::CACTUS),
209            1.0,
210        );
211    }
212
213    fn is_pathfindable(
214        &self,
215        _state: BlockStateId,
216        _computation_type: PathComputationType,
217    ) -> bool {
218        false
219    }
