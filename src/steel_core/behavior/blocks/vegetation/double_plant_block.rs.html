20/// Behavior for vanilla two-block-tall plants.
21#[block_behavior]
22pub struct DoublePlantBlock {
23    block: BlockRef,
24}
25
26impl DoublePlantBlock {
27    /// Creates a new double plant block behavior.
28    #[must_use]
29    pub const fn new(block: BlockRef) -> Self {
30        Self { block }
31    }
32
33    pub(super) fn copy_waterlogged_from(
34        world: &Arc<World>,
35        pos: BlockPos,
36        state: BlockStateId,
37    ) -> BlockStateId {
38        if state
39            .try_get_value(&BlockStateProperties::WATERLOGGED)
40            .is_some()
41        {
42            state.set_value(
43                &BlockStateProperties::WATERLOGGED,
44                get_fluid_state(world, pos).is_water(),
45            )
46        } else {
47            state
48        }
49    }
50}
51
52impl Vegetation for DoublePlantBlock {}
53
54impl BlockBehavior for DoublePlantBlock {
55    fn update_shape(
56        &self,
57        state: BlockStateId,
58        world: &dyn ScheduledTickAccess,
59        pos: BlockPos,
60        direction: Direction,
61        _neighbor_pos: BlockPos,
62        neighbor_state: BlockStateId,
63    ) -> BlockStateId {
64        let half = state.get_value(&BlockStateProperties::DOUBLE_BLOCK_HALF);
65        let neighbor_is_matching_other_half = neighbor_state.get_block() == self.block
66            && neighbor_state.get_value(&BlockStateProperties::DOUBLE_BLOCK_HALF) != half;
67
68        if direction.get_axis() == Axis::Y
69            && (half == DoubleBlockHalf::Lower) == (direction == Direction::Up)
70            && !neighbor_is_matching_other_half
71        {
72            return vanilla_blocks::AIR.default_state();
73        }
74
75        if half == DoubleBlockHalf::Lower
76            && direction == Direction::Down
77            && !self.can_survive(state, world, pos)
78        {
79            return vanilla_blocks::AIR.default_state();
80        }
81
82        state
83    }
84
85    fn can_survive(&self, state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
86        double_plant_can_survive(self, state, world, pos)
87    }
88
89    fn set_placed_by(
90        &self,
91        _state: BlockStateId,
92        world: &Arc<World>,
93        pos: BlockPos,
94        _player: Option<&Player>,
95        _inv: &InventoryAccess,
96    ) {
97        let upper_pos = pos.above();
98        let upper_state = Self::copy_waterlogged_from(
99            world,
100            upper_pos,
101            self.block.default_state().set_value(
102                &BlockStateProperties::DOUBLE_BLOCK_HALF,
103                DoubleBlockHalf::Upper,
104            ),
105        );
106        world.set_block(upper_pos, upper_state, UpdateFlags::UPDATE_ALL);
107    }
108
109    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
110        if context.relative_pos.y() >= context.world.max_y_exclusive() - 1 {
111            return None;
112        }
113        if !context
114            .world
115            .get_block_state(context.relative_pos.above())
116            .is_replaceable()
117        {
118            return None;
119        }
120        default_surviving_state(self.block, self, context)
121    }
122}
