15/// Behavior for the nether portal block.
16#[block_behavior]
17pub struct NetherPortalBlock {
18    block: BlockRef,
19}
20impl NetherPortalBlock {
21    /// Create a new `NetherPortalBlock`
22    #[must_use]
23    pub const fn new(block: BlockRef) -> Self {
24        Self { block }
25    }
26}
27
28impl BlockBehavior for NetherPortalBlock {
29    fn update_shape(
30        &self,
31        state: BlockStateId,
32        world: &dyn ScheduledTickAccess,
33        pos: BlockPos,
34        direction: Direction,
35        _neighbor_pos: BlockPos,
36        neighbor_state: BlockStateId,
37    ) -> BlockStateId {
38        let update_axis = direction.get_axis();
39        let axis: Axis = state.get_value(&BlockStateProperties::HORIZONTAL_AXIS);
40        let wrong_axis = axis != update_axis && update_axis != Axis::Y;
41
42        if !wrong_axis
43            && neighbor_state.get_block() != self.block
44            && !PortalShape::find_any_shape(world, pos, axis, &nether_portal_config())
45                .is_some_and(|s| s.is_complete())
46        {
47            return AIR.default_state();
48        }
49        state
50    }
51
52    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
53        None // TODO: add this functionality but has low priority
54    }
