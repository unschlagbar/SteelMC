28/// Converts a rotation in degrees to a 16-segment rotation value (0-15).
29///
30/// This is equivalent to vanilla's `RotationSegment.convertToSegment(float)`.
31/// Each segment is 22.5 degrees, and rotation is measured clockwise from south.
32fn convert_to_rotation_segment(degrees: f32) -> u8 {
33    // Normalize to 0-360
34    let normalized = degrees.rem_euclid(360.0);
35    // Convert to segment (each segment is 22.5 degrees)
36    // Round to nearest segment
37    (((normalized / 22.5) + 0.5) as u8) & 15
38}

40/// Gets the nearest looking directions from the player's rotation.
41///
42/// Returns horizontal directions in order of how closely they match the player's look direction.
43fn get_nearest_looking_directions(rotation: f32, clicked_face: Direction) -> Vec<Direction> {
44    // Build list of directions in order of preference
45    // Start with the opposite of the clicked face (most natural for wall signs)
46    // Then add directions based on player facing
47    let mut directions = Vec::with_capacity(4);
48
49    // Add horizontal directions in order of how closely they match player's look
50    let all_horizontal = [
51        Direction::North,
52        Direction::East,
53        Direction::South,
54        Direction::West,
55    ];
56
57    // Calculate angle for each direction and sort by distance to player's rotation
58    let mut scored: Vec<(Direction, f32)> = all_horizontal
59        .iter()
60        .map(|&dir| {
61            let dir_angle = dir.to_yaw();
62            let diff = (rotation - dir_angle + 180.0).rem_euclid(360.0) - 180.0;
63            (dir, diff.abs())
64        })
65        .collect();
66
67    scored.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
68
69    for (dir, _) in scored {
70        directions.push(dir);
71    }
72
73    // If clicked face is horizontal, prefer placing on the opposite side
74    if clicked_face.is_horizontal() {
75        let opposite = clicked_face.opposite();
76        if let Some(pos) = directions.iter().position(|&d| d == opposite) {
77            directions.remove(pos);
78            directions.insert(0, opposite);
79        }
80    }
81
82    directions
83}

85/// Calculates whether the player is facing the front of a sign.
86///
87/// Uses the sign's rotation (from block state) and the player's position
88/// relative to the sign to determine which side they're looking at.
89fn is_facing_front_text(state: BlockStateId, pos: BlockPos, player: &Player) -> bool {
90    // Get the sign's Y rotation in degrees from the block state
91    let sign_y_rot = get_sign_rotation_degrees(state);
92
93    // Calculate player's angle relative to the sign center
94    let player_pos = player.position();
95    let dx = player_pos.x - (f64::from(pos.0.x) + 0.5);
96    let dz = player_pos.z - (f64::from(pos.0.z) + 0.5);
97
98    // Calculate angle from sign to player (in degrees, -90 to account for Minecraft's coordinate system)
99    let player_angle = (dz.atan2(dx) * 180.0 / PI) as f32 - 90.0;
100
101    // Front text if the angle difference is <= 90 degrees
102    let diff = (sign_y_rot - player_angle + 180.0).rem_euclid(360.0) - 180.0;
103    diff.abs() <= 90.0
104}

106/// Gets the Y rotation of a sign in degrees from its block state.
107fn get_sign_rotation_degrees(state: BlockStateId) -> f32 {
108    // Standing signs use "rotation" property (0-15, each step is 22.5 degrees)
109    if let Some(rotation) = state.try_get_value(&BlockStateProperties::ROTATION_16) {
110        return f32::from(rotation) * 22.5;
111    }
112
113    // Wall signs use "facing" property
114    if let Some(facing) = state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING) {
115        return facing.to_yaw();
116    }
117
118    0.0
119}

121/// Checks if a block state can support a standing sign.
122///
123/// Vanilla uses `isSolid()` which checks if the collision shape is a full cube.
124/// This means signs cannot be placed on other signs, fences, walls, etc.
125fn can_support_standing_sign(world: &dyn LevelReader, pos: BlockPos) -> bool {
126    let below_pos = BlockPos::new(pos.x(), pos.y() - 1, pos.z());
127    let below_state = world.get_block_state(below_pos);
128    below_state.is_solid()
129}

131/// Checks if a wall sign can survive at the given position with the given facing.
132///
133/// Vanilla uses `isSolid()` which allows wall signs to be placed on other signs
134/// (since signs have `forceSolidOn`).
135fn can_wall_sign_survive(world: &dyn LevelReader, pos: BlockPos, facing: Direction) -> bool {
136    // Wall sign needs a solid block behind it
137    let behind_pos = facing.opposite().relative(pos);
138    let behind_state = world.get_block_state(behind_pos);
139    behind_state.is_solid()
140}

142/// Checks if a ceiling hanging sign can survive at the given position.
143fn can_ceiling_hanging_sign_survive(world: &dyn LevelReader, pos: BlockPos) -> bool {
144    let above_pos = BlockPos::new(pos.x(), pos.y() + 1, pos.z());
145    let above_state = world.get_block_state(above_pos);
146    above_state.is_face_sturdy_for_at(above_pos, Direction::Down, SupportType::Center)
147}

149/// Checks if a wall hanging sign can attach to a neighboring block.
150///
151/// Vanilla's `WallHangingSignBlock.canAttachTo` checks:
152/// 1. If the neighbor is a wall hanging sign on the same axis, allow attachment
153/// 2. Otherwise, check if the face is sturdy with FULL support type
154fn can_attach_to(
155    world: &dyn LevelReader,
156    sign_facing: Direction,
157    attach_pos: BlockPos,
158    attach_face: Direction,
159) -> bool {
160    let attach_state = world.get_block_state(attach_pos);
161    let attach_block = REGISTRY.blocks.by_state_id(attach_state);
162
163    // Check if it's another wall hanging sign (vanilla uses BlockTags.WALL_HANGING_SIGNS)
164    if let Some(block) = attach_block
165        && block.key.path.contains("wall_hanging_sign")
166    {
167        // Wall hanging signs can chain if they're on the same axis
168        if let Some(neighbor_facing) =
169            attach_state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
170        {
171            return neighbor_facing.axis() == sign_facing.axis();
172        }
173    }
174
175    // Otherwise, check for sturdy face with FULL support
176    attach_state.is_face_sturdy_for_at(attach_pos, attach_face, SupportType::Full)
177}

179/// Checks if a wall hanging sign can survive at the given position.
180///
181/// Wall hanging signs need support on at least one side perpendicular to facing.
182/// This matches vanilla's `WallHangingSignBlock.canPlace`.
183fn can_wall_hanging_sign_survive(
184    world: &dyn LevelReader,
185    pos: BlockPos,
186    facing: Direction,
187) -> bool {
188    let clockwise = facing.rotate_y_clockwise();
189    let counter_clockwise = facing.rotate_y_counter_clockwise();
190
191    let can_attach_clockwise = {
192        let attach_pos = clockwise.relative(pos);
193        can_attach_to(world, facing, attach_pos, counter_clockwise)
194    };
195
196    let can_attach_counter = {
197        let attach_pos = counter_clockwise.relative(pos);
198        can_attach_to(world, facing, attach_pos, clockwise)
199    };
200
201    can_attach_clockwise || can_attach_counter
202}

226/// Attempts to open the sign editor for a player.
227///
228/// Checks all conditions required by vanilla:
229/// 1. Block entity exists and is a sign
230/// 2. Sign is not waxed
231/// 3. No other player is currently editing
232/// 4. Player has build permission (`may_build`)
233///
234/// Returns `Success` if the editor was opened, `Pass` otherwise.
235fn try_open_sign_editor(
236    state: BlockStateId,
237    world: &Arc<World>,
238    pos: BlockPos,
239    player: &Player,
240) -> InteractionResult {
241    // Get the block entity
242    let Some(block_entity) = world.get_block_entity(pos) else {
243        return InteractionResult::Pass;
244    };
245
246    let mut guard = block_entity.lock();
247    let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() else {
248        return InteractionResult::Pass;
249    };
250
251    // Check 1: Is the sign waxed?
252    if sign.is_waxed {
253        // TODO: Play waxed sign interaction fail sound
254        return InteractionResult::Success; // Vanilla returns SUCCESS even when waxed
255    }
256
257    // Check 2: Is another player editing?
258    if sign.is_other_player_editing(player.gameprofile.id) {
259        return InteractionResult::Pass;
260    }
261
262    // Check 3: Player must have build permission
263    // TODO: Implement may_build check properly
264    // if !player.may_build() {
265    //     return InteractionResult::Pass;
266    // }
267
268    // Determine which side the player is facing
269    let is_front_text = is_facing_front_text(state, pos, player);
270
271    // Set the editing player lock
272    sign.set_player_who_may_edit(Some(player.gameprofile.id));
273
274    // Release lock before calling player method
275    drop(guard);
276
277    // Open the editor
278    player.open_sign_editor(pos, is_front_text);
279    InteractionResult::Success
280}

296impl BlockBehavior for StandingSignBlock {
297    fn update_shape(
298        &self,
299        state: BlockStateId,
300        world: &dyn ScheduledTickAccess,
301        pos: BlockPos,
302        direction: Direction,
303        _neighbor_pos: BlockPos,
304        _neighbor_state: BlockStateId,
305    ) -> BlockStateId {
306        // Standing signs break when the block below is removed
307        if direction == Direction::Down && !can_support_standing_sign(world, pos) {

309        }
310        state
311    }

314        // Check if we can place on the block below
315        if !can_support_standing_sign(context.world, context.relative_pos) {
316            return None;
317        }
318
319        // Calculate rotation from player's yaw
320        // Vanilla: RotationSegment.convertToSegment(context.getRotation() + 180.0F)
321        let rotation = convert_to_rotation_segment(context.rotation + 180.0);
322
323        Some(
324            self.block
325                .default_state()
326                .set_value(&BlockStateProperties::ROTATION_16, rotation),
327        )
328    }
329
330    fn has_block_entity(&self) -> bool {
331        true
332    }
333
334    fn new_block_entity(
335        &self,
336        level: Weak<World>,
337        pos: BlockPos,
338        state: BlockStateId,
339    ) -> Option<SharedBlockEntity> {
340        Some(Arc::new(SyncMutex::new(SignBlockEntity::new(
341            level, pos, state,
342        ))))
343    }
344
345    fn should_keep_block_entity(&self, _old_state: BlockStateId, _new_state: BlockStateId) -> bool {
346        // Signs don't keep their block entity when changing to a different block

348    }
349
350    fn use_without_item(
351        &self,
352        state: BlockStateId,
353        world: &Arc<World>,
354        pos: BlockPos,
355        player: &Player,
356        _hit_result: &BlockHitResult,
357        _inv: &mut InventoryAccess,
358    ) -> InteractionResult {
359        try_open_sign_editor(state, world, pos, player)
360    }
361}

404            let facing = direction.opposite();

406            // Check if sign can survive with this facing

409                    self.block
410                        .default_state()
411                        .set_value(&BlockStateProperties::HORIZONTAL_FACING, facing),

452/// Behavior for ceiling hanging sign blocks.
453#[block_behavior]
454pub struct CeilingHangingSignBlock {
455    block: BlockRef,
456}
457
458impl CeilingHangingSignBlock {
459    /// Creates a new ceiling hanging sign block behavior.
460    #[must_use]
461    pub const fn new(block: BlockRef) -> Self {
462        Self { block }
463    }
464}
465
466impl BlockBehavior for CeilingHangingSignBlock {
467    fn update_shape(
468        &self,
469        state: BlockStateId,
470        world: &dyn ScheduledTickAccess,
471        pos: BlockPos,
472        direction: Direction,
473        _neighbor_pos: BlockPos,
474        _neighbor_state: BlockStateId,
475    ) -> BlockStateId {
476        // Ceiling hanging signs break when the block above is removed
477        if direction == Direction::Up && !can_ceiling_hanging_sign_survive(world, pos) {
478            return REGISTRY.blocks.get_default_state_id(&vanilla_blocks::AIR);
479        }
480        state
481    }
482
483    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
484        // Check if we can hang from the block above
485        if !can_ceiling_hanging_sign_survive(context.world, context.relative_pos) {
486            return None;
487        }
488
489        let above_pos = BlockPos::new(
490            context.relative_pos.x(),
491            context.relative_pos.y() + 1,
492            context.relative_pos.z(),
493        );
494        let above_state = context.world.get_block_state(above_pos);
495
496        // Determine if we should attach to the middle or not based on block above
497        let direction = Direction::from_yaw(context.rotation);
498        let is_above_full =
499            above_state.is_face_sturdy_for_at(above_pos, Direction::Down, SupportType::Full);
500
501        // Check if block above is also a hanging sign
502        let above_block = REGISTRY.blocks.by_state_id(above_state);
503        let is_below_hanging_sign =
504            above_block.is_some_and(|b| b.key.path.contains("hanging_sign"));
505
506        // Determine if attached to middle based on vanilla logic
507        let attached_to_middle = if is_below_hanging_sign {
508            // When below another hanging sign, check if we can chain
509            if let Some(above_facing) =
510                above_state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
511            {
512                // Wall hanging sign above - check axis alignment
513                above_facing.axis() != direction.axis()
514            } else if let Some(above_rotation) =
515                above_state.try_get_value(&BlockStateProperties::ROTATION_16)
516            {
517                // Ceiling hanging sign above - check if we can align
518                let above_direction = rotation_to_direction(above_rotation);
519                above_direction.is_none_or(|d| d.axis() != direction.axis())
520            } else {
521                !is_above_full
522            }
523        } else {
524            !is_above_full
525        };
526
527        // Calculate rotation
528        let rotation = if attached_to_middle {
529            // Attached to middle - use player rotation
530            convert_to_rotation_segment(context.rotation + 180.0)
531        } else {
532            // Attached to chains - align with direction
533            convert_to_rotation_segment(direction.opposite().to_yaw())
534        };
535
536        Some(
537            self.block
538                .default_state()
539                .set_value(&BlockStateProperties::ROTATION_16, rotation)
540                .set_value(&BlockStateProperties::ATTACHED, attached_to_middle),
541        )
542    }
543
544    fn has_block_entity(&self) -> bool {
545        true
546    }
547
548    fn new_block_entity(
549        &self,
550        level: Weak<World>,
551        pos: BlockPos,
552        state: BlockStateId,
553    ) -> Option<SharedBlockEntity> {
554        Some(Arc::new(SyncMutex::new(SignBlockEntity::new_hanging(
555            level, pos, state,
556        ))))
557    }
558
559    fn should_keep_block_entity(&self, _old_state: BlockStateId, _new_state: BlockStateId) -> bool {
560        false
561    }
562
563    fn use_without_item(
564        &self,
565        state: BlockStateId,
566        world: &Arc<World>,
567        pos: BlockPos,
568        player: &Player,
569        _hit_result: &BlockHitResult,
570        _inv: &mut InventoryAccess,
571    ) -> InteractionResult {
572        try_open_sign_editor(state, world, pos, player)
573    }
574}

576/// Converts a rotation segment (0-15) to a cardinal direction, if applicable.
577const fn rotation_to_direction(rotation: u8) -> Option<Direction> {
578    match rotation {
579        0 => Some(Direction::South),
580        4 => Some(Direction::West),
581        8 => Some(Direction::North),
582        12 => Some(Direction::East),
583        _ => None,
584    }

608        _neighbor_pos: BlockPos,

612        // and they can no longer survive

614            // Check if the change is on the perpendicular axis (clockwise/counterclockwise)
615            if direction.axis() == facing.rotate_y_clockwise().axis()
616                && !can_wall_hanging_sign_survive(world, pos, facing)
617            {
618                return REGISTRY.blocks.get_default_state_id(&vanilla_blocks::AIR);
619            }
620        }

625        // Try each horizontal direction based on player's look direction
626        let directions = get_nearest_looking_directions(context.rotation, context.clicked_face);
627
628        for direction in directions {
629            // Wall hanging signs face perpendicular to the wall they're attached to
630            // Skip if the clicked face is on the same axis
631            if direction.axis() == context.clicked_face.axis() {
632                continue;
633            }
634
635            let facing = direction.opposite();
636
637            // Check if sign can survive with this facing
638            if can_wall_hanging_sign_survive(context.world, context.relative_pos, facing) {

640                    self.block
641                        .default_state()
