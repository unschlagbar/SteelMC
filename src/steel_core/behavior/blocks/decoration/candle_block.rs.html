27const CANDLES_PROPERTY: IntProperty = BlockStateProperties::CANDLES;
28const LIT_PROPERTY: BoolProperty = BlockStateProperties::LIT;
29const WATERLOGGED: BoolProperty = BlockStateProperties::WATERLOGGED;
30const MAX_CANDLES: u8 = 4;
31
32/// Behavior for all Candle type blocks
33#[block_behavior]
34pub struct CandleBlock {
35    block: BlockRef,
36}
37
38impl CandleBlock {
39    /// Creates a new candle block behavior for the given block
40    #[must_use]
41    pub const fn new(block: BlockRef) -> Self {
42        Self { block }
43    }
44}
45
46impl BlockBehavior for CandleBlock {
47    /// Checks if the candle block can survive at the given position.
48    fn can_survive(
49        &self,
50        _state: steel_utils::BlockStateId,
51        world: &dyn LevelReader,
52        pos: BlockPos,
53    ) -> bool {
54        let below_pos = pos.below();
55        world.get_block_state(below_pos).is_face_sturdy_for_at(
56            below_pos,
57            Direction::Up,
58            SupportType::Center,
59        )
60    }
61
62    fn get_state_for_placement(
63        &self,
64        context: &BlockPlaceContext<'_>,
65    ) -> Option<steel_utils::BlockStateId> {
66        let default_state = self.block.default_state();
67        if self.can_survive(default_state, context.world, context.relative_pos) {
68            return Some(default_state.set_value(&WATERLOGGED, context.is_water_source()));
69        }
70        None
71    }
72
73    fn update_shape(
74        &self,
75        state: steel_utils::BlockStateId,
76        world: &dyn ScheduledTickAccess,
77        pos: BlockPos,
78        _direction: Direction,
79        _neighbor_pos: BlockPos,
80        _neighbor_state: steel_utils::BlockStateId,
81    ) -> steel_utils::BlockStateId {
82        if !self.can_survive(state, world, pos) {
83            return REGISTRY.blocks.get_default_state_id(&vanilla_blocks::AIR);
84        }
85        state
86    }
87
88    fn use_item_on(
89        &self,
90        state: steel_utils::BlockStateId,
91        world: &Arc<World>,
92        pos: BlockPos,
93        _player: &player::Player,
94        _hand: types::InteractionHand,
95        _hit_result: &BlockHitResult,
96        inv: &mut InventoryAccess,
97    ) -> InteractionResult {
98        let item_is_empty = inv.with_item(|item_stack| item_stack.is_empty());
99        if item_is_empty {
100            if !state.get_value(&LIT_PROPERTY) {
101                return InteractionResult::Pass;
102            }
103            let new_state = state.set_value(&LIT_PROPERTY, false);
104            world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL_IMMEDIATE);
105            return InteractionResult::Success;
106        }
107
108        if self
109            .get_clone_item_stack(self.block, state, false)
110            .is_some_and(|it| inv.with_item(|item_stack| it.is(item_stack.item)))
111        {
112            let candles_amount = state.get_value(&CANDLES_PROPERTY);
113            if candles_amount < MAX_CANDLES {
114                let new_state = state.set_value(&CANDLES_PROPERTY, candles_amount + 1);
115                world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL_IMMEDIATE);
116                return InteractionResult::Success;
117            }
118        }
119
120        InteractionResult::TryEmptyHandInteraction
121    }
122}
