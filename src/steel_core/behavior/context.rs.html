19/// Result of an interaction (item use, block use, etc.)
20#[derive(Debug, Clone, Copy, PartialEq, Eq)]
21pub enum InteractionResult {
22    /// The interaction succeeded and consumed the action.
23    Success,
24    /// The interaction succeeded and the server should broadcast the swing.
25    SuccessServer,
26    /// The interaction consumed the action without swinging.
27    Consume,
28    /// The interaction failed and consumed the action.
29    Fail,
30    /// The interaction did not apply; try the next handler.
31    Pass,
32    /// Try the empty-hand interaction on the block.
33    TryEmptyHandInteraction,
34}
35
36impl InteractionResult {
37    /// Returns true if this result consumes the action (Success or Fail).
38    /// Pass and `TryEmptyHandInteraction` do not consume the action.
39    #[must_use]
40    pub const fn consumes_action(self) -> bool {
41        matches!(
42            self,
43            InteractionResult::Success
44                | InteractionResult::SuccessServer
45                | InteractionResult::Consume
46                | InteractionResult::Fail
47        )
48    }
49
50    /// Returns true when vanilla requests the server to broadcast the swing.
51    #[must_use]
52    pub const fn should_swing_server(self) -> bool {
53        matches!(self, InteractionResult::SuccessServer)
54    }
55}

57/// Context for placing a block.
58pub struct BlockPlaceContext<'a> {
59    /// The position that was clicked.
60    pub clicked_pos: BlockPos,
61    /// The face of the block that was clicked.
62    pub clicked_face: Direction,
63    /// The exact location where the click occurred.
64    pub click_location: DVec3,
65    /// Whether the click was inside the block.
66    pub inside: bool,
67    /// The position where the block will be placed.
68    pub relative_pos: BlockPos,
69    /// Whether the clicked block is being replaced.
70    pub replace_clicked: bool,
71    /// The horizontal direction the player is facing.
72    pub horizontal_direction: Direction,
73    /// The player's rotation (yaw).
74    pub rotation: f32,
75    /// The player's pitch (vertical look angle).
76    pub pitch: f32,
77    /// Whether the player is using the secondary action, normally sneaking.
78    pub is_secondary_use_active: bool,
79    /// The world where the block is being placed.
80    pub world: &'a Arc<World>,
81}
82
83impl BlockPlaceContext<'_> {
84    /// Returns the direction the player is looking at most directly.
85    ///
86    /// This considers both yaw and pitch to determine the nearest direction
87    /// among all 6 directions (UP, DOWN, NORTH, SOUTH, EAST, WEST).
88    ///
89    /// Based on Java's `Direction.orderedByNearest(Entity)[0]`.
90    #[must_use]
91    pub fn get_nearest_looking_direction(&self) -> Direction {
92        self.get_nearest_looking_directions()[0]
93    }
94
95    /// Returns the vertical direction the player is looking toward.
96    ///
97    /// Based on Java's `BlockPlaceContext.getNearestLookingVerticalDirection()`.
98    #[must_use]
99    pub const fn get_nearest_looking_vertical_direction(&self) -> Direction {
100        if self.pitch < 0.0 {
101            Direction::Up
102        } else {
103            Direction::Down
104        }
105    }
106
107    /// Returns all 6 directions ordered by how closely the player is looking at them.
108    ///
109    /// Based on Java's `BlockPlaceContext.getNearestLookingDirections()`.
110    /// When not replacing the clicked block, the opposite of the clicked face
111    /// is moved to the front of the array.
112    #[must_use]
113    pub fn get_nearest_looking_directions(&self) -> [Direction; 6] {
114        let mut directions = Direction::ordered_by_nearest(self.rotation, self.pitch);
115
116        // If not replacing the clicked block, prioritize the opposite of clicked face
117        if !self.replace_clicked {
118            let clicked_opposite = self.clicked_face.opposite();
119            if let Some(index) = directions.iter().position(|&d| d == clicked_opposite)
120                && index > 0
121            {
122                directions.copy_within(0..index, 1);
123                directions[0] = clicked_opposite;
124            }
125        }
126
127        directions
128    }
129
130    /// Returns true if the block at the relative position is a water source
131    #[must_use]
132    pub fn is_water_source(&self) -> bool {
133        use crate::fluid::get_fluid_state;
134        let fluid_state = get_fluid_state(self.world, self.relative_pos);
135        fluid_state.is_source() && fluid_state.is_water()
136    }
137}

139/// Access to the player's inventory.
140///
141/// This handle does not hold the inventory lock by itself. Use the closure
142/// methods to keep lock scopes short and avoid carrying an inventory guard
143/// through block behavior, world mutation, or menu opening.
144pub struct InventoryAccess {
145    inventory: SyncPlayerInv,
146    hand: InteractionHand,
147}
148
149impl InventoryAccess {
150    /// Creates a new `InventoryAccess` instance.
151    pub const fn new(inventory: SyncPlayerInv, hand: InteractionHand) -> Self {
152        Self { inventory, hand }
153    }
154
155    /// Runs `f` with mutable access to the item in the player's hand.
156    pub fn with_item<R>(&self, f: impl FnOnce(&mut ItemStack) -> R) -> R {
157        let mut inventory = self.inventory.lock();
158        f(inventory.get_item_in_hand_mut(self.hand))
159    }
160
161    /// Runs `f` with mutable access to the player's inventory.
162    pub fn with_inventory<R>(&self, f: impl FnOnce(&mut PlayerInventory) -> R) -> R {
163        let mut inventory = self.inventory.lock();
164        f(&mut inventory)
165    }
166
167    /// Runs `f` with a container guard containing the player's inventory.
168    ///
169    /// Prefer [`Self::with_item`] or [`Self::with_inventory`] unless an operation
170    /// must interoperate with APIs that require `ContainerLockGuard`.
171    pub fn with_guard<R>(&self, f: impl FnOnce(&mut ContainerLockGuard) -> R) -> R {
172        let inv_ref = ContainerRef::PlayerInventory(self.inventory.clone());
173        let mut guard = ContainerLockGuard::lock_all(&[&inv_ref]);
174        f(&mut guard)
175    }
176}

178/// Context for using an item on a block.
179///
180/// Immutable fields (`player`, `hand`, `world`, `hit_result`) can be accessed
181/// freely while `inv` is mutably borrowed — the borrow checker tracks them as
182/// disjoint fields.
183pub struct UseOnContext<'a> {
184    /// The player using the item.
185    pub player: &'a Player,
186    /// Which hand the item is in.
187    pub hand: InteractionHand,
188    /// Information about where the block was hit.
189    pub hit_result: BlockHitResult,
190    /// The world where the interaction is happening.
191    pub world: &'a Arc<World>,
192    /// Mutable inventory access.
193    pub inv: InventoryAccess,
194}
195
196impl<'a> UseOnContext<'a> {
197    /// Creates a new `UseOnContext`.
198    #[must_use]
199    pub const fn new(
200        player: &'a Player,
201        hand: InteractionHand,
202        hit_result: BlockHitResult,
203        world: &'a Arc<World>,
204        inventory: SyncPlayerInv,
205    ) -> Self {
206        Self {
207            player,
208            hand,
209            hit_result,
210            world,
211            inv: InventoryAccess::new(inventory, hand),
212        }
213    }
214
215    /// Builds a [`BlockPlaceContext`] from this interaction context.
216    ///
217    /// Returns `None` if placement is invalid (out of bounds or non-replaceable target).
218    /// This is the common prefix of vanilla's `BlockItem.useOn`.
219    #[must_use]
220    pub fn build_place_context(&self) -> Option<BlockPlaceContext<'a>> {
221        let clicked_pos = self.hit_result.block_pos;
222        let clicked_state = self.world.get_block_state(clicked_pos);
223        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
224        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
225
226        let (place_pos, replace_clicked) = if clicked_replaceable {
227            (clicked_pos, true)
228        } else {
229            (self.hit_result.direction.relative(clicked_pos), false)
230        };
231
232        if !self.world.is_in_valid_bounds(place_pos) {
233            return None;
234        }
235
236        let existing_state = self.world.get_block_state(place_pos);
237        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
238        if !existing_block.is_some_and(|b| b.config.replaceable) {
239            return None;
240        }
241
242        let (yaw, pitch) = self.player.rotation();
243
244        Some(BlockPlaceContext {
245            clicked_pos,
246            clicked_face: self.hit_result.direction,
247            click_location: self.hit_result.location,
248            inside: self.hit_result.inside,
249            relative_pos: place_pos,
250            replace_clicked,
251            horizontal_direction: Direction::from_yaw(yaw),
252            rotation: yaw,
253            pitch,
254            is_secondary_use_active: self.player.is_secondary_use_active(),
255            world: self.world,
256        })
257    }
258}
