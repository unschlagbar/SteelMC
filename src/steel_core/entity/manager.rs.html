22/// Error returned when adding an entity to the runtime world fails.
23#[derive(Debug, Clone, PartialEq, Eq)]
24pub enum AddEntityError {
25    /// The entity is in a chunk that is not active in the world entity manager.
26    ChunkNotLoaded {
27        /// Entity network ID.
28        entity_id: i32,
29        /// Chunk containing the entity.
30        chunk: ChunkPos,
31    },
32    /// Another live entity with the same persistent UUID is already registered.
33    DuplicateUuid {
34        /// Entity network ID.
35        entity_id: i32,
36        /// Duplicate persistent UUID.
37        uuid: Uuid,
38    },
39    /// The entity is already removed and cannot be added to the live world.
40    RemovedEntity {
41        /// Entity network ID.
42        entity_id: i32,
43    },
44}
45
46impl fmt::Display for AddEntityError {
47    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
48        match self {
49            Self::ChunkNotLoaded { entity_id, chunk } => {
50                write!(f, "entity {entity_id} is in non-loaded chunk {chunk:?}")
51            }
52            Self::DuplicateUuid { entity_id, uuid } => {
53                write!(f, "entity {entity_id} has duplicate UUID {uuid}")
54            }
55            Self::RemovedEntity { entity_id } => {
56                write!(f, "entity {entity_id} is already removed")
57            }
58        }
59    }
60}

116/// Entity visibility for a chunk column.
117///
118/// Mirrors vanilla `Visibility`: hidden chunks keep entity data inactive,
119/// tracked chunks expose entities to lookup/tracking, and ticking chunks also
120/// run manager-owned entity ticks.
121#[derive(Debug, Clone, Copy, PartialEq, Eq)]
122pub enum EntityVisibility {
123    /// Not accessible to entity lookup/tracking and not ticking.
124    Hidden,
125    /// Accessible to entity lookup/tracking but not ticking.
126    Tracked,
127    /// Accessible to entity lookup/tracking and ticking.
128    Ticking,
129}
130
131impl EntityVisibility {
132    /// Returns whether entities in this visibility are accessible to queries and tracking.
133    #[must_use]
134    pub const fn is_accessible(self) -> bool {
135        matches!(self, Self::Tracked | Self::Ticking)
136    }
137
138    /// Returns whether entities in this visibility are eligible for ticking.
139    #[must_use]
140    pub const fn is_ticking(self) -> bool {
141        matches!(self, Self::Ticking)
142    }
143}

201impl EntityMoveUpdate {

231

237
238    /// Returns whether this move made a previously ticking entity stop ticking.

240    pub const fn became_non_ticking(&self) -> bool {

242    }

289impl EntityEntry {
290    fn new(entity: SharedEntity, ownership: EntityOwnership) -> Self {
291        let section = SectionPos::from_entity_pos(entity.position());

302    #[must_use]
303    fn should_save(&self) -> bool {
304        self.ownership == EntityOwnership::ManagerOwned
305            && (!self.entity.is_removed()
306                || self
307                    .entity
308                    .removal_reason()
309                    .is_some_and(RemovalReason::should_save))
310            && !self.entity.is_passenger()
311            && !self.entity.has_exactly_one_player_passenger()
312            && self.entity.entity_type().can_serialize
313    }

328#[derive(Default)]
329struct EntityTickList {
330    active: FxHashMap<i32, SharedEntity>,
331    order: Vec<i32>,
332}
333
334impl EntityTickList {
335    fn add(&mut self, entity: &SharedEntity) -> bool {
336        let entity_id = entity.id();
337        if self.active.insert(entity_id, entity.clone()).is_some() {
338            return false;
339        }
340        self.order.push(entity_id);
341        true
342    }
343
344    fn remove(&mut self, entity_id: i32) -> Option<SharedEntity> {
345        let removed = self.active.remove(&entity_id)?;
346        self.order.retain(|id| *id != entity_id);
347        Some(removed)
348    }
349
350    fn contains(&self, entity_id: i32) -> bool {
351        self.active.contains_key(&entity_id)
352    }
353
354    fn snapshot(&self) -> Vec<SharedEntity> {
355        self.order
356            .iter()
357            .filter_map(|id| self.active.get(id))
358            .cloned()
359            .collect()
360    }
361}

388    /// Marks a chunk as loaded and reactivates retained unloading entities.
389    pub fn on_chunk_loaded(&self, pos: ChunkPos) -> ChunkEntityLoadResult {
390        let mut state = self.state.write();
391        state
392            .chunk_visibility
393            .entry(pos)
394            .or_insert(EntityVisibility::Hidden);
395
396        let mut result = ChunkEntityLoadResult::default();
397        if let Some(entries) = state.unloading_by_chunk.remove(&pos) {
398            result.restored.reserve(entries.len());
399            for entry in entries {
400                if entry.entity.is_removed() {
401                    if entry.should_save() {
402                        result.needs_save = true;
403                        state
404                            .save_pending_by_chunk
405                            .entry(pos)
406                            .or_default()
407                            .push(entry);
408                    }
409                    continue;
410                }
411
412                let entity = entry.entity.clone();
413                Self::insert_live_entry(&mut state, entry);
414                let lifecycle = Self::apply_entity_lifecycle_after_insert(&mut state, entity.id());
415                result.tracking_started.extend(lifecycle.tracking_started);
416                result.ticking_started.extend(lifecycle.ticking_started);
417                result.restored.push(entity);
418            }
419        }
420
421        result
422    }

453    /// Moves manager-owned root entities in `pos` out of live world membership while
454    /// retaining them for possible chunk recovery.
455    pub fn begin_chunk_unload(&self, pos: ChunkPos) -> ChunkEntityUnloadStart {
456        let mut state = self.state.write();
457        let previous_visibility = state
458            .chunk_visibility
459            .remove(&pos)
460            .unwrap_or(EntityVisibility::Hidden);
461
462        let ids = state
463            .by_chunk
464            .get(&pos)
465            .map(|set| set.iter().copied().collect::<Vec<_>>())
466            .unwrap_or_default();
467
468        let mut result = ChunkEntityUnloadStart::default();
469        let lifecycle = Self::apply_chunk_visibility_change(
470            &mut state,
471            pos,
472            previous_visibility,
473            EntityVisibility::Hidden,
474        );
475        let mut tracking_stopped_ids = lifecycle
476            .tracking_stopped
477            .iter()
478            .map(|entity| entity.id())
479            .collect::<FxHashSet<_>>();
480        result.tracking_stopped = lifecycle.tracking_stopped;
481        result.ticking_stopped = lifecycle.ticking_stopped;
482
483        let mut root_ids = Vec::new();
484        for entity_id in ids {
485            let Some(entry) = state.live_by_id.get(&entity_id) else {
486                continue;
487            };
488            if entry.ownership != EntityOwnership::ManagerOwned {
489                continue;
490            }
491
492            Self::push_unique_entity(
493                &entry.entity,
494                &mut tracking_stopped_ids,
495                &mut result.tracking_stopped,
496            );
497            if !entry.entity.is_passenger() {
498                root_ids.push(entity_id);
499            }
500        }
501
502        let mut retained = Vec::new();
503        let mut visited = FxHashSet::default();
504        for entity_id in root_ids {
505            Self::retain_unloading_entity_tree(
506                &mut state,
507                entity_id,
508                &mut visited,
509                &mut retained,
510                &mut result.retained,
511                &mut tracking_stopped_ids,
512                &mut result.tracking_stopped,
513            );
514        }
515
516        if !retained.is_empty() {
517            state
518                .unloading_by_chunk
519                .entry(pos)
520                .or_default()
521                .extend(retained);
522        }
523
524        result
525    }

527    fn retain_unloading_entity_tree(
528        state: &mut ManagerState,
529        entity_id: i32,
530        visited: &mut FxHashSet<i32>,
531        retained: &mut Vec<EntityEntry>,
532        retained_entities: &mut Vec<SharedEntity>,
533        tracking_stopped_ids: &mut FxHashSet<i32>,
534        tracking_stopped: &mut Vec<SharedEntity>,
535    ) {
536        if !visited.insert(entity_id) {
537            return;
538        }
539
540        let Some(entry) = Self::remove_live_entry(state, entity_id) else {
541            return;
542        };
543
544        if entry.ownership != EntityOwnership::ManagerOwned {
545            let restored_id = entry.entity.id();
546            Self::insert_live_entry(state, entry);
547            let entity_to_tick = state.live_by_id.get(&restored_id).and_then(|entry| {
548                let visibility = Self::lifecycle_visibility_for(
549                    entry,
550                    Self::chunk_visibility(state, entry.chunk),
551                );
552                visibility.is_ticking().then(|| entry.entity.clone())
553            });
554            if let Some(entity) = entity_to_tick {
555                state.tick_list.add(&entity);
556            }
557            return;
558        }
559
560        let passengers = entry.entity.passengers();
561        Self::push_unique_entity(&entry.entity, tracking_stopped_ids, tracking_stopped);
562        retained_entities.push(Arc::clone(&entry.entity));
563        retained.push(entry);
564        for passenger in passengers {
565            Self::retain_unloading_entity_tree(
566                state,
567                passenger.id(),
568                visited,
569                retained,
570                retained_entities,
571                tracking_stopped_ids,
572                tracking_stopped,
573            );
574        }
575    }

594    /// Registers a live runtime entity.
595    ///
596    /// # Panics
597    ///
598    /// Panics if an entity with the same session network ID is already present. Duplicate runtime
599    /// IDs indicate corrupted manager ownership and cannot be recovered without losing identity.
600    pub fn add_live_entity(
601        &self,
602        entity: SharedEntity,
603        ownership: EntityOwnership,
604    ) -> Result<EntityLifecycleChanges, AddEntityError> {
605        let entry = Self::checked_live_entry(entity, ownership)?;
606        let entity_id = entry.entity.id();
607        let mut state = self.state.write();
608        Self::validate_live_entries(&state, slice::from_ref(&entry), ownership, true)?;
609        Self::insert_live_entry(&mut state, entry);
610        Ok(Self::apply_entity_lifecycle_after_insert(
611            &mut state, entity_id,
612        ))
613    }

615    /// Adds a related group of live entities atomically.
616    ///
617    /// Use this for persisted vehicle/passenger trees so registration either
618    /// publishes the whole tree or leaves world indexes unchanged.
619    ///
620    /// # Panics
621    ///
622    /// Panics if the entity tree contains the same session network ID more
623    /// than once. Duplicate runtime IDs indicate corrupted ownership.
624    pub fn add_live_entity_tree(
625        &self,
626        entities: &[SharedEntity],
627        ownership: EntityOwnership,
628    ) -> Result<EntityLifecycleChanges, AddEntityError> {
629        let mut entries = Vec::with_capacity(entities.len());
630        for entity in entities {
631            entries.push(Self::checked_live_entry(Arc::clone(entity), ownership)?);
632        }
633
634        let mut seen_ids = FxHashSet::default();
635        let mut seen_uuids = FxHashSet::default();
636        for entry in &entries {
637            let entity_id = entry.entity.id();
638            assert!(
639                seen_ids.insert(entity_id),
640                "entity id {entity_id} appears more than once in a live entity tree"
641            );
642            if !seen_uuids.insert(entry.uuid) {
643                return Err(AddEntityError::DuplicateUuid {
644                    entity_id,
645                    uuid: entry.uuid,
646                });
647            }
648        }
649
650        let mut state = self.state.write();
651        Self::validate_live_entries(&state, &entries, ownership, false)?;
652        let entity_ids = entries
653            .iter()
654            .map(|entry| entry.entity.id())
655            .collect::<Vec<_>>();
656        for entry in entries {
657            Self::insert_live_entry(&mut state, entry);
658        }
659        let mut lifecycle = EntityLifecycleChanges::default();
660        for entity_id in entity_ids {
661            lifecycle.extend(Self::apply_entity_lifecycle_after_insert(
662                &mut state, entity_id,
663            ));
664        }
665        Ok(lifecycle)
666    }

668    fn checked_live_entry(
669        entity: SharedEntity,
670        ownership: EntityOwnership,
671    ) -> Result<EntityEntry, AddEntityError> {
672        if entity.is_removed() {
673            return Err(AddEntityError::RemovedEntity {
674                entity_id: entity.id(),
675            });
676        }

681    fn validate_live_entries(
682        state: &ManagerState,
683        entries: &[EntityEntry],
684        ownership: EntityOwnership,
685        require_loaded_chunks: bool,
686    ) -> Result<(), AddEntityError> {
687        for entry in entries {
688            let entity_id = entry.entity.id();
689            assert!(
690                !Self::contains_id(state, entity_id),
691                "entity id {entity_id} is already registered in the world entity manager"
692            );
693            if Self::contains_uuid(state, entry.uuid) {
694                return Err(AddEntityError::DuplicateUuid {
695                    entity_id,
696                    uuid: entry.uuid,
697                });
698            }
699            if require_loaded_chunks
700                && ownership == EntityOwnership::ManagerOwned
701                && !state.chunk_visibility.contains_key(&entry.chunk)
702            {
703                return Err(AddEntityError::ChunkNotLoaded {
704                    entity_id,
705                    chunk: entry.chunk,
706                });
707            }
708        }
709        Ok(())
710    }

782    /// Commits manager indexes after a live entity position change.
783    pub fn commit_move(
784        &self,
785        entity_id: i32,
786        new_pos: DVec3,
787    ) -> Result<EntityMoveUpdate, EntityMoveError> {
788        let mut state = self.state.write();
789        let Some(current) = state.live_by_id.get(&entity_id) else {
790            return Err(EntityMoveError::NotLive { entity_id });
791        };
792
793        let new_section = SectionPos::from_entity_pos(new_pos);
794        let new_chunk = ChunkPos::new(new_section.x(), new_section.z());
795        if current.ownership == EntityOwnership::ManagerOwned
796            && !Self::can_move_manager_owned_to_chunk(&state, current, new_chunk)
797        {
798            return Err(EntityMoveError::UnloadedDestination {
799                entity_id,
800                chunk: new_chunk,
801            });
802        }
803
804        let old_section = current.section;
805        let old_chunk = current.chunk;
806        let old_accessible = Self::is_accessible(&state, current);
807        let new_accessible = Self::is_accessible_at(&state, current.ownership, new_chunk);
808        let old_visibility =
809            Self::lifecycle_visibility_for(current, Self::chunk_visibility(&state, old_chunk));
810        let new_visibility =
811            Self::lifecycle_visibility_for(current, Self::chunk_visibility(&state, new_chunk));
812        let old_ticking = old_visibility.is_ticking();
813        let new_ticking = new_visibility.is_ticking();
814        let entity = current.entity.clone();
815        if old_section == new_section && old_chunk == new_chunk {
816            return Ok(EntityMoveUpdate {
817                entity_id,
818                old_section,
819                new_section,
820                old_chunk,
821                new_chunk,
822                old_accessible,
823                new_accessible,
824                old_ticking,
825                new_ticking,
826            });
827        }
828
829        Self::remove_from_section(&mut state, old_section, entity_id);
830        Self::remove_from_chunk(&mut state, old_chunk, entity_id);
831
832        if let Some(entry) = state.live_by_id.get_mut(&entity_id) {
833            entry.section = new_section;
834            entry.chunk = new_chunk;
835        }
836
837        state
838            .by_section
839            .entry(new_section)
840            .or_default()
841            .insert(entity_id);
842        state
843            .by_chunk
844            .entry(new_chunk)
845            .or_default()
846            .insert(entity_id);
847
848        if old_ticking && !new_ticking {
849            state.tick_list.remove(entity_id);
850        } else if !old_ticking && new_ticking {
851            state.tick_list.add(&entity);
852        }
853
854        Ok(EntityMoveUpdate {
855            entity_id,
856            old_section,
857            new_section,
858            old_chunk,
859            new_chunk,
860            old_accessible,
861            new_accessible,
862            old_ticking,
863            new_ticking,
864        })
865    }

877    fn has_live_loaded_root_vehicle(state: &ManagerState, entity: &SharedEntity) -> bool {
878        let mut visited = FxHashSet::default();
879        visited.insert(entity.id());
880
881        let mut passenger = Arc::clone(entity);
882        let Some(mut vehicle) = passenger.vehicle() else {
883            return false;
884        };
885
886        loop {
887            assert!(
888                visited.insert(vehicle.id()),
889                "cyclic passenger relationship involving entity {}",
890                entity.id()
891            );
892            if vehicle.is_removed() || !vehicle.has_passenger(passenger.as_ref()) {
893                return false;
894            }
895
896            let Some(vehicle_entry) = state.live_by_id.get(&vehicle.id()) else {
897                return false;
898            };
899
900            let Some(next_vehicle) = vehicle.vehicle() else {
901                return match vehicle_entry.ownership {
902                    EntityOwnership::External => true,
903                    EntityOwnership::ManagerOwned => {
904                        state.chunk_visibility.contains_key(&vehicle_entry.chunk)
905                    }
906                };
907            };
908
909            passenger = vehicle;
910            vehicle = next_vehicle;
911        }
912    }

1184    /// Ticks live entities currently in the ticking visibility set.
1185    pub fn tick_entities(&self, _tick_count: i32, runs_normally: bool) -> FxHashSet<ChunkPos> {
1186        let mut dirty_chunks = FxHashSet::default();
1187        let mut ticked_entities = FxHashSet::default();
1188        let tick_candidates = self.ticking_entities_snapshot();
1189        for entity in tick_candidates {
1190            if !self.can_tick_entity_now(entity.id()) {
1191                continue;
1192            }
1193
1194            if entity.is_removed() {
1195                continue;
1196            }
1197

1212                continue;
1213            }
1214
1215            if !ticked_entities.insert(entity.id()) {
1216                continue;
1217            }
1218
1219            self.tick_non_passenger(&entity, &mut ticked_entities, &mut dirty_chunks);
1220        }

1287    fn apply_chunk_visibility_change(
1288        state: &mut ManagerState,
1289        chunk: ChunkPos,
1290        previous: EntityVisibility,
1291        new: EntityVisibility,
1292    ) -> EntityLifecycleChanges {
1293        let entity_ids = state
1294            .by_chunk
1295            .get(&chunk)
1296            .map(|ids| ids.iter().copied().collect::<Vec<_>>())
1297            .unwrap_or_default();
1298        let mut lifecycle = EntityLifecycleChanges::default();
1299
1300        for entity_id in entity_ids {
1301            let Some(entry) = state.live_by_id.get(&entity_id) else {
1302                continue;
1303            };
1304            if entry.ownership != EntityOwnership::ManagerOwned {
1305                continue;
1306            }
1307
1308            let old_visibility = Self::lifecycle_visibility_for(entry, previous);
1309            let new_visibility = Self::lifecycle_visibility_for(entry, new);
1310            if old_visibility == new_visibility {
1311                continue;
1312            }
1313
1314            let entity = entry.entity.clone();
1315            if old_visibility.is_ticking()
1316                && !new_visibility.is_ticking()
1317                && state.tick_list.remove(entity_id).is_some()
1318            {
1319                lifecycle.ticking_stopped.push(entity.clone());
1320            }
1321
1322            if old_visibility.is_accessible() && !new_visibility.is_accessible() {
1323                lifecycle.tracking_stopped.push(entity.clone());
1324            } else if !old_visibility.is_accessible() && new_visibility.is_accessible() {
1325                lifecycle.tracking_started.push(entity.clone());
1326            }
1327
1328            if !old_visibility.is_ticking()
1329                && new_visibility.is_ticking()
1330                && state.tick_list.add(&entity)
1331            {
1332                lifecycle.ticking_started.push(entity);
1333            }
1334        }
1335
1336        lifecycle
1337    }
