22/// Factory function type for creating entities.
23///
24/// Takes the entity type, entity ID, spawn position, and world reference.
25/// Returns a new entity instance. The entity ID should be obtained from
26/// `next_entity_id()`.
27pub type EntityFactory = fn(EntityTypeRef, i32, DVec3, Weak<World>) -> SharedEntity;
28
29/// Factory function type for loading entities from disk.
30///
31/// Takes the entity type and all base entity fields needed for reconstruction.
32pub type EntityLoadFactory = fn(EntityTypeRef, EntityBaseLoad) -> SharedEntity;
33
34/// Entity load request before the registry assigns a runtime ID.
35pub struct EntityLoadRequest {
36    /// Entity type to instantiate.
37    pub entity_type: EntityTypeRef,
38    /// Restored entity position.
39    pub position: DVec3,
40    /// Persisted entity UUID.
41    pub uuid: Uuid,
42    /// Restored velocity.
43    pub velocity: DVec3,
44    /// Restored yaw and pitch.
45    pub rotation: (f32, f32),
46    /// Restored accumulated fall distance.
47    pub fall_distance: f64,
48    /// Restored vanilla fire/freeze state.
49    pub fire_freeze: EntityFireFreezeState,
50    /// Restored ground-contact flag.
51    pub on_ground: bool,
52    /// Restored shared vanilla save data.
53    pub save_data: EntityBaseSaveData,
54    /// World reference for the loaded entity.
55    pub world: Weak<World>,
56}
57
58impl EntityLoadRequest {
59    fn into_base_load(self) -> (EntityTypeRef, EntityBaseLoad) {
60        (
61            self.entity_type,
62            EntityBaseLoad {
63                id: next_entity_id(),
64                position: self.position,
65                uuid: self.uuid,
66                velocity: self.velocity,
67                rotation: self.rotation,
68                fall_distance: self.fall_distance,
69                fire_freeze: self.fire_freeze,
70                on_ground: self.on_ground,
71                save_data: self.save_data,
72                world: self.world,
73            },
74        )
75    }
76}

78/// Registry entry for an entity type.
79struct EntityEntry {
80    /// Factory function to create new instances.
81    factory: Option<EntityFactory>,
82    /// Factory function to load instances from disk.
83    load_factory: Option<EntityLoadFactory>,
84}
85
86/// Registry for entity factories.
87///
88/// Maps `EntityType` to factory functions that can create entity instances.
89/// This is used when loading entities from disk or when entities are spawned.
90pub struct EntityRegistry {
91    entries: Vec<EntityEntry>,
92}
93
94impl EntityRegistry {
95    /// Creates a new empty registry with entries for all entity types.
96    #[must_use]
97    pub fn new() -> Self {
98        let count = REGISTRY.entity_types.len();
99        let entries = (0..count)
100            .map(|_| EntityEntry {
101                factory: None,
102                load_factory: None,
103            })
104            .collect();
105
106        Self { entries }
107    }
108
109    /// Registers a factory function for an entity type.
110    ///
111    /// # Panics
112    ///
113    /// Panics if a factory is already registered for the entity type.
114    pub fn register(&mut self, entity_type: EntityTypeRef, factory: EntityFactory) {
115        let id = entity_type.id();
116        assert!(
117            self.entries[id].factory.is_none(),
118            "entity factory for {} is already registered",
119            entity_type.key
120        );
121        self.entries[id].factory = Some(factory);
122    }
123
124    /// Registers a load factory function for an entity type.
125    ///
126    /// The load factory is used when loading entities from disk.
127    ///
128    /// # Panics
129    ///
130    /// Panics if a load factory is already registered for the entity type.
131    pub fn register_load(&mut self, entity_type: EntityTypeRef, factory: EntityLoadFactory) {
132        let id = entity_type.id();
133        assert!(
134            self.entries[id].load_factory.is_none(),
135            "entity load factory for {} is already registered",
136            entity_type.key
137        );
138        self.entries[id].load_factory = Some(factory);
139    }
140
141    /// Creates a new entity instance.
142    ///
143    /// Returns `None` if no factory is registered for the given type.
144    #[must_use]
145    pub fn create(
146        &self,
147        entity_type: EntityTypeRef,
148        entity_id: i32,
149        pos: DVec3,
150        world: Weak<World>,
151    ) -> Option<SharedEntity> {
152        let id = entity_type.id();
153        self.entries
154            .get(id)?
155            .factory
156            .map(|f| f(entity_type, entity_id, pos, world))
157    }
158
159    /// Creates an entity from persisted data and loads its type-specific NBT.
160    ///
161    /// Returns `None` if no load factory is registered for the entity type.
162    #[must_use]
163    pub fn create_and_load(
164        &self,
165        request: EntityLoadRequest,
166        nbt: &BorrowedNbtCompound<'_>,
167    ) -> Option<SharedEntity> {
168        let (entity_type, load) = request.into_base_load();
169        let id = entity_type.id();
170        let load_factory = self.entries.get(id)?.load_factory?;
171
172        let entity = load_factory(entity_type, load);
173        let nbt: BorrowedNbtCompoundView<'_, '_> = nbt.into();
174        entity.load_additional(nbt);
175        entity.sync_base_entity_data();
176        Some(entity)
177    }
178
179    /// Creates an entity from persisted data, falling back to raw NBT preservation.
180    #[must_use]
181    pub fn create_and_load_or_raw(
182        &self,
183        request: EntityLoadRequest,
184        nbt: &BorrowedNbtCompound<'_>,
185    ) -> SharedEntity {
186        let (entity_type, load) = request.into_base_load();
187        let id = entity_type.id();
188        if let Some(load_factory) = self.entries.get(id).and_then(|entry| entry.load_factory) {
189            let entity = load_factory(entity_type, load);
190            let nbt: BorrowedNbtCompoundView<'_, '_> = nbt.into();
191            entity.load_additional(nbt);
192            entity.sync_base_entity_data();
193            return entity;
194        }
195
196        let entity: SharedEntity = Arc::new(RawEntity::from_saved(load, entity_type));
197        let nbt: BorrowedNbtCompoundView<'_, '_> = nbt.into();
198        entity.load_additional(nbt);
199        entity
200    }
201
202    /// Returns whether a factory is registered for the given type.
203    #[must_use]
204    pub fn has_factory(&self, entity_type: EntityTypeRef) -> bool {
205        let id = entity_type.id();
206        self.entries.get(id).is_some_and(|e| e.factory.is_some())
207    }
208
209    /// Returns whether a load factory is registered for the given type.
210    #[must_use]
211    pub fn has_load_factory(&self, entity_type: EntityTypeRef) -> bool {
212        let id = entity_type.id();
213        self.entries
214            .get(id)
215            .is_some_and(|e| e.load_factory.is_some())
216    }
217}

225/// Wrapper for the global entity registry that implements `Deref`.
226pub struct EntityRegistryLock(OnceLock<EntityRegistry>);
227
228impl Deref for EntityRegistryLock {
229    type Target = EntityRegistry;
230
231    fn deref(&self) -> &Self::Target {
232        self.0.get().expect("Entity registry not initialized")
233    }
234}
235
236impl EntityRegistryLock {
237    /// Sets the registry. Returns `Err` if already initialized.
238    pub fn set(&self, registry: EntityRegistry) -> Result<(), EntityRegistry> {
239        self.0.set(registry)
240    }
241
242    /// Returns the initialized registry, if entity factories have been installed.
243    #[must_use]
244    pub fn get(&self) -> Option<&EntityRegistry> {
245        self.0.get()
246    }
247}
248
249/// Global entity registry.
250///
251/// Access via deref: `ENTITIES.create(type, entity_id, pos)`
252pub static ENTITIES: EntityRegistryLock = EntityRegistryLock(OnceLock::new());
253
254/// Initializes the global entity registry.
255///
256/// This should be called once after the main registry is frozen.
257///
258/// # Panics
259///
260/// Panics if called more than once.
261pub fn init_entities() {
262    let mut registry = EntityRegistry::new();
263    register_entity_factories(&mut registry);
264
265    assert!(
266        ENTITIES.set(registry).is_ok(),
267        "Entity registry already initialized"
268    );
269}

297    #[test]
298    fn create_and_load_or_raw_preserves_unregistered_entity_data() {
299        init_test_registry();
300        let registry = EntityRegistry::new();
301        let mut nbt = NbtCompound::new();
302        nbt.insert("SteelRawMarker", "raw");
303        let mut bytes = Vec::new();
304        nbt.write(&mut bytes);
305        let borrowed =
306            read_borrowed_compound(&mut Cursor::new(&bytes)).expect("test nbt should reborrow");
307
308        let entity = registry.create_and_load_or_raw(
309            EntityLoadRequest {
310                entity_type: &vanilla_entities::VILLAGER,
311                position: DVec3::new(1.0, 2.0, 3.0),
312                uuid: Uuid::from_u128(1),
313                velocity: DVec3::new(0.1, 0.0, 0.2),
314                rotation: (45.0, 10.0),
315                fall_distance: 2.25,
316                fire_freeze: EntityFireFreezeState::new(),
317                on_ground: true,
318                save_data: EntityBaseSaveData {
319                    no_gravity: true,
320                    invulnerable: true,
321                    ..EntityBaseSaveData::new()
322                },
323                world: Weak::new(),
324            },
325            &borrowed,
326        );
327
328        assert_eq!(&entity.entity_type().key, &vanilla_entities::VILLAGER.key);
329        assert_eq!(entity.position(), DVec3::new(1.0, 2.0, 3.0));
330        assert_eq!(entity.velocity(), DVec3::new(0.1, 0.0, 0.2));
331        assert_eq!(entity.rotation(), (45.0, 10.0));
332        assert!((entity.fall_distance() - 2.25).abs() <= f64::EPSILON);
333        assert!(entity.on_ground());
334        assert!(entity.is_no_gravity());
335        assert!(entity.is_invulnerable());
336
337        let mut saved = NbtCompound::new();
338        entity.save_additional(&mut saved);
339        assert_eq!(
340            saved.string("SteelRawMarker").map(ToString::to_string),
341            Some("raw".to_owned())
342        );
343    }
