79#[derive(Debug, Clone, Copy, PartialEq)]
80struct DropChances {
81    by_equipment: [f32; EquipmentSlot::ALL.len()],
82}
83
84impl DropChances {
85    const DEFAULT: Self = Self {
86        by_equipment: [DEFAULT_EQUIPMENT_DROP_CHANCE; EquipmentSlot::ALL.len()],
87    };
88
89    #[must_use]
90    const fn by_equipment(self, slot: EquipmentSlot) -> f32 {
91        self.by_equipment[slot.index()]
92    }
93
94    const fn set_guaranteed_drop(&mut self, slot: EquipmentSlot) {
95        self.by_equipment[slot.index()] = PRESERVE_ITEM_DROP_CHANCE;
96    }
97
98    fn set_equipment_chance(&mut self, slot: EquipmentSlot, chance: f32) -> bool {
99        if chance < 0.0 {
100            return false;
101        }
102
103        self.by_equipment[slot.index()] = chance;
104        true
105    }
106
107    #[must_use]
108    fn is_preserved(self, slot: EquipmentSlot) -> bool {
109        self.by_equipment(slot) > PRESERVE_ITEM_DROP_CHANCE_THRESHOLD
110    }
111
112    fn save(self, nbt: &mut NbtCompound) {
113        if self == Self::DEFAULT {
114            return;
115        }
116
117        let mut drop_chances = NbtCompound::new();
118        for slot in EquipmentSlot::ALL {
119            let chance = self.by_equipment(slot);
120            if chance.to_bits() != DEFAULT_EQUIPMENT_DROP_CHANCE.to_bits() {
121                drop_chances.insert(slot.name(), chance);
122            }
123        }
124
125        nbt.insert("drop_chances", NbtTag::Compound(drop_chances));
126    }
127
128    fn load(nbt: BorrowedNbtCompoundView<'_, '_>) -> Self {
129        let Some(drop_chances) = nbt.compound("drop_chances") else {
130            return Self::DEFAULT;
131        };
132
133        let mut loaded = Self::DEFAULT;
134        for slot in EquipmentSlot::ALL {
135            let Some(chance) = drop_chances.float(slot.name()) else {
136                continue;
137            };
138            if !loaded.set_equipment_chance(slot, chance) {
139                return Self::DEFAULT;
140            }
141        }
142
143        loaded
144    }
145}

273            .map(|position| {
274                Self::from_delayed_attachment(LeashAttachment::FenceKnot(BlockPos::new(
275                    position[0],
276                    position[1],
277                    position[2],
278                )))
279            })

309fn rotate_y(vector: DVec3, radians: f32) -> DVec3 {
310    let cos = f64::from(radians.cos());
311    let sin = f64::from(radians.sin());
312    DVec3::new(
313        vector.x * cos + vector.z * sin,
314        vector.y,
315        vector.z * cos - vector.x * sin,
316    )
317}

323fn compute_elastic_interaction(
324    entity: &dyn Entity,
325    holder: &dyn Entity,
326    slack_distance: f64,
327) -> Option<LeashWrench> {
328    let entity_y_rot = entity.rotation().0 * PI / 180.0;
329    let entity_attach_vector = rotate_y(
330        ENTITY_LEASH_ATTACHMENT_POINT * leash_dimensions(entity),
331        -entity_y_rot,
332    );
333    let entity_attach_pos = entity.position() + entity_attach_vector;
334
335    let holder_y_rot = holder.rotation().0 * PI / 180.0;
336    let holder_attach_vector = rotate_y(
337        LEASHER_ATTACHMENT_POINT * leash_dimensions(holder),
338        -holder_y_rot,
339    );
340    let holder_attach_pos = holder.position() + holder_attach_vector;
341
342    compute_dampened_spring_interaction(
343        holder_attach_pos,
344        entity_attach_pos,
345        slack_distance,
346        leash_holder_movement(entity),
347        entity_attach_vector,
348    )
349}

351fn compute_dampened_spring_interaction(
352    pivot_point: DVec3,
353    object_position: DVec3,
354    spring_slack: f64,
355    object_motion: DVec3,
356    lever_arm: DVec3,
357) -> Option<LeashWrench> {
358    let distance = object_position.distance(pivot_point);
359    if distance < spring_slack {
360        return None;
361    }
362
363    let mut displacement = (pivot_point - object_position).normalize() * (distance - spring_slack);
364    let torque = torque_from_force(lever_arm, displacement);
365    if object_motion.dot(displacement) >= 0.0 {
366        displacement *= 1.0 - LEASH_SPRING_DAMPENING;
367    }
368
369    Some(LeashWrench::new(displacement, torque))
370}

372fn torque_from_force(lever_arm: DVec3, force: DVec3) -> f64 {
373    lever_arm.z * force.x - lever_arm.x * force.z
374}

652        _spawn_reason: EntitySpawnReason,
653        group_data: Option<SpawnGroupData>,

655        let needs_random_spawn_bonus = !self
656            .attributes()
657            .lock()
658            .has_modifier(vanilla_attributes::FOLLOW_RANGE, &RANDOM_SPAWN_BONUS_ID);
659        let (random_spawn_bonus, left_handed) = {
660            let mut random = world.random().lock();
661            let random_spawn_bonus =
662                needs_random_spawn_bonus.then(|| random.triangle(0.0, RANDOM_SPAWN_BONUS_SCALE));
663            let left_handed = random.next_f32() < LEFT_HANDED_SPAWN_CHANCE;
664            (random_spawn_bonus, left_handed)
665        };
666
667        if let Some(amount) = random_spawn_bonus {
668            self.attributes().lock().add_modifier(
669                vanilla_attributes::FOLLOW_RANGE,
670                AttributeModifier {
671                    id: RANDOM_SPAWN_BONUS_ID,
672                    amount,
673                    operation: AttributeModifierOperation::AddMultipliedBase,
674                },
675                true,
676            );
677        }
678        self.set_left_handed(left_handed);
679        group_data
680    }

770        if self.level().is_none() {
771            return;
772        }
773
774        for slot in EquipmentSlot::ALL {
775            let drop_chance = self.equipment_drop_chance(slot);
776            let preserve = self.is_equipment_drop_preserved(slot);
777            if !can_attempt_equipment_drop(drop_chance, preserve, killed_by_player) {
778                continue;
779            }
780
781            let can_drop_item = {
782                let equipment = self.living_base().equipment().lock();
783                let item_stack = equipment.get_ref(slot);
784                !item_stack.is_empty()
785                    && !item_stack
786                        .has_enchantment_effect(EnchantmentEffectComponent::PreventEquipmentDrop)
787            };
788            if !can_drop_item {
789                continue;
790            }
791
792            // TODO: Apply EquipmentDrops enchantment value effects once damage
793            // sources can resolve their living attacker context.
794            let random_roll = self.base().random().lock().next_f32();
795            if random_roll >= drop_chance {
796                continue;
797            }
798
799            let mut item_stack = {
800                let mut equipment = self.living_base().equipment().lock();
801                let item_stack = equipment.get_ref(slot);
802                if item_stack.is_empty()
803                    || item_stack
804                        .has_enchantment_effect(EnchantmentEffectComponent::PreventEquipmentDrop)
805                {
806                    continue;
807                }
808
809                equipment.take(slot)
810            };
811            if !preserve && item_stack.is_damageable_item() {
812                let max_damage = item_stack.get_max_damage();
813                let damage = {
814                    let mut random = self.base().random().lock();
815                    let inner = random.next_i32_bounded((max_damage - 3).max(1));
816                    max_damage - random.next_i32_bounded(1 + inner)
817                };
818                item_stack.set_damage_value(damage);
819            }
820
821            self.spawn_at_location(item_stack, 0.0);
822        }
823    }

1084    fn tick_leash(&self) {
1085        if let Some(holder) = self.leash_holder() {
1086            if !self.can_interact_with_level() || !holder.can_interact_with_level() {
1087                if let Some(world) = self.level()
1088                    && world.get_game_rule(&ENTITY_DROPS).as_bool() == Some(true)
1089                {
1090                    self.drop_leash();
1091                } else {
1092                    self.remove_leash();
1093                }
1094                return;
1095            }
1096
1097            let distance_to = self.leash_distance_to(holder.as_ref());
1098            self.when_leashed_to(holder.as_ref());
1099            let angular_momentum_before_distance_action = self.leash_angular_momentum();
1100            if distance_to > self.leash_snap_distance() {
1101                if let Some(world) = self.level() {
1102                    world.play_sound_at(
1103                        &sound_events::ITEM_LEAD_BREAK,
1104                        SoundSource::Neutral,
1105                        holder.position(),
1106                        1.0,
1107                        1.0,
1108                        None,
1109                    );
1110                }
1111                self.leash_too_far_behaviour();
1112            } else if distance_to
1113                > self.leash_elastic_distance()
1114                    - f64::from(holder.base().dimensions().width)
1115                    - f64::from(self.base().dimensions().width)
1116                && self.check_elastic_interactions(holder.as_ref())
1117            {
1118                self.on_elastic_leash_pull();
1119            } else {
1120                self.close_range_leash_behaviour(holder.as_ref());
1121            }
1122            if !self.apply_leash_angular_momentum()
1123                && let Some(angular_momentum) = angular_momentum_before_distance_action
1124            {
1125                self.rotate_by_leash_angular_momentum(angular_momentum);
1126            }
1127            return;
1128        }
1129
1130        let Some(attachment) = self.leash_attachment() else {
1131            return;
1132        };
1133
1134        let Some(world) = self.level() else {
1135            return;
1136        };
1137
1138        match attachment {
1139            LeashAttachment::Entity(uuid) => {
1140                if let Some(holder) = world.get_entity_by_uuid(&uuid) {
1141                    let _ = self.set_leashed_to(&holder);
1142                    return;
1143                }
1144
1145                if self.tick_count() > DELAYED_LEASH_DROP_TICKS {
1146                    let _ = self.spawn_at_location(ItemStack::new(&vanilla_items::ITEMS.lead), 0.0);
1147                    self.remove_leash_state();
1148                }
1149            }
1150            LeashAttachment::FenceKnot(pos) => {
1151                if let Some(holder) = LeashFenceKnotEntity::get_or_create_knot(&world, pos) {
1152                    let _ = self.set_leashed_to(&holder);
1153                    return;
1154                }
1155
1156                if self.tick_count() > DELAYED_LEASH_DROP_TICKS {
1157                    let _ = self.spawn_at_location(ItemStack::new(&vanilla_items::ITEMS.lead), 0.0);
1158                    self.remove_leash_state();
1159                }
1160            }
1161        }
1162    }

1228    fn check_mob_despawn(&self) {
1229        if self
1230            .level()
1231            .is_some_and(|world| world.difficulty() == Difficulty::Peaceful)
1232            && !self.entity_type().allowed_in_peaceful
1233        {
1234            self.set_removed(RemovalReason::Discarded);
1235            return;
1236        }
1237
1238        if self.is_persistence_required() || self.requires_custom_persistence() {
1239            self.set_no_action_time(0);
1240            return;
1241        }
1242
1243        let Some(nearest_player_dist_sqr) = self.nearest_player_distance_sqr() else {
1244            return;
1245        };
1246
1247        let mob_category = self.entity_type().mob_category;
1248        let despawn_distance = mob_category.despawn_distance();
1249        let despawn_distance_sqr = despawn_distance * despawn_distance;
1250        if nearest_player_dist_sqr > f64::from(despawn_distance_sqr)
1251            && self.remove_when_far_away(nearest_player_dist_sqr)
1252        {
1253            self.set_removed(RemovalReason::Discarded);
1254            return;
1255        }
1256
1257        let no_despawn_distance = mob_category.no_despawn_distance();
1258        let no_despawn_distance_sqr = no_despawn_distance * no_despawn_distance;
1259        if self.no_action_time() > 600
1260            && nearest_player_dist_sqr > f64::from(no_despawn_distance_sqr)
1261            && self.remove_when_far_away(nearest_player_dist_sqr)
1262        {
1263            let should_discard = {
1264                let mut random = self.base().random().lock();
1265                random.next_i32_bounded(800) == 0
1266            };
1267            if should_discard {
1268                self.set_removed(RemovalReason::Discarded);

1656    fn tick_strafe_control(&self, forward: f32, right: f32) {
1657        let movement_speed = self
1658            .attributes()
1659            .lock()
1660            .required_value(vanilla_attributes::MOVEMENT_SPEED) as f32;
1661        let speed = movement_speed * 0.25;
1662        let mut strafe_forward = forward;
1663        let mut strafe_right = right;
1664
1665        let mut distance = strafe_forward
1666            .mul_add(strafe_forward, strafe_right * strafe_right)
1667            .sqrt();
1668        if distance < 1.0 {
1669            distance = 1.0;
1670        }
1671        distance = speed / distance;
1672        let xa = strafe_forward * distance;
1673        let za = strafe_right * distance;
1674        let yaw_radians = self.rotation().0 * PI / 180.0;
1675        let sin = yaw_radians.sin();
1676        let cos = yaw_radians.cos();
1677        let dx = xa.mul_add(cos, -(za * sin));
1678        let dz = za.mul_add(cos, xa * sin);
1679        if !self.is_strafe_walkable(dx, dz) {
1680            strafe_forward = 1.0;
1681            strafe_right = 0.0;
1682        }
1683
1684        self.set_speed(speed);
1685        self.set_travel_input(LivingTravelInput::new(strafe_right, 0.0, strafe_forward));
1686        self.mob_base().controls().lock().move_control.set_wait();
1687    }
