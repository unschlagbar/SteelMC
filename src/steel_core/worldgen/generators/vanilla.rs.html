41/// A chunk generator for vanilla (normal) world generation.
42///
43/// Matches vanilla's `NoiseBasedChunkGenerator`. The biome source is pluggable
44/// per-dimension — overworld, nether, and end each provide a different
45/// [`BiomeSourceKind`] variant.
46///
47/// Generic over `N: DimensionNoises` to support different dimensions with
48/// their own transpiled density functions and noise settings.
49pub struct VanillaGenerator<N: DimensionNoises> {
50    /// Biome source for this dimension. Determines biomes at each quart position.
51    biome_source: BiomeSourceKind,
52    /// Representative biome for source-carver lookup when every possible
53    /// biome from `biome_source` has the same carver list.
54    ///
55    /// Vanilla still samples each source biome in `apply_carvers`; Steel skips
56    /// that sampling only when the source's full possible-biome set proves the
57    /// carver list is uniform. If future biome sources can produce mixed
58    /// carver lists this remains `None` and the vanilla per-source lookup is
59    /// used.
60    uniform_carver_biome: Option<BiomeRef>,
61    /// Noise generators for this dimension's density functions.
62    /// Boxed because noise structs can be large.
63    noises: Box<N>,
64    /// Seed positional splitter for per-chunk construction of aquifers.
65    splitter: RandomSplitter,
66    /// Ore vein generator for replacing stone with ore blocks.
67    ore_veinifier: Option<OreVeinifier>,
68    /// Surface system for biome-specific block replacement.
69    surface_system: SurfaceSystem,
70    /// Which vanilla surface extension biomes this source can produce.
71    surface_extension_biomes: SurfaceExtensionBiomes,
72    /// Block state ID for the default block, cached at construction time.
73    default_block_id: BlockStateId,
74    /// Obfuscated seed for `BiomeManager` biome zoom fuzzing.
75    biome_zoom_seed: i64,
76    /// World seed as i64 (matching Java's long), used for structures and carver seeding.
77    seed: i64,
78    /// Shared structure placement/selection engine.
79    structure_generator: StructureGenerator,
80    /// Cached placed-feature order for biome decoration.
81    feature_runner: FeatureDecorationRunner,
82    _phantom: PhantomData<N>,
83}

105impl<N: DimensionNoises> VanillaGenerator<N> {
106    /// Creates a new `VanillaGenerator` with the given biome source and seed.
107    ///
108    /// # Panics
109    /// Panics if SHA-256 hash output is shorter than 8 bytes (cannot happen).
110    #[must_use]
111    pub fn new(biome_source: BiomeSourceKind, seed: u64) -> Self {
112        // Nether uses Java's LCG; overworld/end use Xoroshiro.
113        let splitter = if N::Settings::LEGACY_RANDOM_SOURCE {
114            LegacyRandom::from_seed(seed).next_positional()
115        } else {
116            Xoroshiro::from_seed(seed).next_positional()
117        };
118        let noise_params = get_noise_parameters();
119        let noises = N::create(seed, &splitter, &noise_params);
120
121        let ore_veinifier = if N::Settings::ORE_VEINS_ENABLED {
122            Some(OreVeinifier::new(&splitter))
123        } else {
124            None
125        };
126
127        let default_block_id = N::Settings::default_block_id();
128        let surface_system = SurfaceSystem::new(
129            &splitter,
130            &noise_params,
131            N::surface_noise_ids(),
132            N::surface_gradient_ids(),
133            default_block_id,
134            N::Settings::SEA_LEVEL,
135        );
136
137        let biome_zoom_seed = obfuscate_biome_seed(seed as i64);
138
139        let possible_biome_refs = biome_source.possible_biome_refs();
140        let possible_biomes = biome_source.possible_biomes();
141        let surface_extension_biomes = SurfaceExtensionBiomes::from_possible(&possible_biomes);
142        let structure_generator = StructureGenerator::vanilla(seed as i64, &biome_source);
143        let uniform_carver_biome = Self::uniform_carver_biome(&possible_biomes);
144        let feature_runner = FeatureDecorationRunner::new(&possible_biome_refs, &REGISTRY);
145
146        Self {
147            biome_source,
148            uniform_carver_biome,
149            noises: Box::new(noises),
150            splitter,
151            ore_veinifier,
152            surface_system,
153            surface_extension_biomes,
154            default_block_id,
155            biome_zoom_seed,
156            seed: seed as i64,
157            structure_generator,
158            feature_runner,
159            _phantom: PhantomData,
160        }
161    }
162
163    fn uniform_carver_biome(possible_biomes: &FxHashSet<Identifier>) -> Option<BiomeRef> {
164        let mut possible_biomes = possible_biomes.iter();
165        let first_key = possible_biomes.next()?;
166        let first = REGISTRY.biomes.by_key(first_key)?;
167
168        possible_biomes
169            .all(|key| {
170                REGISTRY
171                    .biomes
172                    .by_key(key)
173                    .is_some_and(|biome| biome.carvers == first.carvers)
174            })
175            .then_some(first)
176    }
177}

196    fn create_structures(&self, chunk: &ChunkAccess) {
197        let pos = chunk.pos();
198        let chunk_x = pos.0.x;
199        let chunk_z = pos.0.y;
200
201        let mut sampler = self.biome_source.chunk_sampler();
202        let chunk_min_x = chunk_x * 16;
203        let chunk_min_z = chunk_z * 16;
204
205        let mut height_cache = N::ColumnCache::default();
206        let sea_level = N::Settings::SEA_LEVEL;
207
208        // No eager `init_grid`: most chunks' structures (mineshaft, village)
209        // use their own caches, and the 1–4 column probes of the remainder
210        // hit this cache's lazy single-entry mode cheaply. Eager 5×5 quart
211        // init cost ~36µs per chunk with no payoff.
212        let mut aquifer = LazyAquifer::new(chunk_min_x, chunk_min_z, &self.splitter, &*self.noises);
213        let mut surface_y_cache: Option<i32> = None;
214        let mut height_cache_grid_ready = false;
215        let mut ctx = GenerationContext::<'_, '_, N>::new(
216            self.seed,
217            chunk_x,
218            chunk_z,
219            sea_level,
220            &self.noises,
221            &self.splitter,
222            self.structure_generator.template_pools(),
223            self.structure_generator.templates(),
224            &mut sampler,
225            &mut height_cache,
226            &mut aquifer,
227            &mut surface_y_cache,
228            &mut height_cache_grid_ready,
229        );
230
231        create_structures(&self.structure_generator, chunk, &mut ctx);
232    }

234    fn create_biomes(&self, chunk: &ChunkAccess) {
235        let pos = chunk.pos();
236        let min_y = chunk.min_y();
237        let section_count = chunk.sections().sections.len();
238
239        let chunk_x = pos.0.x;
240        let chunk_z = pos.0.y;
241
242        let mut sampler = self.biome_source.chunk_sampler();
243        // Pre-compute the flat (xz-only) climate-noise grid for this chunk so the
244        // per-cell sampling below does O(1) column lookups instead of recomputing
245        // the flat noise for all 1536 cells (the noise stage's `fill_from_noise`
246        // already does this). Values are bit-identical — same functions, same quart
247        // coordinates — so biome selection is unchanged.
248        sampler.init_grid(chunk_x * 16, chunk_z * 16);
249
250        // Match vanilla's iteration order: Section(Y) → X → Y → Z.
251        // This is critical because the R-tree biome cache (persistent warm-start)
252        // determines tie-breaking for equal-distance entries, and the cache state
253        // depends on the order of biome lookups.
254        for section_index in 0..section_count {
255            let section_y = (min_y / 16) + section_index as i32;
256            let section = &chunk.sections().sections[section_index];
257            let mut section_guard = section.write();
258
259            for local_quart_x in 0..4i32 {
260                let quart_x = chunk_x * 4 + local_quart_x;
261
262                for local_quart_y in 0..4i32 {
263                    let quart_y = section_y * 4 + local_quart_y;
264
265                    for local_quart_z in 0..4i32 {
266                        let quart_z = chunk_z * 4 + local_quart_z;
267
268                        let biome = sampler.sample(quart_x, quart_y, quart_z);
269                        let biome_id = biome.id() as u16;
270
271                        section_guard.biomes.set(
272                            local_quart_x as usize,
273                            local_quart_y as usize,
274                            local_quart_z as usize,
275                            biome_id,
276                        );
277                    }
278                }
279            }
280        }
281
282        chunk.mark_dirty();
283    }

285    fn fill_from_noise(&self, chunk: &ChunkAccess, beardifier: Option<&Beardifier>) {
286        let pos = chunk.pos();
287        let chunk_min_x = pos.0.x * 16;
288        let chunk_min_z = pos.0.y * 16;
289
290        let min_y = N::Settings::MIN_Y;
291        let height = N::Settings::HEIGHT;
292
293        let mut noise_chunk = NoiseChunk::<N>::new(chunk_min_x, chunk_min_z);
294        let noises = &*self.noises;
295
296        let mut column_cache = N::ColumnCache::default();
297        column_cache.init_grid(chunk_min_x, chunk_min_z, noises);
298
299        let default_block_id = self.default_block_id;
300        let ore_veinifier = &self.ore_veinifier;
301        let mut aquifer = Aquifer::<N>::new(
302            chunk_min_x,
303            chunk_min_z,
304            min_y,
305            height,
306            &self.splitter,
307            noises,
308            // Aquifer samples at arbitrary (x,z) outside the chunk, so it needs its own cache
309            column_cache.clone(),
310        );
311
312        // Collect writes per (x,z) column and flush in batch to avoid per-block
313        // write lock acquisition on sections.
314        let mut pending_writes: Vec<(usize, usize, usize, BlockStateId)> = Vec::new();
315        let mut prev_x: usize = usize::MAX;
316        let mut prev_z: usize = usize::MAX;
317        let mut ocean_floor_wg =
318            Heightmap::new(HeightmapType::OceanFloorWg, min_y, N::Settings::HEIGHT);
319        let mut world_surface_wg =
320            Heightmap::new(HeightmapType::WorldSurfaceWg, min_y, N::Settings::HEIGHT);
321
322        noise_chunk.fill(
323            noises,
324            &mut column_cache,
325            beardifier,
326            |local_x, world_y, local_z, density, interpolated, cache| {
327                // Flush when we move to a new column
328                if local_x != prev_x || local_z != prev_z {
329                    if !pending_writes.is_empty() {
330                        chunk.write_block_batch_for_generation(&pending_writes);
331                        pending_writes.clear();
332                    }
333                    prev_x = local_x;
334                    prev_z = local_z;
335                }
336
337                let relative_y = (world_y - min_y) as usize;
338                let world_x = chunk_min_x + local_x as i32;
339                let world_z = chunk_min_z + local_z as i32;
340
341                match aquifer.compute_substance(noises, world_x, world_y, world_z, density) {
342                    AquiferResult::Solid => {
343                        let block = ore_veinifier
344                            .as_ref()
345                            .and_then(|ov| {
346                                ov.compute_interpolated(
347                                    noises,
348                                    cache,
349                                    interpolated,
350                                    world_x,
351                                    world_y,
352                                    world_z,
353                                )
354                            })
355                            .unwrap_or(default_block_id);
356                        pending_writes.push((local_x, relative_y, local_z, block));
357                        ocean_floor_wg.update_for_initial_fill(local_x, world_y, local_z, block);
358                        world_surface_wg.update_for_initial_fill(local_x, world_y, local_z, block);
359                    }
360                    AquiferResult::Fluid(id) => {
361                        pending_writes.push((local_x, relative_y, local_z, id));
362                        ocean_floor_wg.update_for_initial_fill(local_x, world_y, local_z, id);
363                        world_surface_wg.update_for_initial_fill(local_x, world_y, local_z, id);
364                        if aquifer.should_schedule_fluid_update() && id.has_fluid() {
365                            chunk.mark_pos_for_postprocessing(BlockPos::new(
366                                world_x, world_y, world_z,
367                            ));
368                        }
369                    }
370                    AquiferResult::Air => {}
371                }
372            },
373        );
374
375        // Flush remaining writes
376        if !pending_writes.is_empty() {
377            chunk.write_block_batch_for_generation(&pending_writes);
378        }
379
380        let ChunkAccess::Proto(proto) = chunk else {
381            return;
382        };
383        let mut heightmaps = proto.heightmaps.write();
384        heightmaps.replace(ocean_floor_wg);
385        heightmaps.replace(world_surface_wg);
386    }

388    #[expect(clippy::too_many_lines, reason = "splitting would hurt readability")]
389    fn build_surface(&self, chunk: &ChunkAccess, neighbor_biomes: &dyn Fn(IVec3) -> u16) {
390        let min_y = N::Settings::MIN_Y;
391        let pos = chunk.pos();
392        let chunk_min_x = pos.0.x * 16;
393        let chunk_min_z = pos.0.y * 16;
394        let default_block_id = self.default_block_id;
395        let noises = &*self.noises;
396        let surface_rule_block_states = N::surface_rule_block_states();
397        let surface_rule_uses_biome = N::surface_rule_uses_biome();
398        let surface_rule_uses_preliminary_surface = N::surface_rule_uses_preliminary_surface();
399        let surface_rule_uses_surface_secondary = N::surface_rule_uses_surface_secondary();
400        let surface_rule_uses_steep = N::surface_rule_uses_steep();
401        let lazy_surface_rule_biome =
402            surface_rule_uses_biome && surface_rule_uses_preliminary_surface;
403        let surface_needs_min_surface_level =
404            surface_rule_uses_preliminary_surface || self.surface_extension_biomes.frozen_ocean;
405        let surface_needs_biomes =
406            surface_rule_uses_biome || self.surface_extension_biomes.needs_surface_biome();
407        let chunk_quart_x = pos.0.x * 4;
408        let chunk_quart_z = pos.0.y * 4;
409
410        chunk.prime_heightmaps(&[HeightmapType::WorldSurfaceWg]);
411
412        // Pre-compute preliminary surface corners only for rules/extensions that read them.
413        let preliminary_surface_corners = surface_needs_min_surface_level.then(|| {
414            let mut psl_cache = N::ColumnCache::default();
415            let p00 =
416                preliminary_surface_level::<N>(noises, &mut psl_cache, chunk_min_x, chunk_min_z);
417            let p10 = preliminary_surface_level::<N>(
418                noises,
419                &mut psl_cache,
420                chunk_min_x + 16,
421                chunk_min_z,
422            );
423            let p01 = preliminary_surface_level::<N>(
424                noises,
425                &mut psl_cache,
426                chunk_min_x,
427                chunk_min_z + 16,
428            );
429            let p11 = preliminary_surface_level::<N>(
430                noises,
431                &mut psl_cache,
432                chunk_min_x + 16,
433                chunk_min_z + 16,
434            );
435            (p00, p10, p01, p11)
436        });
437
438        let eroded_badlands_id = (*vanilla_biomes::ERODED_BADLANDS).id() as u16;
439        let frozen_ocean_id = (*vanilla_biomes::FROZEN_OCEAN).id() as u16;
440        let deep_frozen_ocean_id = (*vanilla_biomes::DEEP_FROZEN_OCEAN).id() as u16;
441
442        // Pre-extract biome palette values only if surface rules/extensions need them.
443        let biome_data = surface_needs_biomes.then(|| chunk.sections().read_all_biomes());
444        let section_count = chunk.sections().sections.len();
445
446        let mut pending_writes: Vec<(usize, BlockStateId)> = Vec::new();
447        let mut column_buf: Vec<BlockStateId> = Vec::new();
448        let condition_noise_values = N::surface_noise_ids()
449            .iter()
450            .map(|_| Cell::new(0.0))
451            .collect::<Vec<_>>();
452        let condition_noise_initialized = N::surface_noise_ids()
453            .iter()
454            .map(|_| Cell::new(false))
455            .collect::<Vec<_>>();
456        let condition_noise_cache =
457            SurfaceConditionNoiseCache::new(&condition_noise_values, &condition_noise_initialized);
458
459        for local_x in 0..16usize {
460            for local_z in 0..16usize {
461                let block_x = chunk_min_x + local_x as i32;
462                let block_z = chunk_min_z + local_z as i32;
463
464                // Start scanning from one above the highest non-air block
465                let mut start_height =
466                    chunk.height_at(HeightmapType::WorldSurfaceWg, local_x, local_z);
467
468                // Column-local Voronoi cache for fuzzed biome lookups.
469                let mut biome_col = biome_data.as_deref().map(|biome_data| {
470                    FuzzedBiomeColumn::new(
471                        biome_data,
472                        section_count,
473                        self.biome_zoom_seed,
474                        block_x,
475                        block_z,
476                        min_y,
477                        chunk_quart_x,
478                        chunk_quart_z,
479                        neighbor_biomes,
480                    )
481                });
482
483                // Eroded badlands extension: add terracotta pillars above surface
484                let surface_biome_id = if self.surface_extension_biomes.needs_surface_biome() {
485                    biome_col
486                        .as_mut()
487                        .map(|biome_col| biome_col.get(start_height))
488                } else {
489                    None
490                };
491                if self.surface_extension_biomes.eroded_badlands
492                    && surface_biome_id == Some(eroded_badlands_id)
493                {
494                    start_height = self.surface_system.eroded_badlands_extension(
495                        chunk,
496                        local_x,
497                        local_z,
498                        block_x,
499                        block_z,
500                        start_height,
501                        min_y,
502                    );
503                }
504
505                // Snapshot the column once — avoids per-block section locking in the Y scan.
506                // Taken after eroded_badlands_extension which may write blocks above the surface.
507                chunk
508                    .sections()
509                    .read_column_into(local_x, local_z, &mut column_buf);
510
511                // Surface depth for this column
512                let surface_depth = self.surface_system.get_surface_depth(block_x, block_z);
513
514                let surface_secondary = if surface_rule_uses_surface_secondary {
515                    self.surface_system.get_surface_secondary(block_x, block_z)
516                } else {
517                    0.0
518                };
519                condition_noise_cache.reset();
520
521                let min_surface_level =
522                    if let Some((p00, p10, p01, p11)) = preliminary_surface_corners {
523                        // Vanilla: (float)(blockX & 15) / 16.0F — exact for 0-15.
524                        let t_x = f64::from(local_x as u8) / 16.0;
525                        let t_z = f64::from(local_z as u8) / 16.0;
526                        let interp = lerp2(
527                            t_x,
528                            t_z,
529                            f64::from(p00),
530                            f64::from(p10),
531                            f64::from(p01),
532                            f64::from(p11),
533                        );
534                        interp.floor() as i32 + surface_depth - 8
535                    } else {
536                        0
537                    };
538
539                // Steep condition: vanilla only checks south >= north + 4 and
540                // west >= east + 4 (asymmetric, not absolute difference).
541                let steep = surface_rule_uses_steep && {
542                    let z_north = local_z.saturating_sub(1);
543                    let z_south = (local_z + 1).min(15);
544                    let h_north =
545                        chunk.height_at(HeightmapType::WorldSurfaceWg, local_x, z_north) - 1;
546                    let h_south =
547                        chunk.height_at(HeightmapType::WorldSurfaceWg, local_x, z_south) - 1;
548                    if h_south >= h_north + 4 {
549                        true
550                    } else {
551                        let x_west = local_x.saturating_sub(1);
552                        let x_east = (local_x + 1).min(15);
553                        let h_west =
554                            chunk.height_at(HeightmapType::WorldSurfaceWg, x_west, local_z) - 1;
555                        let h_east =
556                            chunk.height_at(HeightmapType::WorldSurfaceWg, x_east, local_z) - 1;
557                        h_west >= h_east + 4
558                    }
559                };
560
561                let mut stone_depth_above: i32 = 0;
562                let mut water_height: i32 = i32::MIN;
563                let mut next_ceiling_stone_y: i32 = i32::MAX;
564                pending_writes.clear();
565
566                for y in (min_y..=start_height).rev() {
567                    let relative_y = (y - min_y) as usize;
568                    let state = column_buf[relative_y];
569
570                    if state.is_air() {
571                        stone_depth_above = 0;
572                        water_height = i32::MIN;
573                        continue;
574                    }
575
576                    if state.get_block().config.liquid {
577                        if water_height == i32::MIN {
578                            water_height = y + 1;
579                        }
580                        continue;
581                    }
582
583                    // Solid block — scan for stone_depth_below (lookahead)
584                    if next_ceiling_stone_y >= y {
585                        next_ceiling_stone_y = i32::MIN;
586                        for la_y in (min_y - 1..y).rev() {
587                            if la_y < min_y {
588                                next_ceiling_stone_y = la_y + 1;
589                                break;
590                            }
591                            let la_rel = (la_y - min_y) as usize;
592                            let la_state = column_buf[la_rel];
593                            // isStone = !isAir && !isLiquid
594                            if la_state.is_air() || la_state.get_block().config.liquid {
595                                next_ceiling_stone_y = la_y + 1;
596                                break;
597                            }
598                        }
599                    }
600
601                    stone_depth_above += 1;
602                    let stone_depth_below = y - next_ceiling_stone_y + 1;
603
604                    // Only apply surface rules to the default block
605                    if state == default_block_id {
606                        let eager_biome_id = if surface_rule_uses_biome && !lazy_surface_rule_biome
607                        {
608                            biome_col.as_mut().map(|biome_col| biome_col.get(y))
609                        } else {
610                            None
611                        };
612                        let biome_provider = if lazy_surface_rule_biome {
613                            biome_col
614                                .as_mut()
615                                .map(|biome_col| biome_col as &mut dyn SurfaceBiomeProvider)
616                        } else {
617                            None
618                        };
619
620                        let mut ctx = SurfaceRuleContext::new(
621                            block_x,
622                            block_z,
623                            surface_depth,
624                            surface_secondary,
625                            min_surface_level,
626                            steep,
627                            y,
628                            stone_depth_above,
629                            stone_depth_below,
630                            water_height,
631                            eager_biome_id,
632                            biome_provider,
633                            &self.surface_system,
634                            &condition_noise_cache,
635                            surface_rule_block_states,
636                        );
637
638                        let rule_result = N::try_apply_surface_rule(&mut ctx);
639
640                        if let Some(new_block) = rule_result {
641                            pending_writes.push((relative_y, new_block));
642                        }
643                    }
644                }
645
646                // Flush batched writes — holds each section's write guard once
647                if !pending_writes.is_empty() {
648                    chunk.write_column_blocks_for_generation(local_x, local_z, &pending_writes);
649                    for &(relative_y, state) in &pending_writes {
650                        column_buf[relative_y] = state;
651                    }
652                    chunk.update_heightmaps_after_direct_column_writes(
653                        local_x,
654                        local_z,
655                        &pending_writes,
656                    );
657                    chunk.mark_dirty();
658                }
659
660                // Frozen ocean iceberg extension: add packed ice and snow
661                if self.surface_extension_biomes.frozen_ocean
662                    && let Some(surface_biome_id) = surface_biome_id
663                        .filter(|id| *id == frozen_ocean_id || *id == deep_frozen_ocean_id)
664                {
665                    pending_writes.clear();
666                    self.surface_system.collect_frozen_ocean_extension_writes(
667                        surface_biome_id,
668                        block_x,
669                        block_z,
670                        start_height,
671                        min_surface_level,
672                        min_y,
673                        &column_buf,
674                        &mut pending_writes,
675                    );
676                    if !pending_writes.is_empty() {
677                        chunk.write_column_blocks_for_generation(local_x, local_z, &pending_writes);
678                        chunk.update_heightmaps_after_direct_column_writes(
679                            local_x,
680                            local_z,
681                            &pending_writes,
682                        );
683                        chunk.mark_dirty();
684                    }
685                }
686            }
687        }
688    }

690    #[expect(clippy::too_many_lines, reason = "matches vanilla carver setup flow")]
691    fn apply_carvers(&self, chunk: &ChunkAccess) {
692        // Carvers only run on proto chunks.
693        let ChunkAccess::Proto(proto) = chunk else {
694            return;
695        };
696
697        if self
698            .uniform_carver_biome
699            .is_some_and(|biome| biome.carvers.is_empty())
700        {
701            return;
702        }
703
704        chunk.prime_heightmaps(&[HeightmapType::WorldSurfaceWg]);
705
706        let pos = chunk.pos();
707        let chunk_min_x = pos.0.x * 16;
708        let chunk_min_z = pos.0.y * 16;
709        let min_y = N::Settings::MIN_Y;
710        let height = N::Settings::HEIGHT;
711        let noises = &*self.noises;
712
713        // Fresh aquifer (vanilla caches NoiseChunk across stages; see TODO on
714        // ProtoChunk::carving_mask for why we rebuild instead).
715        let mut column_cache = N::ColumnCache::default();
716        if N::Settings::AQUIFERS_ENABLED {
717            column_cache.init_grid(chunk_min_x, chunk_min_z, noises);
718        }
719        let aquifer = Aquifer::<N>::new(
720            chunk_min_x,
721            chunk_min_z,
722            min_y,
723            height,
724            &self.splitter,
725            noises,
726            column_cache,
727        );
728
729        // Preliminary surface level at the chunk's 4 corners — used by
730        // top_material min_surface_level interpolation.
731        let mut psl_cache = N::ColumnCache::default();
732        let psl_corners = PreliminarySurfaceCorners {
733            nw: preliminary_surface_level::<N>(noises, &mut psl_cache, chunk_min_x, chunk_min_z),
734            ne: preliminary_surface_level::<N>(
735                noises,
736                &mut psl_cache,
737                chunk_min_x + 16,
738                chunk_min_z,
739            ),
740            sw: preliminary_surface_level::<N>(
741                noises,
742                &mut psl_cache,
743                chunk_min_x,
744                chunk_min_z + 16,
745            ),
746            se: preliminary_surface_level::<N>(
747                noises,
748                &mut psl_cache,
749                chunk_min_x + 16,
750                chunk_min_z + 16,
751            ),
752        };
753
754        let mut ctx = CarvingContext {
755            min_y,
756            gen_depth: height,
757            surface_system: &self.surface_system,
758            aquifer,
759            default_block_id: self.default_block_id,
760            psl_corners,
761            chunk_min_x,
762            chunk_min_z,
763        };
764
765        let ids = CarverBlockIds::load();
766
767        // Pre-fetch the 17×17 source-chunk carver lists. Done up front so we
768        // can later close over `biome_sampler` mutably inside `biome_getter`.
769        // Vanilla samples every source biome here; when this generator's full
770        // possible-biome set has a uniform carver list, the representative
771        // biome gives the same carver keys without 289 climate lookups.
772        let mut biome_sampler = self.biome_source.chunk_sampler();
773        let mut source_biomes: SmallVec<[SourceChunk; CARVER_SOURCE_CHUNK_COUNT]> = SmallVec::new();
774        for dx in -8i32..=8 {
775            for dz in -8i32..=8 {
776                let sx = pos.0.x + dx;
777                let sz = pos.0.y + dz;
778                let biome = if let Some(biome) = self.uniform_carver_biome {
779                    biome
780                } else {
781                    let qx = (sx * 16) >> 2;
782                    let qz = (sz * 16) >> 2;
783                    biome_sampler.sample(qx, 0, qz)
784                };
785                source_biomes.push(SourceChunk {
786                    pos: ChunkPos::new(sx, sz),
787                    biome,
788                });
789            }
790        }
791
792        // Grab (and lazily create) the carving mask on the proto chunk.
793        let mut mask_guard = proto.get_or_create_carving_mask();
794        let mask = &mut *mask_guard;
795
796        // `WorldgenRandom(LegacyRandomSource(generateUniqueSeed()))` — initial
797        // seed is irrelevant; every carver overwrites it via
798        // `set_large_feature_seed` before its probability check.
799        let mut random = LegacyRandom::from_seed(0);
800        let seed_i64 = self.seed;
801
802        let biome_zoom_seed = self.biome_zoom_seed;
803        // BiomeManager-fuzzed lookup — matches vanilla's `BiomeManager.getBiome`
804        // used by the carver's top-material path. An unfuzzed quart lookup
805        // would mismatch vanilla at quart-cell boundaries.
806        let mut biome_getter = |pos: BlockPos| -> u16 {
807            fuzzed_biome_at_block(biome_zoom_seed, pos, |q_pos| {
808                biome_sampler.sample(q_pos.x, q_pos.y, q_pos.z).id() as u16
809            })
810        };
811
812        let mut run = CarveRun {
813            ctx: &mut ctx,
814            noises,
815            chunk,
816            chunk_min_x,
817            chunk_min_z,
818            biome_getter: &mut biome_getter,
819            mask,
820            ids,
821        };
822
823        run.run_all(&source_biomes, seed_i64, &mut random);
824    }
825
826    fn create_worldgen_region_random(&self, _world_seed: i64, center: ChunkPos) -> RandomSource {
827        worldgen_region_random_from_splitter(&self.splitter, center)
828    }
829
830    fn apply_biome_decorations(&self, region: &mut WorldGenRegion<'_>) {
831        self.feature_runner
832            .decorate(region, &REGISTRY, self.seed, self.biome_zoom_seed);
833    }

836impl<N, F> CarveRun<'_, '_, N, F>
837where
838    N: DimensionNoises,
839    F: FnMut(BlockPos) -> u16,
840{
841    /// Drive the 17×17 source-chunk carver loop. Each carver in each source
842    /// biome is seeded via `set_large_feature_seed`, probability-checked,
843    /// then dispatched to the appropriate `carve_*` method.
844    fn run_all(&mut self, source_biomes: &[SourceChunk], seed_i64: i64, random: &mut LegacyRandom) {
845        for source in source_biomes {
846            for (index, carver_key) in source.biome.carvers.iter().enumerate() {
847                let Some(carver) = REGISTRY.configured_carvers.by_key(carver_key) else {
848                    panic!(
849                        "biome {} references unknown configured carver {}",
850                        source.biome.key, carver_key
851                    );
852                };
853                let index_i64 = index as i64;
854                random.set_large_feature_seed(
855                    seed_i64.wrapping_add(index_i64),
856                    source.pos.0.x,
857                    source.pos.0.y,
858                );
859
860                let probability = carver.base().probability;
861                if random.next_f32() > probability {
862                    continue;
863                }
864
865                match &carver.kind {
866                    ConfiguredCarverKind::Cave(cfg) => {
867                        self.carve_cave(cfg, cave::CaveKind::Overworld, source.pos, random);
868                    }
869                    ConfiguredCarverKind::NetherCave(cfg) => {
870                        self.carve_cave(cfg, cave::CaveKind::Nether, source.pos, random);
871                    }
872                    ConfiguredCarverKind::Canyon(cfg) => {
873                        self.carve_canyon(cfg, source.pos, random);
874                    }
875                }
876            }
877        }
878    }
879}

883/// Vanilla's `LinearCongruentialGenerator.next()`.
884#[inline]
885const fn lcg_next(mut rval: i64, c: i64) -> i64 {
886    rval = rval.wrapping_mul(
887        rval.wrapping_mul(6_364_136_223_846_793_005)
888            .wrapping_add(1_442_695_040_888_963_407),
889    );
890    rval = rval.wrapping_add(c);
891    rval
892}

894/// Vanilla's `BiomeManager.getFiddle()`.
895#[inline]
896fn get_fiddle(rval: i64) -> f64 {
897    let uniform = ((rval >> 24).rem_euclid(1024)) as f64 / 1024.0;
898    (uniform - 0.5) * 0.9
899}

901/// Single-shot fuzzed biome lookup at a block position. Matches vanilla's
902/// `BiomeManager.getBiome(BlockPos)`: the block is shifted by `-2`, snapped to
903/// the enclosing quart cell, and the winning biome is chosen from the 8
904/// corners of that cell by `get_fiddle`-perturbed squared distance.
905///
906/// `quart_biome` returns the unfuzzed biome at a quart-coordinate — typically
907/// `biome_sampler.sample(qx, qy, qz).id()`.
908///
909/// Used by carver top-material lookups where a simple unfuzzed lookup would
910/// differ from vanilla at the quart-cell boundaries.
911pub(crate) fn fuzzed_biome_at_block<F: FnMut(IVec3) -> u16>(
912    biome_zoom_seed: i64,
913    pos: BlockPos,
914    mut quart_biome: F,
915) -> u16 {
916    let abs = pos.0 - IVec3::splat(2);
917    let parent = IVec3::new(abs.x >> 2, abs.y >> 2, abs.z >> 2);
918    let fract = DVec3::new(
919        f64::from(abs.x & 3),
920        f64::from(abs.y & 3),
921        f64::from(abs.z & 3),
922    ) / 4.0;
923
924    let mut min_i = 0usize;
925    let mut min_dist = f64::INFINITY;
926
927    for i in 0..8usize {
928        let x_even = (i & 4) == 0;
929        let y_even = (i & 2) == 0;
930        let z_even = (i & 1) == 0;
931        let cx = if x_even { parent.x } else { parent.x + 1 };
932        let cy = if y_even { parent.y } else { parent.y + 1 };
933        let cz = if z_even { parent.z } else { parent.z + 1 };
934        let dx = if x_even { fract.x } else { fract.x - 1.0 };
935        let dy = if y_even { fract.y } else { fract.y - 1.0 };
936        let dz = if z_even { fract.z } else { fract.z - 1.0 };
937
938        // BiomeManager.getFiddledDistance — identical sequence to
939        // FuzzedBiomeColumn::compute_cy_group but without the column cache.
940        let mut rval = lcg_next(biome_zoom_seed, i64::from(cx));
941        rval = lcg_next(rval, i64::from(cy));
942        rval = lcg_next(rval, i64::from(cz));
943        rval = lcg_next(rval, i64::from(cx));
944        rval = lcg_next(rval, i64::from(cy));
945        rval = lcg_next(rval, i64::from(cz));
946        let fx = get_fiddle(rval);
947        rval = lcg_next(rval, biome_zoom_seed);
948        let fy = get_fiddle(rval);
949        rval = lcg_next(rval, biome_zoom_seed);
950        let fz = get_fiddle(rval);
951
952        let dist = (dx + fx).powi(2) + (dy + fy).powi(2) + (dz + fz).powi(2);
953        if min_dist > dist {
954            min_i = i;
955            min_dist = dist;
956        }
957    }
958
959    let b = IVec3::new(
960        if (min_i & 4) == 0 {
961            parent.x
962        } else {
963            parent.x + 1
964        },
965        if (min_i & 2) == 0 {
966            parent.y
967        } else {
968            parent.y + 1
969        },
970        if (min_i & 1) == 0 {
971            parent.z
972        } else {
973            parent.z + 1
974        },
975    );
976    quart_biome(b)
977}

979/// Column-local cache for fuzzed biome lookups (vanilla `BiomeManager.getBiome()`).
980///
981/// Within a column, `parent_x`, `parent_z`, `fract_x`, `fract_z` are constant.
982/// The 8 Voronoi candidate fiddle values (computed via 8 serial LCG calls each)
983/// only change when `parent_y` changes (every 4 blocks). This cache precomputes
984/// the fiddle values and X/Z distance components per `parent_y` group, reducing
985/// per-block work to 8 additions + 8 multiplies + 8 comparisons.
986struct FuzzedBiomeColumn<'a> {
987    biome_data: &'a [u16],
988    section_count: usize,
989    biome_zoom_seed: i64,
990    parent_x: i32,
991    parent_z: i32,
992    fract_x: f64,
993    fract_z: f64,
994    min_y: i32,
995    chunk_quart_x: i32,
996    chunk_quart_z: i32,
997    neighbor_biomes: &'a dyn Fn(IVec3) -> u16,
998    cached_parent_y: i32,
999    /// Per-candidate cached values: (`fy`, `xz_partial_distance`).
1000    candidates: [(f64, f64); 8],
1001    /// Precomputed `lcg_next(seed, parent_x)` and `lcg_next(seed, parent_x + 1)`.
1002    rval_after_cx: [i64; 2],
1003}
1004
1005impl<'a> FuzzedBiomeColumn<'a> {
1006    #[expect(
1007        clippy::too_many_arguments,
1008        reason = "matches vanilla BiomeManager.getBiome signature"
1009    )]
1010    fn new(
1011        biome_data: &'a [u16],
1012        section_count: usize,
1013        biome_zoom_seed: i64,
1014        block_x: i32,
1015        block_z: i32,
1016        min_y: i32,
1017        chunk_quart_x: i32,
1018        chunk_quart_z: i32,
1019        neighbor_biomes: &'a dyn Fn(IVec3) -> u16,
1020    ) -> Self {
1021        let abs_x = block_x - 2;
1022        let abs_z = block_z - 2;
1023        let parent_x = abs_x >> 2;
1024        let parent_z = abs_z >> 2;
1025        Self {
1026            biome_data,
1027            section_count,
1028            biome_zoom_seed,
1029            parent_x,
1030            parent_z,
1031            fract_x: f64::from(abs_x & 3) / 4.0,
1032            fract_z: f64::from(abs_z & 3) / 4.0,
1033            min_y,
1034            chunk_quart_x,
1035            chunk_quart_z,
1036            neighbor_biomes,
1037            cached_parent_y: i32::MIN,
1038            candidates: [(0.0, 0.0); 8],
1039            rval_after_cx: [
1040                lcg_next(biome_zoom_seed, i64::from(parent_x)),
1041                lcg_next(biome_zoom_seed, i64::from(parent_x + 1)),
1042            ],
1043        }
1044    }
1045
1046    /// Compute candidates for a given `cy`, writing to either the low (bit1=0)
1047    /// or high (bit1=1) slots. Shares the `lcg_next(seed, cx)` precomputation
1048    /// and the `lcg_next(_, cy)` step within each cx group.
1049    #[inline]
1050    fn compute_cy_group(&mut self, cy: i32, high: bool) {
1051        let base_idx = if high { 2 } else { 0 };
1052        for cx_idx in 0..2usize {
1053            let cx = self.parent_x + cx_idx as i32;
1054            let dx = if cx_idx == 0 {
1055                self.fract_x
1056            } else {
1057                self.fract_x - 1.0
1058            };
1059            let rval_cy = lcg_next(self.rval_after_cx[cx_idx], i64::from(cy));
1060            for cz_off in 0..2usize {
1061                let cz = self.parent_z + cz_off as i32;
1062                let dz = if cz_off == 0 {
1063                    self.fract_z
1064                } else {
1065                    self.fract_z - 1.0
1066                };
1067
1068                let mut rval = lcg_next(rval_cy, i64::from(cz));
1069                rval = lcg_next(rval, i64::from(cx));
1070                rval = lcg_next(rval, i64::from(cy));
1071                rval = lcg_next(rval, i64::from(cz));
1072                let fx = get_fiddle(rval);
1073                rval = lcg_next(rval, self.biome_zoom_seed);
1074                let fy = get_fiddle(rval);
1075                rval = lcg_next(rval, self.biome_zoom_seed);
1076                let fz = get_fiddle(rval);
1077
1078                let xz_partial = (dx + fx) * (dx + fx) + (dz + fz) * (dz + fz);
1079                self.candidates[cx_idx * 4 + base_idx + cz_off] = (fy, xz_partial);
1080            }
1081        }
1082    }
1083
1084    /// Recompute the 8 candidate fiddle values and X/Z distance for a new `parent_y`.
1085    ///
1086    /// When scanning downward (`parent_y` decreases by 1), the old low-cy candidates
1087    /// (`cy=old_parent_y`) match the new high-cy slots (`cy=new_parent_y+1`), so only
1088    /// the 4 new low-cy candidates need fresh LCG computation.
1089    fn recompute_candidates(&mut self, parent_y: i32) {
1090        if self.cached_parent_y != i32::MIN && parent_y == self.cached_parent_y - 1 {
1091            // Reuse: old low-cy group → new high-cy group
1092            self.candidates[2] = self.candidates[0];
1093            self.candidates[3] = self.candidates[1];
1094            self.candidates[6] = self.candidates[4];
1095            self.candidates[7] = self.candidates[5];
1096            self.compute_cy_group(parent_y, false);
1097        } else {
1098            self.compute_cy_group(parent_y, false);
1099            self.compute_cy_group(parent_y + 1, true);
1100        }
1101        self.cached_parent_y = parent_y;
1102    }
1103
1104    /// Fuzzed biome lookup for a given `block_y`.
1105    #[expect(
1106        clippy::similar_names,
1107        reason = "matches vanilla variable names: fract_x/y/z, parent_x/y/z"
1108    )]
1109    #[inline]
1110    fn get(&mut self, block_y: i32) -> u16 {
1111        let abs_y = block_y - 2;
1112        let parent_y = abs_y >> 2;
1113        let fract_y = f64::from(abs_y & 3) / 4.0;
1114
1115        if parent_y != self.cached_parent_y {
1116            self.recompute_candidates(parent_y);
1117        }
1118
1119        let mut min_i = 0usize;
1120        let mut min_dist = f64::INFINITY;
1121        for i in 0..8usize {
1122            let (fy, xz_partial) = self.candidates[i];
1123            let dy = if (i & 2) == 0 { fract_y } else { fract_y - 1.0 };
1124            let dist = xz_partial + (dy + fy) * (dy + fy);
1125            if min_dist > dist {
1126                min_i = i;
1127                min_dist = dist;
1128            }
1129        }
1130
1131        let biome_quart = IVec3::new(
1132            if (min_i & 4) == 0 {
1133                self.parent_x
1134            } else {
1135                self.parent_x + 1
1136            },
1137            if (min_i & 2) == 0 {
1138                parent_y
1139            } else {
1140                parent_y + 1
1141            },
1142            if (min_i & 1) == 0 {
1143                self.parent_z
1144            } else {
1145                self.parent_z + 1
1146            },
1147        );
1148
1149        let in_chunk = biome_quart.x >= self.chunk_quart_x
1150            && biome_quart.x < self.chunk_quart_x + 4
1151            && biome_quart.z >= self.chunk_quart_z
1152            && biome_quart.z < self.chunk_quart_z + 4;
1153
1154        if in_chunk {
1155            let min_qy = self.min_y >> 2;
1156            let total_quarts_y = self.section_count * 4;
1157            let local_qx = (biome_quart.x - self.chunk_quart_x) as usize;
1158            let local_qz = (biome_quart.z - self.chunk_quart_z) as usize;
1159            let qy_in_chunk = (biome_quart.y - min_qy).clamp(0, total_quarts_y as i32 - 1) as usize;
1160            let section_idx = qy_in_chunk / 4;
1161            let local_qy = qy_in_chunk % 4;
1162            self.biome_data[section_idx * 64 + local_qy * 16 + local_qz * 4 + local_qx]
1163        } else {
1164            (self.neighbor_biomes)(biome_quart)
1165        }
1166    }
1167}
