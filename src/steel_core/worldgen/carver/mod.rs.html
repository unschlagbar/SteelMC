35/// The four preliminary-surface-level samples at a chunk's block corners, in
36/// world Y. Indexed by local `(x, z)` corner as `(0,0)`, `(16,0)`, `(0,16)`,
37/// `(16,16)`.
38#[derive(Debug, Clone, Copy)]
39pub struct PreliminarySurfaceCorners {
40    /// Corner at `(chunk_min_x, chunk_min_z)`.
41    pub nw: i32,
42    /// Corner at `(chunk_min_x + 16, chunk_min_z)`.
43    pub ne: i32,
44    /// Corner at `(chunk_min_x, chunk_min_z + 16)`.
45    pub sw: i32,
46    /// Corner at `(chunk_min_x + 16, chunk_min_z + 16)`.
47    pub se: i32,
48}

61/// Runtime context for a single `apply_carvers` invocation on one chunk.
62///
63/// Mirrors vanilla's `CarvingContext`. Owns the freshly-built [`Aquifer`] for
64/// this chunk; the aquifer is regenerated per carver invocation rather than
65/// cached on the [`ProtoChunk`] — see the TODO on `ProtoChunk::carving_mask`
66/// for discussion.
67pub struct CarvingContext<'a, N: DimensionNoises> {
68    /// Dimension minimum Y (inclusive).
69    pub min_y: i32,
70    /// Dimension vertical extent in blocks (`max_y = min_y + gen_depth - 1`).
71    pub gen_depth: i32,
72    /// Surface system (biome-specific surface noise + clay bands).
73    pub surface_system: &'a SurfaceSystem,
74    /// Owned aquifer for this chunk. Built fresh from the dimension's noises
75    /// at the start of `apply_carvers`.
76    pub aquifer: Aquifer<N>,
77    /// Default solid block for this dimension (stone / netherrack /
78    /// `end_stone`).
79    pub default_block_id: BlockStateId,
80    /// Preliminary surface levels at the 4 corners of this chunk, used for
81    /// bilinear interpolation of `min_surface_level` during top-material
82    /// lookup.
83    pub psl_corners: PreliminarySurfaceCorners,
84    /// Chunk NW block X — anchors `psl_corners`.
85    pub chunk_min_x: i32,
86    /// Chunk NW block Z — anchors `psl_corners`.
87    pub chunk_min_z: i32,
88}

90impl<N: DimensionNoises> CarvingContext<'_, N> {
91    /// Bilinear interpolation of the 4 preliminary-surface-level corners at
92    /// the given in-chunk position. Matches vanilla's
93    /// `SurfaceRules.Context.updateXZ` path.
94    #[must_use]
95    pub fn min_surface_level(&self, block_x: i32, block_z: i32) -> i32 {
96        let local_x = (block_x - self.chunk_min_x).clamp(0, 16);
97        let local_z = (block_z - self.chunk_min_z).clamp(0, 16);
98        // Vanilla: (float)(blockX & 15) / 16.0F — float intermediate is exact for 0-15
99        let t_x = f64::from(local_x as u8) / 16.0;
100        let t_z = f64::from(local_z as u8) / 16.0;
101        let c = self.psl_corners;
102        let interp = lerp2(
103            t_x,
104            t_z,
105            f64::from(c.nw),
106            f64::from(c.ne),
107            f64::from(c.sw),
108            f64::from(c.se),
109        );
110        interp.floor() as i32
111    }
112
113    /// Runs surface rules at a single position to pick the "top material"
114    /// block (grass / podzol / mycelium / sand / ...). Called by the carver
115    /// when it uncovers dirt beneath a grass block so the exposed surface gets
116    /// rewritten to the biome-appropriate surface block.
117    ///
118    /// Mirrors vanilla's `SurfaceSystem.topMaterial` (the `@Deprecated`
119    /// carver-specific variant). Vanilla hardcodes
120    /// `stone_depth_above = stone_depth_below = 1` here, and the water height
121    /// depends on whether the carved block was replaced with a fluid.
122    pub fn top_material(
123        &self,
124        biome_id: u16,
125        block_x: i32,
126        block_y: i32,
127        block_z: i32,
128        steep: bool,
129        under_fluid: bool,
130    ) -> Option<BlockStateId> {
131        // Surface noise inputs (same helpers build_surface uses per column).
132        let surface_depth = self.surface_system.get_surface_depth(block_x, block_z);
133        let surface_secondary = self.surface_system.get_surface_secondary(block_x, block_z);
134        let min_surface_level = self.min_surface_level(block_x, block_z) + surface_depth - 8;
135
136        let water_height = if under_fluid { block_y + 1 } else { i32::MIN };
137        let condition_noise_values: SmallVec<[Cell<f64>; 8]> = N::surface_noise_ids()
138            .iter()
139            .map(|_| Cell::new(0.0))
140            .collect();
141        let condition_noise_initialized: SmallVec<[Cell<bool>; 8]> = N::surface_noise_ids()
142            .iter()
143            .map(|_| Cell::new(false))
144            .collect();
145        let condition_noise_cache =
146            SurfaceConditionNoiseCache::new(&condition_noise_values, &condition_noise_initialized);
147
148        let mut ctx = SurfaceRuleContext::new(
149            block_x,
150            block_z,
151            surface_depth,
152            surface_secondary,
153            min_surface_level,
154            steep,
155            block_y,
156            1,
157            1,
158            water_height,
159            Some(biome_id),
160            None,
161            self.surface_system,
162            &condition_noise_cache,
163            N::surface_rule_block_states(),
164        );
165
166        N::try_apply_surface_rule(&mut ctx)
167    }
168}

170/// Vanilla's `WorldCarver.canReplaceBlock`: a carver may only replace blocks
171/// in its config's `replaceable` tag.
172#[must_use]
173pub fn can_replace_block(state: BlockStateId, tag: &Identifier) -> bool {
174    if state.is_air() {
175        return false;
176    }
177    let Some(block) = REGISTRY.blocks.by_state_id(state) else {
178        return false;
179    };
180    block.has_tag(tag)
181}

183/// Per-state membership cache for a carver's replaceable block tag.
184///
185/// Vanilla tests a block-state predicate for every candidate block. Steel's
186/// registry stores tags by block key, so resolving that predicate once into a
187/// state-id table avoids repeated tag/hash lookups in the carve loop while
188/// preserving the configured tag as the source of truth.
189#[derive(Debug)]
190pub struct CarverReplaceableStates {
191    states: Box<[bool]>,
192}
193
194impl CarverReplaceableStates {
195    fn build(tag: &Identifier) -> Self {
196        let states = REGISTRY
197            .blocks
198            .state_to_block_lookup
199            .iter()
200            .map(|&block| block.has_tag(tag))
201            .collect();
202        Self { states }
203    }
204
205    /// Returns whether `state` belongs to this cached replaceable set.
206    #[inline]
207    #[must_use]
208    pub fn contains(&self, state: BlockStateId) -> bool {
209        self.states.get(state.0 as usize).copied().unwrap_or(false)
210    }
211}
212
213static CARVER_REPLACEABLE_STATES: LazyLock<FxHashMap<Identifier, CarverReplaceableStates>> =
214    LazyLock::new(|| {
215        let mut states_by_tag = FxHashMap::default();
216        for (_, carver) in REGISTRY.configured_carvers.iter() {
217            let tag = &carver.base().replaceable_tag;
218            if !states_by_tag.contains_key(tag) {
219                states_by_tag.insert(tag.clone(), CarverReplaceableStates::build(tag));
220            }
221        }
222        states_by_tag
223    });
224
225/// Returns the cached replaceable-state set for a configured carver tag.
226#[must_use]
227pub fn cached_replaceable_states(tag: &Identifier) -> Option<&'static CarverReplaceableStates> {
228    CARVER_REPLACEABLE_STATES.get(tag)
229}

231/// Which carver family dictates the per-block decision inside
232/// [`CarveRun::carve_ellipsoid`]. Overworld carvers (cave + canyon) use the
233/// aquifer to pick air / water / lava; the nether variant hardcodes lava
234/// below `min_gen_y + 31` and cave-air elsewhere, with no aquifer lookups.
235#[derive(Debug, Clone, Copy, PartialEq, Eq)]
236pub enum CarverStyle {
237    /// Overworld / end: aquifer-driven fluid/air.
238    Overworld,
239    /// Nether: lava below `min_gen_y + 31` else `CAVE_AIR`; no aquifer check.
240    Nether,
241}

298/// Predicate called inside the carver's Y scan to decide whether a block is
299/// outside the carved shape for a given ellipsoid (cave floor cutoff, canyon
300/// width-by-height, etc). Matches vanilla's `WorldCarver.CarveSkipChecker`.
301pub trait CarveSkipChecker {
302    /// `xd`, `yd`, `zd` are the ellipsoid-normalized offsets from the carver
303    /// origin to this block's center (see `CarveRun::carve_ellipsoid`);
304    /// `world_y` is the absolute Y coordinate of the current block.
305    fn should_skip(&mut self, xd: f64, yd: f64, zd: f64, world_y: i32) -> bool;
306}
307
308impl<F: FnMut(f64, f64, f64, i32) -> bool> CarveSkipChecker for F {
309    fn should_skip(&mut self, xd: f64, yd: f64, zd: f64, world_y: i32) -> bool {
310        self(xd, yd, zd, world_y)
311    }
312}

329/// Vanilla cave/canyon tunnel radius calculation.
330#[inline]
331#[must_use]
332pub(super) fn horizontal_tunnel_radius(progress_arg: f32, thickness: f32) -> f64 {
333    let radius_offset = trig::sin(f64::from(progress_arg)) * thickness;
334    1.5 + f64::from(radius_offset)
335}

345/// The references every carver method needs. Bundled so `carve_ellipsoid`,
346/// `carve_block`, `create_tunnel`, `create_room`, `carve_cave`,
347/// `carve_canyon`, and `do_carve` can all be `&mut self` methods instead of
348/// repeating the same 7–8 arguments.
349pub struct CarveRun<'a, 'b, N, F>
350where
351    N: DimensionNoises,
352    F: FnMut(BlockPos) -> u16,
353{
354    /// Dimension-level context (aquifer, surface system, bounds, psl).
355    pub ctx: &'a mut CarvingContext<'b, N>,
356    /// Noise generators for this dimension.
357    pub noises: &'a N,
358    /// Chunk being carved into.
359    pub chunk: &'a ChunkAccess,
360    /// Chunk NW block X (cached; `ctx.chunk_min_x` mirrors this).
361    pub chunk_min_x: i32,
362    /// Chunk NW block Z (cached; `ctx.chunk_min_z` mirrors this).
363    pub chunk_min_z: i32,
364    /// Biome lookup (vanilla `BiomeManager.getBiome`-style, fuzzed).
365    pub biome_getter: &'a mut F,
366    /// Carving mask for the chunk (lazily created on the proto chunk).
367    pub mask: &'a mut CarvingMask,
368    /// Block IDs cached once per carver session.
369    pub ids: CarverBlockIds,
370}

372impl<N, F> CarveRun<'_, '_, N, F>
373where
374    N: DimensionNoises,
375    F: FnMut(BlockPos) -> u16,
376{
377    /// Carve every block inside the given ellipsoid that falls in this chunk.
378    /// Mirrors vanilla's `WorldCarver.carveEllipsoid`.
379    ///
380    /// Returns `true` if at least one block was carved.
381    #[expect(
382        clippy::similar_names,
383        reason = "min_x_idx / min_z_idx / max_x_idx / max_z_idx mirror vanilla"
384    )]
385    #[expect(
386        clippy::too_many_arguments,
387        reason = "params + x/y/z/horizontal_radius/vertical_radius + skip_checker mirrors vanilla"
388    )]
389    pub fn carve_ellipsoid<S: CarveSkipChecker>(
390        &mut self,
391        params: &CarveParams<'_>,
392        x: f64,
393        y: f64,
394        z: f64,
395        horizontal_radius: f64,
396        vertical_radius: f64,
397        mut skip_checker: S,
398    ) -> bool {
399        let middle_x = f64::from(self.chunk_min_x) + 8.0;
400        let middle_z = f64::from(self.chunk_min_z) + 8.0;
401        let max_delta = 16.0 + horizontal_radius * 2.0;
402        if (x - middle_x).abs() > max_delta || (z - middle_z).abs() > max_delta {
403            return false;
404        }
405
406        let min_x_idx = ((x - horizontal_radius).floor() as i32 - self.chunk_min_x - 1).max(0);
407        let max_x_idx = ((x + horizontal_radius).floor() as i32 - self.chunk_min_x).min(15);
408        let min_y = ((y - vertical_radius).floor() as i32 - 1).max(self.ctx.min_y + 1);
409        // Vanilla: `chunk.isUpgrading() ? 0 : 7`. No chunk upgrade path yet,
410        // so always 7 — matches extractor config.
411        let protected_blocks_on_top = 7;
412        let max_y = ((y + vertical_radius).floor() as i32 + 1)
413            .min(self.ctx.min_y + self.ctx.gen_depth - 1 - protected_blocks_on_top);
414        let min_z_idx = ((z - horizontal_radius).floor() as i32 - self.chunk_min_z - 1).max(0);
415        let max_z_idx = ((z + horizontal_radius).floor() as i32 - self.chunk_min_z).min(15);
416
417        let mut carved = false;
418
419        for x_idx in min_x_idx..=max_x_idx {
420            let world_x = self.chunk_min_x + x_idx;
421            let xd = (f64::from(world_x) + 0.5 - x) / horizontal_radius;
422
423            for z_idx in min_z_idx..=max_z_idx {
424                let world_z = self.chunk_min_z + z_idx;
425                let zd = (f64::from(world_z) + 0.5 - z) / horizontal_radius;
426                if xd * xd + zd * zd >= 1.0 {
427                    continue;
428                }
429
430                let mut has_grass = false;
431
432                // Scan top-down; range is exclusive of min_y (matches vanilla's
433                // `worldY > minY`).
434                for world_y in (min_y + 1..=max_y).rev() {
435                    let yd = (f64::from(world_y) - 0.5 - y) / vertical_radius;
436                    if skip_checker.should_skip(xd, yd, zd, world_y) {
437                        continue;
438                    }
439                    if !self.mask.set_if_unset(x_idx, world_y, z_idx) {
440                        continue;
441                    }
442                    if self.carve_block(params, world_x, world_y, world_z, &mut has_grass) {
443                        carved = true;
444                    }
445                }
446            }
447        }
448
449        carved
450    }
451
452    /// Per-block carve decision + placement. Mirrors vanilla's
453    /// `WorldCarver.carveBlock` (and the `NetherWorldCarver` override).
454    fn carve_block(
455        &mut self,
456        params: &CarveParams<'_>,
457        world_x: i32,
458        world_y: i32,
459        world_z: i32,
460        has_grass: &mut bool,
461    ) -> bool {
462        let pos = BlockPos::new(world_x, world_y, world_z);
463        let existing = self.chunk.get_block_state(pos);
464
465        // Track grass/mycelium for the top-material rewrite later.
466        if existing == self.ids.grass_block || existing == self.ids.mycelium {
467            *has_grass = true;
468        }
469
470        if !Self::can_replace(params, existing) {
471            return false;
472        }
473
474        let state = match self.get_carve_state(params, world_x, world_y, world_z) {
475            CarveState::Place(id) => id,
476            CarveState::Skip => return false,
477        };
478
479        self.chunk.set_block_state(pos, state, UpdateFlags::empty());
480        if params.style == CarverStyle::Overworld
481            && self.ctx.aquifer.should_schedule_fluid_update()
482            && state.has_fluid()
483        {
484            self.chunk.mark_pos_for_postprocessing(pos);
485        }
486
487        // Top-material rewrite: only when we just turned a grass/mycelium
488        // block into something carved, and the block directly below is plain
489        // dirt. Nether carver skips this entirely (its override of carveBlock
490        // doesn't run this branch).
491        if params.style == CarverStyle::Overworld && *has_grass {
492            let below_pos = BlockPos::new(world_x, world_y - 1, world_z);
493            if self.chunk.get_block_state(below_pos) == self.ids.dirt {
494                let under_fluid = !self.ids.is_air_like(state);
495                let steep = self.steep_material_condition(world_x, world_z);
496                let biome_id =
497                    (self.biome_getter)(BlockPos(IVec3::new(world_x, world_y - 1, world_z)));
498                if let Some(top) = self.ctx.top_material(
499                    biome_id,
500                    world_x,
501                    world_y - 1,
502                    world_z,
503                    steep,
504                    under_fluid,
505                ) {
506                    self.chunk
507                        .set_block_state(below_pos, top, UpdateFlags::empty());
508                    if top.has_fluid() {
509                        self.chunk.mark_pos_for_postprocessing(below_pos);
510                    }
511                }
512            }
513        }
514
515        true
516    }
517
518    #[inline]
519    fn can_replace(params: &CarveParams<'_>, state: BlockStateId) -> bool {
520        if state.is_air() {
521            return false;
522        }
523        if let Some(states) = params.replaceable_states {
524            return states.contains(state);
525        }
526        can_replace_block(state, params.replaceable_tag)
527    }
528
529    fn steep_material_condition(&self, world_x: i32, world_z: i32) -> bool {
530        let heightmaps = self.chunk.proto_heightmaps();
531        if let Some(worldgen_surface) = heightmaps.get(HeightmapType::WorldSurfaceWg) {
532            return steep_material_condition(worldgen_surface, world_x, world_z);
533        }
534        drop(heightmaps);
535
536        self.chunk
537            .prime_heightmaps(&[HeightmapType::WorldSurfaceWg]);
538        let heightmaps = self.chunk.proto_heightmaps();
539        if let Some(worldgen_surface) = heightmaps.get(HeightmapType::WorldSurfaceWg) {
540            return steep_material_condition(worldgen_surface, world_x, world_z);
541        }
542
543        log::error!("WorldSurfaceWg heightmap missing during carver top-material lookup");
544        false
545    }
546
547    /// Vanilla's `WorldCarver.getCarveState` + the nether override dispatch.
548    fn get_carve_state(&mut self, params: &CarveParams<'_>, x: i32, y: i32, z: i32) -> CarveState {
549        match params.style {
550            CarverStyle::Overworld => {
551                if y <= params.lava_level_y {
552                    return CarveState::Place(self.ids.lava);
553                }
554                match self
555                    .ctx
556                    .aquifer
557                    .compute_substance(self.noises, x, y, z, 0.0)
558                {
559                    AquiferResult::Solid => CarveState::Skip,
560                    AquiferResult::Fluid(id) => CarveState::Place(id),
561                    AquiferResult::Air => CarveState::Place(self.ids.air),
562                }
563            }
564            CarverStyle::Nether => {
565                if y <= self.ctx.min_y + 31 {
566                    CarveState::Place(self.ids.lava)
567                } else {
568                    CarveState::Place(self.ids.cave_air)
569                }
570            }
571        }
572    }
573}

575/// Vanilla's `SurfaceRules.steep()` condition. It is asymmetric: only
576/// south-vs-north and west-vs-east deltas are checked.
577#[must_use]
578fn steep_material_condition(worldgen_surface: &Heightmap, block_x: i32, block_z: i32) -> bool {
579    let local_x = (block_x & 15) as usize;
580    let local_z = (block_z & 15) as usize;
581
582    let z_north = local_z.saturating_sub(1);
583    let z_south = (local_z + 1).min(15);
584    let h_north = worldgen_surface.get_highest_taken(local_x, z_north);
585    let h_south = worldgen_surface.get_highest_taken(local_x, z_south);
586    if h_south >= h_north + 4 {
587        return true;
588    }
589
590    let x_west = local_x.saturating_sub(1);
591    let x_east = (local_x + 1).min(15);
592    let h_west = worldgen_surface.get_highest_taken(x_west, local_z);
593    let h_east = worldgen_surface.get_highest_taken(x_east, local_z);
594    h_west >= h_east + 4
595}

597/// Vanilla's `WorldCarver.canReach` — prunes carver steps that can't touch
598/// any block in the given chunk (used by cave/canyon tunnel loops before
599/// carving an ellipsoid).
600#[must_use]
601pub fn can_reach(
602    chunk_min_x: i32,
603    chunk_min_z: i32,
604    x: f64,
605    z: f64,
606    current_step: i32,
607    total_steps: i32,
608    thickness: f32,
609) -> bool {
610    let x_mid = f64::from(chunk_min_x) + 8.0;
611    let z_mid = f64::from(chunk_min_z) + 8.0;
612    let xd = x - x_mid;
613    let zd = z - z_mid;
614    let remaining = f64::from(total_steps - current_step);
615    let rr = f64::from(thickness + 2.0_f32 + 16.0_f32);
616    xd * xd + zd * zd - remaining * remaining <= rr * rr
617}
