18pub(crate) fn initialize(
19    _context: Arc<WorldGenContext>,
20    _step: &ChunkStep,
21    _cache: &Arc<StaticCache2D<Arc<ChunkHolder>>>,
22    holder: Arc<ChunkHolder>,
23) {
24    let Some(chunk) = holder.try_chunk(ChunkStatus::Features) else {
25        panic!("Chunk not found at status Features");
26    };
27
28    chunk.initialize_light_sources();
29}

31pub(crate) fn generate(
32    context: Arc<WorldGenContext>,
33    _step: &ChunkStep,
34    cache: &Arc<StaticCache2D<Arc<ChunkHolder>>>,
35    holder: Arc<ChunkHolder>,
36) {
37    let (sky_updates, block_updates) = run_light_stage(
38        cache,
39        holder.as_ref(),
40        context.world().dimension_type.has_skylight,
41    );
42    publish_light_updates(&context, LightLayer::Sky, sky_updates);
43    publish_light_updates(&context, LightLayer::Block, block_updates);
44}

61fn run_light_stage(
62    cache: &StaticCache2D<Arc<ChunkHolder>>,
63    holder: &ChunkHolder,
64    has_skylight: bool,
65) -> (Vec<SectionPos>, Vec<SectionPos>) {
66    let center = holder.get_pos();
67    assert!(
68        holder.try_chunk(ChunkStatus::InitializeLight).is_some(),
69        "Chunk not found at status InitializeLight"
70    );
71
72    let Ok(range) = LightSectionRange::from_world_height(holder.min_y(), holder.height()) else {
73        panic!("invalid world height for light stage");
74    };
75
76    let layout = LightCacheLayout::new(center, range);
77    let Ok(workset) = LightWorkset::setup_with_scopes(
78        layout,
79        LightCacheSetupRadius::Full,
80        true,
81        |pos| {
82            let holder = cache.try_get(pos.0.x, pos.0.y)?;
83            holder
84                .try_chunk(ChunkStatus::InitializeLight)
85                .is_some()
86                .then(|| Arc::clone(holder))
87        },
88        |cached_chunk, holder, _chunk| {
89            let status = holder.persisted_status();
90            let center_chunk = cached_chunk.chunk_pos == center;
91            let initialized = status.is_some_and(|status| status >= ChunkStatus::InitializeLight);
92            let lit = status.is_some_and(|status| status >= ChunkStatus::Light);
93            (center_chunk || initialized, center_chunk || lit)
94        },
95    ) else {
96        panic!("required light-stage chunk is missing");
97    };
98
99    let sky_updates = if has_skylight {
100        match propagate_sky_light_chunk(&workset, SkyLightChunkEdgeChecks::Required) {
101            Ok(result) => result.updated_sections,
102            Err(error) => panic!("sky light chunk propagation failed: {error:?}"),
103        }
104    } else {
105        Vec::new()
106    };
107    let block_result =
108        match propagate_block_light_chunk(&workset, BlockLightChunkEdgeChecks::Required) {
109            Ok(result) => result,
110            Err(error) => panic!("block light chunk propagation failed: {error:?}"),
111        };
112
113    (sky_updates, block_result.updated_sections)
114}

116fn run_loaded_light_stage(
117    cache: &StaticCache2D<Arc<ChunkHolder>>,
118    holder: &ChunkHolder,
119    has_skylight: bool,
120) -> (Vec<SectionPos>, Vec<SectionPos>) {
121    let center = holder.get_pos();
122    assert!(
123        holder.try_chunk(ChunkStatus::Light).is_some(),
124        "Chunk not found at status Light"
125    );
126
127    let Ok(range) = LightSectionRange::from_world_height(holder.min_y(), holder.height()) else {
128        panic!("invalid world height for loaded light stage");
129    };
130
131    let layout = LightCacheLayout::new(center, range);
132    let Ok(workset) = LightWorkset::setup(
133        layout,
134        LightCacheSetupRadius::Full,
135        true,
136        |pos| {
137            let holder = cache.try_get(pos.0.x, pos.0.y)?;
138            holder
139                .try_chunk(ChunkStatus::Light)
140                .is_some()
141                .then(|| Arc::clone(holder))
142        },
143        |_| true,
144    ) else {
145        panic!("required loaded light-stage chunk is missing");
146    };
147
148    let mut sky_updates = if has_skylight {
149        match force_load_sky_light_chunk(&workset) {
150            Ok(result) => result.updated_sections,
151            Err(error) => panic!("loaded sky light force-load failed: {error:?}"),
152        }
153    } else {
154        Vec::new()
155    };
156    let mut block_updates = match force_load_block_light_chunk(&workset) {
157        Ok(result) => result.updated_sections,
158        Err(error) => panic!("loaded block light force-load failed: {error:?}"),
159    };
160
161    if has_skylight {
162        match check_sky_light_chunk_edges(&workset) {
163            Ok(result) => sky_updates.extend(result.updated_sections),
164            Err(error) => panic!("loaded sky light edge validation failed: {error:?}"),
165        }
166    }
167    match check_block_light_chunk_edges(&workset) {
168        Ok(result) => block_updates.extend(result.updated_sections),
169        Err(error) => panic!("loaded block light edge validation failed: {error:?}"),
170    }
171
172    (sky_updates, block_updates)
173}

175fn publish_light_updates(
176    context: &WorldGenContext,
177    layer: LightLayer,
178    updated_sections: Vec<SectionPos>,
179) {
180    if updated_sections.is_empty() {
181        return;
182    }
183
184    let world = context.world();
185    for section_pos in updated_sections {
186        world.chunk_map.light_changed(layer, section_pos);
187    }
188}

212    fn holder_with_section(
213        pos: ChunkPos,
214        status: ChunkStatus,
215        section: ChunkSection,
216    ) -> Arc<ChunkHolder> {
217        let sections = Sections::from_owned(vec![section].into_boxed_slice());
218        let proto = ProtoChunk::new(sections, pos, 0, 16, Weak::new());
219        proto.initialize_light_sources();
220        let holder = Arc::new(ChunkHolder::new(
221            pos,
222            ChunkTicketLevel::FULL_CHUNK,
223            Some(ChunkTicketLevel::FULL_CHUNK),
224            0,
225            16,
226        ));
227        holder.insert_chunk(ChunkAccess::Proto(proto), status);
228        holder
229    }

231    fn empty_holder(pos: ChunkPos, status: ChunkStatus) -> Arc<ChunkHolder> {
232        holder_with_section(pos, status, ChunkSection::new_empty())
233    }

235    fn cache_with_center(
236        center_pos: ChunkPos,
237        center_holder: &Arc<ChunkHolder>,
238        neighbor_status: ChunkStatus,
239    ) -> StaticCache2D<Arc<ChunkHolder>> {
240        let center_holder = Arc::clone(center_holder);
241        StaticCache2D::create(center_pos.0.x, center_pos.0.y, 2, move |x, z| {
242            let pos = ChunkPos::new(x, z);
243            if pos == center_pos {
244                return Arc::clone(&center_holder);
245            }
246
247            empty_holder(pos, neighbor_status)
248        })
249    }
