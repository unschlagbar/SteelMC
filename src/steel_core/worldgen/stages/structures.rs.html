11/// Generates structure starts.
12///
13/// # Panics
14/// Panics if the chunk is not at `ChunkStatus::Empty` or higher.
15pub(crate) fn generate_starts(
16    context: Arc<WorldGenContext>,
17    _step: &ChunkStep,
18    _cache: &Arc<StaticCache2D<Arc<ChunkHolder>>>,
19    holder: Arc<ChunkHolder>,
20) {
21    let chunk = holder
22        .try_chunk(ChunkStatus::Empty)
23        .expect("Chunk not found at status Empty");
24
25    context.generator.create_structures(&chunk);
26}

28/// Collects structure references from surrounding chunks' starts.
29///
30/// # Panics
31/// Panics if the chunk is not at `ChunkStatus::StructureStarts` or higher.
32pub(crate) fn generate_references(
33    _context: Arc<WorldGenContext>,
34    _step: &ChunkStep,
35    cache: &Arc<StaticCache2D<Arc<ChunkHolder>>>,
36    holder: Arc<ChunkHolder>,
37) {
38    let chunk = holder
39        .try_chunk(ChunkStatus::StructureStarts)
40        .expect("Chunk not found at status StructureStarts");
41    let target_pos = chunk.pos();
42    let target_x = target_pos.0.x;
43    let target_z = target_pos.0.y;
44    let target_block_x = target_x * 16;
45    let target_block_z = target_z * 16;
46    drop(chunk);
47
48    let mut references = StructureReferenceMap::default();
49
50    // Radius-8 scan for starts whose BB intersects this chunk.
51    for source_x in (target_x - 8)..=(target_x + 8) {
52        for source_z in (target_z - 8)..=(target_z + 8) {
53            let source_holder = cache.get(source_x, source_z);
54            let Some(source_chunk) = source_holder.try_chunk(ChunkStatus::StructureStarts) else {
55                continue;
56            };
57
58            for (structure_id, start) in source_chunk.structure_starts().iter() {
59                // Empty-pieces starts have no BB and are not valid. `start.bounding_box`
60                // is already inflated by `bb_inflate`.
61                let Some(bb) = start.bounding_box else {
62                    continue;
63                };
64                if bb.intersects_xz(
65                    target_block_x,
66                    target_block_z,
67                    target_block_x + 15,
68                    target_block_z + 15,
69                ) {
70                    references
71                        .entry(structure_id.clone())
72                        .or_default()
73                        .insert(steel_utils::ChunkPos::new(source_x, source_z));
74                }
75            }
76        }
77    }
78
79    if !references.is_empty() {
80        let target_chunk = holder
81            .try_chunk(ChunkStatus::StructureStarts)
82            .expect("Chunk not found");
83        let mut target_references = target_chunk.structure_references_mut();
84        for (structure_id, source_chunks) in references {
85            target_references
86                .entry(structure_id)
87                .or_default()
88                .extend(source_chunks);
89        }
90        target_chunk.mark_dirty();
91    }
92}
