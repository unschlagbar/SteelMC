20/// A trait for generating chunks.
21#[enum_dispatch]
22pub trait ChunkGenerator: Send + Sync {
23    /// Returns the generator's minimum Y coordinate.
24    fn min_y(&self) -> i32;
25
26    /// Returns the generator's vertical generation depth in blocks.
27    fn gen_depth(&self) -> i32;
28
29    /// Returns the climate-selected origin used by vanilla before searching for a safe spawn chunk.
30    fn initial_spawn_search_origin(&self) -> BlockPos {
31        BlockPos::new(0, 0, 0)
32    }
33
34    /// Returns the generator-provided spawn height used before falling back to the surface heightmap.
35    fn spawn_height(&self, min_y: i32, _height: i32) -> i32 {
36        let _ = min_y;
37        64
38    }
39
40    /// Returns the structure generator used for placement and locate queries.
41    fn structure_generator(&self) -> Option<&StructureGenerator> {
42        None
43    }
44
45    /// Creates the structures in a chunk.
46    fn create_structures(&self, chunk: &ChunkAccess);
47
48    /// Creates the biomes in a chunk.
49    fn create_biomes(&self, chunk: &ChunkAccess);
50
51    /// Fills the chunk with noise.
52    ///
53    /// `beardifier` carries pre-collected structure-piece terrain adaptation. The caller
54    /// (production: noise stage; tests: harness) is responsible for walking the chunk's
55    /// structure references and building the beardifier — this trait stays free of any
56    /// cross-chunk lookup. `None` skips the integration entirely (cheaper than passing
57    /// an empty beardifier).
58    fn fill_from_noise(&self, chunk: &ChunkAccess, beardifier: Option<&Beardifier>);
59
60    /// Builds the surface of the chunk.
61    ///
62    /// `neighbor_biomes` maps `(quart_x, quart_y, quart_z)` to a biome palette ID,
63    /// reading from neighbor chunk palettes for out-of-chunk biome lookups (matching
64    /// vanilla's `WorldGenRegion.getNoiseBiome`).
65    fn build_surface(&self, chunk: &ChunkAccess, neighbor_biomes: &dyn Fn(IVec3) -> u16);
66
67    /// Applies carvers to the chunk.
68    fn apply_carvers(&self, chunk: &ChunkAccess);
69
70    /// Creates the per-region random source exposed by vanilla `WorldGenRegion.getRandom()`.
71    fn create_worldgen_region_random(&self, world_seed: i64, center: ChunkPos) -> RandomSource;
72
73    /// Applies structure piece placement and biome feature decorations.
74    fn apply_biome_decorations(&self, region: &mut WorldGenRegion<'_>);
75}

77pub(crate) fn worldgen_region_random_from_splitter(
78    splitter: &RandomSplitter,
79    center: ChunkPos,
80) -> RandomSource {
81    const WORLDGEN_REGION_RANDOM: NameHash = NameHash::new("minecraft:worldgen_region_random");
82
83    let mut named_random = splitter.with_hash_of(&WORLDGEN_REGION_RANDOM);
84    let region_factory = named_random.next_positional();
85    region_factory.at(center.0.x * 16, 0, center.0.y * 16)
86}

88pub(crate) fn xoroshiro_worldgen_region_random(world_seed: i64, center: ChunkPos) -> RandomSource {
89    let splitter = Xoroshiro::from_seed(world_seed as u64).next_positional();
90    worldgen_region_random_from_splitter(&splitter, center)
