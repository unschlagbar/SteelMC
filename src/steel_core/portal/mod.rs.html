14/// Describes a teleport transition to another loaded world.
15///
16/// Vanilla names loaded worlds "dimensions" in packets and saves. Steel uses
17/// "world" for runtime loaded world instances, reserving "dimension type" for
18/// the vanilla registry entry that defines height, skylight, ceiling, etc.
19pub struct TeleportTransition {
20    /// The target world to teleport into.
21    pub target_world: Arc<World>,
22    /// The position in the target world.
23    pub position: DVec3,
24    /// The rotation (yaw, pitch) in the target world.
25    pub rotation: (f32, f32),
26    /// Portal cooldown in ticks (prevents immediate re-entry).
27    pub portal_cooldown: i32,
28}
29
30/// A queued request to move an entity between loaded worlds.
31///
32/// Vanilla calls these world changes "dimension changes". Steel keeps the
33/// runtime API named after loaded worlds to avoid confusing worlds with vanilla
34/// dimension types.
35pub enum WorldChangeRequest {
36    /// Pre-computed transition (players after chunk pre-warming).
37    Computed(TeleportTransition),
38    /// Command-driven world change to the target world's spawn.
39    WorldSpawn {
40        /// The target world to teleport into.
41        target_world: Arc<World>,
42    },
43    /// Portal position — server computes destination at processing time.
44    /// TODO: implement portal destination calculation (`nether_portal::calculate_destination`)
45    Portal {
46        /// The world the entity is currently in.
47        source_world: Arc<World>,
48        /// The portal block position.
49        portal_pos: BlockPos,
50    },
51}
