1//! Thread-safe player storage with dual indexing by UUID and entity ID.
2
3use std::sync::Arc;
4
5use scc::HashMap;
6use uuid::Uuid;
7
8use crate::{entity::Entity, player::Player};
9
10/// Thread-safe player storage with dual indexing.
11///
12/// Maintains two synchronized maps for O(1) lookup by either UUID or entity ID.
13/// All operations keep both maps in sync automatically.
14pub struct PlayerMap {
15    /// Primary index by UUID (persistent identifier)
16    by_uuid: HashMap<Uuid, Arc<Player>>,
17    /// Secondary index by entity ID (session-local identifier)
18    by_entity_id: HashMap<i32, Arc<Player>>,
19}
20
21impl Default for PlayerMap {
22    fn default() -> Self {
23        Self::new()
24    }
25}
26
27impl PlayerMap {
28    /// Creates a new empty player map.
29    #[must_use]
30    pub fn new() -> Self {
31        Self {
32            by_uuid: HashMap::new(),
33            by_entity_id: HashMap::new(),
34        }
35    }
36
37    /// Inserts a player into both maps.
38    ///
39    /// Returns `true` if the player was inserted, `false` if a player with the same UUID already exists.
40    ///
41    /// # Panics
42    ///
43    /// Panics if another player already has the same entity ID. Entity IDs are
44    /// session-unique; accepting a duplicate would break entity lookup and
45    /// packet routing invariants.
46    pub fn insert(&self, player: Arc<Player>) -> bool {
47        let uuid = player.gameprofile.id;
48        let entity_id = player.id();
49
50        if self.by_uuid.insert_sync(uuid, player.clone()).is_err() {
51            return false;
52        }
53
54        if self.by_entity_id.insert_sync(entity_id, player).is_err() {
55            let _ = self.by_uuid.remove_sync(&uuid);
56            panic!("player entity id {entity_id} is already registered");
57        }
58        true
59    }
60
61    /// Removes a player by UUID from both maps.
62    ///
63    /// Returns the removed player if found.
64    pub async fn remove(&self, uuid: &Uuid) -> Option<Arc<Player>> {
65        if let Some((_, player)) = self.by_uuid.remove_async(uuid).await {
66            let _ = self.by_entity_id.remove_async(&player.id()).await;
67            Some(player)
68        } else {
69            None
70        }
71    }
72
73    /// Removes this exact player from both maps.
74    ///
75    /// Returns the removed player if the UUID still maps to this same player
76    /// handle. A stale duplicate-login cleanup must not remove the accepted
77    /// player that owns the UUID.
78    pub async fn remove_player(&self, player: &Arc<Player>) -> Option<Arc<Player>> {
79        let uuid = player.gameprofile.id;
80        let (_, removed) = self
81            .by_uuid
82            .remove_if_async(&uuid, |current| Arc::ptr_eq(current, player))
83            .await?;
84        let _ = self
85            .by_entity_id
86            .remove_if_async(&removed.id(), |current| Arc::ptr_eq(current, &removed))
87            .await;
88        Some(removed)
89    }
90
91    /// Removes a player by UUID from both maps synchronously.
92    ///
93    /// Returns the removed player if found. Use this when async is not available
94    /// (e.g., during world changes on the tick thread).
95    pub fn remove_sync(&self, uuid: &Uuid) -> Option<Arc<Player>> {
96        if let Some((_, player)) = self.by_uuid.remove_sync(uuid) {
97            let _ = self.by_entity_id.remove_sync(&player.id());
98            Some(player)
99        } else {
100            None
101        }
102    }
103
104    /// Removes this exact player from both maps synchronously.
105    pub fn remove_player_sync(&self, player: &Arc<Player>) -> Option<Arc<Player>> {
106        let uuid = player.gameprofile.id;
107        let (_, removed) = self
108            .by_uuid
109            .remove_if_sync(&uuid, |current| Arc::ptr_eq(current, player))?;
110        let _ = self
111            .by_entity_id
112            .remove_if_sync(&removed.id(), |current| Arc::ptr_eq(current, &removed));
113        Some(removed)
114    }
115
116    /// Gets a player by UUID.
117    #[must_use]
118    pub fn get_by_uuid(&self, uuid: &Uuid) -> Option<Arc<Player>> {
119        self.by_uuid.read_sync(uuid, |_, p| p.clone())
120    }
121
122    /// Gets a player by entity ID.
123    #[must_use]
124    pub fn get_by_entity_id(&self, entity_id: i32) -> Option<Arc<Player>> {
125        self.by_entity_id.read_sync(&entity_id, |_, p| p.clone())
126    }
127
128    /// Iterates over all players.
129    ///
130    /// The callback returns `true` to continue iteration, `false` to stop.
131    pub fn iter_players<F>(&self, mut f: F)
132    where
133        F: FnMut(&Uuid, &Arc<Player>) -> bool,
134    {
135        self.by_uuid.iter_sync(|uuid, player| f(uuid, player));
136    }
137
138    /// Returns the number of players.
139    #[must_use]
140    pub fn len(&self) -> usize {
141        self.by_uuid.len()
142    }
143
144    /// Returns true if there are no players.
145    #[must_use]
146    pub fn is_empty(&self) -> bool {
147        self.by_uuid.is_empty()
148    }
149}
