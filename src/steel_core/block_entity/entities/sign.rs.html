26/// Maximum distance (in blocks) a player can be from a sign while editing.
27/// If they move further away, the edit lock is released.
28const MAX_EDIT_DISTANCE: f64 = 4.0;
29
30/// Number of text lines on each side of a sign.
31pub const SIGN_LINES: usize = 4;
32
33/// Text and styling for one side of a sign.
34#[derive(Debug, Clone)]
35pub struct SignText {
36    /// The 4 lines of text (raw, unfiltered).
37    pub messages: [TextComponent; SIGN_LINES],
38    /// Text color (dye color applied to the sign).
39    pub color: DyeColor,
40    /// Whether the text has a glowing effect (from glow ink sac).
41    pub has_glowing_text: bool,
42}
43
44impl Default for SignText {
45    fn default() -> Self {
46        Self::new()
47    }
48}
49
50impl SignText {
51    /// Creates a new empty sign text with default color (black) and no glow.
52    #[must_use]
53    pub fn new() -> Self {
54        Self {
55            messages: array::from_fn(|_| TextComponent::new()),
56            color: DyeColor::Black,
57            has_glowing_text: false,
58        }
59    }
60
61    /// Gets a message line by index.
62    #[must_use]
63    pub fn get_message(&self, index: usize) -> Option<&TextComponent> {
64        self.messages.get(index)
65    }
66
67    /// Sets a message line by index.
68    pub fn set_message(&mut self, index: usize, message: TextComponent) {
69        if index < SIGN_LINES {
70            self.messages[index] = message;
71        }
72    }
73
74    /// Checks if any line has text content.
75    #[must_use]
76    pub fn has_message(&self) -> bool {
77        self.messages.iter().any(|msg| {
78            // Check if the text component has any actual content
79            match &msg.content {
80                Content::Text { text } => !text.is_empty(),
81                _ => true, // Translations, etc. count as having a message
82            }
83        })
84    }
85
86    /// Loads sign text from borrowed NBT.
87    pub fn load(&mut self, nbt: BorrowedNbtCompoundView<'_, '_>) {
88        // Load messages - they are stored as a list of compounds (text components)
89        if let Some(messages) = nbt.list("messages")
90            && let Some(compounds) = messages.compounds()
91        {
92            for (i, compound) in compounds.into_iter().enumerate().take(SIGN_LINES) {
93                if let Some(text) = TextComponent::from_nbt(&NbtTag::Compound(compound.to_owned()))
94                {
95                    self.messages[i] = text;
96                }
97            }
98        }
99
100        // Load color
101        if let Some(color_str) = nbt.string("color") {
102            self.color = dye_color_from_str(&color_str.to_str());
103        }
104
105        // Load glow
106        if let Some(glow) = nbt.byte("has_glowing_text") {
107            self.has_glowing_text = glow != 0;
108        }
109    }
110
111    /// Saves sign text to NBT.
112    pub fn save(&self, nbt: &mut NbtCompound) {
113        // Save messages as a list of compounds (text components)
114        let compounds: Vec<NbtCompound> = self
115            .messages
116            .iter()
117            .map(|msg| msg.to_nbt_tag().into_compound().unwrap_or_default())
118            .collect();
119        nbt.insert("messages", NbtList::Compound(compounds));
120
121        // Save color
122        nbt.insert("color", dye_color_to_str(self.color));
123
124        // Save glow
125        nbt.insert("has_glowing_text", i8::from(self.has_glowing_text));
126    }
127}

129/// Sign block entity.
130///
131/// Stores text on both front and back sides of the sign.
132pub struct SignBlockEntity {
133    /// Weak reference to the world for marking chunks dirty.
134    level: Weak<World>,
135    /// Block entity type (sign or `hanging_sign`).
136    block_entity_type: BlockEntityTypeRef,
137    /// Position in the world.
138    pos: BlockPos,
139    /// Current block state.
140    state: BlockStateId,
141    /// Whether this entity has been marked for removal.
142    removed: bool,
143    /// Text on the front side.
144    pub front_text: SignText,
145    /// Text on the back side.
146    pub back_text: SignText,
147    /// Whether the sign is waxed (prevents editing).
148    pub is_waxed: bool,
149    /// UUID of the player currently allowed to edit this sign.
150    /// Used to prevent multiple players from editing simultaneously.
151    player_who_may_edit: Option<Uuid>,
152}
153
154impl SignBlockEntity {
155    /// Creates a new sign block entity.
156    #[must_use]
157    pub fn new(level: Weak<World>, pos: BlockPos, state: BlockStateId) -> Self {
158        Self::with_type(level, &vanilla_block_entity_types::SIGN, pos, state)
159    }
160
161    /// Creates a new hanging sign block entity.
162    #[must_use]
163    pub fn new_hanging(level: Weak<World>, pos: BlockPos, state: BlockStateId) -> Self {
164        Self::with_type(level, &vanilla_block_entity_types::HANGING_SIGN, pos, state)
165    }
166
167    /// Creates a sign block entity with a specific type.
168    #[must_use]
169    pub fn with_type(
170        level: Weak<World>,
171        block_entity_type: BlockEntityTypeRef,
172        pos: BlockPos,
173        state: BlockStateId,
174    ) -> Self {
175        Self {
176            level,
177            block_entity_type,
178            pos,
179            state,
180            removed: false,
181            front_text: SignText::new(),
182            back_text: SignText::new(),
183            is_waxed: false,
184            player_who_may_edit: None,
185        }
186    }
187
188    /// Gets the UUID of the player currently allowed to edit this sign.
189    #[must_use]
190    pub const fn get_player_who_may_edit(&self) -> Option<Uuid> {
191        self.player_who_may_edit
192    }
193
194    /// Sets the player allowed to edit this sign.
195    pub const fn set_player_who_may_edit(&mut self, player: Option<Uuid>) {
196        self.player_who_may_edit = player;
197    }
198
199    /// Checks if another player (not the given one) is currently editing this sign.
200    #[must_use]
201    pub fn is_other_player_editing(&self, player_uuid: Uuid) -> bool {
202        self.player_who_may_edit
203            .is_some_and(|editor| editor != player_uuid)
204    }
205
206    /// Gets the text for a side.
207    #[must_use]
208    pub const fn get_text(&self, front: bool) -> &SignText {
209        if front {
210            &self.front_text
211        } else {
212            &self.back_text
213        }
214    }
215
216    /// Gets mutable text for a side.
217    pub const fn get_text_mut(&mut self, front: bool) -> &mut SignText {
218        if front {
219            &mut self.front_text
220        } else {
221            &mut self.back_text
222        }
223    }
224
225    /// Sets the text for a side.
226    pub fn set_text(&mut self, text: SignText, front: bool) {
227        if front {
228            self.front_text = text;
229        } else {
230            self.back_text = text;
231        }
232    }
233}

258    }

260    fn is_removed(&self) -> bool {
261        self.removed
262    }

264    fn set_removed(&mut self) {
265        self.removed = true;
266    }

268    fn clear_removed(&mut self) {
269        self.removed = false;
270    }

272    fn get_level(&self) -> Option<Arc<World>> {

274    }
275
276    fn load_additional(&mut self, nbt: &BorrowedNbtCompound<'_>) {
277        // Convert to NbtCompound view for accessing methods
278        let nbt_view: BorrowedNbtCompoundView<'_, '_> = nbt.into();
279
280        // Load front text
281        if let Some(front_nbt) = nbt_view.compound("front_text") {
282            self.front_text.load(front_nbt);
283        }
284
285        // Load back text
286        if let Some(back_nbt) = nbt_view.compound("back_text") {
287            self.back_text.load(back_nbt);
288        }
289
290        // Load waxed state
291        if let Some(waxed) = nbt_view.byte("is_waxed") {
292            self.is_waxed = waxed != 0;
293        }
294    }
295
296    fn save_additional(&self, nbt: &mut NbtCompound) {
297        // Save front text
298        let mut front_nbt = NbtCompound::new();
299        self.front_text.save(&mut front_nbt);
300        nbt.insert("front_text", front_nbt);
301
302        // Save back text
303        let mut back_nbt = NbtCompound::new();
304        self.back_text.save(&mut back_nbt);
305        nbt.insert("back_text", back_nbt);
306
307        // Save waxed state
308        nbt.insert("is_waxed", i8::from(self.is_waxed));
309    }
310
311    fn get_update_tag(&self) -> Option<NbtCompound> {
312        // Send full sign data to client
313        let mut nbt = NbtCompound::new();
314        self.save_additional(&mut nbt);
315        Some(nbt)
316    }
317
318    fn is_ticking(&self) -> bool {
319        // Signs tick to clear the edit lock if the player moves away
320        self.player_who_may_edit.is_some()
321    }
322
323    fn tick(&mut self, world: &Arc<World>) -> Option<BlockEntityTickAction> {
324        // Clear the edit lock if the editing player is too far away or gone
325        if let Some(editor_uuid) = self.player_who_may_edit {
326            let should_clear = world
327                .players
328                .get_by_uuid(&editor_uuid)
329                .is_none_or(|player| {
330                    let pos = self.pos;
331                    let player_pos = player.position();
332                    let dx = player_pos.x - f64::from(pos.0.x) - 0.5;
333                    let dy = player_pos.y - f64::from(pos.0.y) - 0.5;
334                    let dz = player_pos.z - f64::from(pos.0.z) - 0.5;
335                    let distance_sq = dx * dx + dy * dy + dz * dz;
336                    distance_sq > MAX_EDIT_DISTANCE * MAX_EDIT_DISTANCE
337                });
338
339            if should_clear {
340                self.player_who_may_edit = None;
341            }
342        }
343        None
344    }
345}

347/// Converts a dye color to its string representation.
348const fn dye_color_to_str(color: DyeColor) -> &'static str {
349    match color {
350        DyeColor::White => "white",
351        DyeColor::Orange => "orange",
352        DyeColor::Magenta => "magenta",
353        DyeColor::LightBlue => "light_blue",
354        DyeColor::Yellow => "yellow",
355        DyeColor::Lime => "lime",
356        DyeColor::Pink => "pink",
357        DyeColor::Gray => "gray",
358        DyeColor::LightGray => "light_gray",
359        DyeColor::Cyan => "cyan",
360        DyeColor::Purple => "purple",
361        DyeColor::Blue => "blue",
362        DyeColor::Brown => "brown",
363        DyeColor::Green => "green",
364        DyeColor::Red => "red",
365        DyeColor::Black => "black",
366    }
367}

369/// Parses a dye color from its string representation.
370fn dye_color_from_str(s: &str) -> DyeColor {
371    match s {
372        "white" => DyeColor::White,
373        "orange" => DyeColor::Orange,
374        "magenta" => DyeColor::Magenta,
375        "light_blue" => DyeColor::LightBlue,
376        "yellow" => DyeColor::Yellow,
377        "lime" => DyeColor::Lime,
378        "pink" => DyeColor::Pink,
379        "gray" => DyeColor::Gray,
380        "light_gray" => DyeColor::LightGray,
381        "cyan" => DyeColor::Cyan,
382        "purple" => DyeColor::Purple,
383        "blue" => DyeColor::Blue,
384        "brown" => DyeColor::Brown,
385        "green" => DyeColor::Green,
386        "red" => DyeColor::Red,
387        _ => DyeColor::Black,
388    }
