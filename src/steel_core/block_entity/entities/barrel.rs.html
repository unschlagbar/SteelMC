1//! Barrel block entity implementation.
2//!
3//! Barrels are container block entities with 27 slots (3x9 grid),
4//! functioning similarly to chests but without double-chest behavior.
5
6use std::any::Any;
7use std::sync::{Arc, Weak};
8
9use simdnbt::ToNbtTag;
10use simdnbt::borrow::{BaseNbtCompound as BorrowedNbtCompound, NbtCompound as NbtCompoundView};
11use simdnbt::owned::{NbtCompound, NbtList, NbtTag};
12use steel_registry::block_entity_type::BlockEntityTypeRef;
13use steel_registry::blocks::block_state_ext::BlockStateExt;
14use steel_registry::item_stack::ItemStack;
15use steel_registry::vanilla_block_entity_types;
16use steel_registry::vanilla_blocks;

18

24/// Number of slots in a barrel (3 rows of 9).
25pub const BARREL_SLOTS: usize = 27;
26
27/// Barrel block entity.
28///
29/// A simple container with 27 slots, using the same menu as chests.
30pub struct BarrelBlockEntity {
31    /// Weak reference to the world for marking chunks dirty.
32    level: Weak<World>,
33    /// Position in the world.
34    pos: BlockPos,
35    /// Current block state.
36    state: BlockStateId,
37    /// Whether this entity has been marked for removal.
38    removed: bool,
39    /// The 27 item slots.
40    items: Vec<ItemStack>,
41}
42
43impl BarrelBlockEntity {
44    /// Creates a new barrel block entity.
45    #[must_use]
46    pub fn new(level: Weak<World>, pos: BlockPos, state: BlockStateId) -> Self {
47        Self {
48            level,
49            pos,
50            state,
51            removed: false,
52            items: vec![ItemStack::empty(); BARREL_SLOTS],
53        }
54    }
55}
56
57impl BlockEntity for BarrelBlockEntity {
58    fn as_any(&self) -> &dyn Any {
59        self
60    }
61
62    fn as_any_mut(&mut self) -> &mut dyn Any {
63        self
64    }
65
66    fn get_type(&self) -> BlockEntityTypeRef {
67        &vanilla_block_entity_types::BARREL
68    }
69
70    fn get_block_pos(&self) -> BlockPos {
71        self.pos
72    }
73
74    fn get_block_state(&self) -> BlockStateId {
75        self.state
76    }
77
78    fn set_block_state(&mut self, state: BlockStateId) {
79        self.state = state;
80    }
81
82    fn is_removed(&self) -> bool {
83        self.removed
84    }
85
86    fn set_removed(&mut self) {
87        self.removed = true;
88    }
89
90    fn clear_removed(&mut self) {
91        self.removed = false;
92    }
93
94    fn get_level(&self) -> Option<Arc<World>> {
95        self.level.upgrade()
96    }
97
98    fn pre_remove_side_effects(&mut self, pos: BlockPos, _state: BlockStateId) {
99        // Drop all items when the barrel is broken
100        if let Some(world) = self.level.upgrade() {
101            for item in self.items.drain(..) {
102                world.drop_item_stack(pos, item);
103            }
104        }
105    }
106
107    fn load_additional(&mut self, nbt: &BorrowedNbtCompound<'_>) {
108        // Convert to NbtCompound view for accessing methods
109        let nbt_view: NbtCompoundView<'_, '_> = nbt.into();
110
111        // Load items from NBT using borrowed NBT for proper ItemStack parsing
112        if let Some(items_list) = nbt_view.list("Items")
113            && let Some(compounds) = items_list.compounds()
114        {
115            for compound in compounds {
116                // Each item has a "Slot" byte and item data
117                if let Some(slot) = compound.byte("Slot") {
118                    let slot = slot as usize;
119                    if slot < BARREL_SLOTS {
120                        // Parse item directly from the borrowed compound
121                        if let Some(item) = ItemStack::from_borrowed_compound(&compound) {
122                            self.items[slot] = item;
123                        }
124                    }
125                }
126            }
127        }
128    }
129
130    fn save_additional(&self, nbt: &mut NbtCompound) {
131        // Save items to NBT (only non-empty slots)
132        let mut items: Vec<NbtCompound> = Vec::new();
133        for (slot, item) in self.items.iter().enumerate() {
134            if !item.is_empty() {
135                // Use ItemStack's ToNbtTag implementation for proper component serialization
136                if let NbtTag::Compound(mut item_nbt) = item.clone().to_nbt_tag() {
137                    item_nbt.insert("Slot", slot as i8);
138                    items.push(item_nbt);
139                }
140            }
141        }
142        nbt.insert("Items", NbtList::Compound(items));
143    }
144
145    fn get_update_tag(&self) -> Option<NbtCompound> {
146        // Barrels don't need to send inventory to clients on chunk load
147        // (unlike signs which display text)
148        None
149    }
150
151    fn as_container(&self) -> Option<&(dyn Container + 'static)> {
152        Some(self)
153    }
154
155    fn as_container_mut(&mut self) -> Option<&mut (dyn Container + 'static)> {
156        Some(self)
157    }
158}
159
160impl Container for BarrelBlockEntity {
161    fn get_container_size(&self) -> usize {
162        BARREL_SLOTS
163    }
164
165    fn get_item(&self, slot: usize) -> &ItemStack {
166        &self.items[slot]
167    }
168
169    fn get_item_mut(&mut self, slot: usize) -> &mut ItemStack {
170        &mut self.items[slot]
171    }
172
173    fn set_item(&mut self, slot: usize, stack: ItemStack) {
174        if slot < BARREL_SLOTS {
175            self.items[slot] = stack;
176            self.set_changed();
177        }
178    }
179
180    fn get_max_stack_size(&self) -> i32 {
181        64
182    }
183
184    fn still_valid(&self, player: &Player) -> bool {
185        if self.removed {
186            return false;
187        }
188
189        let Some(level) = self.level.upgrade() else {
190            return false;
191        };
192
193        level.get_block_state(self.pos).get_block() == &vanilla_blocks::BARREL
194            && player.is_within_block_interaction_range_with_buffer(self.pos, 4.0)
195    }
196
197    fn set_changed(&mut self) {
198        BlockEntity::set_changed(self);
199    }
200}
