342fn homogeneous_packed_light_value(data: &[u8; DATA_LAYER_SIZE]) -> Option<u8> {
343    let first = data[0];
344    let value = first & 0x0F;
345    if first >> 4 != value {
346        return None;
347    }
348    data.iter().all(|byte| *byte == first).then_some(value)
349}

369/// Builder for creating a persistent chunk with its own palettes.
370struct ChunkBuilder<'a> {
371    block_states: Vec<PersistentBlockState>,
372    biomes: Vec<Identifier>,
373    registry: &'a Registry,
374}
375
376impl<'a> ChunkBuilder<'a> {
377    const fn new(registry: &'a Registry) -> Self {
378        Self {
379            block_states: Vec::new(),
380            biomes: Vec::new(),
381            registry,
382        }
383    }
384
385    /// Ensures a block state exists in the chunk's palette, returning its index.
386    fn ensure_block_state(&mut self, block_id: BlockStateId) -> u16 {
387        // Get block and properties from registry
388        let block = self
389            .registry
390            .blocks
391            .by_state_id(block_id)
392            .expect("Invalid block state ID");
393        let properties = self.registry.blocks.get_properties(block_id);
394
395        let persistent = PersistentBlockState {
396            name: block.key.clone(),
397            properties,
398        };
399
400        // Check if already exists
401        if let Some(idx) = self.block_states.iter().position(|s| s == &persistent) {
402            return idx as u16;
403        }
404
405        // Add new entry
406        let idx = self.block_states.len();
407        self.block_states.push(persistent);
408        idx as u16
409    }
410
411    /// Ensures a biome exists in the chunk's palette, returning its index.
412    fn ensure_biome(&mut self, biome_id: u16) -> u16 {
413        // Get biome identifier from registry
414        let biome = self
415            .registry
416            .biomes
417            .by_id(biome_id as usize)
418            .expect("Invalid biome ID");
419        let identifier = biome.key.clone();
420
421        if let Some(idx) = self.biomes.iter().position(|b| b == &identifier) {
422            return idx as u16;
423        }
424
425        let idx = self.biomes.len();
426        self.biomes.push(identifier);
427        idx as u16
428    }
429}

529    /// Saves a chunk to the appropriate region.
530    ///
531    /// The chunk is serialized, compressed, and written to disk immediately.
532    /// If the region was already open (has loaded chunks), the header update is
533    /// deferred. If this call opened the region, it will be closed after saving.
534    ///
535    /// If the chunk is not dirty and `force` is false, this is a no-op.
536    /// Returns `Ok(true)` if the chunk was saved.
537    /// Prepares chunk data for saving. Call this while holding the chunk lock,
538    /// then pass the result to `save_chunk_data` after releasing the lock.
539    #[must_use]
540    #[expect(
541        clippy::similar_names,
542        reason = "`pois` vs `pos` are semantically distinct"
543    )]
544    #[expect(
545        clippy::too_many_lines,
546        reason = "chunk save preparation keeps related serialization setup in one pass"
547    )]
548    pub fn prepare_chunk_save(
549        chunk: &ChunkAccess,
550        runtime_entities: &[SharedEntity],
551        force: bool,
552    ) -> Option<PreparedChunkSave> {
553        if !force && !chunk.is_dirty() {
554            return None;
555        }
556
557        // Finalize any sections still in worldgen Building mode. Proto chunks
558        // can be saved before being upgraded to `LevelChunk::from_proto`
559        // (which is where `recalculate_counts` normally runs and implicitly
560        // finalizes). Without this, `section_to_persistent` would panic on
561        // the Building variant.
562        for section_holder in &chunk.sections().sections {
563            let mut guard = section_holder.write();
564            if matches!(&guard.states, PalettedContainer::Building(_)) {
565                guard.recalculate_counts();
566            }
567        }
568
569        let pos = chunk.pos();
570
571        let block_entities = chunk.get_block_entities();
572
573        let mut seen_entity_ids = FxHashSet::default();
574        let mut seen_entity_uuids = FxHashSet::default();
575        let mut entities = Vec::new();
576        for entity in chunk.get_saveable_entities() {
577            if !Self::entity_position_is_finite(entity.as_ref()) {
578                Self::warn_skipping_non_finite_entity(entity.as_ref());
579                continue;
580            }
581            if seen_entity_ids.insert(entity.id()) {
582                Self::assert_unique_save_uuid(
583                    &mut seen_entity_uuids,
584                    entity.uuid(),
585                    entity.id(),
586                    pos,
587                );
588                entities.push(entity);
589            }
590        }
591        let mut handled_runtime_entity_ids = Vec::new();
592        for entity in runtime_entities {
593            handled_runtime_entity_ids.push(entity.id());
594            if !Self::entity_position_is_finite(entity.as_ref()) {
595                Self::warn_skipping_non_finite_entity(entity.as_ref());
596                continue;
597            }
598            if seen_entity_ids.insert(entity.id()) {
599                Self::assert_unique_save_uuid(
600                    &mut seen_entity_uuids,
601                    entity.uuid(),
602                    entity.id(),
603                    pos,
604                );
605                entities.push(Arc::clone(entity));
606            }
607        }
608
609        // Serialize scheduled ticks
610        let (block_ticks, fluid_ticks) = match chunk {
611            ChunkAccess::Full(c) => {
612                let bt = Self::block_ticks_to_persistent(&c.block_ticks.lock(), pos);
613                let ft = Self::fluid_ticks_to_persistent(&c.fluid_ticks.lock(), pos);
614                (bt, ft)
615            }
616            ChunkAccess::Proto(c) => {
617                let bt = Self::block_ticks_to_persistent(&c.block_ticks.lock(), pos);
618                let ft = Self::fluid_ticks_to_persistent(&c.fluid_ticks.lock(), pos);
619                (bt, ft)
620            }
621            ChunkAccess::Unloaded => unreachable!(),
622        };
623
624        // Serialize heightmaps
625        let heightmaps = chunk
626            .as_full()
627            .map(|c| Self::heightmaps_to_persistent(&c.heightmaps.read()))
628            .unwrap_or_default();
629
630        let light = match chunk {
631            ChunkAccess::Full(c) => Self::light_to_persistent(&c.light.read()),
632            ChunkAccess::Proto(c) => Self::light_to_persistent(&c.light.read()),
633            ChunkAccess::Unloaded => unreachable!(),
634        };
635
636        // Serialize structure data (works for both proto and full chunks)
637        let structure_starts = Self::structure_starts_to_persistent(&chunk.structure_starts());
638        let structure_references =
639            Self::structure_references_to_persistent(&chunk.structure_references());
640
641        // Collect POI occupancy data from world storage
642        let pois = chunk
643            .as_full()

659                proto.postprocessing.read().iter().map(Vec::clone).collect()
660            }
661            ChunkAccess::Full(_) => Vec::new(),

663        };

666            chunk.sections(),
667            &block_entities,

672            light,
673            carving_mask,
674            postprocessing,

676            structure_references,

679        );

857        layer: &mut ChunkLightLayerStorage,
858        persistent: &[PersistentLightSection],
859        layer_name: &str,
860    ) {
861        for section in persistent {
862            let Ok(section_index) = usize::try_from(section.section_index()) else {
863                tracing::warn!(
864                    layer = layer_name,
865                    section_index = section.section_index(),
866                    "Persisted light section index does not fit this platform"
867                );
868                continue;
869            };
870
871            let Some(target) = layer.sections_mut().get_mut(section_index) else {
872                tracing::warn!(
873                    layer = layer_name,
874                    section_index,
875                    "Persisted light section index is outside world light range"
876                );
877                continue;
878            };
879
880            let Some(restored) = Self::persistent_to_light_section(section, layer_name) else {
881                continue;
882            };
883            *target = restored;
884        }

1142    /// Converts a runtime section to persistent format.
1143    fn section_to_persistent(
1144        section: &SectionHolder,
1145        builder: &mut ChunkBuilder,
1146    ) -> PersistentSection {
1147        let section = section.read();
1148        let biomes = Self::biomes_to_persistent(&section.biomes, builder);
1149
1150        match &section.states {
1151            PalettedContainer::Homogeneous(block_id) => {
1152                let block_idx = builder.ensure_block_state(*block_id);
1153                PersistentSection::Homogeneous {
1154                    block_state: block_idx,
1155                    biomes,
1156                }
1157            }
1158            PalettedContainer::Heterogeneous(data) => {
1159                // Build section-local palette (indices into chunk's block_states)
1160                let palette: Vec<u16> = data
1161                    .palette
1162                    .iter()
1163                    .map(|(block_id, _)| builder.ensure_block_state(*block_id))
1164                    .collect();
1165
1166                // Pack block indices (indices into section-local palette)
1167                let bits = bits_for_palette_len(palette.len())
1168                    .expect("Heterogeneous section should have palette length >= 2");
1169                let indices: Vec<u32> = data
1170                    .cube
1171                    .iter()
1172                    .flatten()
1173                    .flatten()
1174                    .map(|block_id| {
1175                        data.palette
1176                            .iter()
1177                            .position(|(v, _)| v == block_id)
1178                            .unwrap_or(0) as u32
1179                    })
1180                    .collect();
1181
1182                let block_data = pack_indices(&indices, bits);
1183
1184                PersistentSection::Heterogeneous {
1185                    palette,
1186                    bits_per_entry: bits,
1187                    block_data,
1188                    biomes,
1189                }
1190            }
1191            PalettedContainer::Building(_) => panic!(
1192                "section_to_persistent called on a section still in worldgen Building mode; \
1193                 finalize_building must be called before serialization"
1194            ),
1195        }
1196    }

1246    /// Converts a persistent chunk to runtime format.
1247    /// The returned chunk is not dirty (freshly loaded from disk).
1248    ///
1249    /// # Arguments
1250    /// * `persistent` - The persistent chunk data
1251    /// * `pos` - The chunk position
1252    /// * `status` - The chunk status
1253    /// * `min_y` - The minimum Y coordinate of the world
1254    /// * `height` - The total height of the world
1255    /// * `level` - Weak reference to the world for `LevelChunk`
1256    #[expect(
1257        clippy::too_many_lines,
1258        reason = "chunk persistence conversion is a linear field-by-field transform"
1259    )]
1260    pub(crate) fn persistent_to_chunk(
1261        persistent: &PersistentChunk,
1262        pos: ChunkPos,
1263        status: ChunkStatus,
1264        min_y: i32,
1265        height: i32,
1266        level: Weak<World>,
1267    ) -> LoadedChunk {
1268        let sections: Vec<ChunkSection> = persistent
1269            .sections
1270            .iter()
1271            .map(|section| Self::persistent_to_section(section, persistent))
1272            .collect();
1273
1274        // Reconstruct structure data
1275        let structure_starts = Self::persistent_to_structure_starts(&persistent.structure_starts);
1276        let structure_references =
1277            Self::persistent_to_structure_references(&persistent.structure_references);
1278        let light = Self::persistent_to_light(&persistent.light, min_y, height, status);
1279
1280        if status == ChunkStatus::Full {
1281            // Reconstruct scheduled ticks from persistent data
1282            let block_ticks = Self::persistent_to_block_ticks(&persistent.block_ticks, pos);
1283            let fluid_ticks = Self::persistent_to_fluid_ticks(&persistent.fluid_ticks, pos);
1284
1285            // Reconstruct heightmaps from persistent data
1286            let heightmaps = Self::persistent_to_heightmaps(&persistent.heightmaps, min_y, height);
1287
1288            let chunk = LevelChunk::from_disk(
1289                Sections::from_owned(sections.into_boxed_slice()),
1290                pos,
1291                min_y,
1292                height,
1293                level.clone(),
1294                block_ticks,
1295                fluid_ticks,
1296                heightmaps,
1297                structure_starts,
1298                structure_references,
1299                light,
1300            );
1301
1302            // Load block entities
1303            for persistent_be in &persistent.block_entities {
1304                if let Some(block_entity) =
1305                    Self::persistent_to_block_entity(persistent_be, pos, &chunk)
1306                {
1307                    chunk.add_and_register_block_entity(block_entity);
1308                }
1309            }
1310
1311            let mut pending_entities = Vec::with_capacity(persistent.entities.len());
1312            let level_weak = chunk.level_weak();
1313            for persistent_entity in &persistent.entities {
1314                let mut loaded_entities =
1315                    Self::persistent_to_entity_tree_at_level(persistent_entity, pos, &level_weak);
1316                pending_entities.append(&mut loaded_entities);
1317            }
1318
1319            // Restore POI ticket state (populate_poi ran in from_disk, now apply saved occupancy)
1320            if !persistent.pois.is_empty()
1321                && let Some(world) = level.upgrade()
1322            {
1323                let tickets: Vec<_> = persistent
1324                    .pois
1325                    .iter()
1326                    .map(|p| {
1327                        let block_pos = BlockPos::new(
1328                            pos.0.x * 16 + i32::from(p.x),
1329                            i32::from(p.y),
1330                            pos.0.y * 16 + i32::from(p.z),
1331                        );
1332                        (block_pos, p.free_tickets)
1333                    })
1334                    .collect();
1335                world.poi_storage.lock().restore_tickets(pos, &tickets);
1336            }
1337
1338            // Clear dirty flag since we just loaded (add_and_register marks dirty)
1339            chunk.dirty.store(false, Ordering::Release);
1340
1341            LoadedChunk {
1342                chunk: ChunkAccess::Full(chunk),
1343                status,
1344                pending_entities,
1345            }
1346        } else {
1347            let block_ticks = Self::persistent_to_block_ticks(&persistent.block_ticks, pos);
1348            let fluid_ticks = Self::persistent_to_fluid_ticks(&persistent.fluid_ticks, pos);
1349            let carving_mask = persistent
1350                .carving_mask
1351                .as_deref()
1352                .map(|packed| CarvingMask::from_packed_u64s(height, min_y, packed));
1353
1354            let chunk = ProtoChunk::from_disk(
1355                Sections::from_owned(sections.into_boxed_slice()),
1356                pos,
1357                status,
1358                min_y,
1359                height,
1360                structure_starts,
1361                structure_references,
1362                carving_mask,
1363                persistent.postprocessing.iter().map(Vec::clone).collect(),
1364                block_ticks,
1365                fluid_ticks,
1366                level.clone(),
1367                light,
1368            );
1369
1370            for persistent_be in &persistent.block_entities {
1371                let block_entity_pos = Self::persistent_block_entity_pos(persistent_be, pos);
1372                let state = chunk.get_block_state(block_entity_pos);
1373                if let Some(block_entity) = Self::persistent_to_block_entity_at(
1374                    persistent_be,
1375                    block_entity_pos,
1376                    level.clone(),
1377                    state,
1378                ) {
1379                    chunk.add_and_register_block_entity(block_entity);
1380                }
1381            }
1382
1383            for persistent_entity in &persistent.entities {
1384                let loaded_entities =
1385                    Self::persistent_to_entity_tree_at_level(persistent_entity, pos, &level);
1386                for entity in loaded_entities {
1387                    chunk.add_entity(entity);
1388                }
1389            }
1390
1391            chunk.dirty.store(false, Ordering::Release);
1392
1393            LoadedChunk {
1394                chunk: ChunkAccess::Proto(chunk),
1395                status,
1396                pending_entities: Vec::new(),
1397            }
1398        }
1399    }

1496    /// Converts one persistent entity to runtime format without loading passengers.
1497    fn persistent_to_entity_at_level(
1498        persistent: &PersistentEntity,
1499        chunk_pos: ChunkPos,
1500        level: &Weak<World>,
1501    ) -> Option<SharedEntity> {
1502        use uuid::Uuid;
1503
1504        // Reconstruct base fields
1505        let stored_pos = DVec3::new(persistent.pos[0], persistent.pos[1], persistent.pos[2]);
1506        let mut velocity = DVec3::new(
1507            persistent.motion[0],
1508            persistent.motion[1],
1509            persistent.motion[2],
1510        );
1511        let rotation = (persistent.rotation[0], persistent.rotation[1]);
1512        let uuid = Uuid::from_bytes(persistent.uuid);
1513
1514        // Validate position is finite
1515        if !stored_pos.x.is_finite() || !stored_pos.y.is_finite() || !stored_pos.z.is_finite() {
1516            tracing::warn!(
1517                ?uuid,
1518                "Entity has non-finite position {:?}, skipping load",
1519                stored_pos
1520            );
1521            return None;
1522        }
1523
1524        if !rotation.0.is_finite() || !rotation.1.is_finite() {
1525            tracing::warn!(
1526                ?uuid,
1527                "Entity has non-finite rotation {rotation:?}, skipping load"
1528            );
1529            return None;
1530        }
1531
1532        let pos = Self::clamp_loaded_entity_position(stored_pos);
1533
1534        // Validate position is within expected chunk (sanity check)
1535        let expected_chunk = ChunkPos::from_entity_pos(pos);
1536        if chunk_pos != expected_chunk {
1537            tracing::warn!(
1538                ?uuid,
1539                "Entity position {:?} doesn't match chunk {:?}, loading anyway",
1540                pos,
1541                chunk_pos
1542            );
1543        }
1544
1545        // Clamp motion values > 10.0 to 0 (vanilla behavior to prevent corruption)
1546        if velocity.x.abs() > 10.0 {
1547            velocity.x = 0.0;
1548        }
1549        if velocity.y.abs() > 10.0 {
1550            velocity.y = 0.0;
1551        }
1552        if velocity.z.abs() > 10.0 {
1553            velocity.z = 0.0;
1554        }
1555
1556        // Look up entity type
1557        let entity_type = REGISTRY.entity_types.by_key(&persistent.entity_type)?;
1558        let save_data = Self::save_data_from_persistent(persistent, uuid);
1559
1560        // Parse NBT from bytes (or use empty compound data)
1561        let nbt_bytes = if persistent.nbt_data.is_empty() {
1562            // Empty compound body for `simdnbt::borrow::read_compound`.
1563            &[0x00][..]
1564        } else {
1565            &persistent.nbt_data[..]
1566        };
1567
1568        let Ok(nbt) = read_borrowed_compound(&mut Cursor::new(nbt_bytes)) else {
1569            tracing::warn!(?uuid, "Failed to parse entity NBT, skipping");
1570            return None;
1571        };
1572
1573        Some(ENTITIES.create_and_load_or_raw(
1574            EntityLoadRequest {
1575                entity_type,
1576                position: pos,
1577                uuid,
1578                velocity,
1579                rotation,
1580                fall_distance: persistent.fall_distance,
1581                fire_freeze: EntityFireFreezeState::from_parts(
1582                    persistent.remaining_fire_ticks,
1583                    persistent.ticks_frozen,
1584                    persistent.is_in_powder_snow,
1585                    persistent.was_in_powder_snow,
1586                    persistent.has_visual_fire,
1587                ),
1588                on_ground: persistent.on_ground,
1589                save_data,
1590                world: Weak::clone(level),
1591            },
1592            &nbt,
1593        ))
1594    }

2800        chunk: &PersistentChunk,

2802        match persistent {
2803            PersistentSection::Homogeneous {
2804                block_state,

2818                let runtime_palette: Vec<BlockStateId> = palette
2819                    .iter()
2820                    .map(|&idx| Self::resolve_block_state(chunk, idx))
2821                    .collect();
2822                let mut cube = Box::new([[[BlockStateId(0); 16]; 16]; 16]);
2823                for plane in cube.iter_mut() {
2824                    for row in plane {
2825                        for cell in row {
2826                            *cell = runtime_palette[indices.next().expect(
2827                                "this should never fail, we know the iterator is long enough",
2828                            ) as usize];
2829                        }
2830                    }
2831                }
2832                let states = PalettedContainer::from_cube(cube);
2833                let biome_data = Self::persistent_to_biomes(biomes, chunk);
2834                ChunkSection::new_with_biomes(states, biome_data)
