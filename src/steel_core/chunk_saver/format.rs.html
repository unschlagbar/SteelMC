1//! Data structures for the chunk persistence format.
2//!
3//! ## Format Overview
4//!
5//! Region files use a sector-based format with a fixed header for fast random access:
6//!
7//! ```text
8//! ┌─────────────────────────────────────────────────────┐
9//! │ Magic (4 bytes): "STLR"                             │
10//! │ Version (2 bytes): u16                              │
11//! │ Padding (2 bytes): reserved                         │
12//! ├─────────────────────────────────────────────────────┤
13//! │ Header: 1024 entries × 8 bytes = 8KB                │
14//! │   Each entry: offset (u32) + size (u24) + flags (u8)│
15//! ├─────────────────────────────────────────────────────┤
16//! │ Chunk data in 4KB sectors                           │
17//! │   [chunk data padded to 4KB boundary]               │
18//! │   [chunk data padded to 4KB boundary]               │
19//! │   ...                                               │
20//! └─────────────────────────────────────────────────────┘
21//! ```
22//!
23//! ## Design
24//!
25//! Each chunk stores its own block state and biome palettes, making chunks
26//! self-contained and avoiding expensive region-wide table rebuilds.
27//!
28//! Block data uses power-of-2 bit packing (1, 2, 4, 8, 16 bits) to avoid entries
29//! spanning u64 boundaries.

61/// Number of chunks per region side (32×32 = 1024 chunks per region).
62pub const REGION_SIZE: usize = 32;

64/// Total chunks in a region.
65pub const CHUNKS_PER_REGION: usize = REGION_SIZE * REGION_SIZE;

67/// Number of blocks per section side (16×16×16 = 4096 blocks per section).
68pub const SECTION_SIZE: usize = 16;

70/// Total blocks in a section.
71pub const BLOCKS_PER_SECTION: usize = SECTION_SIZE * SECTION_SIZE * SECTION_SIZE;

73/// Number of biome cells per section side (4×4×4 = 64 biomes per section).
74pub const BIOME_SIZE: usize = 4;

76/// Total biome cells in a section.
77pub const BIOMES_PER_SECTION: usize = BIOME_SIZE * BIOME_SIZE * BIOME_SIZE;

79/// Sector size in bytes (4KB, matches modern disk physical sectors).
80pub const SECTOR_SIZE: usize = 4096;

82/// Size of the file header (magic + version + padding).
83pub const FILE_HEADER_SIZE: usize = 8;

85/// Size of the chunk location table (1024 entries × 8 bytes).
86pub const CHUNK_TABLE_SIZE: usize = CHUNKS_PER_REGION * 8;

91/// First sector where chunk data can be stored.
92/// Header takes `ceil(TOTAL_HEADER_SIZE` / `SECTOR_SIZE`) = 3 sectors (8 + 8192 = 8200 bytes).
93pub const FIRST_DATA_SECTOR: u32 = 3;
94
95/// Maximum chunk size in bytes (16MB - should be plenty).
96pub const MAX_CHUNK_SIZE: usize = 16 * 1024 * 1024;

98/// Entry in the chunk location table.
99///
100/// Layout (8 bytes total):
101/// - offset: u32 - sector offset (0 = chunk doesn't exist)
102/// - size: u24 - compressed size in bytes
103/// - flags: u8 - status and flags
104#[derive(Clone, Copy)]
105pub struct ChunkEntry {
106    /// Sector offset from start of file. 0 means chunk doesn't exist.
107    /// Multiply by `SECTOR_SIZE` to get byte offset.
108    pub sector_offset: u32,
109    /// Size of compressed chunk data in bytes (stored as u24, max ~16MB).
110    pub size_bytes: u32,
111    /// Chunk status (generation state).
112    pub status: ChunkStatus,
113}
114
115impl ChunkEntry {
116    /// Creates a new chunk entry.
117    #[must_use]
118    pub const fn new(sector_offset: u32, size_bytes: u32, status: ChunkStatus) -> Self {
119        Self {
120            sector_offset,
121            size_bytes,
122            status,
123        }
124    }
125
126    /// Returns true if this entry represents an existing chunk.
127    #[must_use]
128    pub const fn exists(&self) -> bool {
129        self.sector_offset != 0
130    }
131
132    /// Creates an empty/non-existent chunk entry.
133    #[must_use]
134    pub const fn empty() -> Self {
135        Self {
136            sector_offset: 0,
137            size_bytes: 0,
138            status: ChunkStatus::Empty,
139        }
140    }
141
142    /// Calculates the number of sectors this chunk occupies.
143    #[must_use]
144    pub const fn sector_count(&self) -> u32 {
145        if self.size_bytes == 0 {
146            0
147        } else {
148            (self.size_bytes as usize).div_ceil(SECTOR_SIZE) as u32
149        }
150    }
151
152    /// Serializes to 8 bytes: [offset: 4][size: 3][flags: 1]
153    #[must_use]
154    pub const fn to_bytes(self) -> [u8; 8] {
155        let offset_bytes = self.sector_offset.to_le_bytes();
156        let size_bytes = self.size_bytes.to_le_bytes();
157        let flags = self.status.get_index() as u8;
158        [
159            offset_bytes[0],
160            offset_bytes[1],
161            offset_bytes[2],
162            offset_bytes[3],
163            size_bytes[0],
164            size_bytes[1],
165            size_bytes[2],
166            flags,
167        ]
168    }
169
170    /// Deserializes from 8 bytes.
171    #[must_use]
172    pub fn from_bytes(bytes: [u8; 8]) -> Self {
173        let sector_offset = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
174        let size_bytes = u32::from_le_bytes([bytes[4], bytes[5], bytes[6], 0]);
175        let status = ChunkStatus::from_index(bytes[7] as usize).unwrap_or(ChunkStatus::Empty);
176        Self {
177            sector_offset,
178            size_bytes,
179            status,
180        }
181    }
182}
183
184impl Default for ChunkEntry {
185    fn default() -> Self {
186        Self::empty()
187    }
188}

190/// Region header containing chunk location table.
191pub struct RegionHeader {
192    /// Chunk entries (1024 = 32×32).
193    pub entries: Box<[ChunkEntry; CHUNKS_PER_REGION]>,
194}
195
196impl RegionHeader {
197    /// Creates an empty header with no chunks.
198    #[must_use]
199    pub fn new() -> Self {
200        Self {
201            entries: Box::new([ChunkEntry::default(); CHUNKS_PER_REGION]),
202        }
203    }
204
205    /// Gets the local index for a chunk position within this region.
206    #[must_use]
207    pub const fn chunk_index(local_x: usize, local_z: usize) -> usize {
208        debug_assert!(local_x < REGION_SIZE);
209        debug_assert!(local_z < REGION_SIZE);
210        local_z * REGION_SIZE + local_x
211    }
212
213    /// Converts a chunk index back to local coordinates.
214    #[must_use]
215    pub const fn index_to_local(index: usize) -> (usize, usize) {
216        debug_assert!(index < CHUNKS_PER_REGION);
217        (index % REGION_SIZE, index / REGION_SIZE)
218    }
219
220    /// Serializes the header to bytes.
221    #[must_use]
222    pub fn to_bytes(&self) -> Vec<u8> {
223        let mut bytes = Vec::with_capacity(CHUNK_TABLE_SIZE);
224        for entry in self.entries.iter() {
225            bytes.extend_from_slice(&entry.to_bytes());
226        }
227        bytes
228    }
229
230    /// Deserializes the header from bytes.
231    ///
232    /// # Panics
233    /// Panics if bytes length is not exactly `CHUNK_TABLE_SIZE`.
234    #[must_use]
235    pub fn from_bytes(bytes: &[u8]) -> Self {
236        assert_eq!(bytes.len(), CHUNK_TABLE_SIZE);
237        let mut entries = Box::new([ChunkEntry::default(); CHUNKS_PER_REGION]);
238        for (i, chunk) in bytes.chunks_exact(8).enumerate() {
239            entries[i] = ChunkEntry::from_bytes(chunk.try_into().expect("chunk entry is 8 bytes"));
240        }
241        Self { entries }
242    }
243
244    /// Finds a contiguous range of free sectors for allocation.
245    ///
246    /// Returns the starting sector offset, or `None` if no suitable range exists.
247    #[must_use]
248    pub fn find_free_sectors(&self, sectors_needed: u32, file_sectors: u32) -> u32 {
249        if sectors_needed == 0 {
250            return FIRST_DATA_SECTOR;
251        }
252
253        // Build a list of (start, end) ranges for used sectors
254        let mut used_ranges: Vec<(u32, u32)> = self
255            .entries
256            .iter()
257            .filter(|e| e.exists())
258            .map(|e| (e.sector_offset, e.sector_offset + e.sector_count()))
259            .collect();
260        used_ranges.sort_by_key(|r| r.0);
261
262        // Try to find a gap between used ranges
263        let mut current_sector = FIRST_DATA_SECTOR;
264        for (start, end) in used_ranges {
265            if start >= current_sector + sectors_needed {
266                // Found a gap
267                return current_sector;
268            }
269            current_sector = current_sector.max(end);
270        }
271
272        // No gap found, append at the end
273        current_sector.max(file_sectors)
274    }
275}
276
277impl Default for RegionHeader {
278    fn default() -> Self {
279        Self::new()
280    }
281}

313/// One persisted chunk-owned light section.
314#[derive(SchemaWrite, SchemaRead)]
315pub enum PersistentLightSection {
316    /// Present zero-filled section without backing bytes.
317    Uninitialized {
318        /// Index in the chunk's padded light-section array.
319        section_index: u32,
320    },
321    /// Visible initialized light bytes.
322    Initialized {
323        /// Index in the chunk's padded light-section array.
324        section_index: u32,
325        /// Packed 4-bit light values.
326        data: Vec<u8>,
327    },
328    /// Initialized bytes hidden from vanilla packet conversion.
329    Internal {
330        /// Index in the chunk's padded light-section array.
331        section_index: u32,
332        /// Packed 4-bit light values.
333        data: Vec<u8>,
334    },
335}
336
337impl PersistentLightSection {
338    /// Returns the padded light-section index.
339    #[must_use]
340    pub const fn section_index(&self) -> u32 {
341        match self {
342            Self::Uninitialized { section_index }
343            | Self::Initialized { section_index, .. }
344            | Self::Internal { section_index, .. } => *section_index,
345        }
346    }
347}

1153/// Position of a region in region coordinates.
1154#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
1155pub struct RegionPos {
1156    /// Region X coordinate (`chunk_x` / 32).
1157    pub x: i32,
1158    /// Region Z coordinate (`chunk_z` / 32).
1159    pub z: i32,
1160}
1161
1162impl RegionPos {
1163    /// Creates a new region position.
1164    #[must_use]
1165    pub const fn new(x: i32, z: i32) -> Self {
1166        Self { x, z }
1167    }
1168
1169    /// Converts a chunk position to a region position.
1170    #[must_use]
1171    pub const fn from_chunk(chunk_x: i32, chunk_z: i32) -> Self {
1172        Self {
1173            x: chunk_x.div_euclid(REGION_SIZE as i32),
1174            z: chunk_z.div_euclid(REGION_SIZE as i32),
1175        }
1176    }
1177
1178    /// Gets the local chunk coordinates within this region for a global chunk position.
1179    #[must_use]
1180    pub const fn local_chunk_pos(chunk_x: i32, chunk_z: i32) -> (usize, usize) {
1181        (
1182            chunk_x.rem_euclid(REGION_SIZE as i32) as usize,
1183            chunk_z.rem_euclid(REGION_SIZE as i32) as usize,
1184        )
1185    }
1186
1187    /// Returns the filename for this region (e.g., "r.0.-1.srg").
1188    #[must_use]
1189    pub fn filename(self) -> String {
1190        format!("r.{}.{}.srg", self.x, self.z)
1191    }
1192}
