1//! Shared logic for flowing fluids (Water, Lava).
2//!
3//! Provides the `FlowingFluid` trait, which contains the mathematical spread
4//! algorithms derived from vanilla's `FlowingFluid.java`. Individual fluids
5//! like `WaterFluid` and `LavaFluid` implement this trait to inherit behavior.

23/// Trait providing the base algorithm for flowing fluids (Water, Lava).
24/// In vanilla Minecraft, this is the `FlowingFluid` abstract class.
25pub trait FlowingFluid: FluidBehavior {
26    /// The base tick logic
27    fn base_tick(&self, world: &Arc<World>, pos: BlockPos) {
28        let mut current_fluid = world.get_block_state(pos).get_fluid_state();
29
30        if current_fluid.is_empty() || !self.is_same(current_fluid.fluid_id) {
31            return;
32        }
33
34        // TODO: animate_tick (ambient sounds, particles) belongs in a client-side
35        // ambient tick dispatcher (equivalent to Level.animateTick), not here.
36        // It should fire at render rate for nearby blocks, not per scheduled fluid tick.
37
38        if !current_fluid.is_source() {
39            let new_fluid = get_new_liquid(world, pos, self.fluid_type(), self.drop_off(world));
40
41            if new_fluid.is_empty() {
42                current_fluid = new_fluid;
43                // Vanilla: unconditionally sets Blocks.AIR when fluid empties
44                let air = REGISTRY.blocks.get_default_state_id(&vanilla_blocks::AIR);
45                world.set_block(pos, air, UpdateFlags::UPDATE_ALL);
46            } else if new_fluid != current_fluid {
47                let old_fluid = current_fluid;
48                current_fluid = new_fluid;
49                let existing_state = world.get_block_state(pos);
50                let block_state = fluid_state_to_block_with_existing(new_fluid, existing_state);
51                world.set_block(pos, block_state, UpdateFlags::UPDATE_ALL);
52
53                world.schedule_fluid_tick_default(
54                    pos,
55                    self.fluid_type(),
56                    self.get_spread_delay(world, pos, old_fluid, new_fluid),
57                );
58            }
59        }
60
61        self.spread(world, pos, current_fluid);
62    }
63
64    /// The base spread logic.
65    ///
66    /// Vanilla equivalent: `FlowingFluid.spread()`.
67    fn base_spread(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) {
68        if fluid_state.is_empty() {
69            return;
70        }
71
72        let below = pos.below();
73        let below_state = world.get_block_state(below);
74        let below_fluid = below_state.get_fluid_state();
75
76        // Vanilla: canMaybePassThrough (source check + canHoldAnyFluid + wall check)
77        //          + canBeReplacedWith + canHoldSpecificFluid
78        let can_spread_down = world.is_in_valid_bounds(below)
79            && !(self.is_same(below_fluid.fluid_id) && below_fluid.is_source())
80            && can_hold_any_fluid(world, below)
81            && can_pass_through_wall(world, pos, below, Direction::Down);
82
83        if can_spread_down {
84            let new_below_fluid =
85                get_new_liquid(world, below, self.fluid_type(), self.drop_off(world));
86
87            if !new_below_fluid.is_empty() {
88                let existing_behavior = FLUID_BEHAVIORS.get_behavior(below_fluid.fluid_id);
89                let can_replace = existing_behavior.can_be_replaced_with(
90                    below_fluid,
91                    world,
92                    below,
93                    new_below_fluid.fluid_id,
94                    Direction::Down,
95                );
96
97                if can_replace && can_hold_specific_fluid(below_state, new_below_fluid.fluid_id) {
98                    self.spread_to(world, below, new_below_fluid, Direction::Down);
99
100                    if self.source_neighbor_count(world, pos) >= 3 {
101                        self.spread_to_sides(world, pos, fluid_state);
102                    }
103                    return;
104                }
105            }
106        }
107
108        if fluid_state.is_source() || !is_hole(world, pos, self.fluid_type()) {
109            self.spread_to_sides(world, pos, fluid_state);
110        }
111    }
112
113    /// The base logic for placing a fluid into a specific adjacent block.
114    ///
115    /// Vanilla equivalent: `FlowingFluid.spreadTo()`.
116    /// Note: vanilla's spreadTo does NOT schedule ticks — that's handled by
117    /// LiquidBlockContainer.placeLiquid or the new block's onPlace callback.
118    fn base_spread_to(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) {
119        let target_state = world.get_block_state(pos);
120
121        if target_state
122            .try_get_value(&BlockStateProperties::WATERLOGGED)
123            .is_some()
124        {
125            let behavior = BLOCK_BEHAVIORS.get_behavior(target_state.get_block());
126            if behavior.place_liquid(world, pos, target_state, fluid_state) {
127                return;
128            }
129        }
130
131        // Non-LiquidBlockContainer path: destroy the block and place the raw fluid.
132        let target_block = target_state.get_block();
133        if !target_block.config.is_air {
134            self.before_destroying_block(world, pos, target_state);
135        }
136
137        let block_state = fluid_state_to_block(fluid_state);
138        // Vanilla uses flag 3 (UPDATE_ALL). Tick scheduling is handled by
139        // LiquidBlock.on_place which fires from set_block.
140        world.set_block(pos, block_state, UpdateFlags::UPDATE_ALL);
141    }
142
143    /// Performs the actual placement of fluid and schedules the tick.
144    fn spread_to(
145        &self,
146        world: &Arc<World>,
147        pos: BlockPos,
148        fluid_state: FluidState,
149        _direction: Direction,
150    ) {
151        self.base_spread_to(world, pos, fluid_state);
152    }
153
154    /// Returns the number of fluid sources in the 4-directional neighborhood of the given position.
155    fn source_neighbor_count(&self, world: &Arc<World>, pos: BlockPos) -> u8 {
156        let mut count = 0u8;
157        for dir in Direction::HORIZONTAL {
158            let neighbor = dir.relative(pos);
159            let f = world.get_block_state(neighbor).get_fluid_state();
160            if self.is_same(f.fluid_id) && f.is_source() {
161                count += 1;
162            }
163        }
164        count
165    }
166
167    /// Spreads the fluid to horizontal neighbors.
168    ///
169    /// Vanilla equivalent: `FlowingFluid.spreadToSides()`.
170    /// Computes outgoing amount, overrides to 7 for falling fluids, and skips
171    /// if the outgoing amount is zero.
172    fn spread_to_sides(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) {
173        // Vanilla: neighbor = amount - dropOff; if (falling) neighbor = 7; if (neighbor <= 0) skip
174        let mut neighbor = fluid_state.amount.saturating_sub(self.drop_off(world));
175        if fluid_state.falling {
176            neighbor = 7;
177        }
178        if neighbor == 0 {
179            return;
180        }
181
182        let spreads = get_spread(
183            world,
184            pos,
185            self.fluid_type(),
186            self.drop_off(world),
187            self.slope_find_distance(world),
188        );
189
190        for (direction, new_fluid) in spreads {
191            let target: BlockPos = direction.relative(pos);
192            self.spread_to(world, target, new_fluid, direction);
193        }
194    }
195}
