22/// Checks if fluid can pass through a wall between two positions.
23#[must_use]
24pub fn can_pass_through_wall(
25    world: &Arc<World>,
26    from: BlockPos,
27    to: BlockPos,
28    direction: Direction,
29) -> bool {
30    if !world.is_in_valid_bounds(to) {
31        return false;
32    }
33
34    let from_shape = world.get_block_state(from).get_collision_shape_at(from);
35    let to_shape = world.get_block_state(to).get_collision_shape_at(to);
36
37    !merged_offset_face_occludes(from_shape, to_shape, direction)
38}

40/// Checks if a block at the given world position can hold any fluid.
41///
42/// Vanilla equivalent: `FlowingFluid.canHoldAnyFluid(BlockState)`.
43#[must_use]
44pub fn can_hold_any_fluid(world: &Arc<World>, pos: BlockPos) -> bool {
45    let state = world.get_block_state(pos);
46    can_hold_any_fluid_state(state)
47}

49/// Checks if a block state can hold any fluid, without world access.
50///
51/// Vanilla equivalent: `FlowingFluid.canHoldAnyFluid(BlockState)`.
52/// Uses `blocksMotion()` check instead of `has_collision` — vanilla's
53/// `blocksMotion()` = `block != Cobweb && block != BambooSapling && isSolid()`.
54#[must_use]
55pub fn can_hold_any_fluid_state(state: BlockStateId) -> bool {
56    let block = state.get_block();
57
58    // Vanilla: block instanceof LiquidBlockContainer → true
59    // Our equivalent: waterloggable blocks always accept fluid.
60    if state
61        .try_get_value(&BlockStateProperties::WATERLOGGED)
62        .is_some()
63    {
64        return true;
65    }
66
67    // Vanilla: state.blocksMotion() ? false : !(exclusion list)
68    if state.blocks_motion() {
69        return false;
70    }
71
72    // Non-solid blocks that still reject fluid.
73    !is_fluid_excluded_block(block)
74}
75
76/// Returns true if a block is in the vanilla fluid exclusion list.
77fn is_fluid_excluded_block(block: BlockRef) -> bool {
78    block == &vanilla_blocks::LADDER
79        || block == &vanilla_blocks::SUGAR_CANE
80        || block == &vanilla_blocks::BUBBLE_COLUMN
81        || block == &vanilla_blocks::NETHER_PORTAL
82        || block == &vanilla_blocks::END_PORTAL
83        || block == &vanilla_blocks::END_GATEWAY
84        || block == &vanilla_blocks::STRUCTURE_VOID
85        || block.has_tag(&BlockTag::SIGNS)
86        || block.has_tag(&BlockTag::DOORS)
87}

89/// Vanilla equivalent: `FlowingFluid.canHoldSpecificFluid(BlockGetter, BlockPos, BlockState, Fluid)`.
90///
91/// For `LiquidBlockContainer` blocks (blocks with WATERLOGGED), delegates to
92/// `canPlaceLiquid(null, ...)`. For other blocks, always returns true.
93#[must_use]
94pub fn can_hold_specific_fluid(state: BlockStateId, fluid: FluidRef) -> bool {
95    if state
96        .try_get_value(&BlockStateProperties::WATERLOGGED)
97        .is_some()
98    {
99        let behavior = BLOCK_BEHAVIORS.get_behavior(state.get_block());
100        behavior.can_place_liquid(state, fluid)
101    } else {
102        true
103    }
104}
105
106/// Vanilla equivalent: `FlowingFluid.canHoldFluid(BlockGetter, BlockPos, BlockState, Fluid)`.
107///
108/// Combined check: `canHoldAnyFluid(state) && canHoldSpecificFluid(state, fluid)`.
109#[must_use]
110pub fn can_hold_fluid(state: BlockStateId, fluid: FluidRef) -> bool {
111    can_hold_any_fluid_state(state) && can_hold_specific_fluid(state, fluid)
112}

114/// Checks if fluid can pass through to a position horizontally.
115///
116/// This is the world-querying entry point. It reads the block state at `pos`
117/// and delegates entirely to [`can_pass_horizontally_internal`]
118#[must_use]
119pub fn can_pass_horizontally(world: &Arc<World>, pos: BlockPos, target_fluid_id: FluidRef) -> bool {
120    if !world.is_in_valid_bounds(pos) {
121        return false;
122    }
123    let state = world.get_block_state(pos);
124    can_pass_horizontally_internal(state, target_fluid_id)
125}

127/// Core passability logic for horizontal fluid spread.
128///
129/// Vanilla equivalent: `!isSourceBlockOfThisType(testFluidState) && canHoldAnyFluid(testState)`.
130///
131/// Single source of truth used by both the world-querying
132/// [`can_pass_horizontally`] and [`SpreadContext`] (which supplies a
133/// cached `BlockStateId` to avoid redundant world lookups).
134#[must_use]
135pub fn can_pass_horizontally_internal(state: BlockStateId, target_fluid_id: FluidRef) -> bool {
136    // Vanilla: !isSourceBlockOfThisType — reject same-type source blocks
137    let fluid_state = state.get_fluid_state();
138    if fluid_state.fluid_id == target_fluid_id && fluid_state.is_source() {
139        return false;
140    }
141
142    // Vanilla: canHoldAnyFluid
143    can_hold_any_fluid_state(state)
144}
