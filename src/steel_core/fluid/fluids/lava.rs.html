31/// Lava fluid implementation.
32///
33/// Implements [`FluidBehavior`] with lava-specific parameters and
34/// behaviors (world-dependent spread, uphill delay, lava/water chemistry,
35/// fizz sounds).
36pub struct LavaFluid;
37
38impl LavaFluid {
39    /// Returns true if this world uses fast lava (nether-like).
40    fn is_fast_lava(world: &Arc<World>) -> bool {
41        world.dimension_type.fast_lava
42    }
43
44    /// Returns vanilla's lava current scale for entity fluid pushing.
45    pub(crate) fn entity_flow_scale(world: &Arc<World>) -> f64 {
46        if Self::is_fast_lava(world) {
47            FAST_LAVA_ENTITY_FLOW_SCALE
48        } else {
49            NORMAL_LAVA_ENTITY_FLOW_SCALE
50        }
51    }
52}
53
54impl FluidBehavior for LavaFluid {
55    fn fluid_type(&self) -> FluidRef {
56        lava_id()
57    }
58
59    fn is_same(&self, fluid: FluidRef) -> bool {
60        is_lava_fluid(fluid)
61    }
62
63    fn tick_delay(&self, world: &Arc<World>) -> i32 {
64        if Self::is_fast_lava(world) { 10 } else { 30 }
65    }
66
67    fn drop_off(&self, world: &Arc<World>) -> u8 {
68        if Self::is_fast_lava(world) { 1 } else { 2 }
69    }
70
71    fn slope_find_distance(&self, world: &Arc<World>) -> u8 {
72        if Self::is_fast_lava(world) { 4 } else { 2 }
73    }
74
75    fn explosion_resistance(&self) -> f32 {
76        100.0
77    }
78
79    /// Lava is randomly ticking for fire spread.
80    fn is_randomly_ticking(&self) -> bool {
81        true
82    }
83
84    fn can_convert_to_source(&self, world: &Arc<World>) -> bool {
85        match world.get_game_rule(&LAVA_SOURCE_CONVERSION) {
86            GameRuleValue::Bool(val) => val,
87            GameRuleValue::Int(_) => false,
88        }
89    }
90
91    fn get_flow(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) -> DVec3 {
92        flowing_fluid_flow(world, pos, fluid_state)
93    }
94
95    /// Vanilla parity: `LavaFluid.canBeReplacedWith()`.
96    /// Lava can be replaced if its effective height >= 0.444 and the replacer is water.
97    ///
98    /// Uses `get_height()` (not raw `amount / 9.0`) so that falling lava
99    /// (same fluid directly above → height = 1.0) is also correctly handled.
100    fn can_be_replaced_with(
101        &self,
102        fluid_state: FluidState,
103        world: &Arc<World>,
104        pos: BlockPos,
105        other_fluid: FluidRef,
106        _direction: Direction,
107    ) -> bool {
108        get_height(world, pos, fluid_state) >= 0.444_444_45 && is_water_fluid(other_fluid)
109    }
110
111    /// Vanilla parity: `LavaFluid.getSpreadDelay`.
112    /// Uphill lava spreads 4× slower with 3/4 probability.
113    ///
114    /// "Uphill" means the target position (`new_state`) has a greater effective
115    /// height than the source (`old_state`). Uses `get_height()` so that a
116    /// falling lava source (height = 1.0) is correctly treated as "tall".
117    fn get_spread_delay(
118        &self,
119        world: &Arc<World>,
120        pos: BlockPos,
121        old_state: FluidState,
122        new_state: FluidState,
123    ) -> i32 {
124        let base = self.tick_delay(world);
125        if !old_state.is_empty()
126            && !new_state.is_empty()
127            && !old_state.falling
128            && !new_state.falling
129            && get_height(world, pos, new_state) > get_height(world, pos, old_state)
130            && rand::random_range(0u32..4) != 0
131        {
132            base * 4
133        } else {
134            base
135        }
136    }
137
138    /// Vanilla parity: `LavaFluid.beforeDestroyingBlock()` → fizz sound.
139    /// Lava does NOT drop block items (unlike water).
140    fn before_destroying_block(&self, world: &Arc<World>, pos: BlockPos, _state: BlockStateId) {
141        world.level_event(level_events::LAVA_FIZZ, pos, 0, None);
142    }
143
144    /// Vanilla parity: `LavaFluid.entityInside()` clears freezing, ignites, then applies lava damage.
145    fn entity_inside(
146        &self,
147        _world: &Arc<World>,
148        _pos: BlockPos,
149        _entity: &dyn Entity,
150        effect_collector: &mut InsideBlockEffectCollector,
151    ) {
152        effect_collector.apply(InsideBlockEffectType::ClearFreeze);
153        effect_collector.apply(InsideBlockEffectType::LavaIgnite);
154        effect_collector.run_after(
155            InsideBlockEffectType::LavaIgnite,
156            Box::new(|entity| entity.lava_hurt()),
157        );
158    }
159
160    /// Vanilla parity: `LavaFluid.animateTick()`.
161    /// Plays pop (1/100) and ambient (1/200) sounds when air is above.
162    fn animate_tick(&self, world: &Arc<World>, pos: BlockPos, _state: FluidState) {
163        let above_pos = pos.above();
164        let above_block = world.get_block_state(above_pos).get_block();
165
166        if above_block.config.is_air {
167            if rand::random_range(0u32..100) == 0 {
168                let volume: f32 = rand::random::<f32>() * 0.2 + 0.2;
169                let pitch: f32 = rand::random::<f32>() * 0.15 + 0.9;
170                world.play_block_sound(&sound_events::BLOCK_LAVA_POP, pos, volume, pitch, None);
171            }
172
173            if rand::random_range(0u32..200) == 0 {
174                let volume: f32 = rand::random::<f32>() * 0.2 + 0.2;
175                let pitch: f32 = rand::random::<f32>() * 0.15 + 0.9;
176                world.play_block_sound(&sound_events::BLOCK_LAVA_AMBIENT, pos, volume, pitch, None);
177            }
178        }
179    }
180
181    fn tick(&self, world: &Arc<World>, pos: BlockPos) {
182        self.base_tick(world, pos);
183    }
184
185    fn spread(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) {
186        self.base_spread(world, pos, fluid_state);
187    }
188}

190// Marker impl to provide base FlowingFluid logic
191impl FlowingFluid for LavaFluid {
192    fn spread_to(
193        &self,
194        world: &Arc<World>,
195        pos: BlockPos,
196        fluid_state: FluidState,
197        direction: Direction,
198    ) {
199        if direction == Direction::Down {
200            let below_fluid = get_fluid_state(world, pos);
201            if below_fluid.is_water() {
202                // Vanilla: fizz always plays when lava meets water going down,
203                // regardless of whether stone is formed.
204                world.level_event(level_events::LAVA_FIZZ, pos, 0, None);
205
206                // Vanilla: stone only forms when the target is a pure water LiquidBlock,
207                // not a waterlogged block (stairs, slabs, etc.).
208                let below_block = world.get_block_state(pos).get_block();
209                if below_block == &vanilla_blocks::WATER {
210                    world.set_block(
211                        pos,
212                        REGISTRY.blocks.get_default_state_id(&vanilla_blocks::STONE),
213                        UpdateFlags::UPDATE_ALL_IMMEDIATE,
214                    );
215                }
216                return;
217            }
218        }
219
220        self.base_spread_to(world, pos, fluid_state);
221    }
