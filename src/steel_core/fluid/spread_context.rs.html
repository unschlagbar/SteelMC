1//! Spread calculation context for fluid flow optimization.
2//!
3//! Based on vanilla's FlowingFluid.SpreadContext, this provides local caching
4//! of block states and hole checks during fluid spread calculations.
5//!
6//! This avoids repeatedly querying the world for the same positions during
7//! the recursive slope-finding algorithm.
8
9use std::sync::Arc;
10
11use rustc_hash::FxHashMap;
12use steel_registry::fluid::FluidRef;
13use steel_utils::BlockPos;
14use steel_utils::BlockStateId;
15
16use crate::fluid::collision::can_pass_horizontally_internal;
17use crate::fluid::is_hole;
18use crate::world::World;
19/// Context for fluid spread calculations with local caching.
20///
21/// This is created fresh for each `get_spread()` call and caches:
22/// - `BlockState` lookups by relative position
23/// - Hole check results by relative position
24pub(super) struct SpreadContext<'a> {
25    /// Cache for block states by encoded relative position
26    state_cache: FxHashMap<i16, BlockStateId>,
27    /// Cache for hole check results by encoded relative position
28    hole_cache: FxHashMap<i16, bool>,
29    /// Reference to world for cache misses
30    world: &'a Arc<World>,
31    /// The block from which spreading originates — used to compute relative cache keys.
32    origin: BlockPos,
33}
34
35impl<'a> SpreadContext<'a> {
36    /// Creates a new `SpreadContext` anchored at `origin`.
37    ///
38    /// `origin` must be the block that triggered the spread (the block passed to
39    /// `get_spread()`), matching vanilla's `new FlowingFluid.SpreadContext(level, blockPos)`.
40    #[must_use]
41    pub(super) fn new(world: &'a Arc<World>, origin: BlockPos) -> Self {
42        Self {
43            state_cache: FxHashMap::default(),
44            hole_cache: FxHashMap::default(),
45            world,
46            origin,
47        }
48    }
49
50    /// Encodes a world position into a short cache key relative to the spread origin.
51    fn encode_key(&self, pos: BlockPos) -> i16 {
52        // Positions in the slope-finding algorithm stay within slopeFindDistance (<=4)
53        // of the origin, so the difference always fits in i8.
54        let dx = (pos.0.x - self.origin.0.x) as i8;
55        let dz = (pos.0.z - self.origin.0.z) as i8;
56        ((i16::from(dx) + 128) << 8) | (i16::from(dz) + 128)
57    }
58
59    /// Gets the cached block state at the given position, querying the world if not cached.
60    #[must_use]
61    pub fn get_block_state(&mut self, pos: BlockPos) -> BlockStateId {
62        let key = self.encode_key(pos);
63        *self
64            .state_cache
65            .entry(key)
66            .or_insert_with(|| self.world.get_block_state(pos))
67    }
68
69    /// Checks if the position is a hole (can fluid flow down into it?), with caching.
70    #[must_use]
71    pub fn is_hole(&mut self, pos: BlockPos, fluid_id: FluidRef) -> bool {
72        let key = self.encode_key(pos);
73        *self
74            .hole_cache
75            .entry(key)
76            .or_insert_with(|| is_hole(self.world, pos, fluid_id))
77    }
78
79    /// Checks if fluid can pass horizontally to the given position.
80    ///
81    /// This uses the cached block state for efficiency.
82    #[must_use]
83    pub fn can_pass_horizontally(&mut self, pos: BlockPos, fluid_id: FluidRef) -> bool {
84        let state = self.get_block_state(pos);
85        can_pass_horizontally_internal(state, fluid_id)
86    }
87
88    /// Returns a reference to the world.
89    #[must_use]
90    pub(super) const fn world(&self) -> &'a Arc<World> {
91        self.world
92    }
93}
