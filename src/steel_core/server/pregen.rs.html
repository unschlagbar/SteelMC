25const PREGEN_SIZE_ENV: &str = "PREGEN_SIZE";
26const VANILLA_PLAYER_SPAWN_SIZE_CHUNKS: i32 = 7;
27const PREGEN_WINDOW_SIZE: i32 = 32;
28const PREGEN_ACTIVE_WINDOWS: usize = 2;
29const PREGEN_UNLOAD_BACKPRESSURE_HIGH: usize = 8192;
30const PREGEN_UNLOAD_BACKPRESSURE_LOW: usize = 4096;
31const FULL_DEPENDENCY_RADIUS: i32 = GENERATION_PYRAMID
32    .get_step_to(ChunkStatus::Full)
33    .accumulated_dependencies
34    .get_radius_of(ChunkStatus::Empty) as i32;

36#[derive(Clone, Copy, Debug)]
37struct PregenWindow {
38    min_x: i32,
39    max_x: i32,
40    min_z: i32,
41    max_z: i32,
42}
43
44impl PregenWindow {
45    fn positions(self) -> Vec<ChunkPos> {
46        let mut positions = Vec::with_capacity(self.chunk_count());
47        for z in self.min_z..=self.max_z {
48            for x in self.min_x..=self.max_x {
49                positions.push(ChunkPos::new(x, z));
50            }
51        }
52        positions
53    }
54
55    const fn chunk_count(self) -> usize {
56        (self.width() * self.height()) as usize
57    }
58
59    const fn width(self) -> i32 {
60        self.max_x - self.min_x + 1
61    }
62
63    const fn height(self) -> i32 {
64        self.max_z - self.min_z + 1
65    }
66
67    const fn protected_rect(self) -> PregenRect {
68        PregenRect {
69            min_x: self.min_x - FULL_DEPENDENCY_RADIUS,
70            max_x: self.max_x + FULL_DEPENDENCY_RADIUS,
71            min_z: self.min_z - FULL_DEPENDENCY_RADIUS,
72            max_z: self.max_z + FULL_DEPENDENCY_RADIUS,
73        }
74    }
75}

77#[derive(Clone, Copy)]
78struct PregenRect {
79    min_x: i32,
80    max_x: i32,
81    min_z: i32,
82    max_z: i32,
83}
84
85impl PregenRect {
86    const fn overlaps(self, other: Self) -> bool {
87        self.min_x <= other.max_x
88            && self.max_x >= other.min_x
89            && self.min_z <= other.max_z
90            && self.max_z >= other.min_z
91    }
92}

94struct ActivePregenWindow {
95    window: PregenWindow,
96    handle: ChunkRequestHandle,
97    ready_chunks: usize,
98    ready: bool,
99    counted: bool,
100}
101
102impl ActivePregenWindow {
103    fn new(world: &Arc<World>, window: PregenWindow) -> Self {
104        let handle = world.chunk_map.request_chunks(ChunkRequest {
105            status: ChunkStatus::Full,
106            positions: window.positions(),
107            ticket_kind: ChunkTicketKind::Pregen,
108        });
109        Self {
110            window,
111            handle,
112            ready_chunks: 0,
113            ready: false,
114            counted: false,
115        }
116    }
117
118    fn poll(&mut self, world: &Arc<World>) {
119        match self.handle.poll() {
120            ChunkRequestState::Ready => {
121                self.ready_chunks = self.window.chunk_count();
122                self.ready = true;
123            }
124            ChunkRequestState::Pending { ready, .. } => {
125                self.ready_chunks = ready;
126            }
127            ChunkRequestState::Cancelled => {
128                self.handle = world.chunk_map.request_chunks(ChunkRequest {
129                    status: ChunkStatus::Full,
130                    positions: self.window.positions(),
131                    ticket_kind: ChunkTicketKind::Pregen,
132                });
133                self.ready_chunks = 0;
134                self.ready = false;
135                self.counted = false;
136            }
137        }
138    }
139}

141#[derive(Clone, Copy, Debug, PartialEq, Eq)]
142struct PregenSize {
143    side_length: i32,
144    radius: i32,
145}
146
147impl PregenSize {
148    fn from_side_length(side_length: i32) -> Result<Option<Self>, String> {
149        if side_length == 0 {
150            return Ok(None);
151        }
152        if side_length < 0 {
153            return Err(format!(
154                "{PREGEN_SIZE_ENV} must be 0 or a positive odd integer"
155            ));
156        }
157        if side_length % 2 == 0 {
158            return Err(format!(
159                "{PREGEN_SIZE_ENV} must be odd so the area has a single center chunk"
160            ));
161        }
162
163        Ok(Some(Self {
164            side_length,
165            radius: side_length / 2,
166        }))
167    }
168}

170impl Server {
171    /// Generates the startup spawn area for the server default world.
172    ///
173    /// Set `PREGEN_SIZE` to an odd chunk side length, or `0` to skip custom pregen.
174    pub async fn prepare_spawn_area(&self) -> bool {
175        let overworld = self.overworld();
176        let pregen_size = match get_pregen_size() {
177            Ok(Some(size)) => size,
178            Ok(None) => {
179                log::info!("Skipping custom startup spawn-area pregeneration");
180                return true;
181            }
182            Err(error) => {
183                log::error!("{error}");
184                return false;
185            }
186        };
187
188        let center_chunk = if pregen_size.side_length > VANILLA_PLAYER_SPAWN_SIZE_CHUNKS {
189            ChunkPos::new(0, 0)
190        } else {
191            let spawn_pos = overworld.level_data.read().data().spawn_pos();
192            ChunkPos::new(
193                SectionPos::block_to_section_coord(spawn_pos.0.x),
194                SectionPos::block_to_section_coord(spawn_pos.0.z),
195            )
196        };
197
198        pregen_overworld(overworld, center_chunk, pregen_size, &self.cancel_token).await
199    }
200}
201
202fn get_pregen_size() -> Result<Option<PregenSize>, String> {
203    let side_length = match env::var(PREGEN_SIZE_ENV) {
204        Ok(value) => value
205            .parse::<i32>()
206            .map_err(|e| format!("{PREGEN_SIZE_ENV} must be 0 or a positive odd integer: {e}"))?,
207        Err(env::VarError::NotPresent) => return Ok(None),
208        Err(env::VarError::NotUnicode(_)) => {
209            return Err(format!("{PREGEN_SIZE_ENV} must be valid unicode"));
210        }
211    };
212
213    PregenSize::from_side_length(side_length)
214}

216async fn pregen_overworld(
217    world: &Arc<World>,
218    center_chunk: ChunkPos,
219    pregen_size: PregenSize,
220    cancel_token: &CancellationToken,
221) -> bool {
222    let total_chunks = total_chunks(pregen_size.side_length);
223
224    log::info!(
225        "Preparing spawn area: {} chunks ({}x{}) around chunk ({}, {})",
226        total_chunks,
227        pregen_size.side_length,
228        pregen_size.side_length,
229        center_chunk.0.x,
230        center_chunk.0.y,
231    );
232
233    #[cfg(feature = "slow_chunk_gen")]
234    SLOW_CHUNK_GEN.store(true, Ordering::Relaxed);
235
236    let elapsed = {
237        let start = Instant::now();
238        let completed = generate_pregen(world, center_chunk, pregen_size, cancel_token).await;
239        (start.elapsed(), completed)
240    };
241
242    #[cfg(feature = "slow_chunk_gen")]
243    SLOW_CHUNK_GEN.store(false, Ordering::Relaxed);
244
245    let elapsed_secs = elapsed.0.as_secs_f64();
246    let chunks_per_second = if elapsed_secs > 0.0 {
247        total_chunks as f64 / elapsed_secs
248    } else {
249        0.0
250    };
251    if elapsed.1 {
252        log::info!(
253            "Spawn area prepared: {total_chunks} chunks in {elapsed_secs:.2}s ({chunks_per_second:.1} chunks/s)",
254        );
255    } else {
256        log::info!("Spawn area preparation cancelled after {elapsed_secs:.2}s");
257    }
258    elapsed.1
259}

261fn build_pregen_windows(center_chunk: ChunkPos, radius: i32) -> VecDeque<PregenWindow> {
262    let min_x = center_chunk.0.x - radius;
263    let max_x = center_chunk.0.x + radius;
264    let min_z = center_chunk.0.y - radius;
265    let max_z = center_chunk.0.y + radius;
266    let mut windows = VecDeque::new();
267
268    let mut z = min_z;
269    while z <= max_z {
270        let window_max_z = (z + PREGEN_WINDOW_SIZE - 1).min(max_z);
271        let mut x = min_x;
272        while x <= max_x {
273            let window_max_x = (x + PREGEN_WINDOW_SIZE - 1).min(max_x);
274            windows.push_back(PregenWindow {
275                min_x: x,
276                max_x: window_max_x,
277                min_z: z,
278                max_z: window_max_z,
279            });
280            x = window_max_x + 1;
281        }
282        z = window_max_z + 1;
283    }
284
285    windows
286}

288async fn generate_pregen(
289    world: &Arc<World>,
290    center_chunk: ChunkPos,
291    pregen_size: PregenSize,
292    cancel_token: &CancellationToken,
293) -> bool {
294    let total_chunks = total_chunks(pregen_size.side_length);
295    let mut pending_windows = build_pregen_windows(center_chunk, pregen_size.radius);
296    let mut active_windows = Vec::with_capacity(PREGEN_ACTIVE_WINDOWS + 1);
297    let mut last_report = Instant::now();
298    let mut last_completed = 0usize;
299    let mut completed = 0usize;
300    let mut unload_backpressure = false;
301    let start = Instant::now();
302
303    log::info!(
304        "Pregeneration windowing: {PREGEN_WINDOW_SIZE}x{PREGEN_WINDOW_SIZE} target chunks, {PREGEN_ACTIVE_WINDOWS} active windows, dependency halo {FULL_DEPENDENCY_RADIUS} chunks",
305    );
306
307    fill_active_windows(world, &mut pending_windows, &mut active_windows);
308
309    while completed < total_chunks {
310        if cancel_token.is_cancelled() {
311            release_all_windows(world, &mut active_windows);
312            return false;
313        }
314
315        drain_pregen_broadcasts(world);
316        world.chunk_map.tick_scheduling();
317        update_unload_backpressure(world, &mut unload_backpressure);
318
319        for active in &mut active_windows {
320            active.poll(world);
321        }
322
323        if !unload_backpressure {
324            let newly_ready_count = active_windows
325                .iter()
326                .filter(|active| active.ready && !active.counted)
327                .count();
328            for _ in 0..newly_ready_count {
329                activate_next_window(world, &mut pending_windows, &mut active_windows);
330            }
331        }
332
333        for active in &mut active_windows {
334            if active.ready && !active.counted {
335                completed += active.window.chunk_count();
336                active.counted = true;
337            }
338        }
339        if !unload_backpressure {
340            fill_active_windows(world, &mut pending_windows, &mut active_windows);
341        }
342        drain_pregen_broadcasts(world);
343        world.chunk_map.tick_scheduling();
344        update_unload_backpressure(world, &mut unload_backpressure);
345        release_unneeded_completed_windows(world, &mut active_windows);
346
347        if completed == total_chunks {
348            break;
349        }
350
351        if pregen_size.side_length > VANILLA_PLAYER_SPAWN_SIZE_CHUNKS
352            && last_report.elapsed() >= Duration::from_secs(5)
353        {
354            let report_elapsed = last_report.elapsed().as_secs_f64();
355            let ready_in_active = active_windows
356                .iter()
357                .filter(|active| !active.counted)
358                .map(|active| active.ready_chunks)
359                .sum::<usize>();
360            let current_completed = (completed + ready_in_active).min(total_chunks);
361            let elapsed = start.elapsed().as_secs_f64();
362            let chunks_per_sec = if elapsed > 0.0 {
363                (current_completed.saturating_sub(last_completed)) as f64 / report_elapsed
364            } else {
365                0.0
366            };
367            let percent = (current_completed as f64 / total_chunks as f64) * 100.0;
368            let remaining = total_chunks.saturating_sub(current_completed);
369            let eta = if chunks_per_sec > 0.0 && remaining > 0 {
370                remaining as f64 / chunks_per_sec
371            } else {
372                0.0
373            };
374            log::info!(
375                "Progress: {current_completed}/{total_chunks} ({percent:.1}%), {chunks_per_sec:.1} chunks/s, ETA: {eta:.0}s",
376            );
377            last_report = Instant::now();
378            last_completed = current_completed;
379        }
380
381        tokio::select! {
382            () = cancel_token.cancelled() => {
383                release_all_windows(world, &mut active_windows);
384                return false;
385            }
386            () = sleep(Duration::from_millis(10)) => {}
387        }
388    }
389
390    release_all_windows(world, &mut active_windows);
391    true
392}

394fn update_unload_backpressure(world: &Arc<World>, unload_backpressure: &mut bool) {
395    let unloading_chunks = world.chunk_map.unloading_chunks.len();
396    if *unload_backpressure {
397        if unloading_chunks <= PREGEN_UNLOAD_BACKPRESSURE_LOW {
398            *unload_backpressure = false;
399            log::info!(
400                "Pregen unload backpressure released: unloading_chunks={unloading_chunks}, low_watermark={PREGEN_UNLOAD_BACKPRESSURE_LOW}",
401            );
402        }
403        return;
404    }
405
406    if unloading_chunks >= PREGEN_UNLOAD_BACKPRESSURE_HIGH {
407        *unload_backpressure = true;
408        log::info!(
409            "Pregen unload backpressure active: unloading_chunks={unloading_chunks}, high_watermark={PREGEN_UNLOAD_BACKPRESSURE_HIGH}, low_watermark={PREGEN_UNLOAD_BACKPRESSURE_LOW}",
410        );
411    }
412}

414fn drain_pregen_broadcasts(world: &Arc<World>) {
415    world.chunk_map.broadcast_changed_chunks();
416}

418fn total_chunks(side_length: i32) -> usize {
419    let side_length = i64::from(side_length);
420    (side_length * side_length) as usize
421}

423fn fill_active_windows(
424    world: &Arc<World>,
425    pending_windows: &mut VecDeque<PregenWindow>,
426    active_windows: &mut Vec<ActivePregenWindow>,
427) {
428    while active_windows.iter().filter(|active| !active.ready).count() < PREGEN_ACTIVE_WINDOWS {
429        if !activate_next_window(world, pending_windows, active_windows) {
430            break;
431        }
432    }
433}
434
435fn activate_next_window(
436    world: &Arc<World>,
437    pending_windows: &mut VecDeque<PregenWindow>,
438    active_windows: &mut Vec<ActivePregenWindow>,
439) -> bool {
440    let Some(window) = pending_windows.pop_front() else {
441        return false;
442    };
443
444    active_windows.push(ActivePregenWindow::new(world, window));
445    true
446}
447
448fn release_unneeded_completed_windows(
449    world: &Arc<World>,
450    active_windows: &mut Vec<ActivePregenWindow>,
451) {
452    let incomplete_windows = active_windows
453        .iter()
454        .filter(|active| !active.ready)
455        .map(|active| active.window)
456        .collect::<Vec<_>>();
457
458    active_windows.retain(|active| {
459        if !active.ready {
460            return true;
461        }
462
463        let protected = active.window.protected_rect();
464
465        incomplete_windows
466            .iter()
467            .any(|window| protected.overlaps(window.protected_rect()))
468    });
469
470    world.chunk_map.tick_scheduling();
471}
472
473fn release_all_windows(world: &Arc<World>, active_windows: &mut Vec<ActivePregenWindow>) {
474    active_windows.clear();
475    world.chunk_map.tick_scheduling();
476}
