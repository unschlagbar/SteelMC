15const MAX_FLY_SPEED: f32 = 30f32;

17/// Handler for the "flyspeed" command.
18#[must_use]
19pub fn command_handler() -> impl CommandHandlerDyn {
20    CommandHandlerBuilder::new(
21        &["fly"],
22        "Sets the target's flying parameters (may_fly, speed).",
23        "minecraft:command.fly",
24    )
25    .executes(|(), ctx: &mut CommandContext| {
26        let player = ctx
27            .sender
28            .get_player()
29            .ok_or(CommandError::InvalidRequirement)?;
30
31        toggle_fly(slice::from_ref(player));
32
33        Ok(())
34    })
35    .then(
36        argument("target", PlayerArgument::multiple())
37            .executes(
38                |((), targets): ((), Vec<Arc<Player>>), _ctx: &mut CommandContext| {
39                    toggle_fly(&targets);
40                    Ok(())
41                },
42            )
43            .then(argument("value", BoolArgument).executes(
44                |(((), targets), value): (((), Vec<Arc<Player>>), bool),
45                 _ctx: &mut CommandContext| {
46                    set_fly(&targets, value);
47                    Ok(())
48                },
49            ))
50            .then(
51                literal("speed")
52                    .executes(
53                        |((), targets): ((), Vec<Arc<Player>>), ctx: &mut CommandContext| {
54                            query_flying_speed(&targets, &ctx.sender);
55                            Ok(())
56                        },
57                    )
58                    .then(
59                        argument(
60                            "speed",
61                            FloatArgument::bounded(Some(0.0), Some(MAX_FLY_SPEED)),
62                        )
63                        .executes(
64                            |(((), targets), speed): (((), Vec<Arc<Player>>), f32),
65                             ctx: &mut CommandContext| {
66                                set_flying_speed(&targets, speed, &ctx.sender);
67
68                                Ok(())
69                            },
70                        ),
71                    ),
72            ),
73    )
74    .then(
75        literal("speed")
76            .executes(|(), ctx: &mut CommandContext| {
77                let player = ctx
78                    .sender
79                    .get_player()
80                    .ok_or(CommandError::InvalidRequirement)?;
81
82                query_flying_speed(slice::from_ref(player), &ctx.sender);
83
84                Ok(())
85            })
86            .then(
87                argument(
88                    "speed",
89                    FloatArgument::bounded(Some(0.0), Some(MAX_FLY_SPEED)),
90                )
91                .executes(|((), speed): ((), f32), ctx: &mut CommandContext| {
92                    let player = ctx
93                        .sender
94                        .get_player()
95                        .ok_or(CommandError::InvalidRequirement)?;
96                    set_flying_speed(slice::from_ref(player), speed, &ctx.sender);
97                    Ok(())
98                }),
99            ),
100    )
101}

103fn toggle_fly(targets: &[Arc<Player>]) {
104    for target in targets {
105        {
106            let mut lock = target.abilities.lock();
107            lock.may_fly = !lock.may_fly;
108            if !lock.may_fly {
109                lock.flying = false;
110            }
111        }
112        target.send_abilities();
113    }
114}
115
116fn set_fly(targets: &[Arc<Player>], value: bool) {
117    for target in targets {
118        {
119            let mut lock = target.abilities.lock();
120            lock.may_fly = value;
121            if !value {
122                lock.flying = false;
123            }
124        }
125        target.send_abilities();
126    }
127}
128
129fn set_flying_speed(targets: &[Arc<Player>], multiplier: f32, sender: &CommandSender) {
130    let speed = multiplier * 0.05;
131    for target in targets {
132        target.set_flying_speed(speed);
133        target.send_abilities();
134        sender.send_message(&TextComponent::from(format!(
135            "Set flying speed for player '{}' to {multiplier:.1}x ({speed:.3})",
136            target.gameprofile.name.clone()
137        )));
138    }
139}
140
141fn query_flying_speed(targets: &[Arc<Player>], sender: &CommandSender) {
142    for target in targets {
143        let speed = target.get_flying_speed();
144        let multiplier = speed / 0.05; // Show as multiplier of default speed
145
146        sender.send_message(&TextComponent::from(format!(
147            "Current flying speed for player '{}': {multiplier:.1}x ({speed:.3})",
148            target.gameprofile.name.clone()
149        )));
150    }
151}
