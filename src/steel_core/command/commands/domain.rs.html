1//! Handler for the "domain" command.
2
3use crate::command::arguments::domain::DomainArgument;
4use crate::command::commands::{CommandHandlerBuilder, CommandHandlerDyn, argument};
5use crate::command::context::CommandContext;
6use crate::command::error::CommandError;
7use text_components::TextComponent;
8
9/// Handler for switching to another configured domain.
10#[must_use]
11pub fn command_handler() -> impl CommandHandlerDyn {
12    CommandHandlerBuilder::new(
13        &["domain"],
14        "Switches to another configured domain.",
15        "minecraft:command.domain",
16    )
17    .then(argument("domain", DomainArgument).executes(
18        |((), domain): ((), String), context: &mut CommandContext| -> Result<(), CommandError> {
19            let player = context
20                .sender
21                .get_player()
22                .cloned()
23                .ok_or(CommandError::InvalidRequirement)?;
24            let server = context.server.clone();
25            server
26                .queue_domain_switch(player, domain.clone())
27                .map_err(|error| {
28                    CommandError::CommandFailed(Box::new(TextComponent::plain(error)))
29                })?;
30
31            context.sender.send_message(&TextComponent::plain(format!(
32                "Switching to domain {domain}"
33            )));
34            Ok(())
35        },
36    ))
37}
