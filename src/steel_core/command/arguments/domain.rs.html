1//! Argument that resolves a configured domain name.
2
3use steel_protocol::packets::game::{ArgumentType, SuggestionEntry, SuggestionType};
4
5use crate::command::{
6    arguments::{CommandArgument, SuggestionContext},
7    context::CommandContext,
8};
9
10/// Parses a domain name.
11pub struct DomainArgument;
12
13impl CommandArgument for DomainArgument {
14    type Output = String;
15
16    fn parse<'a>(
17        &self,
18        arg: &'a [&'a str],
19        context: &mut CommandContext,
20    ) -> Option<(&'a [&'a str], Self::Output)> {
21        let domain = *arg.first()?;
22        if !context.server.worlds.has_domain(domain) {
23            return None;
24        }
25        Some((&arg[1..], domain.to_owned()))
26    }
27
28    fn usage(&self) -> (ArgumentType, Option<SuggestionType>) {
29        (
30            ArgumentType::ResourceLocation,
31            Some(SuggestionType::AskServer),
32        )
33    }
34
35    fn suggest(&self, prefix: &str, suggestion_ctx: &SuggestionContext) -> Vec<SuggestionEntry> {
36        suggestion_ctx
37            .server
38            .worlds
39            .domain_names()
40            .filter(|domain| domain.starts_with(prefix))
41            .map(SuggestionEntry::new)
42            .collect()
43    }
44}
