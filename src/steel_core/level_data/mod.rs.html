137fn describe_generation_settings(settings: &WorldGenerationSettings) -> String {
138    format!(
139        "generator {}, dimension_type {}, min_y {}, height {}, config {}",
140        settings.generator,
141        settings.dimension_type,
142        settings.min_y,
143        settings.height,
144        generation_config_string(&settings.config),
145    )
146}

148fn generation_config_string(config: &toml::Value) -> String {
149    match toml::to_string(config) {
150        Ok(value) => value.trim().to_owned(),
151        Err(_) => "<invalid generator config>".to_owned(),
152    }

179/// Vanilla default respawn data.
180#[derive(Debug, Clone, PartialEq)]
181pub struct RespawnData {
182    /// Dimension and block position of the default respawn.
183    pub global_pos: GlobalPos,
184    /// Spawn yaw, wrapped with vanilla `Mth.wrapDegrees`.
185    pub yaw: f32,
186    /// Spawn pitch, clamped to vanilla's player pitch range.
187    pub pitch: f32,
188}
189
190impl RespawnData {
191    /// Creates respawn data for the given global position.
192    #[must_use]
193    pub fn new(global_pos: GlobalPos, yaw: f32, pitch: f32) -> Self {
194        Self {
195            global_pos,
196            yaw: wrap_degrees(yaw),
197            pitch: pitch.clamp(-90.0, 90.0),
198        }
199    }
200
201    /// Creates respawn data for a dimension and block position.
202    #[must_use]
203    pub fn of(dimension: Identifier, pos: BlockPos, yaw: f32, pitch: f32) -> Self {
204        Self::new(GlobalPos::new(dimension, pos), yaw, pitch)
205    }
206
207    /// Returns the respawn dimension.
208    #[must_use]
209    pub const fn dimension(&self) -> &Identifier {
210        &self.global_pos.dimension
211    }
212
213    /// Returns the respawn block position.
214    #[must_use]
215    pub const fn pos(&self) -> BlockPos {
216        self.global_pos.pos
217    }
218}

220fn wrap_degrees(mut degrees: f32) -> f32 {
221    degrees %= 360.0;
222    if degrees >= 180.0 {
223        degrees -= 360.0;
224    }
225    if degrees < -180.0 {
226        degrees += 360.0;
227    }
228    degrees
229}

231#[derive(Serialize, Deserialize)]
232struct SerializedRespawnData {
233    dimension: Identifier,
234    x: i32,
235    y: i32,
236    z: i32,
237    yaw: f32,
238    pitch: f32,
239}
240
241impl Serialize for RespawnData {
242    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
243    where
244        S: Serializer,
245    {
246        SerializedRespawnData {
247            dimension: self.global_pos.dimension.clone(),
248            x: self.global_pos.pos.x(),
249            y: self.global_pos.pos.y(),
250            z: self.global_pos.pos.z(),
251            yaw: self.yaw,
252            pitch: self.pitch,
253        }
254        .serialize(serializer)
255    }
256}
257
258impl<'de> Deserialize<'de> for RespawnData {
259    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
260    where
261        D: Deserializer<'de>,
262    {
263        let data = SerializedRespawnData::deserialize(deserializer)?;
264        Ok(Self::of(
265            data.dimension,
266            BlockPos::new(data.x, data.y, data.z),
267            data.yaw,
268            data.pitch,
269        ))
270    }

321    /// Verifies saved generator metadata against the current config.
322    ///
323    /// Returns whether missing metadata was adopted and should be saved.
324    pub fn validate_generation_settings(
325        &mut self,
326        expected: WorldGenerationSettings,
327    ) -> io::Result<bool> {
328        match self.generation.as_ref() {
329            Some(saved) if saved == &expected => Ok(false),
330            Some(saved) => Err(io::Error::new(
331                io::ErrorKind::InvalidData,
332                format!(
333                    "world generation settings do not match saved level data: saved {}; configured {}. Delete or regenerate this world's saved chunks, or restore the previous generator config.",
334                    describe_generation_settings(saved),
335                    describe_generation_settings(&expected),
336                ),
337            )),
338            None => {
339                self.generation = Some(expected);
340                Ok(true)
341            }
342        }
343    }
344
345    /// Loads game rules from the serialized map into the runtime values.
346    pub fn load_game_rules(&mut self) {
347        self.game_rules_values = GameRuleValues::new(&REGISTRY.game_rules);
348        for (name, value) in &self.game_rules {
349            self.game_rules_values
350                .set_by_name(name, *value, &REGISTRY.game_rules);
351        }
352    }
353
354    /// Saves game rules from the runtime values to the serialized map.
355    pub fn save_game_rules(&mut self) {
356        self.game_rules.clear();
357        for (_, rule) in REGISTRY.game_rules.iter() {
358            let name = rule.key.path.to_string();
359            let value = self.game_rules_values.get(rule, &REGISTRY.game_rules);
360            self.game_rules.insert(name, value);
361        }
362    }

368    }

391/// Manages level data persistence for a world.
392pub struct LevelDataManager {
393    /// Path to the level.toml file.
394    path: Option<PathBuf>,
395    /// Cached level data.
396    data: LevelData,
397    /// Whether data has been modified since last save.
398    dirty: bool,
399}
400
401impl LevelDataManager {
402    /// Creates a new level data manager for the given world directory.
403    ///
404    /// If `level.toml` exists, it will be loaded (the provided seed is ignored).
405    /// Otherwise, new data will be created with the provided seed.
406    pub async fn new(
407        world_dir: Option<impl AsRef<Path>>,
408        seed: i64,
409        difficulty: Difficulty,
410        generation: WorldGenerationSettings,
411    ) -> io::Result<Self> {
412        let (data, path, dirty) = if let Some(dir) = &world_dir {
413            let path = dir.as_ref().join("level.toml");
414
415            let (data, dirty) = if path.exists() {
416                // Load existing level data (seed from file takes precedence)
417                let content = fs::read_to_string(&path).await?;
418                let mut loaded: LevelData = toml::from_str(&content).map_err(|e| {
419                    io::Error::new(
420                        io::ErrorKind::InvalidData,
421                        format!("Invalid level.toml: {e}"),
422                    )
423                })?;
424                // Initialize runtime game rules from serialized values
425                loaded.load_game_rules();
426                let adopted_generation = loaded.validate_generation_settings(generation)?;
427                (loaded, adopted_generation)
428            } else {
429                // Create new level data with the provided defaults.
430                let mut data = LevelData::new_with_seed_and_difficulty(seed, difficulty);
431                data.generation = Some(generation);
432                (data, true)
433            };
434            (data, Some(path), dirty)
435        } else {
436            let mut data = LevelData::new_with_seed_and_difficulty(seed, difficulty);
437            data.generation = Some(generation);
438            (data, None, false)
439        };
440
441        Ok(Self { path, data, dirty })
442    }
443
444    /// Loads the saved world seed from `level.toml`, or returns the provided default.
445    pub async fn load_seed_or_default(
446        world_dir: Option<impl AsRef<Path>>,
447        default_seed: i64,
448    ) -> io::Result<i64> {
449        let Some(dir) = world_dir else {
450            return Ok(default_seed);
451        };
452
453        let path = dir.as_ref().join("level.toml");
454        if !path.exists() {
455            return Ok(default_seed);
456        }
457
458        let content = fs::read_to_string(path).await?;
459        let saved: SavedLevelSeed = toml::from_str(&content).map_err(|e| {
460            io::Error::new(
461                io::ErrorKind::InvalidData,
462                format!("Invalid level.toml: {e}"),
463            )
464        })?;
465        Ok(saved.seed)
466    }

483        self.dirty

486    /// Marks the data as dirty (needs saving).
487    pub const fn mark_dirty(&mut self) {
488        self.dirty = true;
489    }
490
491    /// Saves the level data to disk if it has been modified.
492    pub async fn save(&mut self) -> io::Result<()> {
493        if !self.dirty {
494            return Ok(());
495        }
496
497        let Some(world_path) = &self.path else {
498            self.dirty = false;
499            return Ok(());
500        };
501        if let Some(parent) = world_path.parent() {
502            fs::create_dir_all(parent).await?;
503        }
504
505        // Export runtime game rules to serializable format before saving
506        self.data.save_game_rules();
507
508        let content = toml::to_string_pretty(&self.data)
509            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))?;
510        fs::write(world_path, content).await?;
511        self.dirty = false;
512
513        log::debug!("Saved level data to {}", world_path.display());
514        Ok(())
515    }
