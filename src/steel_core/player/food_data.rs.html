101/// Computes the absolute saturation value from a nutrition count and a modifier.
102#[must_use]
103pub fn saturation_by_modifier(nutrition: i32, modifier: f32) -> f32 {
104    nutrition as f32 * modifier * 2.0
105}

107/// Tracks a player's hunger, saturation, and exhaustion state.
108/// One instance is stored per player behind a `SyncMutex`.
109#[derive(Debug, Clone)]
110pub struct FoodData {
111    /// Current food level (0–20). Displayed as the hunger bar on the client.
112    pub food_level: i32,
113    /// Saturation buffer — consumed before the food level drops.
114    pub saturation_level: f32,
115    /// Accumulated exhaustion from actions (sprinting, jumping, damage, …).
116    pub exhaustion_level: f32,
117    /// Internal tick counter shared between regeneration and starvation logic.
118    pub tick_timer: i32,
119}
120
121impl Default for FoodData {
122    fn default() -> Self {
123        Self::new()
124    }
125}
126
127impl FoodData {
128    /// Creates a new `FoodData` with default values (full hunger bar).
129    #[must_use]
130    pub const fn new() -> Self {
131        Self {
132            food_level: DEFAULT_FOOD_LEVEL,
133            saturation_level: DEFAULT_SATURATION,
134            exhaustion_level: 0.0,
135            tick_timer: 0,
136        }
137    }
138
139    /// Returns `true` if the player's food level is below maximum and they
140    /// could benefit from eating.
141    #[must_use]
142    pub const fn needs_food(&self) -> bool {
143        self.food_level < MAX_FOOD_LEVEL
144    }
145
146    /// Returns `true` if the player has enough food to perform exhaustive
147    /// maneuvers (for example sprinting).
148    #[must_use]
149    pub const fn has_enough_food(&self) -> bool {
150        self.food_level > SPRINT_LEVEL
151    }
152
153    /// Adds exhaustion from an action (sprinting, jumping, taking damage, …).
154    /// The value is clamped so `exhaustion_level` never exceeds 40.
155    pub fn add_exhaustion(&mut self, amount: f32) {
156        self.exhaustion_level = (self.exhaustion_level + amount).min(MAX_EXHAUSTION);
157    }
158
159    /// Applies food and saturation
160    fn add(&mut self, food: i32, saturation: f32) {
161        self.food_level = (self.food_level + food).clamp(0, MAX_FOOD_LEVEL);
162        self.saturation_level =
163            (self.saturation_level + saturation).clamp(0.0, self.food_level as f32);
164    }
165
166    /// Applies the nutrition from eating food, given a **saturation modifier**.
167    pub fn eat(&mut self, nutrition: i32, saturation_modifier: f32) {
168        self.add(
169            nutrition,
170            saturation_by_modifier(nutrition, saturation_modifier),
171        );
172    }
173
174    /// Runs one tick of the hunger system.
175    /// Returns a [`FoodTickResult`] describing what happened this tick so the
176    /// caller (`Player::tick`) can apply healing or starvation damage
177    #[must_use]
178    pub fn tick(
179        &mut self,
180        difficulty: Difficulty,
181        natural_regen: bool,
182        current_health: f32,
183        max_health: f32,
184    ) -> FoodTickResult {
185        if self.exhaustion_level > EXHAUSTION_DROP {
186            self.exhaustion_level -= EXHAUSTION_DROP;
187
188            if self.saturation_level > 0.0 {
189                self.saturation_level = (self.saturation_level - 1.0).max(0.0);
190            } else if difficulty != Difficulty::Peaceful {
191                self.food_level = (self.food_level - 1).max(0);
192            }
193        }
194
195        let food = self.food_level;
196        let is_hurt = current_health > 0.0 && current_health < max_health;
197
198        // Fast regen
199        if natural_regen && self.saturation_level > 0.0 && is_hurt && food >= MAX_FOOD_LEVEL {
200            self.tick_timer += 1;
201
202            if self.tick_timer >= HEALTH_TICK_COUNT_SATURATED {
203                let saturation_spent = self.saturation_level.min(EXHAUSTION_HEAL);
204                let heal_amount = saturation_spent / EXHAUSTION_HEAL;
205                self.tick_timer = 0;
206                return FoodTickResult::Heal {
207                    amount: heal_amount,
208                    exhaustion: saturation_spent,
209                };
210            }
211        }
212        // Slow regen
213        else if natural_regen && food >= HEAL_LEVEL && is_hurt {
214            self.tick_timer += 1;
215
216            if self.tick_timer >= HEALTH_TICK_COUNT {
217                self.tick_timer = 0;
218                return FoodTickResult::Heal {
219                    amount: 1.0,
220                    exhaustion: EXHAUSTION_HEAL,
221                };
222            }
223        }
224        // Starving
225        else if food <= 0 {
226            self.tick_timer += 1;
227
228            if self.tick_timer >= HEALTH_TICK_COUNT {
229                self.tick_timer = 0;
230
231                if current_health > 10.0
232                    || difficulty == Difficulty::Hard
233                    || (current_health > 1.0 && difficulty == Difficulty::Normal)
234                {
235                    return FoodTickResult::Starve;
236                }
237            }
238        } else {
239            self.tick_timer = 0;
240        }
241
242        FoodTickResult::None
243    }
244}

246/// Describes what the hunger tick determined should happen this tick
247#[derive(Debug, Clone, Copy, PartialEq)]
248pub enum FoodTickResult {
249    /// Nothing happened this tick
250    None,
251    /// The player should be healed by `amount` HP and `exhaustion` should be added back
252    Heal {
253        /// Health points to restore
254        amount: f32,
255        /// Exhaustion to add as cost of this regeneration
256        exhaustion: f32,
257    },
258    /// The player should take 1 point of starvation damage
259    Starve,
260}

262impl Player {
263    /// Ticks food/hunger regeneration and starvation.
264    pub(super) fn tick_regeneration(&self) {
265        let world = self.get_world();
266        let difficulty = world.level_data.read().data().difficulty;
267        let natural_regen =
268            world.get_game_rule(&NATURAL_HEALTH_REGENERATION) == GameRuleValue::Bool(true);
269        let tick = self.tick_count();
270
271        if difficulty == Difficulty::Peaceful && natural_regen {
272            if tick % 20 == 0 {
273                if self.is_hurt() {
274                    self.heal(1.0);
275                }
276
277                let mut food = self.food_data.lock();
278                if food.saturation_level < food_constants::MAX_SATURATION {
279                    food.saturation_level += 1.0;
280                }
281            }
282
283            if tick % 10 == 0 {
284                let mut food = self.food_data.lock();
285                if food.needs_food() {
286                    food.food_level += 1;
287                }
288            }
289        }
290
291        let current_health = self.get_health();
292        let max_health = self.get_max_health();
293
294        let mut food = self.food_data.lock();
295        let result = food.tick(difficulty, natural_regen, current_health, max_health);
296
297        match result {
298            FoodTickResult::Heal { amount, exhaustion } => {
299                food.add_exhaustion(exhaustion);
300
301                drop(food);
302                self.heal(amount);
303            }
304            FoodTickResult::Starve => {
305                drop(food);
306
307                self.hurt(
308                    &DamageSource::environment(&vanilla_damage_types::STARVE),
309                    1.0,
310                );
311            }
312            FoodTickResult::None => {}
313        }
314    }
315
316    /// Adds food exhaustion, gated by invulnerability.
317    pub fn cause_food_exhaustion(&self, amount: f32) {
318        if !self.abilities.lock().invulnerable {
319            self.food_data.lock().add_exhaustion(amount);
320        }
321    }
322
323    /// Returns `true` if the player is alive but below max health.
324    pub fn is_hurt(&self) -> bool {
325        let health = self.get_health();
326        health > 0.0 && health < self.get_max_health()
327    }
328}

334    #[test]
335    fn exhaustion_drains_saturation_then_food() {
336        let mut food = FoodData::new();
337        food.saturation_level = 3.0;
338        food.add_exhaustion(4.5);
339
340        // First drain: saturation drops by 1
341        let _ = food.tick(Difficulty::Normal, false, 20.0, 20.0);
342        assert!((food.saturation_level - 2.0).abs() < f32::EPSILON);
343        assert_eq!(food.food_level, MAX_FOOD_LEVEL);
344
345        // Now empty saturation and trigger again — food level should drop
346        food.saturation_level = 0.0;
347        food.exhaustion_level = 5.0;
348        let _ = food.tick(Difficulty::Normal, false, 20.0, 20.0);
349        assert_eq!(food.food_level, MAX_FOOD_LEVEL - 1);
350    }

352    /// Fast regen: food=20, saturation>0, hurt → heal every 10 ticks.
353    #[test]
354    fn fast_regen() {
355        let mut food = FoodData::new(); // food=20, sat=5.0
356
357        let mut result = FoodTickResult::None;
358        for _ in 0..HEALTH_TICK_COUNT_SATURATED {
359            result = food.tick(Difficulty::Normal, true, 15.0, 20.0);
360        }
361
362        match result {
363            FoodTickResult::Heal { amount, exhaustion } => {
364                assert!((exhaustion - 5.0).abs() < f32::EPSILON);
365                assert!((amount - 5.0 / 6.0).abs() < f32::EPSILON);
366            }
367            other => panic!("Expected Heal, got {other:?}"),
368        }
369    }

392    /// Starvation thresholds per difficulty:
393    /// - Hard: always starves
394    /// - Normal: stops at 1 HP
395    /// - Easy/Peaceful: stops at 10 HP
396    #[test]
397    fn starvation_by_difficulty() {
398        let run = |diff, health| {
399            let mut food = FoodData::new();
400            food.food_level = 0;
401            food.saturation_level = 0.0;
402            let mut result = FoodTickResult::None;
403            for _ in 0..HEALTH_TICK_COUNT {
404                result = food.tick(diff, false, health, 20.0);
405            }
406            result
407        };
408
409        assert_eq!(run(Difficulty::Hard, 1.0), FoodTickResult::Starve);
410        assert_eq!(run(Difficulty::Normal, 1.0), FoodTickResult::None);
411        assert_eq!(run(Difficulty::Normal, 2.0), FoodTickResult::Starve);
412        assert_eq!(run(Difficulty::Easy, 10.0), FoodTickResult::None);
413        assert_eq!(run(Difficulty::Easy, 11.0), FoodTickResult::Starve);
414    }
