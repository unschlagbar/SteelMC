15/// An object-safe trait for player connections.
16///
17/// This abstracts the connection layer so that:
18/// - `JavaConnection` can handle real network traffic
19/// - Test connections (like `FlintConnection`) can record events for assertions
20///
21/// # Object Safety
22///
23/// This trait uses type erasure for packet sending - packets must be pre-encoded
24/// into `EncodedPacket` before being sent. The `Player` struct provides a generic
25/// `send_packet<P: ClientPacket>()` helper that handles encoding.
26#[enum_dispatch]
27pub trait NetworkConnection: Send + Sync {
28    /// Returns compression info for packet encoding.
29    ///
30    /// Returns `None` if compression is disabled (e.g., for test connections).
31    fn compression(&self) -> Option<CompressionInfo>;
32
33    /// Sends a pre-encoded packet.
34    ///
35    /// This is the object-safe method that accepts already-encoded packets.
36    /// Use `Player::send_packet()` for the generic version that handles encoding.
37    fn send_encoded(&self, packet: EncodedPacket);
38
39    /// Sends multiple pre-encoded packets as an atomic bundle.
40    ///
41    /// The implementation wraps the packets with bundle delimiter packets so
42    /// the client processes them together in a single game tick.
43    /// Use `Player::send_bundle()` for the generic version that handles encoding.
44    fn send_encoded_bundle(&self, packets: Vec<EncodedPacket>);
45
46    /// Disconnects the player with a reason.
47    fn disconnect_with_reason(&self, reason: TextComponent);
48
49    /// Performs per-tick connection maintenance (e.g., keep-alive).
50    fn tick(&self);
51
52    /// Returns the current latency in milliseconds.
53    fn latency(&self) -> i32;
54
55    /// Closes the connection.
56    fn close(&self);
57
58    /// Returns whether the connection is closed.
59    fn closed(&self) -> bool;
60}
61
62impl NetworkConnection for Box<dyn NetworkConnection> {
63    fn compression(&self) -> Option<CompressionInfo> {
64        (**self).compression()
65    }
66
67    fn send_encoded(&self, packet: EncodedPacket) {
68        (**self).send_encoded(packet);
69    }
70
71    fn send_encoded_bundle(&self, packets: Vec<EncodedPacket>) {
72        (**self).send_encoded_bundle(packets);
73    }
74
75    fn disconnect_with_reason(&self, reason: TextComponent) {
76        (**self).disconnect_with_reason(reason);
77    }
78
79    fn tick(&self) {
80        (**self).tick();
81    }
82
83    fn latency(&self) -> i32 {
84        (**self).latency()
85    }
86
87    fn close(&self) {
88        (**self).close();
89    }
90
91    fn closed(&self) -> bool {
92        (**self).closed()
93    }
94}

96impl Player {
97    /// Sends a packet to the player's connection.
98    ///
99    /// This is a generic helper that encodes the packet and delegates to the
100    /// connection's `send_encoded` method, enabling object-safe packet sending.
101    ///
102    /// # Panics
103    ///
104    /// Panics if the packet fails to encode.
105    pub fn send_packet<P: ClientPacket>(&self, packet: P) {
106        let encoded = EncodedPacket::from_bare(
107            packet,
108            self.connection.compression(),
109            ConnectionProtocol::Play,
110        )
111        .expect("Failed to encode packet");
112        self.connection.send_encoded(encoded);
113    }
114
115    /// Sends multiple packets as an atomic bundle.
116    ///
117    /// The closure receives a [`BundleBuilder`](networking::BundleBuilder) to add packets to.
118    /// All packets are encoded, then sent wrapped in bundle delimiters so the
119    /// client processes them together in a single game tick.
120    pub fn send_bundle<F>(&self, f: F)
121    where
122        F: FnOnce(&mut networking::BundleBuilder),
123    {
124        let mut builder = networking::BundleBuilder::new(self.connection.compression());
125        f(&mut builder);
126        let packets = builder.into_packets();
127        if !packets.is_empty() {
128            self.connection.send_encoded_bundle(packets);
129        }
130    }
131
132    /// Disconnects the player with a reason message.
133    pub fn disconnect(&self, reason: impl Into<TextComponent>) {
134        self.connection.disconnect_with_reason(reason.into());
135    }
136
137    /// Handles client information updates during play phase.
138    pub fn handle_client_information(&self, packet: SClientInformation) {
139        let old_view_distance = self.view_distance();
140
141        let info = ClientInformation {
142            language: packet.language,
143            view_distance: packet
144                .view_distance
145                .clamp(2, i32::from(self.config.view_distance).max(2))
146                as u8,
147            chat_visibility: packet.chat_visibility,
148            chat_colors: packet.chat_colors,
149            model_customization: packet.model_customization,
150            main_hand: packet.main_hand,
151            text_filtering_enabled: packet.text_filtering_enabled,
152            allows_listing: packet.allows_listing,
153            particle_status: packet.particle_status,
154        };
155        self.set_client_information(info);
156
157        let new_view_distance = self.view_distance();
158        if old_view_distance != new_view_distance {
159            self.send_packet(CSetChunkCacheRadius {
160                radius: i32::from(new_view_distance),
161            });
162            self.get_world().chunk_map.update_player_status(self);
163        }
164    }
165
166    /// Returns the player's client information settings.
167    #[must_use]
168    pub fn client_information(&self) -> ClientInformation {
169        self.client_information.lock().clone()
170    }
171
172    /// Updates the player's client information settings.
173    pub fn set_client_information(&self, info: ClientInformation) {
174        *self.client_information.lock() = info;
175    }
176
177    /// Returns the effective view distance for this player.
178    ///
179    /// This is the minimum of the client's requested view distance and
180    /// the server's configured maximum view distance.
181    #[must_use]
182    pub fn view_distance(&self) -> u8 {
183        let client_view_distance = self.client_information.lock().view_distance;
184        client_view_distance.min(self.world.load().view_distance)
185    }
