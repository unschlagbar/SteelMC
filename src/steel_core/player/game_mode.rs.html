58/// Handles using an item on a block.
59///
60/// This implements the logic from Java's `ServerPlayerGameMode.useItemOn()`.
61///
62/// # Flow
63/// 1. Spectator mode: Only allow opening menus (currently returns Pass)
64/// 2. Check if block interaction should be suppressed (sneaking + holding items)
65/// 3. If not suppressed: Call block's `use_item_on` method
66/// 4. If block returns `TryEmptyHandInteraction` and main hand: Call block's `use_without_item`
67/// 5. If item not empty: Call item behavior's `use_on` for placement
68/// 6. Handle creative mode infinite materials
69pub fn use_item_on(
70    player: &Player,
71    world: &Arc<World>,
72    hand: InteractionHand,
73    hit_result: &BlockHitResult,
74) -> InteractionResult {
75    let pos = hit_result.block_pos;
76    let state = world.get_block_state(pos);
77
78    // Spectator mode: can only open menus
79    // TODO: Implement menu providers for blocks like chests
80    if player.game_mode() == GameType::Spectator {
81        return InteractionResult::Pass;
82    }
83
84    // Check if block interaction should be suppressed (sneaking + holding items in either hand)
85    let have_something = {
86        let inv = player.inventory.lock();
87        !inv.get_item_in_hand(InteractionHand::MainHand).is_empty()
88            || !inv.get_item_in_hand(InteractionHand::OffHand).is_empty()
89    };
90
91    let suppress_block_use = player.is_secondary_use_active() && have_something;
92
93    // Get behavior registries
94    let block_behaviors = &*BLOCK_BEHAVIORS;
95    let item_behaviors = &*ITEM_BEHAVIORS;
96
97    // Try block interaction first (if not suppressed).
98    // No inventory lock held — block behaviors may need inventory access (e.g. opening chests).
99    if !suppress_block_use {
100        let Some(block) = REGISTRY.blocks.by_state_id(state) else {
101            return InteractionResult::Pass;
102        };
103        let behavior = block_behaviors.get_behavior(block);
104
105        let mut inventory_access = InventoryAccess::new(player.inventory.clone(), hand);
106
107        let block_result = behavior.use_item_on(
108            state,
109            world,
110            pos,
111            player,
112            hand,
113            hit_result,
114            &mut inventory_access,
115        );
116
117        if block_result.consumes_action() {
118            return block_result;
119        }
120
121        if matches!(block_result, InteractionResult::TryEmptyHandInteraction)
122            && hand == InteractionHand::MainHand
123        {
124            let empty_result = behavior.use_without_item(
125                state,
126                world,
127                pos,
128                player,
129                hit_result,
130                &mut inventory_access,
131            );
132
133            if empty_result.consumes_action() {
134                return empty_result;
135            }
136        }
137    }
138
139    let inventory_access = InventoryAccess::new(player.inventory.clone(), hand);
140    let (is_empty, original_count, item_ref) =
141        inventory_access.with_item(|item| (item.is_empty(), item.count, item.item));
142
143    if !is_empty {
144        // TODO: Check item cooldowns
145        // if player.getCooldowns().isOnCooldown(item_stack.item) { return Pass }
146
147        let mut context = UseOnContext::new(
148            player,
149            hand,
150            hit_result.clone(),
151            world,
152            player.inventory.clone(),
153        );
154        let item_behavior = item_behaviors.get_behavior(item_ref);
155        let result = item_behavior.use_on(&mut context);
156
157        // Restore count for creative mode (infinite materials)
158        if player.has_infinite_materials() {
159            context.inv.with_item(|item| {
160                if item.count < original_count {
161                    item.count = original_count;
162                }
163            });
164        }
165
166        return result;
167    }
168
169    InteractionResult::Pass
170}

172/// Handles using an item (general usage like right-clicking air).
173///
174/// This implements logic similar to `ServerPlayerGameMode.useItem()`.
175pub fn use_item(player: &Player, world: &Arc<World>, hand: InteractionHand) -> InteractionResult {
176    // Spectator mode: can only open menus
177    if player.game_mode() == GameType::Spectator {
178        return InteractionResult::Pass;
179    }
180
181    // TODO: Check item cooldowns
182    // if player.getCooldowns().isOnCooldown(item_stack) { return InteractionResult::Pass }
183
184    let inventory_access = InventoryAccess::new(player.inventory.clone(), hand);
185    let (is_empty, original_count, item_ref) =
186        inventory_access.with_item(|item| (item.is_empty(), item.count, item.item));
187
188    if !is_empty {
189        let mut context =
190            crate::behavior::UseItemContext::new(player, hand, world, player.inventory.clone());
191
192        // Get behavior registries
193        let item_behaviors = &*ITEM_BEHAVIORS;
194        let item_behavior = item_behaviors.get_behavior(item_ref);
195
196        let result = item_behavior.use_item(&mut context);
197
198        // Restore count for creative mode (infinite materials)
199        if player.has_infinite_materials() {
200            context.inv.with_item(|item| {
201                if item.count < original_count {
202                    item.count = original_count;
203                }
204            });
205        }
206
207        return result;
208    }
209
210    InteractionResult::Pass
211}

223fn piercing_ray_hit_t(
224    world: &World,
225    bounding_box: WorldAabb,
226    from: DVec3,
227    to: DVec3,
228    entity_margin: f64,
229) -> Option<f64> {
230    if let Some(hit_t) = ray_aabb_hit_t(bounding_box, from, to) {
231        return Some(hit_t);
232    }
233    if entity_margin <= 0.0 {
234        return None;
235    }
236
237    let outside_hit_t = ray_aabb_hit_t(bounding_box.inflate(entity_margin), from, to)?;
238    let outside_hit = from + (to - from) * outside_hit_t;
239    let mut towards_target = DVec3::new(
240        f64::midpoint(bounding_box.min_x(), bounding_box.max_x()),
241        f64::midpoint(bounding_box.min_y(), bounding_box.max_y()),
242        f64::midpoint(bounding_box.min_z(), bounding_box.max_z()),
243    );
244    let block_hit = world.clip(
245        outside_hit,
246        towards_target,
247        ClipBlockShape::Collider,
248        ClipFluid::None,
249    );
250    if !block_hit.is_miss() {
251        towards_target = block_hit.location;
252    }
253    ray_aabb_hit_t(bounding_box, outside_hit, towards_target).map(|_| outside_hit_t)
254}

256fn ray_aabb_hit_t(aabb: WorldAabb, from: DVec3, to: DVec3) -> Option<f64> {
257    if aabb.contains(from) {
258        return Some(0.0);
259    }
260
261    let delta = to - from;
262    let mut t_min = 0.0_f64;
263    let mut t_max = 1.0_f64;
264    if !update_ray_axis(
265        from.x,
266        delta.x,
267        aabb.min_x(),
268        aabb.max_x(),
269        &mut t_min,
270        &mut t_max,
271    ) {
272        return None;
273    }
274    if !update_ray_axis(
275        from.y,
276        delta.y,
277        aabb.min_y(),
278        aabb.max_y(),
279        &mut t_min,
280        &mut t_max,
281    ) {
282        return None;
283    }
284    if !update_ray_axis(
285        from.z,
286        delta.z,
287        aabb.min_z(),
288        aabb.max_z(),
289        &mut t_min,
290        &mut t_max,
291    ) {
292        return None;
293    }
294
295    Some(t_min)
296}

298fn update_ray_axis(
299    start: f64,
300    delta: f64,
301    min: f64,
302    max: f64,
303    t_min: &mut f64,
304    t_max: &mut f64,
305) -> bool {
306    if delta.abs() < f64::EPSILON {
307        return start >= min && start <= max;
308    }
309
310    let inverse_delta = 1.0 / delta;
311    let mut enter = (min - start) * inverse_delta;
312    let mut exit = (max - start) * inverse_delta;
313    if enter > exit {
314        swap(&mut enter, &mut exit);
315    }
316
317    *t_min = (*t_min).max(enter);
318    *t_max = (*t_max).min(exit);
319    *t_min <= *t_max
320}

357    /// Ticks vanilla attack-strength recovery and resets it on main-hand item changes.
358    pub(super) fn tick_attack_strength(&self) {
359        self.tick_state.lock().advance_attack_strength_ticker();
360
361        let main_hand_item = {
362            let inventory = self.inventory.lock();
363            let stack = inventory.get_item_in_hand(InteractionHand::MainHand);
364            stack.copy_with_count(stack.count())
365        };
366
367        let mut last_item = self.last_item_in_main_hand.lock();
368        if ItemStack::matches(&last_item, &main_hand_item) {
369            return;
370        }
371
372        if !ItemStack::is_same_item(&last_item, &main_hand_item) {
373            self.reset_attack_strength_ticker();
374        }
375
376        *last_item = main_hand_item;
377    }

428    fn cause_extra_knockback(
429        &self,
430        entity: &dyn Entity,
431        knockback_amount: f64,
432        old_movement: DVec3,
433    ) {
434        if knockback_amount > 0.0 {
435            let yaw_radians = self.rotation().0.to_radians();
436            let yaw_sin = f64::from(yaw_radians.sin());
437            let yaw_cos = f64::from(yaw_radians.cos());
438            if let Some(living_target) = entity.as_living_entity() {
439                living_target.knockback(knockback_amount, yaw_sin, -yaw_cos);
440            } else {
441                entity.push_impulse(DVec3::new(
442                    -yaw_sin * knockback_amount,
443                    0.1,
444                    yaw_cos * knockback_amount,
445                ));
446            }
447
448            let velocity = self.velocity();
449            self.set_velocity(DVec3::new(velocity.x * 0.6, velocity.y, velocity.z * 0.6));
450            self.set_sprinting(false);
451        }
452
453        if entity.entity_type() == &vanilla_entities::PLAYER
454            && entity.hurt_marked()
455            && let Some(player) = self.get_world().players.get_by_entity_id(entity.id())
456        {
457            let velocity = entity.velocity();
458            player.send_packet(CSetEntityMotion::new(entity.id(), velocity));
459            entity.clear_hurt_mark();
460            entity.set_velocity(old_movement);
461        }
462    }

470    /// Returns true if the target box is within the player's attack range for `item_stack`.
471    #[must_use]
472    pub fn is_within_attack_range_with_buffer(
473        &self,
474        item_stack: &ItemStack,
475        aabb: WorldAabb,
476        buffer: f64,
477    ) -> bool {
478        let distance = aabb.distance_to_sqr(self.eye_position()).sqrt();
479        let (min_reach, max_reach, hitbox_margin) =
480            if let Some(attack_range) = item_stack.get_attack_range() {
481                if self.game_mode() == GameType::Creative {
482                    (
483                        attack_range.min_creative_reach,
484                        attack_range.max_creative_reach,
485                        attack_range.hitbox_margin,
486                    )
487                } else {
488                    (
489                        attack_range.min_reach,
490                        attack_range.max_reach,
491                        attack_range.hitbox_margin,
492                    )
493                }
494            } else {
495                (0.0, self.entity_interaction_range() as f32, 0.0)
496            };
497        let min_reach = f64::from(min_reach) - f64::from(hitbox_margin) - buffer;
498        let max_reach = f64::from(max_reach) + f64::from(hitbox_margin) + buffer;
499        distance >= min_reach && distance <= max_reach
500    }

533    fn piercing_hit_entities(&self, item_stack: &ItemStack, world: &World) -> Vec<SharedEntity> {
534        let look = self.look_angle();
535        if look.length_squared() <= f64::EPSILON {
536            return Vec::new();
537        }
538
539        let (min_reach, max_reach, hitbox_margin) = self.attack_range_for_item(item_stack);
540        let eye_position = self.eye_position();
541        let from = eye_position + look * min_reach;
542        let movement_extension = self.known_movement().dot(look).max(0.0);
543        let mut to = eye_position + look * (max_reach + movement_extension);
544
545        let block_hit = world.clip(eye_position, to, ClipBlockShape::Collider, ClipFluid::None);
546        if !block_hit.is_miss() {
547            to = block_hit.location;
548            if eye_position.distance_squared(to) < eye_position.distance_squared(from) {
549                return Vec::new();
550            }
551        }
552
553        let search_area = WorldAabb::new(from.x, from.y, from.z, from.x, from.y, from.z)
554            .inflate_xyz(hitbox_margin, hitbox_margin, hitbox_margin)
555            .expand_towards(to - from)
556            .inflate(1.0);
557        let mut hits = world
558            .get_entities_in_aabb_matching(&search_area, |entity| {
559                self.can_piercing_hit_entity(entity)
560            })
561            .into_iter()
562            .filter_map(|entity| {
563                piercing_ray_hit_t(world, entity.bounding_box(), from, to, hitbox_margin)
564                    .map(|hit_t| (hit_t, entity))
565            })
566            .collect::<Vec<_>>();
567        hits.sort_by(|(left, _), (right, _)| left.total_cmp(right));
568        hits.into_iter().map(|(_, entity)| entity).collect()
569    }

579    fn piercing_attack(&self, item_stack: &ItemStack, piercing_weapon: &PiercingWeapon) {
580        let world = self.get_world();
581        LivingEntity::refresh_equipment_attribute_modifiers(self, EquipmentSlot::MainHand);
582        let base_damage = self
583            .attributes()
584            .lock()
585            .required_value(vanilla_attributes::ATTACK_DAMAGE) as f32;
586        let mut hit_something = false;
587        for target in self.piercing_hit_entities(item_stack, &world) {
588            hit_something |= self.stab_attack(
589                &target,
590                base_damage,
591                true,
592                piercing_weapon.deals_knockback,
593                piercing_weapon.dismounts,
594            );
595        }
596
597        self.reset_attack_strength_ticker();
598        enchantment_helper::do_post_piercing_attack_effects(self);
599        if hit_something {
600            self.play_sound_holder(piercing_weapon.hit_sound.as_ref());
601        }
602        self.play_sound_holder(piercing_weapon.sound.as_ref());
603        self.swing(InteractionHand::MainHand, false);
604    }

606    fn stab_attack(
607        &self,
608        target: &SharedEntity,
609        base_damage: f32,
610        deals_damage: bool,
611        deals_knockback: bool,
612        dismounts: bool,
613    ) -> bool {
614        let entity = target.as_ref();
615        if self.cannot_attack(entity) {
616            return false;
617        }
618
619        let attacking_item = {
620            let inventory = self.inventory.lock();
621            let stack = inventory.get_item_in_hand(InteractionHand::MainHand);
622            stack.copy_with_count(stack.count())
623        };
624        let damage_source = self.attack_damage_source(&attacking_item);
625        let enchantment_context = EnchantmentDamageContext::new(
626            entity.entity_type(),
627            Some(self.entity_type()),
628            Some(self.entity_type()),
629            &damage_source,
630        );
631        let enchanted_damage =
632            enchantment_helper::modify_damage(&attacking_item, &enchantment_context, base_damage);
633        let attack_strength_scale = self.attack_strength_scale(0.5);
634        let magic_boost = attack_strength_scale * (enchanted_damage - base_damage);
635        let base_damage = base_damage * Self::base_damage_scale_factor(attack_strength_scale);
636        let damage = base_damage + magic_boost;
637        let old_movement = entity.velocity();
638        let mut affected = deals_knockback;
639        let damage_dealt = deals_damage && entity.hurt(&damage_source, damage);
640        affected |= damage_dealt;
641        if deals_knockback {
642            self.cause_extra_knockback(
643                entity,
644                0.4 + Self::get_knockback(0.0, &attacking_item, &enchantment_context),
645                old_movement,
646            );
647        }
648        if dismounts && entity.is_passenger() {
649            affected = true;
650            entity.stop_riding();
651        }
652
653        if !affected {
654            return false;
655        }
656
657        self.item_attack_interaction(entity, &damage_source, damage_dealt);
658        self.set_last_hurt_mob(Some(target));
659        self.cause_food_exhaustion(0.1);
660        true
661    }

674    /// Attacks an entity with the player's main-hand base damage.
675    ///
676    /// Returns `true` if the target accepted damage.
677    #[must_use]
678    pub fn attack(&self, target: &SharedEntity) -> bool {
679        let entity = target.as_ref();
680        if self.cannot_attack(entity) {
681            return false;
682        }
683
684        LivingEntity::refresh_equipment_attribute_modifiers(self, EquipmentSlot::MainHand);
685        let attacking_item = {
686            let inventory = self.inventory.lock();
687            let stack = inventory.get_item_in_hand(InteractionHand::MainHand);
688            stack.copy_with_count(stack.count())
689        };
690        let (attack_damage, attack_speed, attack_knockback) = {
691            let attributes = self.attributes().lock();
692            (
693                attributes.required_value(vanilla_attributes::ATTACK_DAMAGE) as f32,
694                attributes.required_value(vanilla_attributes::ATTACK_SPEED),
695                attributes.required_value(vanilla_attributes::ATTACK_KNOCKBACK),
696            )
697        };
698        let attack_strength_delay = Self::attack_strength_delay_from_speed(attack_speed);
699        let attack_strength_scale =
700            self.attack_strength_scale_for_delay(0.5, attack_strength_delay);
701        let damage_source = self.attack_damage_source(&attacking_item);
702        let enchantment_context = EnchantmentDamageContext::new(
703            entity.entity_type(),
704            Some(self.entity_type()),
705            Some(self.entity_type()),
706            &damage_source,
707        );
708        let enchanted_damage =
709            enchantment_helper::modify_damage(&attacking_item, &enchantment_context, attack_damage);
710        let magic_boost = attack_strength_scale * (enchanted_damage - attack_damage);
711        let mut base_damage = attack_damage * Self::base_damage_scale_factor(attack_strength_scale);
712        base_damage += ITEM_BEHAVIORS
713            .get_behavior(attacking_item.item())
714            .get_attack_damage_bonus(self, entity, base_damage, &damage_source);
715        let total_damage = base_damage + magic_boost;
716        let full_strength_attack = attack_strength_scale > 0.9;
717        let knockback_attack = self.is_sprinting() && full_strength_attack;
718        self.reset_attack_strength_ticker();
719
720        if total_damage <= 0.0 {
721            return false;
722        }
723
724        // TODO: Apply crits, sweep attacks, damage stats, and sounds.
725        let old_movement = entity.velocity();
726        let was_hurt = entity.hurt(&damage_source, total_damage);
727        if was_hurt {
728            self.set_last_hurt_mob(Some(target));
729            let sprint_knockback = if knockback_attack { 0.5 } else { 0.0 };
730            self.cause_extra_knockback(
731                entity,
732                Self::get_knockback(attack_knockback, &attacking_item, &enchantment_context)
733                    + sprint_knockback,
734                old_movement,
735            );
736            self.item_attack_interaction(entity, &damage_source, true);
737            self.cause_food_exhaustion(0.1);
738        }
739
740        enchantment_helper::do_post_piercing_attack_effects(self);
741        was_hurt
742    }

744    fn item_attack_interaction(
745        &self,
746        entity: &dyn Entity,
747        damage_source: &DamageSource,
748        apply_to_target: bool,
749    ) {
750        let post_attack_context =
751            EnchantmentPostAttackContext::new(entity, Some(self), Some(self), damage_source);
752        let (source_item, item_hurt_enemy) = {
753            let mut inventory = self.inventory.lock();
754            inventory.mutate_item_in_hand(InteractionHand::MainHand, |stack| {
755                if stack.is_empty() {
756                    return (ItemStack::empty(), false);
757                }
758                let behavior = ITEM_BEHAVIORS.get_behavior(stack.item());
759                if let Some(living_target) = entity.as_living_entity() {
760                    behavior.hurt_enemy(stack, living_target, self);
761                }
762                let source_item = stack.copy_with_count(stack.count());
763                (source_item, stack.get_weapon().is_some())
764            })
765        };
766
767        if apply_to_target {
768            enchantment_helper::do_post_attack_effects_with_item_source(
769                entity,
770                &source_item,
771                &post_attack_context,
772            );
773        }
774
775        if !item_hurt_enemy {
776            return;
777        }
778
779        let Some(living_target) = entity.as_living_entity() else {
780            return;
781        };
782        let has_infinite_materials = self.has_infinite_materials();
783        let mut inventory = self.inventory.lock();
784        inventory.mutate_item_in_hand(InteractionHand::MainHand, |stack| {
785            if stack.is_empty() {
786                return;
787            }
788            let behavior = ITEM_BEHAVIORS.get_behavior(stack.item());
789            behavior.post_hurt_enemy(stack, living_target, self);
790            if let Some(damage) = behavior.item_damage_per_attack(stack) {
791                stack.hurt_and_break(damage, has_infinite_materials);
792            }
793        });
794    }

796    /// Interacts with an entity using the held item.
797    pub fn interact_on(
798        &self,
799        entity: &dyn Entity,
800        hand: InteractionHand,
801        location: DVec3,
802    ) -> InteractionResult {
803        if self.is_spectator() {
804            // TODO: Open entity menu providers in spectator once that foundation exists.
805            return InteractionResult::Pass;
806        }
807
808        let inventory_access = InventoryAccess::new(self.inventory.clone(), hand);
809        let original_count = inventory_access.with_item(|item| item.count);
810        let result = entity.interact(self, hand, location);
811
812        if self.has_infinite_materials() {
813            inventory_access.with_item(|item| {
814                if item.count < original_count {
815                    item.count = original_count;
816                }
817            });
818        }
819
820        if result.consumes_action() {
821            return result;
822        }
823
824        if inventory_access.with_item(|item| item.is_empty()) {
825            return InteractionResult::Pass;
826        }
827        let Some(living_entity) = entity.as_living_entity() else {
828            return InteractionResult::Pass;
829        };
830        let result = living_entity.interact_living_entity_with_equippable(self, hand);
831        if self.has_infinite_materials() {
832            inventory_access.with_item(|item| {
833                if item.count < original_count {
834                    item.count = original_count;
835                }
836            });
837        }
838        if result.consumes_action() {
839            return result;
840        }
841
842        let item_ref = inventory_access.with_item(|item| item.item());
843        let item_behavior = ITEM_BEHAVIORS.get_behavior(item_ref);
844        let result = inventory_access.with_item(|item| {
845            item_behavior.interact_living_entity(item, self, living_entity, hand)
846        });
847        if self.has_infinite_materials() {
848            inventory_access.with_item(|item| {
849                if item.count < original_count {
850                    item.count = original_count;
851                }
852            });
853        }
854        result
855    }

857    /// Handles a client request to attack an entity.
858    pub fn handle_attack(&self, packet: SAttack) {
859        if !self.has_client_loaded() || self.is_spectator() {
860            return;
861        }
862
863        let world = self.get_world();
864        let Some(target) = world.get_accessible_entity_by_id(packet.entity_id) else {
865            return;
866        };
867
868        let target_pos = target.block_position();
869        if !world.world_border_snapshot().is_within_bounds_with_margin(
870            f64::from(target_pos.x()),
871            f64::from(target_pos.z()),
872            0.0,
873        ) {
874            return;
875        }
876
877        let main_hand_item = {
878            let inventory = self.inventory.lock();
879            let stack = inventory.get_item_in_hand(InteractionHand::MainHand);
880            stack.copy_with_count(stack.count())
881        };
882
883        if !self.is_within_attack_range_with_buffer(
884            &main_hand_item,
885            target.bounding_box(),
886            ATTACK_RANGE_BUFFER,
887        ) {
888            return;
889        }
890
891        if main_hand_item.get_piercing_weapon().is_some() {
892            return;
893        }
894
895        if Self::is_invalid_attack_target(self.id(), target.id(), target.entity_type()) {
896            self.disconnect(Self::invalid_entity_attacked_message());
897            log::warn!(
898                "Player {} tried to attack an invalid entity",
899                self.gameprofile.name
900            );
901            return;
902        }
903
904        if self.cannot_attack_with_item(&main_hand_item, 5) {
905            return;
906        }
907
908        let _ = self.attack(&target);
909    }

911    fn cannot_attack_with_item(&self, item_stack: &ItemStack, tolerance: i32) -> bool {
912        let required_strength = item_stack.minimum_attack_charge();
913        if required_strength <= 0.0 {
914            return false;
915        }
916
917        let optimistic_strength = {
918            let ticker = self.tick_state.lock().attack_strength_ticker() + tolerance;
919            ticker as f32 / self.current_item_attack_strength_delay()
920        };
921        optimistic_strength < required_strength
922    }

936        if !self.has_client_loaded() {
937            return;
938        }
939
940        let world = self.get_world();
941        let target = world.get_accessible_entity_by_id(packet.entity_id);

943        let Some(target) = target else {

946
947        let target_pos = target.block_position();
948        if !world.world_border_snapshot().is_within_bounds_with_margin(
949            f64::from(target_pos.x()),

964        if result.should_swing_server() {
965            self.swing(packet.hand, true);
966        }
967        self.broadcast_inventory_changes();
968    }

970    /// Sets the player's game mode and notifies the client.
971    ///
972    /// Returns `true` if the game mode was changed, `false` if the player was already in the requested game mode.
973    pub fn set_game_mode(&self, gamemode: GameType) -> bool {
974        if !self.change_game_mode_state(gamemode) {
975            return false;
976        }
977
978        // Update abilities based on new game mode (mirrors vanilla GameType.updatePlayerAbilities)
979        let flying_after_update = {
980            let mut abilities = self.abilities.lock();
981            abilities.update_for_game_mode(gamemode);
982            abilities.flying
983        };
984        if flying_after_update
985            && gamemode != GameType::Spectator
986            && self.is_in_range_of_ground_for_flight_disable()
987        {
988            self.set_flying(false);
989        }
990        self.send_abilities();
991
992        self.send_packet(CGameEvent {
993            event: GameEventType::ChangeGameMode,
994            data: gamemode.into(),
995        });
996
997        let update_packet =
998            CPlayerInfoUpdate::update_game_mode(self.gameprofile.id, gamemode as i32);
999        self.get_world().broadcast_to_all(update_packet);
1000
1001        if gamemode == GameType::Creative {
1002            self.reset_current_impulse_context();
1003        }
1004
1005        self.send_message(
1006            &translations::COMMANDS_GAMEMODE_SUCCESS_SELF
1007                .message([get_gamemode_translation(gamemode)])
1008                .into(),
1009        );
1010
1011        true
1012    }

1014    fn is_in_range_of_ground_for_flight_disable(&self) -> bool {
1015        let world = self.get_world();
1016        let collision_world = WorldCollisionProvider::for_entity(&world, self);
1017        let bounding_box = self.bounding_box();
1018        let collision_context =
1019            BlockCollisionContext::entity(self.position().y, self.is_descending())
1020                .with_fall_distance(self.fall_distance())
1021                .with_can_walk_on_powder_snow(self.can_walk_on_powder_snow());
1022
1023        if collision_world.has_collision_with_context(&bounding_box, collision_context) {
1024            return false;
1025        }
1026
1027        let below = WorldAabb::new(
1028            bounding_box.min_x(),
1029            bounding_box.min_y() - FLIGHT_DISABLE_RANGE,
1030            bounding_box.min_z(),
1031            bounding_box.max_x(),
1032            bounding_box.min_y(),
1033            bounding_box.max_z(),
1034        );
1035        let colliders = collision_world.get_collisions_with_context(&below, collision_context);
1036        if colliders.is_empty() {
1037            return false;
1038        }
1039
1040        let available_space_below =
1041            -shapes::collide(Axis::Y, &bounding_box, &colliders, -FLIGHT_DISABLE_RANGE);
1042        available_space_below < FLIGHT_DISABLE_RANGE
1043    }

1155    /// Returns true if player is within block interaction range plus a vanilla buffer.
1156    #[must_use]
1157    pub fn is_within_block_interaction_range_with_buffer(
1158        &self,
1159        pos: BlockPos,
1160        buffer: f64,
1161    ) -> bool {
1162        let player_pos = self.position();
1163        let eye_y = player_pos.y + self.get_eye_height();
1164
1165        let min_x = f64::from(pos.x());
1166        let min_y = f64::from(pos.y());
1167        let min_z = f64::from(pos.z());
1168        let max_x = min_x + 1.0;
1169        let max_y = min_y + 1.0;
1170        let max_z = min_z + 1.0;
1171
1172        let dx = f64::max(f64::max(min_x - player_pos.x, player_pos.x - max_x), 0.0);
1173        let dy = f64::max(f64::max(min_y - eye_y, eye_y - max_y), 0.0);
1174        let dz = f64::max(f64::max(min_z - player_pos.z, player_pos.z - max_z), 0.0);
1175        let dist_sq = dx * dx + dy * dy + dz * dz;
1176
1177        let base_range = self
1178            .attributes()
1179            .lock()
1180            .get_value(vanilla_attributes::BLOCK_INTERACTION_RANGE)
1181            .unwrap_or(4.5);
1182        let max_range = base_range + buffer;
1183        dist_sq < max_range * max_range
1184    }

1189        let player_pos = self.position();

1210    }

1268    /// Handles the use of an item on a block.
1269    ///
1270    /// Implements the logic from Java's `ServerGamePacketListenerImpl.handleUseItemOn()`.
1271    pub fn handle_use_item_on(&self, packet: SUseItemOn) {
1272        if !self.has_client_loaded() {
1273            return;
1274        }
1275
1276        self.ack_block_changes_up_to(packet.sequence);
1277
1278        let pos = packet.block_hit.block_pos;
1279        let direction = packet.block_hit.direction;
1280
1281        if !self.is_within_block_interaction_range(pos) {
1282            self.send_block_updates(pos, direction);
1283            return;
1284        }
1285
1286        let center_x = f64::from(pos.x()) + 0.5;
1287        let center_y = f64::from(pos.y()) + 0.5;
1288        let center_z = f64::from(pos.z()) + 0.5;
1289        let location = &packet.block_hit.location;
1290        let limit = 1.000_000_1;
1291
1292        if (location.x - center_x).abs() >= limit
1293            || (location.y - center_y).abs() >= limit
1294            || (location.z - center_z).abs() >= limit
1295        {
1296            log::warn!(
1297                "Rejecting UseItemOnPacket from {}: location {:?} too far from block {:?}",
1298                self.gameprofile.name,
1299                location,
1300                pos
1301            );
1302            self.send_block_updates(pos, direction);
1303            return;
1304        }
1305
1306        let world = self.get_world();
1307
1308        if pos.y() >= world.max_build_height() {
1309            // TODO: Send "build.tooHigh" message to player
1310            self.send_block_updates(pos, direction);
1311            return;
1312        }
1313
1314        if self.is_awaiting_teleport() {
1315            self.send_block_updates(pos, direction);
1316            return;
1317        }
1318
1319        if !world.may_interact(self, pos) {
1320            self.send_block_updates(pos, direction);
1321            return;
1322        }
1323
1324        let result = use_item_on(self, &world, packet.hand, &packet.block_hit);
1325
1326        if result.should_swing_server() {
1327            self.swing(packet.hand, true);
1328        }
1329
1330        self.send_block_updates(pos, direction);
1331        self.broadcast_inventory_changes();
1332    }

1334    /// Handles a player action packet (block breaking, item dropping, etc.).
1335    pub fn handle_player_action(&self, packet: SPlayerAction) {
1336        if !self.has_client_loaded() {
1337            return;
1338        }
1339
1340        let world = self.get_world();
1341        match packet.action {
1342            PlayerAction::StartDestroyBlock => {
1343                self.block_breaking.lock().handle_block_break_action(
1344                    self,
1345                    &world,
1346                    packet.pos,
1347                    BlockBreakAction::Start,
1348                    packet.direction,
1349                );
1350                self.ack_block_changes_up_to(packet.sequence);
1351            }
1352            PlayerAction::StopDestroyBlock => {
1353                self.block_breaking.lock().handle_block_break_action(
1354                    self,
1355                    &world,
1356                    packet.pos,
1357                    BlockBreakAction::Stop,
1358                    packet.direction,
1359                );
1360                self.ack_block_changes_up_to(packet.sequence);
1361            }
1362            PlayerAction::AbortDestroyBlock => {
1363                self.block_breaking.lock().handle_block_break_action(
1364                    self,
1365                    &world,
1366                    packet.pos,
1367                    BlockBreakAction::Abort,
1368                    packet.direction,
1369                );
1370                self.ack_block_changes_up_to(packet.sequence);
1371            }
1372            PlayerAction::DropAllItems => {
1373                self.drop_from_selected(true);
1374            }
1375            PlayerAction::DropItem => {
1376                self.drop_from_selected(false);
1377            }
1378            PlayerAction::ReleaseUseItem => {
1379                // TODO: Implement release use item (releasing bow, etc.)
1380                log::debug!("Player {} released use item", self.gameprofile.name);
1381            }
1382            PlayerAction::SwapItemWithOffhand => {
1383                if self.game_mode() == GameType::Spectator {
1384                    return;
1385                }
1386
1387                let changed = self.inventory.lock().swap_hands();
1388                if changed {
1389                    self.broadcast_entity_event(EntityStatus::SwapHands);
1390                    self.broadcast_inventory_changes();
1391                }
1392                // TODO: Stop active item use once the using-item foundation exists.
1393            }
1394            PlayerAction::Stab => {
1395                if self.game_mode() == GameType::Spectator {
1396                    return;
1397                }
1398
1399                let main_hand_item = {
1400                    let inventory = self.inventory.lock();
1401                    let stack = inventory.get_item_in_hand(InteractionHand::MainHand);
1402                    stack.copy_with_count(stack.count())
1403                };
1404                if self.cannot_attack_with_item(&main_hand_item, 5) {
1405                    return;
1406                }
1407                if let Some(piercing_weapon) = main_hand_item.get_piercing_weapon() {
1408                    self.piercing_attack(&main_hand_item, piercing_weapon);
1409                }
1410            }
1411        }
1412    }

1441    /// Handles the pick block action (middle click on a block).
1442    ///
1443    /// # Panics
1444    ///
1445    /// Panics if the behavior registry has not been initialized.
1446    pub fn handle_pick_item_from_block(&self, packet: SPickItemFromBlock) {
1447        if !self.is_within_block_interaction_range(packet.pos) {
1448            return;
1449        }
1450
1451        let state = self.get_world().get_block_state(packet.pos);
1452        if state.is_air() {
1453            return;
1454        }
1455
1456        let block = state.get_block();
1457        let block_behaviors = &*BLOCK_BEHAVIORS;
1458        let behavior = block_behaviors.get_behavior(block);
1459
1460        let include_data = self.has_infinite_materials() && packet.include_data;
1461
1462        let Some(item_stack) = behavior.get_clone_item_stack(block, state, include_data) else {
1463            return;
1464        };
1465
1466        if item_stack.is_empty() {
1467            return;
1468        }
1469
1470        // TODO: If include_data, add block entity NBT data to the item stack
1471        // This requires block entity support which isn't implemented yet
1472
1473        let mut inventory = self.inventory.lock();
1474
1475        let slot_with_item = inventory.find_slot_matching_item(&item_stack);
1476
1477        if slot_with_item != -1 {
1478            if PlayerInventory::is_hotbar_slot(slot_with_item as usize) {
1479                inventory.set_selected_slot(slot_with_item as u8);
1480            } else {
1481                inventory.pick_slot(slot_with_item);
1482            }
1483        } else if self.has_infinite_materials() {
1484            inventory.add_and_pick_item(item_stack);
1485        } else {
1486            return;
1487        }
1488
1489        self.send_packet(CSetHeldSlot {
1490            slot: i32::from(inventory.get_selected_slot()),
1491        });
1492
1493        drop(inventory);
1494        self.inventory_menu
1495            .lock()
1496            .behavior_mut()
1497            .broadcast_changes(&self.connection);
1498    }

1500    /// Handles a sign update packet from the client.
1501    pub fn handle_sign_update(&self, packet: SSignUpdate) {
1502        if !self.is_within_block_interaction_range(packet.pos) {
1503            return;
1504        }
1505
1506        let world = self.get_world();
1507
1508        let Some(block_entity) = world.get_block_entity(packet.pos) else {
1509            return;
1510        };
1511
1512        let mut guard = block_entity.lock();
1513        let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() else {
1514            return;
1515        };
1516
1517        if sign.is_waxed {
1518            return;
1519        }
1520
1521        if sign.get_player_who_may_edit() != Some(self.gameprofile.id) {
1522            log::warn!(
1523                "Player {} tried to edit sign they're not allowed to edit",
1524                self.gameprofile.name
1525            );
1526            return;
1527        }
1528
1529        let text = sign.get_text_mut(packet.is_front_text);
1530        for (i, line) in packet.lines.iter().enumerate() {
1531            if i < 4 {
1532                let stripped = strip_formatting_codes(line);
1533                text.set_message(i, TextComponent::plain(stripped));
1534            }
1535        }
1536
1537        sign.set_player_who_may_edit(None);
1538        sign.set_changed();
1539
1540        let update_tag = sign.get_update_tag();
1541        let block_entity_type = sign.get_type();
1542        let pos = packet.pos;
1543
1544        drop(guard);
1545
1546        if let Some(nbt) = update_tag {
1547            world.broadcast_block_entity_update(pos, block_entity_type, nbt);
1548        }
1549    }

1551    /// Opens the sign editor for the player.
1552    ///
1553    /// # Arguments
1554    /// * `pos` - Position of the sign block
1555    /// * `is_front_text` - Whether to edit front (true) or back (false) text
1556    pub fn open_sign_editor(&self, pos: BlockPos, is_front_text: bool) {
1557        let world = self.get_world();
1558
1559        if let Some(block_entity) = world.get_block_entity(pos) {
1560            let mut guard = block_entity.lock();
1561            if let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() {
1562                sign.set_player_who_may_edit(Some(self.gameprofile.id));
1563            }
1564        }
1565
1566        let state = world.get_block_state(pos);
1567        self.send_packet(CBlockUpdate {
1568            pos,
1569            block_state: state,
1570        });
1571
1572        self.send_packet(COpenSignEditor { pos, is_front_text });
1573    }

1576/// Strips Minecraft formatting codes (§ followed by a character) from a string.
1577///
1578/// This is equivalent to vanilla's `ChatFormatting.stripFormatting()`.
1579fn strip_formatting_codes(text: &str) -> String {
1580    let mut result = String::with_capacity(text.len());
1581    let mut chars = text.chars().peekable();
1582
1583    while let Some(c) = chars.next() {
1584        if c == '§' {
1585            chars.next();
1586        } else {
1587            result.push(c);
1588        }
1589    }
1590
1591    result
1592}
