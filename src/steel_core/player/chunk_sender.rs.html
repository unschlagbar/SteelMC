1//! This module is responsible for sending chunks to the client.
2//!
3//! Chunk sending runs on its own independent tick loop, separate from the game
4//! tick. The three-phase design (prepare → encode → commit) minimizes lock hold
5//! time on the per-player `ChunkSender` mutex so that game-tick operations like
6//! `mark_chunk_pending_to_send` and `drop_chunk` are never blocked for long.
7use rustc_hash::FxHashSet;
8use std::sync::Arc;

11use steel_protocol::packets::game::{
12    CChunkBatchFinished, CChunkBatchStart, CForgetLevelChunk, CLevelChunkWithLight,
13};
14use steel_protocol::utils::ConnectionProtocol;
15use steel_utils::locks::SyncMutex;
16use steel_utils::{ChunkPos, PackedChunkPos};

28/// Minimum chunks per tick (vanilla: 0.01)
29const MIN_CHUNKS_PER_TICK: f32 = 0.1f32;
30/// Maximum chunks per tick (vanilla: 64.0, we use 500.0 for faster loading)
31const MAX_CHUNKS_PER_TICK: f32 = 500.0;
32/// Starting chunks per tick (vanilla: 9.0)
33const START_CHUNKS_PER_TICK: f32 = 9.0;
34/// Maximum unacknowledged batches after first ack (vanilla: 10)
35const MAX_UNACKNOWLEDGED_BATCHES: u16 = 10;
36
37/// One chunk selected during the prepare phase.
38pub struct PreparedChunk {
39    /// Chunk position.
40    pub pos: ChunkPos,
41    /// Chunk holder to encode.
42    pub holder: Arc<ChunkHolder>,
43}
44
45/// Data collected during the prepare phase, used to encode and then commit.
46pub struct PreparedBatch {
47    /// Chunk holders to encode.
48    pub chunks: Vec<PreparedChunk>,
49    /// Whether the world dimension has a vanilla sky-light layer.
50    pub has_skylight: bool,
51    /// Snapshot of the player's generation counter at prepare time.
52    pub epoch_snapshot: u32,
53}
54
55/// Encoded chunk packet plus the holder content revision it was built from.
56#[derive(Clone)]
57pub struct EncodedChunk {
58    pos: ChunkPos,
59    packet: EncodedPacket,
60    content_revision: u64,
61}

63/// This struct is responsible for sending chunks to the client.
64#[derive(Debug)]
65pub struct ChunkSender {
66    /// A list of chunks that are waiting to be sent to the client.
67    pub pending_chunks: FxHashSet<ChunkPos>,
68    /// Chunks whose initial chunk packet has been queued for this client.
69    sent_chunks: FxHashSet<ChunkPos>,
70    /// The number of batches that have been sent to the client but have not been acknowledged yet.
71    pub unacknowledged_batches: u16,
72    /// The number of chunks that should be sent to the client per tick.
73    /// This is dynamically adjusted based on client feedback.
74    pub desired_chunks_per_tick: f32,
75    /// The number of chunks that can be sent to the client in the current batch.
76    pub batch_quota: f32,
77    /// The maximum number of unacknowledged batches allowed.
78    /// Starts at 1 and increases to `MAX_UNACKNOWLEDGED_BATCHES` after first ack.
79    pub max_unacknowledged_batches: u16,
80}
81
82impl ChunkSender {
83    /// Marks a chunk as pending to be sent to the client.
84    pub fn mark_chunk_pending_to_send(&mut self, pos: ChunkPos) {
85        self.sent_chunks.remove(&pos);
86        self.pending_chunks.insert(pos);
87    }
88
89    /// Drops a chunk from the client's view.
90    pub fn drop_chunk(&mut self, connection: &PlayerConnection, pos: ChunkPos) {
91        self.pending_chunks.remove(&pos);
92        if self.sent_chunks.remove(&pos) && !connection.closed() {
93            Self::send_packet(
94                connection,
95                CForgetLevelChunk {
96                    pos: PackedChunkPos::from(pos),
97                },
98            );
99        }
100    }
101
102    /// Encodes and sends a packet through the connection.
103    fn send_packet<P: ClientPacket>(connection: &PlayerConnection, packet: P) {
104        let encoded =
105            EncodedPacket::from_bare(packet, connection.compression(), ConnectionProtocol::Play)
106                .expect("Failed to encode packet");
107        connection.send_encoded(encoded);
108    }
109
110    /// Phase 1: Lock briefly to drain pending chunks and snapshot state.
111    ///
112    /// Returns `None` if there is nothing to send this tick.
113    pub fn prepare_batch(
114        &mut self,
115        world: &Arc<World>,
116        player_chunk_pos: ChunkPos,
117        chunk_send_epoch: &SyncMutex<u32>,
118    ) -> Option<PreparedBatch> {
119        if self.unacknowledged_batches >= self.max_unacknowledged_batches {
120            return None;
121        }
122
123        let max_batch_size = self.desired_chunks_per_tick.max(1.0);
124        self.batch_quota = (self.batch_quota + self.desired_chunks_per_tick).min(max_batch_size);
125
126        if self.batch_quota < 1.0 || self.pending_chunks.is_empty() {
127            return None;
128        }
129
130        let holders = self.collect_candidates(world, player_chunk_pos);
131        if holders.is_empty() {
132            return None;
133        }
134
135        let epoch_snapshot = *chunk_send_epoch.lock();
136
137        Some(PreparedBatch {
138            chunks: holders,
139            has_skylight: world.dimension_type.has_skylight,
140            epoch_snapshot,
141        })
142    }
143
144    /// Phase 2: Encode chunks without holding any lock. Called between prepare and commit.
145    ///
146    /// Uses a per-tick local cache so multiple players sharing the same chunks
147    /// don't re-encode them within the same sending tick. No mutex needed.
148    ///
149    /// # Panics
150    /// Panics if a chunk packet fails to encode.
151    pub fn encode_batch(
152        batch: &PreparedBatch,
153        cache: &mut rustc_hash::FxHashMap<ChunkPos, EncodedChunk>,
154        compression: Option<CompressionInfo>,
155    ) -> Vec<EncodedChunk> {
156        let mut encoded_chunks = Vec::with_capacity(batch.chunks.len());
157
158        for prepared in &batch.chunks {
159            let holder = &prepared.holder;
160            let pos = prepared.pos;
161
162            if let Some(cached) = cache.get(&pos)
163                && cached.content_revision == holder.packet_content_revision()
164            {
165                encoded_chunks.push(cached.clone());
166                continue;
167            }
168
169            let revision_before = holder.packet_content_revision();
170            let Some(chunk_guard) = holder.try_chunk(ChunkStatus::Full) else {
171                continue;
172            };
173            let ChunkAccess::Full(chunk) = &*chunk_guard else {
174                continue;
175            };
176
177            let encoded = EncodedPacket::from_bare(
178                CLevelChunkWithLight {
179                    x: pos.0.x,
180                    z: pos.0.y,
181                    chunk_data: chunk.extract_chunk_data(),
182                    light_data: chunk.extract_light_data(batch.has_skylight),
183                },
184                compression,
185                ConnectionProtocol::Play,
186            )
187            .expect("Failed to encode chunk packet");
188            let revision_after = holder.packet_content_revision();
189            if revision_before != revision_after {
190                continue;
191            }
192
193            let encoded = EncodedChunk {
194                pos,
195                packet: encoded,
196                content_revision: revision_after,
197            };
198            cache.insert(pos, encoded.clone());
199            encoded_chunks.push(encoded);
200        }
201
202        encoded_chunks
203    }
204
205    /// Phase 3: Lock briefly to verify generation counter and send the batch.
206    ///
207    /// If the player teleported between prepare and commit (generation counter
208    /// changed), the batch is discarded.
209    pub fn commit_batch(
210        &mut self,
211        batch: &PreparedBatch,
212        encoded_chunks: Vec<EncodedChunk>,
213        connection: &PlayerConnection,
214        chunk_send_epoch: &SyncMutex<u32>,
215    ) -> Vec<ChunkPos> {
216        let epoch = chunk_send_epoch.lock();
217        if *epoch != batch.epoch_snapshot {
218            return Vec::new();
219        }
220        drop(epoch);
221
222        let mut valid_chunks = Vec::with_capacity(encoded_chunks.len());
223        for encoded in encoded_chunks {
224            if !self.pending_chunks.contains(&encoded.pos) {
225                continue;
226            }
227            let Some(prepared) = batch.chunks.iter().find(|chunk| chunk.pos == encoded.pos) else {
228                continue;
229            };
230            if prepared.holder.packet_content_revision() != encoded.content_revision {
231                continue;
232            }
233            valid_chunks.push(encoded);
234        }
235
236        if valid_chunks.is_empty() {
237            return Vec::new();
238        }
239
240        self.unacknowledged_batches += 1;
241        self.batch_quota -= valid_chunks.len() as f32;
242
243        Self::send_packet(connection, CChunkBatchStart {});
244
245        let batch_size = valid_chunks.len();
246        for encoded in &valid_chunks {
247            connection.send_encoded(encoded.packet.clone());
248        }
249
250        Self::send_packet(
251            connection,
252            CChunkBatchFinished {
253                batch_size: batch_size as i32,
254            },
255        );
256
257        let mut sent_chunks = Vec::with_capacity(valid_chunks.len());
258        for encoded in valid_chunks {
259            self.pending_chunks.remove(&encoded.pos);
260            self.sent_chunks.insert(encoded.pos);
261            sent_chunks.push(encoded.pos);
262        }
263        sent_chunks
264    }
265
266    fn collect_candidates(
267        &mut self,
268        world: &Arc<World>,
269        player_chunk_pos: ChunkPos,
270    ) -> Vec<PreparedChunk> {
271        let max_batch_size = self.batch_quota.floor() as usize;
272        let mut candidates: Vec<ChunkPos> = self.pending_chunks.iter().copied().collect();
273
274        // Sort by distance to player
275        candidates.sort_by_key(|pos| Self::chunk_distance_squared(*pos, player_chunk_pos));
276
277        let mut chunks_to_send = Vec::new();
278
279        for pos in candidates {
280            if chunks_to_send.len() >= max_batch_size {
281                break;
282            }
283
284            if let Some(holder) = world
285                .chunk_map
286                .chunks
287                .read_sync(&pos, |_, chunk| chunk.clone())
288                && holder.persisted_status() == Some(ChunkStatus::Full)
289            {
290                chunks_to_send.push(PreparedChunk { pos, holder });
291            }
292        }
293        chunks_to_send
294    }
295
296    fn chunk_distance_squared(pos: ChunkPos, player_chunk_pos: ChunkPos) -> u64 {
297        let dx = u64::from(pos.0.x.abs_diff(player_chunk_pos.0.x));
298        let dz = u64::from(pos.0.y.abs_diff(player_chunk_pos.0.y));
299        dx.saturating_mul(dx).saturating_add(dz.saturating_mul(dz))
300    }

302    /// Handles the acknowledgement of a chunk batch from the client.
303    ///
304    /// The client sends back its desired chunks per tick based on how fast it can
305    /// process chunks. We clamp this value and use it to adjust our sending rate.
306    pub const fn on_chunk_batch_received_by_client(
307        &mut self,
308        desired_chunks_per_tick: f32,
309    ) -> bool {
310        if self.unacknowledged_batches == 0 {
311            return false;
312        }
313
314        self.unacknowledged_batches = self.unacknowledged_batches.saturating_sub(1);
315
316        // Handle NaN and clamp to valid range (vanilla uses 0.01-64, we use 0.01-500)
317        self.desired_chunks_per_tick = if desired_chunks_per_tick.is_nan() {
318            MIN_CHUNKS_PER_TICK
319        } else {
320            desired_chunks_per_tick.clamp(MIN_CHUNKS_PER_TICK, MAX_CHUNKS_PER_TICK)
321        };
322
323        // Reset batch quota when all batches are acknowledged
324        if self.unacknowledged_batches == 0 {
325            self.batch_quota = 1.0;
326        }
327
328        // After receiving the first acknowledgement, allow more unacknowledged batches
329        // for better pipelining (vanilla behavior)
330        self.max_unacknowledged_batches = MAX_UNACKNOWLEDGED_BATCHES;
331        true
332    }
333
334    /// Returns whether the client has been queued the initial chunk packet.
335    #[must_use]
336    pub fn is_chunk_sent(&self, pos: ChunkPos) -> bool {
337        self.sent_chunks.contains(&pos)
338    }
339
340    /// Returns a snapshot of all sent chunks for this player.
341    #[must_use]
342    pub fn sent_chunks_snapshot(&self) -> FxHashSet<ChunkPos> {
343        self.sent_chunks.clone()
344    }
345}

347impl Default for ChunkSender {
348    fn default() -> Self {
349        Self {
350            pending_chunks: FxHashSet::default(),
351            sent_chunks: FxHashSet::default(),
352            unacknowledged_batches: 0,
353            desired_chunks_per_tick: START_CHUNKS_PER_TICK,
354            batch_quota: 0.0,
355            max_unacknowledged_batches: 1,
356        }
357    }
358}

364    #[test]
365    fn chunk_batch_ack_without_outstanding_batch_does_not_update_pacing() {
366        let mut sender = ChunkSender::default();
367
368        assert!(!sender.on_chunk_batch_received_by_client(64.0));
369        assert_eq!(sender.unacknowledged_batches, 0);
370        assert_eq!(
371            sender.desired_chunks_per_tick.to_bits(),
372            START_CHUNKS_PER_TICK.to_bits()
373        );
374        assert_eq!(sender.batch_quota.to_bits(), 0.0_f32.to_bits());
375        assert_eq!(sender.max_unacknowledged_batches, 1);
376    }
377
378    #[test]
379    fn chunk_batch_ack_updates_pacing_for_outstanding_batch() {
380        let mut sender = ChunkSender {
381            unacknowledged_batches: 1,
382            ..ChunkSender::default()
383        };
384
385        assert!(sender.on_chunk_batch_received_by_client(f32::NAN));
386        assert_eq!(sender.unacknowledged_batches, 0);
387        assert_eq!(
388            sender.desired_chunks_per_tick.to_bits(),
389            MIN_CHUNKS_PER_TICK.to_bits()
390        );
391        assert_eq!(sender.batch_quota.to_bits(), 1.0_f32.to_bits());
392        assert_eq!(
393            sender.max_unacknowledged_batches,
394            MAX_UNACKNOWLEDGED_BATCHES
395        );
396    }
