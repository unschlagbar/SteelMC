20/// Persistent player data saved by Steel's storage backend.
21///
22/// This is Steel's runtime save snapshot. Vanilla import/export should live outside
23/// server runtime storage so compatibility logic does not constrain the native format.
24#[derive(Debug, Clone)]
25pub struct PersistentPlayerData {
26    /// Position (x, y, z) in absolute world coordinates.
27    pub pos: [f64; 3],
28
29    /// Velocity (x, y, z) in blocks per tick.
30    pub motion: [f64; 3],
31
32    /// Rotation (yaw, pitch) in degrees.
33    pub rotation: [f32; 2],
34
35    /// Whether the player is on the ground.
36    pub on_ground: bool,
37
38    /// Whether the player is elytra gliding.
39    pub fall_flying: bool,
40
41    /// Vanilla `remainingFireTicks`.
42    pub remaining_fire_ticks: i32,
43
44    /// Synchronized vanilla `TicksFrozen`.
45    pub ticks_frozen: i32,
46
47    /// Vanilla `isInPowderSnow`.
48    pub is_in_powder_snow: bool,
49
50    /// Vanilla `wasInPowderSnow`.
51    pub was_in_powder_snow: bool,
52
53    /// Vanilla `hasVisualFire`.
54    pub has_visual_fire: bool,
55
56    /// Current health points.
57    pub health: f32,
58
59    /// Current game mode (0=survival, 1=creative, 2=adventure, 3=spectator).
60    pub game_mode: i32,
61
62    /// Previous game mode of the player, or `None` if vanilla has not recorded one yet.
63    pub prev_game_mode: Option<i32>,
64
65    /// Player abilities (flight, invulnerability, etc.).
66    pub abilities: PersistentAbilities,
67
68    /// Inventory items with slot indices.
69    pub inventory: Vec<PersistentSlot>,
70
71    /// Currently selected hotbar slot (0-8).
72    pub selected_slot: i32,
73
74    /// Loaded world identifier (e.g., "minecraft:overworld").
75    pub world: String,
76
77    /// Current food level (0–20, default 20).
78    pub food_level: i32,
79
80    /// Food saturation level (0.0–`food_level`, default 5.0).
81    pub food_saturation_level: f32,
82
83    /// Accumulated food exhaustion (0.0–40.0, default 0.0).
84    pub food_exhaustion_level: f32,
85
86    /// Internal tick timer for regen/starvation (default 0).
87    pub food_tick_timer: i32,
88
89    /// Data version for format migrations.
90    pub data_version: i32,
91
92    /// Current experience level
93    pub experience_level: i32,
94
95    /// To progress to the next experience level
96    pub experience_progress: f32,
97
98    /// The checked value of the Score, cannot decrease below 0 (???)
99    /// TODO: what exactly is experienceTotal
100    pub experience_total: i32,
101
102    /// A non decreasing value of the experience orbs added (/xp add, picking up orbs and advancements)
103    /// this value can be negative by using (/xp add ... -x)
104    pub score: i32,
105
106    /// Vanilla one-player root vehicle tree stored with the player instead of chunk data.
107    pub root_vehicle: Option<PersistentRootVehicle>,
108}

147impl PersistentPlayerData {
148    /// Extracts persistent data from a live player.
149    #[must_use]
150    pub fn from_player(player: &Player) -> Self {
151        let pos = player.position();
152        let (yaw, pitch) = player.rotation();
153        let delta = player.velocity();
154        let on_ground = player.on_ground();
155        let fall_flying = player.is_fall_flying();
156        let fire_freeze = player.fire_freeze_state();
157        let abilities = player.abilities.lock();
158        let inventory = player.inventory.lock();
159        let food_data = player.food_data.lock();
160
161        // Collect non-empty inventory slots
162        let mut slots = Vec::new();
163        // Main inventory (0-35) and equipment (36-42)
164        for slot in 0..43 {
165            let item = inventory.get_item(slot);
166            if !item.is_empty() {
167                slots.push(PersistentSlot {
168                    slot: slot as i8,
169                    item: item.clone(),
170                });
171            }
172        }
173
174        let (experience_level, experience_progress, experience_total, score) = {
175            let lock = player.experience.lock();
176            (
177                lock.level(),
178                lock.progress() as f32,
179                lock.total_points(),
180                lock.score,
181            )
182        };
183        let root_vehicle = Self::root_vehicle_from_player(player)
184            .or_else(|| player.pending_root_vehicle_for_current_world());
185
186        Self {
187            pos: [pos.x, pos.y, pos.z],
188            motion: [delta.x, delta.y, delta.z],
189            rotation: [yaw, pitch],
190            on_ground,
191            fall_flying,
192            remaining_fire_ticks: fire_freeze.remaining_fire_ticks(),
193            ticks_frozen: fire_freeze.ticks_frozen(),
194            is_in_powder_snow: fire_freeze.is_in_powder_snow(),
195            was_in_powder_snow: fire_freeze.was_in_powder_snow(),
196            has_visual_fire: fire_freeze.has_visual_fire(),
197            health: player.get_health(),
198            game_mode: player.game_mode() as i32,
199            prev_game_mode: player
200                .previous_game_mode()
201                .map(|game_mode| game_mode as i32),
202            abilities: PersistentAbilities {
203                invulnerable: abilities.invulnerable,
204                flying: abilities.flying,
205                may_fly: abilities.may_fly,
206                instabuild: abilities.instabuild,
207                may_build: abilities.may_build,
208                flying_speed: abilities.flying_speed,
209                walking_speed: abilities.walking_speed,
210            },
211            inventory: slots,
212            selected_slot: i32::from(inventory.get_selected_slot()),
213            world: player.get_world().key.to_string(),
214            food_level: food_data.food_level,
215            food_saturation_level: food_data.saturation_level,
216            food_exhaustion_level: food_data.exhaustion_level,
217            food_tick_timer: food_data.tick_timer,
218            data_version: PLAYER_DATA_VERSION,
219            experience_level,
220            experience_progress,
221            experience_total,
222            score,
223            root_vehicle,
224        }
225    }
226
227    fn root_vehicle_from_player(player: &Player) -> Option<PersistentRootVehicle> {
228        let vehicle = player.vehicle()?;
229        let root_vehicle = player.root_vehicle()?;
230        if root_vehicle.id() == player.id() || !root_vehicle.has_exactly_one_player_passenger() {
231            return None;
232        }
233
234        let entity = ChunkStorage::entity_tree_to_persistent(&root_vehicle)?;
235        Some(PersistentRootVehicle {
236            attach: *vehicle.uuid().as_bytes(),
237            entity,
238        })
239    }
240}

284impl PersistentPlayerData {
285    /// Applies the saved data to a player.
286    ///
287    /// This restores position, rotation, inventory, abilities, etc.
288    pub fn apply_to_player(&self, player: &Player) {
289        self.apply_to_player_inner(player, true);
290    }
291
292    /// Applies saved gameplay state without restoring world-local location data.
293    ///
294    /// Used when the saved world no longer exists and the player must spawn at
295    /// the target world's default spawn instead of stale coordinates.
296    pub fn apply_to_player_without_location(&self, player: &Player) {
297        self.apply_to_player_inner(player, false);
298    }
299
300    fn apply_to_player_inner(&self, player: &Player, restore_location: bool) {
301        use glam::DVec3;
302
303        if restore_location {
304            // Position
305            player
306                .base()
307                .set_position_local(DVec3::new(self.pos[0], self.pos[1], self.pos[2]));
308
309            // Rotation
310            player.set_rotation((self.rotation[0], self.rotation[1]));
311
312            // Motion/velocity
313            player.set_velocity(DVec3::new(self.motion[0], self.motion[1], self.motion[2]));
314
315            // Ground state
316            player.set_fall_flying(self.fall_flying);
317            player.set_on_ground(self.on_ground);
318        }
319
320        player
321            .base()
322            .set_fire_freeze_state(EntityFireFreezeState::from_parts(
323                self.remaining_fire_ticks,
324                self.ticks_frozen,
325                self.is_in_powder_snow,
326                self.was_in_powder_snow,
327                self.has_visual_fire,
328            ));
329        player.sync_base_fire_freeze_entity_data();
330
331        // Health
332        player.set_health(self.health);
333
334        // Game mode
335        player.restore_game_modes(
336            self.game_mode.into(),
337            self.prev_game_mode.map(GameType::from),
338        );
339
340        // Abilities
341        *player.abilities.lock() = self.abilities.clone().into();
342
343        // Inventory
344        {
345            let mut inventory = player.inventory.lock();
346            // Clear existing inventory first
347            for slot in 0..43 {
348                inventory.set_item(slot, ItemStack::empty());
349            }
350            // Restore saved items
351            for slot_data in &self.inventory {
352                let slot_index = slot_data.slot as usize;
353                if slot_index < 43 {
354                    inventory.set_item(slot_index, slot_data.item.clone());
355                }
356            }
357            // Restore selected slot
358            let selected = self.selected_slot.clamp(0, 8) as u8;
359            inventory.set_selected_slot(selected);
360        }
361
362        // Food data
363        {
364            let mut food = player.food_data.lock();
365            food.food_level = self.food_level;
366            food.saturation_level = self.food_saturation_level;
367            food.exhaustion_level = self.food_exhaustion_level;
368            food.tick_timer = self.food_tick_timer;
369        }
370
371        {
372            let mut experience = player.experience.lock();
373            experience.set_levels(self.experience_level);
374            experience.set_progress(f64::from(self.experience_progress));
375            experience.score = self.score;
376        }
377    }
