19/// Improved Perlin noise generator.
20///
21/// This implements the improved Perlin noise algorithm as used in Minecraft.
22/// Each instance has a permutation table and offset values initialized from
23/// a random source.
24#[derive(Debug, Clone)]
25pub struct ImprovedNoise {
26    /// Permutation table (256 bytes)
27    p: [u8; 256],
28    /// X offset for the noise coordinates
29    pub xo: f64,
30    /// Y offset for the noise coordinates
31    pub yo: f64,
32    /// Z offset for the noise coordinates
33    pub zo: f64,
34    yo_floor: i32,
35    yo_fraction: f64,
36    zo_floor: i32,
37    zo_fraction: f64,
38}

41    /// Creates a new `ImprovedNoise` from a random source.
42    ///
43    /// Initializes the permutation table using Fisher-Yates shuffle
44    /// and sets random offsets.
45    pub fn new<R: Random>(random: &mut R) -> Self {
46        let xo = random.next_f64() * 256.0;
47        let yo = random.next_f64() * 256.0;
48        let zo = random.next_f64() * 256.0;
49
50        let mut p = [0u8; 256];
51        #[expect(
52            clippy::needless_range_loop,
53            reason = "index is used as the initial permutation value"
54        )]
55        for i in 0..256 {
56            p[i] = i as u8;
57        }
58
59        // Fisher-Yates shuffle matching vanilla's implementation
60        for i in 0..256 {
61            let offset = random.next_i32_bounded((256 - i) as i32) as usize;
62            p.swap(i, i + offset);
63        }
64
65        let yo_floor = fast_floor(yo);
66        let yo_fraction = yo - f64::from(yo_floor);
67        let zo_floor = fast_floor(zo);
68        let zo_fraction = zo - f64::from(zo_floor);
69
70        Self {
71            p,
72            xo,
73            yo,
74            zo,
75            yo_floor,
76            yo_fraction,
77            zo_floor,
78            zo_fraction,
79        }
80    }

82    /// Sample noise at the given coordinates.
83    ///
84    /// This is the standard 3D Perlin noise sampling without Y scaling.
85    #[inline]
86    #[must_use]
87    pub fn noise(&self, x: f64, y: f64, z: f64) -> f64 {
88        let x = x + self.xo;
89        let y = y + self.yo;
90        let z = z + self.zo;
91
92        let xf = fast_floor(x);
93        let yf = fast_floor(y);
94        let zf = fast_floor(z);
95
96        let xr = x - f64::from(xf);
97        let yr = y - f64::from(yf);
98        let zr = z - f64::from(zf);
99
100        self.sample_and_lerp(xf, yf, zf, xr, yr, zr, yr)
101    }

103    /// Calculate Perlin noise using SIMD vectors.
104    #[inline]
105    #[must_use]
106    pub fn noise_simd<F, const N: usize>(
107        &self,
108        x: Simd<F, N>,
109        y: Simd<F, N>,
110        z: Simd<F, N>,
111    ) -> Simd<F, N>
112    where
113        F: SimdElement + SimdCast,
114        Simd<F, N>: SimdFloat<Cast<i32> = Simd<i32, N>>
115            + SimdPartialOrd
116            + SimdPartialEq<Mask = Mask<<F as SimdElement>::Mask, N>>
117            + ops::Add<Output = Simd<F, N>>
118            + ops::Sub<Output = Simd<F, N>>
119            + ops::Mul<Output = Simd<F, N>>
120            + ops::Neg<Output = Simd<F, N>>,
121    {
122        let x = x + Simd::splat(self.xo).cast();
123        let y = y + Simd::splat(self.yo).cast();
124        let z = z + Simd::splat(self.zo).cast();
125
126        let xf = fast_floor_simd::<F, i32, N>(x);
127        let yf = fast_floor_simd::<F, i32, N>(y);
128        let zf = fast_floor_simd::<F, i32, N>(z);
129
130        let xr = x - xf.cast();
131        let yr = y - yf.cast();
132        let zr = z - zf.cast();
133
134        self.sample_and_lerp_simd(xf, yf, zf, xr, yr, zr, yr)
135    }

204    /// Sample noise with Y scale and fudge parameters.
205    ///
206    /// The `y_scale` and `y_fudge` parameters are used for terrain generation
207    /// where vertical noise needs special handling.
208    ///
209    /// # Arguments
210    /// * `x`, `y`, `z` - The coordinates to sample
211    /// * `y_scale` - Y scaling factor (0.0 to disable)
212    /// * `y_fudge` - Y fudge factor for floor snapping
213    #[must_use]
214    #[expect(
215        clippy::similar_names,
216        reason = "yr_fudge and y_fudge match vanilla naming"
217    )]
218    pub fn noise_with_y_scale(&self, x: f64, y: f64, z: f64, y_scale: f64, y_fudge: f64) -> f64 {
219        let x = x + self.xo;
220        let y = y + self.yo;
221        let z = z + self.zo;
222
223        let xf = fast_floor(x);
224        let yf = fast_floor(y);
225        let zf = fast_floor(z);
226
227        let xr = x - f64::from(xf);
228        let yr = y - f64::from(yf);
229        let zr = z - f64::from(zf);
230
231        // Calculate Y fudge for terrain generation
232        #[expect(
233            clippy::if_not_else,
234            reason = "matches vanilla's conditional structure"
235        )]
236        let yr_fudge = if y_scale != 0.0 {
237            let fudge_limit = if y_fudge >= 0.0 && y_fudge < yr {
238                y_fudge
239            } else {
240                yr
241            };
242            // SHIFT_UP_EPSILON = 1.0E-7F in Java (float literal promoted to double)
243            (fudge_limit / y_scale + f64::from(1.0e-7_f32)).floor() * y_scale
244        } else {
245            0.0
246        };
247        self.sample_and_lerp(xf, yf, zf, xr, yr - yr_fudge, zr, yr)
248    }

250    /// Sample noise at grid point and interpolate.
251    ///
252    /// The 8 corner gradient-dot products are evaluated as 2 × `f64x4` so the
253    /// per-lane math stays identical to the scalar path (`((gx*xr) + (gy*yr)) + (gz*zr)`),
254    /// which preserves bit-identical output. Inspired by C2ME's `c2me-opts-math`
255    /// flat-gradient SIMD form.
256    #[expect(clippy::too_many_arguments, reason = "matches vanilla signature")]
257    fn sample_and_lerp(
258        &self,
259        x: i32,
260        y: i32,
261        z: i32,
262        xr: f64,
263        yr: f64,
264        zr: f64,
265        yr_original: f64,
266    ) -> f64 {
267        let x = x as u8;
268        let y = y as u8;
269        let z = z as u8;
270        // Get permutation indices for the 8 corners
271        let x0 = self.p[x as usize];
272        let x1 = self.p[x.wrapping_add(1) as usize];
273
274        let xy00 = self.p[x0.wrapping_add(y) as usize];
275        let xy01 = self.p[x0.wrapping_add(y).wrapping_add(1) as usize];
276        let xy10 = self.p[x1.wrapping_add(y) as usize];
277        let xy11 = self.p[x1.wrapping_add(y).wrapping_add(1) as usize];
278
279        let (d000, d100, d010, d110, d001, d101, d011, d111) = {
280            #[cfg(target_feature = "avx512f")]
281            {
282                // Hashes for the z-face and z+1-face, in (000,100,010,110) order.
283                let h_z0 = [
284                    self.p[xy00.wrapping_add(z) as usize] as usize,
285                    self.p[xy10.wrapping_add(z) as usize] as usize,
286                    self.p[xy01.wrapping_add(z) as usize] as usize,
287                    self.p[xy11.wrapping_add(z) as usize] as usize,
288                ];
289                let h_z1 = [
290                    self.p[xy00.wrapping_add(z).wrapping_add(1) as usize] as usize,
291                    self.p[xy10.wrapping_add(z).wrapping_add(1) as usize] as usize,
292                    self.p[xy01.wrapping_add(z).wrapping_add(1) as usize] as usize,
293                    self.p[xy11.wrapping_add(z).wrapping_add(1) as usize] as usize,
294                ];
295
296                let xr_v = f64x4::from_array([xr, xr - 1.0, xr, xr - 1.0]);
297                let yr_v = f64x4::from_array([yr, yr, yr - 1.0, yr - 1.0]);
298                let zr_v0 = f64x4::splat(zr);
299                let zr_v1 = f64x4::splat(zr - 1.0);
300
301                let [d000, d100, d010, d110] = grad_dot_simd(h_z0, xr_v, yr_v, zr_v0).to_array();
302                let [d001, d101, d011, d111] = grad_dot_simd(h_z1, xr_v, yr_v, zr_v1).to_array();
303                (d000, d100, d010, d110, d001, d101, d011, d111)
304            }
305
306            #[cfg(not(target_feature = "avx512f"))]
307            {
308                // Calculate gradient dot products at each corner
309                let d000 = grad_dot(self.p[xy00.wrapping_add(z) as usize] as usize, xr, yr, zr);
310                let d100 = grad_dot(
311                    self.p[xy10.wrapping_add(z) as usize] as usize,
312                    xr - 1.0,
313                    yr,
314                    zr,
315                );
316                let d010 = grad_dot(
317                    self.p[xy01.wrapping_add(z) as usize] as usize,
318                    xr,
319                    yr - 1.0,
320                    zr,
321                );
322                let d110 = grad_dot(
323                    self.p[xy11.wrapping_add(z) as usize] as usize,
324                    xr - 1.0,
325                    yr - 1.0,
326                    zr,
327                );
328                let d001 = grad_dot(
329                    self.p[xy00.wrapping_add(z).wrapping_add(1) as usize] as usize,
330                    xr,
331                    yr,
332                    zr - 1.0,
333                );
334                let d101 = grad_dot(
335                    self.p[xy10.wrapping_add(z).wrapping_add(1) as usize] as usize,
336                    xr - 1.0,
337                    yr,
338                    zr - 1.0,
339                );
340                let d011 = grad_dot(
341                    self.p[xy01.wrapping_add(z).wrapping_add(1) as usize] as usize,
342                    xr,
343                    yr - 1.0,
344                    zr - 1.0,
345                );
346                let d111 = grad_dot(
347                    self.p[xy11.wrapping_add(z).wrapping_add(1) as usize] as usize,
348                    xr - 1.0,
349                    yr - 1.0,
350                    zr - 1.0,
351                );
352                (d000, d100, d010, d110, d001, d101, d011, d111)
353            }
354        };
355        // Apply smoothstep interpolation
356        let x_alpha = smoothstep(xr);
357        let y_alpha = smoothstep(yr_original);
358        let z_alpha = smoothstep(zr);
359
360        lerp3(
361            x_alpha, y_alpha, z_alpha, d000, d100, d010, d110, d001, d101, d011, d111,
362        )
363    }

365    #[inline]
366    fn p_simd<const N: usize>(&self, idx: Simd<u8, N>) -> Simd<u8, N> {
367        let offset = idx.cast::<usize>();
368        let p = Simd::splat(self.p.as_ptr()).wrapping_add(offset);
369        // SAFETY: `idx` is a `Simd<u8, N>`, meaning each lane's index is at most 255.
370        // `self.p` has length 256, so all offsets are guaranteed to be within bounds of `self.p`.
371        unsafe { Simd::gather_ptr(p) }
372    }

374    /// Sample noise at grid point and interpolate.
375    #[expect(clippy::too_many_arguments, reason = "matches vanilla signature")]
376    fn sample_and_lerp_simd<F, const N: usize>(
377        &self,
378        x: Simd<i32, N>,
379        y: Simd<i32, N>,
380        z: Simd<i32, N>,
381        xr: Simd<F, N>,
382        yr: Simd<F, N>,
383        zr: Simd<F, N>,
384        yr_original: Simd<F, N>,
385    ) -> Simd<F, N>
386    where
387        F: SimdElement + SimdCast,
388        Simd<F, N>: ops::Mul<Output = Simd<F, N>>
389            + ops::Add<Output = Simd<F, N>>
390            + ops::Sub<Output = Simd<F, N>>
391            + ops::Neg<Output = Simd<F, N>>,
392    {
393        let x = x.cast::<u8>();
394        let y = y.cast::<u8>();
395        let z = z.cast::<u8>();
396        // Get permutation indices for the 8 corners
397        let x0 = self.p_simd(x);
398        let x1 = self.p_simd(x + Simd::splat(1));
399
400        let xy00 = self.p_simd(x0 + y);
401        let xy01 = self.p_simd(x0 + y + Simd::splat(1));
402        let xy10 = self.p_simd(x1 + y);
403        let xy11 = self.p_simd(x1 + y + Simd::splat(1));
404
405        let h000 = self.p_simd(xy00 + z).cast::<usize>().to_array();
406        let h100 = self.p_simd(xy10 + z).cast::<usize>().to_array();
407        let h010 = self.p_simd(xy01 + z).cast::<usize>().to_array();
408        let h110 = self.p_simd(xy11 + z).cast::<usize>().to_array();
409        let h001 = self
410            .p_simd(xy00 + z + Simd::splat(1))
411            .cast::<usize>()
412            .to_array();
413        let h101 = self
414            .p_simd(xy10 + z + Simd::splat(1))
415            .cast::<usize>()
416            .to_array();
417        let h011 = self
418            .p_simd(xy01 + z + Simd::splat(1))
419            .cast::<usize>()
420            .to_array();
421        let h111 = self
422            .p_simd(xy11 + z + Simd::splat(1))
423            .cast::<usize>()
424            .to_array();
425
426        // Calculate gradient dot products at each corner
427        let d000 = grad_dot_simd(h000, xr, yr, zr);
428        let d100 = grad_dot_simd(h100, xr - Simd::splat(1.0).cast::<F>(), yr, zr);
429        let d010 = grad_dot_simd(h010, xr, yr - Simd::splat(1.0).cast::<F>(), zr);
430        let d110 = grad_dot_simd(
431            h110,
432            xr - Simd::splat(1.0).cast::<F>(),
433            yr - Simd::splat(1.0).cast::<F>(),
434            zr,
435        );
436        let d001 = grad_dot_simd(h001, xr, yr, zr - Simd::splat(1.0).cast::<F>());
437        let d101 = grad_dot_simd(
438            h101,
439            xr - Simd::splat(1.0).cast::<F>(),
440            yr,
441            zr - Simd::splat(1.0).cast::<F>(),
442        );
443        let d011 = grad_dot_simd(
444            h011,
445            xr,
446            yr - Simd::splat(1.0).cast::<F>(),
447            zr - Simd::splat(1.0).cast::<F>(),
448        );
449        let d111 = grad_dot_simd(
450            h111,
451            xr - Simd::splat(1.0).cast::<F>(),
452            yr - Simd::splat(1.0).cast::<F>(),
453            zr - Simd::splat(1.0).cast::<F>(),
454        );
455
456        // Apply smoothstep interpolation
457        let x_alpha = smoothstep_simd(xr);
458        let y_alpha = smoothstep_simd(yr_original);
459        let z_alpha = smoothstep_simd(zr);
460
461        lerp3_simd(
462            x_alpha, y_alpha, z_alpha, d000, d100, d010, d110, d001, d101, d011, d111,
463        )
464    }

466    /// Generic N-lane form of [`Self::noise_with_y_scale_4x`]. Each lane runs the
467    /// exact per-lane math of the scalar [`Self::noise_with_y_scale`], so any
468    /// supported lane width yields bit-identical per-lane results — only the
469    /// SIMD batch size changes. `f64x4` ≡ `noise_with_y_scale_simd::<4>`.
470    #[inline]
471    #[must_use]
472    pub fn noise_with_y_scale_simd<const N: usize>(
473        &self,
474        x: f64,
475        ys: Simd<f64, N>,
476        z: f64,
477        y_scale: f64,
478        y_fudges: Simd<f64, N>,
479    ) -> Simd<f64, N> {
480        // Shared x/z offset and floor
481        let x = x + self.xo;
482        let z = z + self.zo;
483        let xf = fast_floor(x);
484        let zf = fast_floor(z);
485        let xr = x - f64::from(xf);
486        let zr = z - f64::from(zf);
487
488        // Per-lane y offset and floor
489        let ys = ys + Simd::splat(self.yo);
490        let ys_floor = ys.floor();
491        let yrs = ys - ys_floor;
492
493        // Y fudge (per-lane)
494        let yr_fudge: Simd<f64, N> = if y_scale == 0.0 {
495            Simd::splat(0.0)
496        } else {
497            let y_scale_v: Simd<f64, N> = Simd::splat(y_scale);
498            let zero: Simd<f64, N> = Simd::splat(0.0);
499            let mask = y_fudges.simd_ge(zero) & y_fudges.simd_lt(yrs);
500            let fudge_limits = mask.select(y_fudges, yrs);
501            let epsilon: Simd<f64, N> = Simd::splat(f64::from(1.0e-7_f32));
502            ((fudge_limits / y_scale_v) + epsilon).floor() * y_scale_v
503        };
504
505        let yrs_adjusted = yrs - yr_fudge;
506
507        self.sample_and_lerp_y_simd(xf, zf, xr, zr, ys_floor, yrs_adjusted, yrs)
508    }

510    /// Vectorized sample-and-lerp for N Y values sharing x/z grid position.
511    /// Generic counterpart of [`Self::sample_and_lerp_4x`].
512    #[expect(
513        clippy::too_many_arguments,
514        reason = "mirrors scalar sample_and_lerp with Nx SIMD y-batching"
515    )]
516    #[inline]
517    fn sample_and_lerp_y_simd<const N: usize>(
518        &self,
519        xf: i32,
520        zf: i32,
521        xr: f64,
522        zr: f64,
523        ys_floor: Simd<f64, N>,
524        yrs: Simd<f64, N>,
525        yrs_original: Simd<f64, N>,
526    ) -> Simd<f64, N> {
527        let xf = xf as u8;
528        let zf = zf as u8;
529        // Shared x permutation lookups (2 instead of 2×N)
530        let x0 = self.p[xf as usize];
531        let x1 = self.p[xf.wrapping_add(1) as usize];
532
533        let yf = ys_floor.cast::<i32>().cast();
534
535        // Per-lane y-dependent permutation lookups
536        let mut h000 = [0usize; N];
537        let mut h100 = [0usize; N];
538        let mut h010 = [0usize; N];
539        let mut h110 = [0usize; N];
540        let mut h001 = [0usize; N];
541        let mut h101 = [0usize; N];
542        let mut h011 = [0usize; N];
543        let mut h111 = [0usize; N];
544
545        for i in 0..N {
546            let y = yf[i];
547            let xy00 = self.p[x0.wrapping_add(y) as usize];
548            let xy01 = self.p[x0.wrapping_add(y).wrapping_add(1) as usize];
549            let xy10 = self.p[x1.wrapping_add(y) as usize];
550            let xy11 = self.p[x1.wrapping_add(y).wrapping_add(1) as usize];
551            h000[i] = self.p[xy00.wrapping_add(zf) as usize] as usize;
552            h100[i] = self.p[xy10.wrapping_add(zf) as usize] as usize;
553            h010[i] = self.p[xy01.wrapping_add(zf) as usize] as usize;
554            h110[i] = self.p[xy11.wrapping_add(zf) as usize] as usize;
555            h001[i] = self.p[xy00.wrapping_add(zf).wrapping_add(1) as usize] as usize;
556            h101[i] = self.p[xy10.wrapping_add(zf).wrapping_add(1) as usize] as usize;
557            h011[i] = self.p[xy01.wrapping_add(zf).wrapping_add(1) as usize] as usize;
558            h111[i] = self.p[xy11.wrapping_add(zf).wrapping_add(1) as usize] as usize;
559        }
560
561        // Vectorized gradient dot products
562        let xr_v: Simd<f64, N> = Simd::splat(xr);
563        let zr_v: Simd<f64, N> = Simd::splat(zr);
564        let one: Simd<f64, N> = Simd::splat(1.0);
565        let xr_m1 = xr_v - one;
566        let yr_m1 = yrs - one;
567        let zr_m1 = zr_v - one;
568
569        let d000 = grad_dot_simd(h000, xr_v, yrs, zr_v);
570        let d100 = grad_dot_simd(h100, xr_m1, yrs, zr_v);
571        let d010 = grad_dot_simd(h010, xr_v, yr_m1, zr_v);
572        let d110 = grad_dot_simd(h110, xr_m1, yr_m1, zr_v);
573        let d001 = grad_dot_simd(h001, xr_v, yrs, zr_m1);
574        let d101 = grad_dot_simd(h101, xr_m1, yrs, zr_m1);
575        let d011 = grad_dot_simd(h011, xr_v, yr_m1, zr_m1);
576        let d111 = grad_dot_simd(h111, xr_m1, yr_m1, zr_m1);
577
578        // Smoothstep — x and z are shared across lanes
579        let x_alpha: Simd<f64, N> = Simd::splat(smoothstep(xr));
580        let y_alpha = smoothstep_simd(yrs_original);
581        let z_alpha: Simd<f64, N> = Simd::splat(smoothstep(zr));
582
583        lerp3_simd(
584            x_alpha, y_alpha, z_alpha, d000, d100, d010, d110, d001, d101, d011, d111,
585        )
586    }

588    /// Sample noise at grid point, interpolate, and accumulate derivatives.
589    #[expect(clippy::too_many_arguments, reason = "matches vanilla signature")]
590    fn sample_with_derivative(
591        &self,
592        x: i32,
593        y: i32,
594        z: i32,
595        xr: f64,
596        yr: f64,
597        zr: f64,
598        derivative_out: &mut [f64; 3],
599    ) -> f64 {
600        let x = x as u8;
601        let y = y as u8;
602        let z = z as u8;
603
604        let x0 = self.p[x as usize];
605        let x1 = self.p[x.wrapping_add(1) as usize];
606        let xy00 = self.p[x0.wrapping_add(y) as usize];
607        let xy01 = self.p[x0.wrapping_add(y).wrapping_add(1) as usize];
608        let xy10 = self.p[x1.wrapping_add(y) as usize];
609        let xy11 = self.p[x1.wrapping_add(y).wrapping_add(1) as usize];
610
611        let h000 = self.p[xy00.wrapping_add(z) as usize] as usize;
612        let h100 = self.p[xy10.wrapping_add(z) as usize] as usize;
613        let h010 = self.p[xy01.wrapping_add(z) as usize] as usize;
614        let h110 = self.p[xy11.wrapping_add(z) as usize] as usize;
615        let h001 = self.p[xy00.wrapping_add(z).wrapping_add(1) as usize] as usize;
616        let h101 = self.p[xy10.wrapping_add(z).wrapping_add(1) as usize] as usize;
617        let h011 = self.p[xy01.wrapping_add(z).wrapping_add(1) as usize] as usize;
618        let h111 = self.p[xy11.wrapping_add(z).wrapping_add(1) as usize] as usize;
619
620        let g000 = Simd::from_array(GRADIENT[h000 & 15]);
621        let g100 = Simd::from_array(GRADIENT[h100 & 15]);
622        let g010 = Simd::from_array(GRADIENT[h010 & 15]);
623        let g110 = Simd::from_array(GRADIENT[h110 & 15]);
624        let g001 = Simd::from_array(GRADIENT[h001 & 15]);
625        let g101 = Simd::from_array(GRADIENT[h101 & 15]);
626        let g011 = Simd::from_array(GRADIENT[h011 & 15]);
627        let g111 = Simd::from_array(GRADIENT[h111 & 15]);
628
629        // Gradient dot products at each corner
630        let d000 = grad_dot(h000, xr, yr, zr);
631        let d100 = grad_dot(h100, xr - 1.0, yr, zr);
632        let d010 = grad_dot(h010, xr, yr - 1.0, zr);
633        let d110 = grad_dot(h110, xr - 1.0, yr - 1.0, zr);
634        let d001 = grad_dot(h001, xr, yr, zr - 1.0);
635        let d101 = grad_dot(h101, xr - 1.0, yr, zr - 1.0);
636        let d011 = grad_dot(h011, xr, yr - 1.0, zr - 1.0);
637        let d111 = grad_dot(h111, xr - 1.0, yr - 1.0, zr - 1.0);
638
639        let alpha_x = smoothstep(xr);
640        let alpha_y = smoothstep(yr);
641        let alpha_z = smoothstep(zr);
642
643        // Interpolate gradient components for direct derivative contribution
644        let d1_v = lerp3_simd(
645            Simd::splat(alpha_x),
646            Simd::splat(alpha_y),
647            Simd::splat(alpha_z),
648            g000,
649            g100,
650            g010,
651            g110,
652            g001,
653            g101,
654            g011,
655            g111,
656        );
657
658        // Smoothstep correction terms via differences
659        let d2x = lerp2(
660            alpha_y,
661            alpha_z,
662            d100 - d000,
663            d110 - d010,
664            d101 - d001,
665            d111 - d011,
666        );
667        let d2y = lerp2(
668            alpha_z,
669            alpha_x,
670            d010 - d000,
671            d011 - d001,
672            d110 - d100,
673            d111 - d101,
674        );
675        let d2z = lerp2(
676            alpha_x,
677            alpha_y,
678            d001 - d000,
679            d101 - d100,
680            d011 - d010,
681            d111 - d110,
682        );
683
684        let x_sd = smoothstep_derivative(xr);
685        let y_sd = smoothstep_derivative(yr);
686        let z_sd = smoothstep_derivative(zr);
687
688        // Accumulate derivatives (vanilla uses +=)
689        derivative_out[0] += d1_v[0] + x_sd * d2x;
690        derivative_out[1] += d1_v[1] + y_sd * d2y;
691        derivative_out[2] += d1_v[2] + z_sd * d2z;
692        lerp3(
693            alpha_x, alpha_y, alpha_z, d000, d100, d010, d110, d001, d101, d011, d111,
694        )
695    }
