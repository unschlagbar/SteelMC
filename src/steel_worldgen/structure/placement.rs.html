11/// How structures are spread within their grid cell. Vanilla's `RandomSpreadType`.
12#[derive(Debug, Clone, Copy)]
13pub enum SpreadType {
14    /// Uniform random within range.
15    Linear,
16    /// Average of two uniform samples (center-biased).
17    Triangular,
18}
19
20impl SpreadType {
21    /// Offset in `[0, limit)`.
22    pub fn evaluate(self, rng: &mut LegacyRandom, limit: i32) -> i32 {
23        match self {
24            Self::Linear => rng.next_i32_bounded(limit),
25            // Vanilla: `(nextInt(limit) + nextInt(limit)) / 2`.
26            #[expect(
27                clippy::manual_midpoint,
28                reason = "midpoint would change overflow vs vanilla"
29            )]
30            Self::Triangular => (rng.next_i32_bounded(limit) + rng.next_i32_bounded(limit)) / 2,
31        }
32    }
33}

35/// Vanilla's `StructurePlacement.FrequencyReductionMethod`. Variants differ in
36/// seeding/RNG strategy for historical compatibility.
37#[derive(Debug, Clone, Copy, Default)]
38pub enum FrequencyReductionMethod {
39    /// Seeds with salt, uses `next_f32`.
40    #[default]
41    Default,
42    /// Pillager outpost legacy.
43    LegacyType1,
44    /// Hardcoded salt `10_387_320`.
45    LegacyType2,
46    /// Uses `next_f64` instead of `next_f32`.
47    LegacyType3,
48}
49
50impl FrequencyReductionMethod {
51    /// Args match vanilla's `FrequencyReducer`:
52    /// `(levelSeed, placementSalt, chunkX, chunkZ, frequency)`.
53    #[must_use]
54    pub fn should_generate(
55        self,
56        seed: i64,
57        salt: i32,
58        source_x: i32,
59        source_z: i32,
60        probability: f32,
61    ) -> bool {
62        let mut rng = LegacyRandom::from_seed(0);
63        match self {
64            Self::Default => {
65                rng.set_large_feature_with_salt(seed, salt, source_x, source_z);
66                rng.next_f32() < probability
67            }
68            Self::LegacyType1 => {
69                let cx = source_x >> 4;
70                let cz = source_z >> 4;
71                rng.set_seed(i64::from(cx ^ (cz << 4)) ^ seed);
72                rng.next_i32();
73                #[expect(
74                    clippy::cast_possible_truncation,
75                    reason = "vanilla: truncates reciprocal probability to i32 bound"
76                )]
77                let bound = (1.0_f32 / probability) as i32;
78                rng.next_i32_bounded(bound) == 0
79            }
80            Self::LegacyType2 => {
81                rng.set_large_feature_with_salt(seed, source_x, source_z, 10_387_320);
82                rng.next_f32() < probability
83            }
84            Self::LegacyType3 => {
85                rng.set_large_feature_seed(seed, source_x, source_z);
86                rng.next_f64() < f64::from(probability)
87            }
88        }
89    }
90}

101/// Java's `Math.round(double)` — half-up toward +∞. Rust's `f64::round()` rounds
102/// -0.5 → -1; Java rounds -0.5 → 0.
103fn java_round(v: f64) -> i32 {
104    (v + 0.5).floor() as i32
105}

107/// Vanilla's `ChunkGeneratorStructureState.generateRingPositions`. Positions in
108/// chunk coords. If `snap_biome` is provided it gets `(block_x, block_z, &mut rng)`
109/// and returns `Some((snapped_block_x, snapped_block_z))` to snap, or `None` to keep raw.
110#[must_use]
111pub fn generate_ring_positions<F>(
112    seed: i64,
113    distance: i32,
114    spread: i32,
115    count: i32,
116    mut snap_biome: Option<&mut F>,
117) -> Vec<ChunkPos>
118where
119    F: FnMut(i32, i32, &mut LegacyRandom) -> Option<(i32, i32)>,
120{
121    use std::f64::consts::TAU;
122
123    if count == 0 {
124        return vec![];
125    }
126
127    let mut rng = LegacyRandom::from_seed(seed as u64);
128    let mut angle = rng.next_f64() * TAU;
129    let mut positions = Vec::with_capacity(count as usize);
130    let mut spread = spread;
131    let mut position_in_circle = 0;
132    let mut circle = 0;
133
134    let distance_f = f64::from(distance);
135
136    for i in 0..count {
137        let dist = (4.0 + 6.0 * f64::from(circle)) * distance_f
138            + (rng.next_f64() - 0.5) * distance_f * 2.5;
139        let initial_x = java_round(angle.cos() * dist);
140        let initial_z = java_round(angle.sin() * dist);
141
142        // Vanilla forks the RNG for async biome search; we use the fork for the snap.
143        let mut forked = rng.fork();
144        let chunk_pos = if let Some(snap) = snap_biome.as_deref_mut() {
145            // sectionToBlockCoord(x, 8) = x * 16 + 8; snap result is blocks → >> 4.
146            if let Some((sx, sz)) = snap(initial_x * 16 + 8, initial_z * 16 + 8, &mut forked) {
147                ChunkPos::new(sx >> 4, sz >> 4)
148            } else {
149                ChunkPos::new(initial_x, initial_z)
150            }
151        } else {
152            ChunkPos::new(initial_x, initial_z)
153        };
154        positions.push(chunk_pos);
155
156        angle += TAU / f64::from(spread);
157        position_in_circle += 1;
158        if position_in_circle == spread {
159            circle += 1;
160            position_in_circle = 0;
161            spread += 2 * spread / (circle + 1);
162            spread = spread.min(count - i);
163            angle += rng.next_f64() * TAU;
164        }
165    }
166
167    positions
168}

170/// Kind-specific placement parameters.
171#[derive(Debug, Clone)]
172pub enum PlacementKind {
173    /// Vanilla's `RandomSpreadStructurePlacement`.
174    RandomSpread {
175        /// Chunk spacing between grid-cell centers.
176        spacing: i32,
177        /// Minimum chunk separation within a cell.
178        separation: i32,
179        /// Offset computation within the cell.
180        spread_type: SpreadType,
181    },
182    /// Vanilla's `ConcentricRingsStructurePlacement` (strongholds).
183    ConcentricRings {
184        /// Base distance between rings (chunks).
185        distance: i32,
186        /// Positions per ring.
187        spread: i32,
188        /// Total structure positions.
189        count: i32,
190        /// Preferred snap biomes.
191        preferred_biomes: Vec<Identifier>,
192    },
193}

195/// Structure placement configuration.
196#[derive(Debug, Clone)]
197pub struct StructurePlacement {
198    /// Unique seed modifier.
199    pub salt: i32,
200    /// Probability of generating on a placement-chunk. 1.0 = always.
201    pub frequency: f32,
202    /// Frequency-reduction method.
203    pub frequency_reduction_method: FrequencyReductionMethod,
204    /// Optional exclusion zone against another structure set.
205    pub exclusion_zone: Option<ExclusionZone>,
206    /// Block offset from the placement chunk used by `/locate`.
207    pub locate_offset: IVec3,
208    /// Kind-specific parameters.
209    pub kind: PlacementKind,
210}
211
212impl StructurePlacement {
213    /// Locate result block position for a valid placement chunk.
214    #[must_use]
215    pub const fn locate_pos(&self, chunk_pos: ChunkPos) -> BlockPos {
216        BlockPos::new(
217            chunk_pos.0.x * 16 + self.locate_offset.x,
218            self.locate_offset.y,
219            chunk_pos.0.y * 16 + self.locate_offset.z,
220        )
221    }
222
223    /// Valid placement chunk + frequency check. For `ConcentricRings`,
224    /// `ring_positions` must be pre-computed.
225    #[must_use]
226    pub fn is_structure_chunk(
227        &self,
228        seed: i64,
229        source_x: i32,
230        source_z: i32,
231        ring_positions: Option<&[ChunkPos]>,
232    ) -> bool {
233        if !self.is_placement_chunk(seed, source_x, source_z, ring_positions) {
234            return false;
235        }
236        if self.frequency < 1.0
237            && !self.frequency_reduction_method.should_generate(
238                seed,
239                self.salt,
240                source_x,
241                source_z,
242                self.frequency,
243            )
244        {
245            return false;
246        }
247        true
248    }
249
250    fn is_placement_chunk(
251        &self,
252        seed: i64,
253        source_x: i32,
254        source_z: i32,
255        ring_positions: Option<&[ChunkPos]>,
256    ) -> bool {
257        match &self.kind {
258            PlacementKind::RandomSpread {
259                spacing,
260                separation,
261                spread_type,
262            } => {
263                let potential = Self::get_potential_structure_chunk(
264                    seed,
265                    self.salt,
266                    source_x,
267                    source_z,
268                    *spacing,
269                    *separation,
270                    *spread_type,
271                );
272                potential.0.x == source_x && potential.0.y == source_z
273            }
274            PlacementKind::ConcentricRings { .. } => ring_positions
275                .is_some_and(|positions| positions.contains(&ChunkPos::new(source_x, source_z))),
276        }
277    }
278
279    /// Deterministic structure chunk for the grid cell containing `(source_x, source_z)`.
280    #[must_use]
281    pub fn get_potential_structure_chunk(
282        seed: i64,
283        salt: i32,
284        source_x: i32,
285        source_z: i32,
286        spacing: i32,
287        separation: i32,
288        spread_type: SpreadType,
289    ) -> ChunkPos {
290        let grid_x = source_x.div_euclid(spacing);
291        let grid_z = source_z.div_euclid(spacing);
292
293        let mut rng = LegacyRandom::from_seed(0);
294        rng.set_large_feature_with_salt(seed, grid_x, grid_z, salt);
295
296        let limit = spacing - separation;
297        let spread_x = spread_type.evaluate(&mut rng, limit);
298        let spread_z = spread_type.evaluate(&mut rng, limit);
299        ChunkPos::new(grid_x * spacing + spread_x, grid_z * spacing + spread_z)
300    }
301}

312/// Vanilla's `StructureSet`: weighted structures + one placement.

325impl From<SpreadTypeData> for SpreadType {
326    fn from(data: SpreadTypeData) -> Self {
327        match data {
328            SpreadTypeData::Linear => Self::Linear,
329            SpreadTypeData::Triangular => Self::Triangular,
330        }
331    }
332}
333
334impl From<FrequencyMethodData> for FrequencyReductionMethod {
335    fn from(data: FrequencyMethodData) -> Self {
336        match data {
337            FrequencyMethodData::Default => Self::Default,
338            FrequencyMethodData::LegacyType1 => Self::LegacyType1,
339            FrequencyMethodData::LegacyType2 => Self::LegacyType2,
340            FrequencyMethodData::LegacyType3 => Self::LegacyType3,
341        }
342    }
343}
344
345fn convert_structure_set(data: StructureSetData) -> (Identifier, StructureSet) {
346    let structures = data
347        .structures
348        .into_iter()
349        .map(|e| StructureSelectionEntry {
350            structure: e.structure,
351            weight: e.weight,
352        })
353        .collect();
354
355    let placement = match data.placement {
356        PlacementData::RandomSpread {
357            spacing,
358            separation,
359            spread_type,
360            salt,
361            frequency,
362            frequency_reduction_method,
363            exclusion_zone,
364            locate_offset,
365        } => StructurePlacement {
366            salt,
367            frequency,
368            frequency_reduction_method: frequency_reduction_method.into(),
369            exclusion_zone: exclusion_zone.map(|ez| ExclusionZone {
370                other_set: ez.other_set,
371                chunk_count: ez.chunk_count,
372            }),
373            locate_offset,
374            kind: PlacementKind::RandomSpread {
375                spacing,
376                separation,
377                spread_type: spread_type.into(),
378            },
379        },
380        PlacementData::ConcentricRings {
381            distance,
382            spread,
383            count,
384            preferred_biomes,
385            salt,
386            frequency,
387            frequency_reduction_method,
388            locate_offset,
389        } => StructurePlacement {
390            salt,
391            frequency,
392            frequency_reduction_method: frequency_reduction_method.into(),
393            exclusion_zone: None,
394            locate_offset,
395            kind: PlacementKind::ConcentricRings {
396                distance,
397                spread,
398                count,
399                preferred_biomes,
400            },
401        },
402    };
403
404    (
405        data.key,
406        StructureSet {
407            structures,
408            placement,
409        },
410    )
411}

449            frequency: 1.0,

474        let mut rng = LegacyRandom::from_seed(0);
475        rng.set_large_feature_with_salt(0, 0, 0, 10_387_312);
476        let spread_x = rng.next_i32_bounded(26); // 34 - 8
477        let spread_z = rng.next_i32_bounded(26);
478        assert_eq!(potential, ChunkPos::new(spread_x, spread_z));
479
480        // The structure chunk for (0,0) should only match if we query the

484        // happens to be the potential chunk)

623    #[test]
624    fn test_generate_ring_positions_strongholds() {
625        // Strongholds: distance=32, spread=3, count=128
626        let positions = generate_ring_positions::<
627            fn(i32, i32, &mut LegacyRandom) -> Option<(i32, i32)>,
628        >(0, 32, 3, 128, None);
629        assert_eq!(positions.len(), 128);
630
631        // First ring should be roughly 4*32 = 128 chunks from origin
632        // (with some jitter)
633        let first = positions[0];
634        let dist = (f64::from(first.0.x).powi(2) + f64::from(first.0.y).powi(2)).sqrt();
635        assert!(
636            dist > 80.0 && dist < 200.0,
637            "First stronghold at distance {dist}, expected ~128"
638        );
639
640        // All positions should be unique
641        let mut unique = positions.clone();
642        unique.sort_by_key(|p| (p.0.x, p.0.y));
643        unique.dedup_by_key(|p| (p.0.x, p.0.y));
644        assert_eq!(
645            unique.len(),
646            positions.len(),
647            "Ring positions should be unique"
648        );
649
650        // Deterministic: same seed produces same positions
651        let positions2 = generate_ring_positions::<
652            fn(i32, i32, &mut LegacyRandom) -> Option<(i32, i32)>,
653        >(0, 32, 3, 128, None);
654        assert_eq!(positions, positions2);
655    }
