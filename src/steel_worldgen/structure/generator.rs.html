93impl StructureBiomeProvider for FixedStructureBiomeProvider {
94    fn possible_biomes(&self) -> FxHashSet<Identifier> {
95        FxHashSet::from_iter([self.biome.key.clone()])
96    }
97
98    fn find_biome_horizontal(
99        &self,
100        origin_x: i32,
101        origin_z: i32,
102        search_radius: i32,
103        allowed: &dyn Fn(&Identifier) -> bool,
104        rng: &mut LegacyRandom,
105    ) -> Option<(i32, i32)> {
106        if !allowed(&self.biome.key) {
107            return None;
108        }
109
110        let noise_center_x = origin_x >> 2;
111        let noise_center_z = origin_z >> 2;
112        let noise_radius = search_radius >> 2;
113        let mut result = None;
114        let mut found = 0;
115        for z in -noise_radius..=noise_radius {
116            for x in -noise_radius..=noise_radius {
117                if result.is_none() || rng.next_i32_bounded(found + 1) == 0 {
118                    result = Some(((noise_center_x + x) << 2, (noise_center_z + z) << 2));
119                }
120                found += 1;
121            }
122        }
123        result
124    }
125}

217    /// Returns `true` if the plan has any random-spread placements.
218    #[must_use]
219    pub fn has_random_spread(&self) -> bool {
220        self.placements.iter().any(|placement| {
221            matches!(
222                &placement.placement.kind,
223                PlacementKind::RandomSpread { .. }
224            )
225        })
226    }
227
228    /// Ring-placement candidates ordered by vanilla's stronghold distance pre-check.
229    #[must_use]
230    pub fn ring_candidates(&self, origin: BlockPos) -> Vec<StructureLocateCandidate> {
231        let mut candidates = Vec::new();
232        for (scan_id, placement) in self.placements.iter().enumerate() {
233            let Some(ring_positions) = &placement.ring_positions else {
234                continue;
235            };
236            for &chunk_pos in ring_positions {
237                candidates.push(StructureLocateCandidate::new_ring(
238                    scan_id,
239                    placement.placement.locate_pos(chunk_pos),
240                    chunk_pos,
241                ));
242            }
243        }
244        candidates.sort_by_key(|candidate| candidate.ring_distance_squared(origin));
245        candidates
246    }
247
248    /// Random-spread candidates on the square shell at `radius` around `origin`.
249    ///
250    /// This matches vanilla's `ChunkGenerator.getNearestGeneratedStructure` scan:
251    /// each shell step moves by the placement spacing before resolving the
252    /// potential structure chunk inside that placement cell.
253    #[must_use]
254    pub fn random_spread_candidates_at_radius(
255        &self,
256        origin: BlockPos,
257        radius: i32,
258    ) -> Vec<StructureLocateCandidate> {
259        if radius < 0 {
260            return Vec::new();
261        }
262
263        let chunk_origin_x = origin.0.x >> 4;
264        let chunk_origin_z = origin.0.z >> 4;
265        let mut candidates = Vec::new();
266
267        for (scan_id, locate_placement) in self.placements.iter().enumerate() {
268            let PlacementKind::RandomSpread {
269                spacing,
270                separation,
271                spread_type,
272            } = &locate_placement.placement.kind
273            else {
274                continue;
275            };
276
277            for x in -radius..=radius {
278                let x_edge = x == -radius || x == radius;
279                for z in -radius..=radius {
280                    let z_edge = z == -radius || z == radius;
281                    if !x_edge && !z_edge {
282                        continue;
283                    }
284
285                    let sector_x = chunk_origin_x + *spacing * x;
286                    let sector_z = chunk_origin_z + *spacing * z;
287                    let chunk_pos = StructurePlacement::get_potential_structure_chunk(
288                        self.seed,
289                        locate_placement.placement.salt,
290                        sector_x,
291                        sector_z,
292                        *spacing,
293                        *separation,
294                        *spread_type,
295                    );
296                    let candidate = StructureLocateCandidate::new(
297                        scan_id,
298                        locate_placement.placement.locate_pos(chunk_pos),
299                        chunk_pos,
300                    );
301                    candidates.push(candidate);
302                }
303            }
304        }
305
306        candidates
307    }
308}

340/// Squared block distance using vanilla's three-dimensional `BlockPos.distSqr`.
341#[must_use]
342pub fn squared_distance(a: BlockPos, b: BlockPos) -> i64 {
343    let dx = i64::from(a.0.x) - i64::from(b.0.x);
344    let dy = i64::from(a.0.y) - i64::from(b.0.y);
345    let dz = i64::from(a.0.z) - i64::from(b.0.z);
346    dx * dx + dy * dy + dz * dz
347}

349fn validate_structure_sets(structure_sets: &[(Identifier, StructureSet)]) {
350    for (set_key, set) in structure_sets {
351        assert!(
352            !set.structures.is_empty(),
353            "Structure set {set_key} must have at least one structure"
354        );
355        for (entry_index, entry) in set.structures.iter().enumerate() {
356            assert!(
357                entry.weight > 0,
358                "Structure set {set_key} entry {entry_index} has non-positive weight {}",
359                entry.weight
360            );
361        }
362        assert!(
363            !(!set.placement.frequency.is_finite()
364                || !(0.0..=1.0).contains(&set.placement.frequency)),
365            "Structure set {set_key} has invalid placement frequency {}",
366            set.placement.frequency
367        );
368        if let Some(exclusion) = &set.placement.exclusion_zone
369            && exclusion.chunk_count < 0
370        {
371            panic!(
372                "Structure set {set_key} has negative exclusion chunk_count {}",
373                exclusion.chunk_count
374            );
375        }
376
377        match &set.placement.kind {
378            PlacementKind::RandomSpread {
379                spacing,
380                separation,
381                ..
382            } => {
383                assert!(
384                    *spacing > 0,
385                    "Structure set {set_key} has non-positive spacing {spacing}"
386                );
387                assert!(
388                    *separation >= 0,
389                    "Structure set {set_key} has negative separation {separation}"
390                );
391                assert!(
392                    spacing > separation,
393                    "Structure set {set_key} has spacing {spacing} <= separation {separation}"
394                );
395            }
396            PlacementKind::ConcentricRings {
397                distance,
398                spread,
399                count,
400                ..
401            } => {
402                assert!(
403                    *distance > 0,
404                    "Structure set {set_key} has non-positive ring distance {distance}"
405                );
406                assert!(
407                    *spread > 0,
408                    "Structure set {set_key} has non-positive ring spread {spread}"
409                );
410                assert!(
411                    *count >= 0,
412                    "Structure set {set_key} has negative ring count {count}"
413                );
414            }
415        }
416    }
417}

419fn validate_structure_assets(
420    structure_sets: &[(Identifier, StructureSet)],
421    structure_data: &FxHashMap<Identifier, StructureRef>,
422    structure_impls: &FxHashMap<Identifier, Box<dyn Structure>>,
423) {
424    for (set_key, set) in structure_sets {
425        for entry in &set.structures {
426            let structure = structure_data.get(&entry.structure).unwrap_or_else(|| {
427                panic!(
428                    "Structure set {set_key} references unknown structure {}",
429                    entry.structure
430                )
431            });
432            assert!(
433                structure_impls.contains_key(&structure.structure_type),
434                "Structure set {set_key} references {} with unsupported structure type {}",
435                structure.key,
436                structure.structure_type
437            );
438        }
439    }

497    fn with_assets_for_ring_seed(
498        seed: i64,
499        ring_position_seed: i64,
500        biome_provider: &impl StructureBiomeProvider,
501        structure_sets: Vec<(Identifier, StructureSet)>,
502        assets: StructureGeneratorAssets,
503    ) -> Self {
504        validate_structure_sets(&structure_sets);
505
506        let structure_data: FxHashMap<Identifier, StructureRef> = REGISTRY
507            .structures
508            .iter()
509            .map(|(_, structure)| (structure.key.clone(), structure))
510            .collect();
511        validate_structure_assets(&structure_sets, &structure_data, &assets.structure_impls);
512
513        let possible_biomes = biome_provider.possible_biomes();
514        let structure_sets: Vec<_> = structure_sets
515            .into_iter()
516            .filter(|(_, set)| {
517                set.structures.iter().any(|entry| {
518                    structure_data
519                        .get(&entry.structure)
520                        .is_some_and(|structure| {
521                            structure.allowed_biomes.is_empty()
522                                || structure
523                                    .allowed_biomes
524                                    .iter()
525                                    .any(|biome| possible_biomes.contains(biome))
526                        })
527                })
528            })
529            .collect();
530
531        let structure_set_indices: FxHashMap<Identifier, usize> = structure_sets
532            .iter()
533            .enumerate()
534            .map(|(index, (key, _))| (key.clone(), index))
535            .collect();
536
537        let mut ring_positions = FxHashMap::default();
538        for (key, set) in &structure_sets {
539            if let PlacementKind::ConcentricRings {
540                distance,
541                spread,
542                count,
543                preferred_biomes,
544            } = &set.placement.kind
545            {
546                let mut snap =
547                    |block_x: i32, block_z: i32, rng: &mut LegacyRandom| -> Option<(i32, i32)> {
548                        biome_provider.find_biome_horizontal(
549                            block_x,
550                            block_z,
551                            112,
552                            &|biome| preferred_biomes.contains(biome),
553                            rng,
554                        )
555                    };
556                let positions = generate_ring_positions(
557                    ring_position_seed,
558                    *distance,
559                    *spread,
560                    *count,
561                    Some(&mut snap),
562                );
563                ring_positions.insert(key.clone(), positions);
564            }
565        }
566
567        Self {
568            seed,
569            structure_sets,
570            structure_set_indices,
571            structure_data,
572            ring_positions,
573            template_pools: assets.template_pools,
574            templates: assets.templates,
575            structure_impls: assets.structure_impls,
576        }
577    }

626    /// Generates structure starts for one chunk.
627    pub fn generate_starts_for_chunk(
628        &self,
629        ctx: &mut dyn StructureGenerationContext,
630        mut has_existing_start: impl FnMut(&Identifier) -> bool,
631    ) -> Vec<StructureStart> {
632        let chunk_x = ctx.chunk_x();
633        let chunk_z = ctx.chunk_z();
634        let mut generated_keys = FxHashSet::default();
635        let mut starts = Vec::new();
636
637        for (set_key, set) in &self.structure_sets {
638            if !self.is_structure_chunk_for_set(set_key, chunk_x, chunk_z, &mut Vec::new()) {
639                continue;
640            }
641
642            if set.structures.iter().any(|entry| {
643                generated_keys.contains(&entry.structure) || has_existing_start(&entry.structure)
644            }) {
645                continue;
646            }
647
648            let Some((structure, stub)) = self.select_structure(set, ctx) else {
649                continue;
650            };
651
652            let start = StructureStart::new(
653                structure.key.clone(),
654                ChunkPos::new(chunk_x, chunk_z),
655                stub.pieces,
656                structure.terrain_adjustment,
657            );
658            if !start.pieces.is_empty() {
659                generated_keys.insert(structure.key.clone());
660            }
661            starts.push(start);
662        }
663
664        starts
665    }

671    fn is_structure_chunk_for_set(
672        &self,
673        set_key: &Identifier,
674        source_x: i32,
675        source_z: i32,
676        stack: &mut Vec<Identifier>,
677    ) -> bool {
678        if stack.iter().any(|key| key == set_key) {
679            let chain = stack
680                .iter()
681                .map(ToString::to_string)
682                .chain(iter::once(set_key.to_string()))
683                .collect::<Vec<_>>()
684                .join(" -> ");
685            panic!("Circular structure exclusion zone: {chain}");
686        }
687
688        let Some(&set_index) = self.structure_set_indices.get(set_key) else {
689            return false;
690        };
691        let (_, set) = &self.structure_sets[set_index];
692        let rings = self.rings_for_set(set_key);
693        if !set
694            .placement
695            .is_structure_chunk(self.seed, source_x, source_z, rings)
696        {
697            return false;
698        }
699
700        stack.push(set_key.clone());
701        let excluded = self.is_excluded(&set.placement, source_x, source_z, stack);
702        stack.pop();
703        !excluded
704    }
705
706    fn is_excluded(
707        &self,
708        placement: &StructurePlacement,
709        source_x: i32,
710        source_z: i32,
711        stack: &mut Vec<Identifier>,
712    ) -> bool {
713        let Some(exclusion) = &placement.exclusion_zone else {
714            return false;
715        };
716
717        for dx in (source_x - exclusion.chunk_count)..=(source_x + exclusion.chunk_count) {
718            for dz in (source_z - exclusion.chunk_count)..=(source_z + exclusion.chunk_count) {
719                if self.is_structure_chunk_for_set(&exclusion.other_set, dx, dz, stack) {
720                    return true;
721                }
722            }
723        }
724        false
725    }
726
727    fn select_structure(
728        &self,
729        set: &StructureSet,
730        ctx: &mut dyn StructureGenerationContext,
731    ) -> Option<(StructureRef, GenerationStub)> {
732        if set.structures.len() == 1 {
733            return self.try_generate_entry(&set.structures[0], ctx);
734        }
735
736        let mut rng = LegacyRandom::from_seed(0);
737        rng.set_large_feature_seed(self.seed, ctx.chunk_x(), ctx.chunk_z());
738
739        let mut remaining: Vec<&StructureSelectionEntry> = set.structures.iter().collect();
740        let mut total_weight: i32 = remaining.iter().map(|entry| entry.weight).sum();
741
742        while !remaining.is_empty() {
743            let mut choice = rng.next_i32_bounded(total_weight);
744            let mut selected_idx = 0;
745            for (idx, entry) in remaining.iter().enumerate() {
746                choice -= entry.weight;
747                if choice < 0 {
748                    selected_idx = idx;
749                    break;
750                }
751            }
752
753            let candidate = remaining[selected_idx];
754            if let Some(generated) = self.try_generate_entry(candidate, ctx) {
755                return Some(generated);
756            }
757
758            total_weight -= candidate.weight;
759            remaining.remove(selected_idx);
760        }
761
762        None
763    }
764
765    fn try_generate_entry(
766        &self,
767        entry: &StructureSelectionEntry,
768        ctx: &mut dyn StructureGenerationContext,
769    ) -> Option<(StructureRef, GenerationStub)> {
770        let Some(structure) = self.structure_data.get(&entry.structure).copied() else {
771            tracing::warn!("Missing structure registry data for {}", entry.structure);
772            return None;
773        };
774
775        if let Some(structure_impl) = self.structure_impls.get(&structure.structure_type) {
776            let mut rng = LegacyRandom::from_seed(0);
777            rng.set_large_feature_seed(self.seed, ctx.chunk_x(), ctx.chunk_z());
778            return structure_impl
779                .find_generation_point(ctx, structure, &mut rng)
780                .map(|stub| (structure, stub));
781        }
782
783        tracing::warn!(
784            "Unknown structure type {:?} for {}, skipping structure start",
785            structure.structure_type,
786            structure.key
787        );
788        None
789    }
790}
