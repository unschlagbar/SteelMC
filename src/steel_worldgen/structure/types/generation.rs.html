19/// Per-chunk context shared by every structure's `findGenerationPoint`.
20///
21/// Holds mutable per-chunk state (biome sampler, height cache, aquifer) so structures
22/// don't each allocate their own. Wraps `VanillaGenerator`'s helpers.
23pub struct GenerationContext<'ctx, 'src, N: DimensionNoises>
24where
25    'src: 'ctx,
26{
27    /// World seed.
28    seed: i64,
29    /// Chunk being populated.
30    chunk_x: i32,
31    /// Chunk being populated.
32    chunk_z: i32,
33    /// `chunk_x * 16`.
34    chunk_min_x: i32,
35    /// `chunk_z * 16`.
36    chunk_min_z: i32,
37    /// `chunk_min_x + 8`.
38    center_block_x: i32,
39    /// `chunk_min_z + 8`.
40    center_block_z: i32,
41    /// Sea level for this dimension.
42    sea_level: i32,
43    /// Shared memoisation slot for the chunk-center surface Y.
44    surface_y_cache: &'ctx mut Option<i32>,
45    /// Whether `height_cache`'s 5×5 quart grid has been populated. Shared across
46    /// per-structure contexts in the same chunk.
47    height_cache_grid_ready: &'ctx mut bool,
48
49    /// Dimension noise router.
50    noises: &'src N,
51    /// Positional splitter for per-chunk RNG.
52    splitter: &'src RandomSplitter,
53    /// Template pool registry for jigsaw assembly.
54    template_pools: &'src FxHashMap<Identifier, TemplatePoolData>,
55    /// Template data registry for jigsaw assembly.
56    templates: &'src FxHashMap<Identifier, TemplateData>,
57
58    /// Biome sampler scoped to this chunk.
59    biome_sampler: &'ctx mut ChunkBiomeSampler<'src>,
60    /// Column cache for height/density queries (grid-initialized on demand).
61    height_cache: &'ctx mut N::ColumnCache,
62    /// Aquifer built on first query; skipped on chunks where no structure needs it.
63    aquifer: &'ctx mut LazyAquifer<'src, N>,
64    /// Cache for terrain height checks.
65    terrain_height_cache: RefCell<FxHashMap<(i32, i32, bool), i32>>,
66    /// Cache for terrain opacity checks.
67    terrain_opaque_cache: RefCell<FxHashMap<(i32, i32, i32, bool), bool>>,
68    /// Probes for off-chunk height/opaque checks.
69    terrain_probes: RefCell<FxHashMap<(i32, i32), TerrainProbe<N>>>,
70}

72/// An off-chunk height and opacity probe.
73pub struct TerrainProbe<N: DimensionNoises> {
74    cache: N::ColumnCache,
75    aquifer: Aquifer<N>,
76}

78impl<N: DimensionNoises> TerrainProbe<N> {
79    fn new(chunk_min_x: i32, chunk_min_z: i32, splitter: &RandomSplitter, noises: &N) -> Self {
80        let mut cache = N::ColumnCache::default();
81        cache.init_grid(chunk_min_x, chunk_min_z, noises);
82        let aquifer = Aquifer::<N>::new(
83            chunk_min_x,
84            chunk_min_z,
85            N::Settings::MIN_Y,
86            N::Settings::HEIGHT,
87            splitter,
88            noises,
89            cache.clone(),
90        );
91        Self { cache, aquifer }
92    }
93}

103/// Terrain, biome, and template queries exposed to structure algorithms.
104///
105/// Vanilla calls these through `ChunkGenerator`/`WorldGenLevel`; keeping the
106/// interface here lets structure algorithms stay independent of a concrete
107/// chunk generator while preserving their vanilla query order.
108pub trait StructureGenerationContext {
109    /// World seed.
110    fn seed(&self) -> i64;
111    /// Chunk X being populated.
112    fn chunk_x(&self) -> i32;
113    /// Chunk Z being populated.
114    fn chunk_z(&self) -> i32;
115    /// Minimum block X of the chunk.
116    fn chunk_min_x(&self) -> i32;
117    /// Minimum block Z of the chunk.
118    fn chunk_min_z(&self) -> i32;
119    /// Center block X of the chunk.
120    fn center_block_x(&self) -> i32;
121    /// Center block Z of the chunk.
122    fn center_block_z(&self) -> i32;
123    /// Sea level for this generator/dimension.
124    fn sea_level(&self) -> i32;
125    /// Minimum build Y.
126    fn min_y(&self) -> i32;
127    /// Total build height.
128    fn height(&self) -> i32;
129    /// One-past-maximum build Y.
130    fn max_y(&self) -> i32 {
131        self.min_y() + self.height()
132    }
133    /// Template pool registry for jigsaw assembly.
134    fn template_pools(&self) -> &FxHashMap<Identifier, TemplatePoolData>;
135    /// Structure templates (piece definitions + sizes).
136    fn templates(&self) -> &FxHashMap<Identifier, TemplateData>;
137    /// Base height at a column.
138    fn base_height(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32;
139    /// Full-column base height scan.
140    fn base_height_full(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32;
141    /// Biome at a block position.
142    fn biome_at(&mut self, block_x: i32, block_y: i32, block_z: i32) -> BiomeRef;
143    /// Classify a block in the generator's base terrain.
144    fn column_state(&mut self, x: i32, y: i32, z: i32) -> ColumnBlock;
145    /// Highest solid base-terrain block directly below air in `[min_solid_y, start_y)`.
146    fn solid_block_below_air(
147        &mut self,
148        x: i32,
149        z: i32,
150        start_y: i32,
151        min_solid_y: i32,
152    ) -> Option<i32> {
153        if start_y <= min_solid_y {
154            return None;
155        }
156
157        let mut above = self.column_state(x, start_y, z);
158        for y in (min_solid_y..start_y).rev() {
159            let current = self.column_state(x, y, z);
160            if above == ColumnBlock::Air && current == ColumnBlock::Solid {
161                return Some(y);
162            }
163            above = current;
164        }
165        None
166    }
167    /// Chunk-center surface Y, memoised by the concrete context.
168    fn surface_y(&mut self) -> i32;
169    /// Surface height for off-chunk terrain queries used by piece placement.
170    fn terrain_surface_height(&self, x: i32, z: i32, ocean_floor: bool) -> i32;
171    /// Opaque terrain test for off-chunk terrain queries used by piece placement.
172    fn terrain_is_opaque(&self, x: i32, y: i32, z: i32, ocean_floor: bool) -> bool;
173}

175impl<'ctx, 'src, N: DimensionNoises> GenerationContext<'ctx, 'src, N>
176where
177    'src: 'ctx,
178{
179    /// Creates a per-chunk structure generation context.
180    #[must_use]
181    #[expect(
182        clippy::too_many_arguments,
183        reason = "borrows all mutable per-chunk generation state without owning it"
184    )]
185    pub fn new(
186        seed: i64,
187        chunk_x: i32,
188        chunk_z: i32,
189        sea_level: i32,
190        noises: &'src N,
191        splitter: &'src RandomSplitter,
192        template_pools: &'src FxHashMap<Identifier, TemplatePoolData>,
193        templates: &'src FxHashMap<Identifier, TemplateData>,
194        biome_sampler: &'ctx mut ChunkBiomeSampler<'src>,
195        height_cache: &'ctx mut N::ColumnCache,
196        aquifer: &'ctx mut LazyAquifer<'src, N>,
197        surface_y_cache: &'ctx mut Option<i32>,
198        height_cache_grid_ready: &'ctx mut bool,
199    ) -> Self {
200        let chunk_min_x = chunk_x * 16;
201        let chunk_min_z = chunk_z * 16;
202        Self {
203            seed,
204            chunk_x,
205            chunk_z,
206            chunk_min_x,
207            chunk_min_z,
208            center_block_x: chunk_min_x + 8,
209            center_block_z: chunk_min_z + 8,
210            sea_level,
211            surface_y_cache,
212            height_cache_grid_ready,
213            noises,
214            splitter,
215            template_pools,
216            templates,
217            biome_sampler,
218            height_cache,
219            aquifer,
220            terrain_height_cache: RefCell::default(),
221            terrain_opaque_cache: RefCell::default(),
222            terrain_probes: RefCell::default(),
223        }
224    }
225
226    /// `getBaseHeight(WORLD_SURFACE_WG)` — aquifer-aware, scans from
227    /// `preliminary_surface_level + 16`.
228    ///
229    /// `ocean_floor=false` → opaque is Solid+Fluid; `true` → opaque is Solid only.
230    ///
231    /// In dimensions with a constant `preliminary_surface_level` (End), use
232    /// [`base_height_full`](Self::base_height_full) instead.
233    pub fn base_height(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32 {
234        self.ensure_height_cache_grid();
235        let aq = self.aquifer.ensure(self.height_cache);
236        column_base_height::<N>(self.height_cache, self.noises, aq, x, z, ocean_floor)
237    }
238
239    /// Full-column scan from chunk top. Matches vanilla's `iterateNoiseColumn`.
240    pub fn base_height_full(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32 {
241        self.ensure_height_cache_grid();
242        let aq = self.aquifer.ensure(self.height_cache);
243        iterate_noise_column_with_aquifer::<N>(
244            self.height_cache,
245            self.noises,
246            aq,
247            x,
248            z,
249            ocean_floor,
250        )
251    }
252
253    /// Biome at a block position (quantized to quart).
254    pub fn biome_at(&mut self, block_x: i32, block_y: i32, block_z: i32) -> BiomeRef {
255        self.biome_sampler
256            .sample(block_x >> 2, block_y >> 2, block_z >> 2)
257    }
258
259    /// Classify a single block in the base-noise column.
260    pub fn column_state(&mut self, x: i32, y: i32, z: i32) -> ColumnBlock {
261        self.ensure_height_cache_grid();
262        let cw = N::Settings::CELL_WIDTH;
263        let ch = N::Settings::CELL_HEIGHT;
264        let density =
265            column_interpolated_density::<N>(self.height_cache, self.noises, x, y, z, cw, ch);
266        let aq = self.aquifer.ensure(self.height_cache);
267        match aq.compute_substance(self.noises, x, y, z, density) {
268            AquiferResult::Solid => ColumnBlock::Solid,
269            AquiferResult::Fluid(_) => ColumnBlock::Fluid,
270            AquiferResult::Air => ColumnBlock::Air,
271        }
272    }
273
274    /// Highest solid base-terrain block directly below air in `[min_solid_y, start_y)`.
275    pub fn solid_block_below_air(
276        &mut self,
277        x: i32,
278        z: i32,
279        start_y: i32,
280        min_solid_y: i32,
281    ) -> Option<i32> {
282        self.ensure_height_cache_grid();
283        let aq = self.aquifer.ensure(self.height_cache);
284        find_solid_block_below_air::<N>(
285            self.height_cache,
286            self.noises,
287            aq,
288            x,
289            z,
290            start_y,
291            min_solid_y,
292        )
293    }
294
295    /// Surface Y at chunk center, memoised across per-structure contexts.
296    pub fn surface_y(&mut self) -> i32 {
297        if let Some(y) = *self.surface_y_cache {
298            return y;
299        }
300        let y = self.base_height(self.center_block_x, self.center_block_z, false) - 1;
301        *self.surface_y_cache = Some(y);
302        y
303    }
304
305    fn ensure_height_cache_grid(&mut self) {
306        if *self.height_cache_grid_ready {
307            return;
308        }
309        self.height_cache
310            .init_grid(self.chunk_min_x, self.chunk_min_z, self.noises);
311        *self.height_cache_grid_ready = true;
312    }
313}

394    fn terrain_surface_height(&self, x: i32, z: i32, ocean_floor: bool) -> i32 {
395        if let Some(height) = self
396            .terrain_height_cache
397            .borrow()
398            .get(&(x, z, ocean_floor))
399            .copied()
400        {
401            return height;
402        }
403
404        let cell_w = N::Settings::CELL_WIDTH;
405        let cell_x = x.div_euclid(cell_w) * cell_w;
406        let cell_z = z.div_euclid(cell_w) * cell_w;
407        let aq_chunk_x = (cell_x >> 4) * 16;
408        let aq_chunk_z = (cell_z >> 4) * 16;
409        let height = {
410            let mut probes = self.terrain_probes.borrow_mut();
411            let probe = probes.entry((aq_chunk_x, aq_chunk_z)).or_insert_with(|| {
412                TerrainProbe::<N>::new(aq_chunk_x, aq_chunk_z, self.splitter, self.noises)
413            });
414            iterate_noise_column_with_aquifer::<N>(
415                &mut probe.cache,
416                self.noises,
417                &mut probe.aquifer,
418                x,
419                z,
420                ocean_floor,
421            )
422        };
423        self.terrain_height_cache
424            .borrow_mut()
425            .insert((x, z, ocean_floor), height);
426        height
427    }
428
429    fn terrain_is_opaque(&self, x: i32, y: i32, z: i32, ocean_floor: bool) -> bool {
430        if let Some(opaque) = self
431            .terrain_opaque_cache
432            .borrow()
433            .get(&(x, y, z, ocean_floor))
434            .copied()
435        {
436            return opaque;
437        }
438
439        let cell_w = N::Settings::CELL_WIDTH;
440        let cell_h = N::Settings::CELL_HEIGHT;
441        let cell_x = x.div_euclid(cell_w) * cell_w;
442        let cell_z = z.div_euclid(cell_w) * cell_w;
443        let aq_chunk_x = (cell_x >> 4) * 16;
444        let aq_chunk_z = (cell_z >> 4) * 16;
445        let opaque = {
446            let mut probes = self.terrain_probes.borrow_mut();
447            let probe = probes.entry((aq_chunk_x, aq_chunk_z)).or_insert_with(|| {
448                TerrainProbe::<N>::new(aq_chunk_x, aq_chunk_z, self.splitter, self.noises)
449            });
450            let density = column_interpolated_density::<N>(
451                &mut probe.cache,
452                self.noises,
453                x,
454                y,
455                z,
456                cell_w,
457                cell_h,
458            );
459            match probe
460                .aquifer
461                .compute_substance(self.noises, x, y, z, density)
462            {
463                AquiferResult::Solid => true,
464                AquiferResult::Fluid(_) => !ocean_floor,
465                AquiferResult::Air => false,
466            }
467        };
468        self.terrain_opaque_cache
469            .borrow_mut()
470            .insert((x, y, z, ocean_floor), opaque);
471        opaque
472    }
