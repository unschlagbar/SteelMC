17/// Vanilla `minecraft:attribute_modifiers` item component.
18#[derive(Debug, Clone, PartialEq)]
19pub struct ItemAttributeModifiers {
20    pub modifiers: Vec<ItemAttributeModifierEntry>,
21}
22
23impl ItemAttributeModifiers {
24    #[must_use]
25    pub const fn empty() -> Self {
26        Self {
27            modifiers: Vec::new(),
28        }
29    }
30
31    #[must_use]
32    pub fn is_empty(&self) -> bool {
33        self.modifiers.is_empty()
34    }
35
36    pub fn for_slot(
37        &self,
38        slot: EquipmentSlot,
39    ) -> impl Iterator<Item = &ItemAttributeModifierEntry> {
40        self.modifiers
41            .iter()
42            .filter(move |entry| entry.slot.test(slot))
43    }
44}
45
46impl Default for ItemAttributeModifiers {
47    fn default() -> Self {
48        Self::empty()
49    }
50}

52/// A single item attribute modifier entry.
53#[derive(Debug, Clone)]
54pub struct ItemAttributeModifierEntry {
55    pub attribute: AttributeRef,
56    pub id: Identifier,
57    pub amount: f64,
58    pub operation: AttributeModifierOperation,
59    pub slot: EquipmentSlotGroup,
60    pub display: ItemAttributeModifierDisplay,
61}
62
63impl PartialEq for ItemAttributeModifierEntry {
64    fn eq(&self, other: &Self) -> bool {
65        self.attribute.key == other.attribute.key
66            && self.id == other.id
67            && self.amount == other.amount
68            && self.operation == other.operation
69            && self.slot == other.slot
70            && self.display == other.display
71    }
72}

74/// Tooltip display behavior for an item attribute modifier.
75#[derive(Debug, Clone, PartialEq)]
76pub enum ItemAttributeModifierDisplay {
77    Default,
78    Hidden,
79    OverrideText(Box<TextComponent>),
80}
81
82impl ItemAttributeModifierDisplay {
83    #[must_use]
84    pub const fn id(&self) -> i32 {
85        match self {
86            Self::Default => 0,
87            Self::Hidden => 1,
88            Self::OverrideText(_) => 2,
89        }
90    }
91
92    #[must_use]
93    pub const fn name(&self) -> &'static str {
94        match self {
95            Self::Default => "default",
96            Self::Hidden => "hidden",
97            Self::OverrideText(_) => "override",
98        }
99    }
100
101    fn from_nbt_compound(compound: simdnbt::borrow::NbtCompound) -> Option<Self> {
102        let display_type = compound.get("type")?.string()?.to_str();
103        match display_type.as_ref() {
104            "default" => Some(Self::Default),
105            "hidden" => Some(Self::Hidden),
106            "override" => {
107                let value = compound.get("value")?;
108                Some(Self::OverrideText(Box::new(TextComponent::from_nbt_tag(
109                    value,
110                )?)))
111            }
112            _ => None,
113        }
114    }
115
116    fn to_nbt_tag_ref(&self) -> NbtTag {
117        let mut compound = NbtCompound::new();
118        compound.insert("type", self.name());
119        if let Self::OverrideText(text) = self {
120            compound.insert("value", text.as_ref().to_nbt_tag());
121        }
122        NbtTag::Compound(compound)
123    }
124}

126impl WriteTo for ItemAttributeModifiers {
127    fn write(&self, writer: &mut impl Write) -> Result<()> {
128        VarInt(self.modifiers.len() as i32).write(writer)?;
129        for entry in &self.modifiers {
130            entry.write(writer)?;
131        }
132        Ok(())
133    }
134}

136impl ReadFrom for ItemAttributeModifiers {
137    fn read(data: &mut Cursor<&[u8]>) -> Result<Self> {
138        let count = VarInt::read(data)?.0;
139        if !(0..=1024).contains(&count) {
140            return Err(std::io::Error::other(format!(
141                "Attribute modifier count out of range: {count}"
142            )));
143        }
144
145        let mut modifiers = Vec::with_capacity(count as usize);
146        for _ in 0..count {
147            modifiers.push(ItemAttributeModifierEntry::read(data)?);
148        }
149        Ok(Self { modifiers })
150    }
151}
152
153impl WriteTo for ItemAttributeModifierEntry {
154    fn write(&self, writer: &mut impl Write) -> Result<()> {
155        let attribute_id = self.attribute.try_id().ok_or_else(|| {
156            std::io::Error::other(format!("Unknown attribute: {}", self.attribute.key))
157        })?;
158        VarInt(attribute_id as i32).write(writer)?;
159        self.id.write(writer)?;
160        self.amount.write(writer)?;
161        self.operation.write(writer)?;
162        VarInt(self.slot.id()).write(writer)?;
163        self.display.write(writer)
164    }
165}

167impl ReadFrom for ItemAttributeModifierEntry {
168    fn read(data: &mut Cursor<&[u8]>) -> Result<Self> {
169        let attribute_id = VarInt::read(data)?.0;
170        let attribute_id = usize::try_from(attribute_id)
171            .map_err(|_| std::io::Error::other(format!("Negative attribute id: {attribute_id}")))?;
172        let attribute = REGISTRY.attributes.by_id(attribute_id).ok_or_else(|| {
173            std::io::Error::other(format!("Unknown attribute id: {attribute_id}"))
174        })?;
175        let id = Identifier::read(data)?;
176        let amount = f64::read(data)?;
177        let operation = AttributeModifierOperation::read(data)?;
178        let slot_id = VarInt::read(data)?.0;
179        let slot = EquipmentSlotGroup::by_id(slot_id);
180        let display = ItemAttributeModifierDisplay::read(data)?;
181
182        Ok(Self {
183            attribute,
184            id,
185            amount,
186            operation,
187            slot,
188            display,
189        })
190    }
191}

193impl WriteTo for ItemAttributeModifierDisplay {
194    fn write(&self, writer: &mut impl Write) -> Result<()> {
195        VarInt(self.id()).write(writer)?;
196        if let Self::OverrideText(text) = self {
197            text.write(writer)?;
198        }
199        Ok(())
200    }
201}
202
203impl ReadFrom for ItemAttributeModifierDisplay {
204    fn read(data: &mut Cursor<&[u8]>) -> Result<Self> {
205        let display_id = VarInt::read(data)?.0;
206        match display_id {
207            1 => Ok(Self::Hidden),
208            2 => Ok(Self::OverrideText(Box::new(TextComponent::read(data)?))),
209            _ => Ok(Self::Default),
210        }
211    }
212}

214impl ToNbtTag for ItemAttributeModifiers {
215    fn to_nbt_tag(self) -> NbtTag {
216        NbtTag::List(NbtList::Compound(
217            self.modifiers
218                .into_iter()
219                .map(ItemAttributeModifierEntry::into_nbt_compound)
220                .collect(),
221        ))
222    }
223}

225impl ItemAttributeModifierEntry {
226    fn into_nbt_compound(self) -> NbtCompound {
227        let mut compound = NbtCompound::new();
228        compound.insert("type", self.attribute.key.to_string());
229        compound.insert("id", self.id.to_string());
230        compound.insert("amount", self.amount);
231        compound.insert("operation", self.operation.name());
232        if self.slot != EquipmentSlotGroup::Any {
233            compound.insert("slot", self.slot.name());
234        }
235        if !matches!(self.display, ItemAttributeModifierDisplay::Default) {
236            compound.insert("display", self.display.to_nbt_tag_ref());
237        }
238        compound
239    }
240}

242impl FromNbtTag for ItemAttributeModifiers {
243    fn from_nbt_tag(tag: simdnbt::borrow::NbtTag) -> Option<Self> {
244        let entries = tag.list()?.compounds()?;
245        let mut modifiers = Vec::with_capacity(entries.len());
246
247        for compound in entries {
248            let attribute_key = compound
249                .get("type")
250                .and_then(|tag| tag.string())
251                .and_then(|value| value.to_str().parse::<Identifier>().ok())?;
252            let attribute = REGISTRY.attributes.by_key(&attribute_key)?;
253            let id = compound
254                .get("id")
255                .and_then(|tag| tag.string())
256                .and_then(|value| value.to_str().parse::<Identifier>().ok())?;
257            let amount = compound.get("amount").and_then(|tag| tag.double())?;
258            let operation = compound
259                .get("operation")
260                .and_then(|tag| tag.string())
261                .and_then(|value| AttributeModifierOperation::by_name(value.to_str().as_ref()))?;
262            let slot = compound
263                .get("slot")
264                .and_then(|tag| tag.string())
265                .and_then(|value| EquipmentSlotGroup::by_name(value.to_str().as_ref()))
266                .unwrap_or(EquipmentSlotGroup::Any);
267            let display = compound
268                .get("display")
269                .and_then(|tag| tag.compound())
270                .and_then(ItemAttributeModifierDisplay::from_nbt_compound)
271                .unwrap_or(ItemAttributeModifierDisplay::Default);
272
273            modifiers.push(ItemAttributeModifierEntry {
274                attribute,
275                id,
276                amount,
277                operation,
278                slot,
279                display,
280            });
281        }
282
283        Some(Self { modifiers })
284    }
285}

287impl HashComponent for ItemAttributeModifiers {
288    fn hash_component(&self, hasher: &mut ComponentHasher) {
289        hasher.start_list();
290        for entry in &self.modifiers {
291            entry.hash_component(hasher);
292        }
293        hasher.end_list();
294    }
295}

297impl HashComponent for ItemAttributeModifierEntry {
298    fn hash_component(&self, hasher: &mut ComponentHasher) {
299        let mut entries = Vec::new();
300        push_hash_entry(&mut entries, "type", &self.attribute.key.to_string());
301        push_hash_entry(&mut entries, "id", &self.id.to_string());
302        push_hash_entry(&mut entries, "amount", &self.amount);
303        push_hash_entry(&mut entries, "operation", self.operation.name());
304        if self.slot != EquipmentSlotGroup::Any {
305            push_hash_entry(&mut entries, "slot", self.slot.name());
306        }
307        if !matches!(self.display, ItemAttributeModifierDisplay::Default) {
308            push_hash_entry(&mut entries, "display", &self.display);
309        }
310
311        sort_map_entries(&mut entries);
312        hasher.start_map();
313        for entry in &entries {
314            hasher.put_raw_bytes(&entry.key_bytes);
315            hasher.put_raw_bytes(&entry.value_bytes);
316        }
317        hasher.end_map();
318    }
319}

321impl HashComponent for ItemAttributeModifierDisplay {
322    fn hash_component(&self, hasher: &mut ComponentHasher) {
323        let mut entries = Vec::new();
324        push_hash_entry(&mut entries, "type", self.name());
325        if let Self::OverrideText(text) = self {
326            push_hash_entry(&mut entries, "value", text.as_ref());
327        }
328
329        sort_map_entries(&mut entries);
330        hasher.start_map();
331        for entry in &entries {
332            hasher.put_raw_bytes(&entry.key_bytes);
333            hasher.put_raw_bytes(&entry.value_bytes);
334        }
335        hasher.end_map();
336    }
337}

339fn push_hash_entry<T: HashComponent + ?Sized>(entries: &mut Vec<HashEntry>, key: &str, value: &T) {
340    let mut key_hasher = ComponentHasher::new();
341    key_hasher.put_string(key);
342    let mut value_hasher = ComponentHasher::new();
343    value.hash_component(&mut value_hasher);
344    entries.push(HashEntry::new(key_hasher, value_hasher));
345}

364    #[test]
365    fn generated_diamond_sword_has_main_hand_attack_modifiers() {
366        init_test_registry();
367
368        let stack = ItemStack::new(&ITEMS.diamond_sword);
369        let modifiers = stack
370            .get_attribute_modifiers()
371            .expect("diamond sword should have attribute modifiers");
372        let main_hand_modifiers = modifiers
373            .for_slot(EquipmentSlot::MainHand)
374            .collect::<Vec<_>>();
375
376        assert_eq!(main_hand_modifiers.len(), 2);
377        assert!(main_hand_modifiers.iter().any(|modifier| {
378            modifier.attribute.key == vanilla_attributes::ATTACK_DAMAGE.key
379                && modifier.id == Identifier::vanilla_static("base_attack_damage")
380                && modifier.amount.to_bits() == 6.0f64.to_bits()
381                && modifier.operation == AttributeModifierOperation::AddValue
382        }));
383        assert!(main_hand_modifiers.iter().any(|modifier| {
384            modifier.attribute.key == vanilla_attributes::ATTACK_SPEED.key
385                && modifier.id == Identifier::vanilla_static("base_attack_speed")
386                && modifier.amount.to_bits() == (-2.4000000953674316f64).to_bits()
387                && modifier.operation == AttributeModifierOperation::AddValue
388        }));
389        assert!(modifiers.for_slot(EquipmentSlot::Head).next().is_none());
390    }
