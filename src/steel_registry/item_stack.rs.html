32/// A stack of items with a count and component modifications.
33#[derive(Debug, Clone, PartialEq)]
34pub struct ItemStack {
35    /// The item type. AIR represents an empty stack.
36    pub item: ItemRef,
37    /// The number of items in this stack.
38    pub count: i32,
39    /// Modifications to the prototype components.
40    patch: DataComponentPatch,
41}

203    /// Damages the item and breaks it if durability reaches zero.
204    ///
205    /// Returns `true` if the item broke and should be removed/replaced.
206    pub fn hurt_and_break(&mut self, amount: i32, has_infinite_materials: bool) -> bool {
207        if !self.is_damageable_item() || amount <= 0 {
208            return false;
209        }
210
211        if has_infinite_materials {
212            return false;
213        }
214
215        let unbreaking_level =
216            self.get_enchantment_level(&crate::vanilla_enchantments::UNBREAKING.key);
217        let mut effective_amount = 0;
218        for _ in 0..amount {
219            if should_consume_durability(unbreaking_level) {
220                effective_amount += 1;
221            }
222        }
223
224        if effective_amount == 0 {
225            return false;
226        }
227
228        let new_damage = self.get_damage_value() + effective_amount;
229
230        // TODO: Trigger ITEM_DURABILITY_CHANGED advancement criteria
231
232        self.set_damage_value(new_damage);
233
234        if self.is_broken() {
235            // TODO: Call onEquippedItemBroken callback which:
236            // - Broadcasts entity event (byte 47 for mainhand) for break sound/particles
237            // - Stops location-based effects (removes attribute modifiers)
238            self.shrink(1);
239            return true;
240        }
241
242        false
243    }

453    pub fn has_enchantment_effect(&self, component: EnchantmentEffectComponent) -> bool {
454        let Some(enchantments) = self.get_enchantments() else {
455            return false;
456        };
457
458        for (key, level) in enchantments.iter() {
459            if *level == 0 {
460                continue;
461            }
462            let Some(enchantment) = REGISTRY.enchantments.by_key(key) else {
463                continue;
464            };
465            if enchantment.effects.has(component) {
466                return true;
467            }
468        }
469
470        false
471    }

473    #[must_use]
474    pub fn apply_unconditional_enchantment_value_effects(
475        &self,
476        component: EnchantmentEffectComponent,
477        input: f32,
478    ) -> f32 {
479        let Some(enchantments) = self.get_enchantments() else {
480            return input;
481        };
482
483        let mut value = input;
484        for (key, level) in enchantments.iter() {
485            if *level == 0 {
486                continue;
487            }
488            let Some(enchantment) = REGISTRY.enchantments.by_key(key) else {
489                continue;
490            };
491            let level = *level as i32;
492
493            for effect in enchantment.effects.value_effects(component) {
494                if !effect.is_unconditional() {
495                    continue;
496                }
497                if let Some(updated) = effect.effect.process_without_random(level, value) {
498                    value = updated;
499                }
500            }
501
502            let Some(effect) = enchantment.effects.single_value_effect(component) else {
503                continue;
504            };
505            if let Some(updated) = effect.process_without_random(level, value) {
506                value = updated;
507            }
508        }
509
510        value
511    }

646    pub fn set_enchantments(&mut self, enchantments: &[(Identifier, u32)], add: bool) {
647        let mut current = self
648            .get(ENCHANTMENTS)
649            .cloned()
650            .unwrap_or_else(ItemEnchantments::empty);
651
652        for (key, level) in enchantments {
653            if add {
654                let existing = current.get_level(key);
655                current.set(key.clone(), existing + *level);
656            } else {
657                current.set(key.clone(), *level);
658            }
659        }
660
661        self.set(ENCHANTMENTS, current);
662    }

829    pub fn components_equal(&self, other: &Self) -> bool {
830        let mut all_keys = rustc_hash::FxHashSet::default();
831
832        for key in self.prototype().keys() {
833            if !self.patch.is_removed(key) {
834                all_keys.insert(key);
835            }
836        }
837        for (key, entry) in self.patch.iter() {
838            if matches!(entry, ComponentPatchEntry::Set(_)) {
839                all_keys.insert(key);
840            }
841        }
842        for key in other.prototype().keys() {
843            if !other.patch.is_removed(key) {
844                all_keys.insert(key);
845            }
846        }
847        for (key, entry) in other.patch.iter() {
848            if matches!(entry, ComponentPatchEntry::Set(_)) {
849                all_keys.insert(key);
850            }
851        }
852        for key in all_keys {
853            let val_a = self.get_effective_value_raw(key);
854            let val_b = other.get_effective_value_raw(key);
855
856            match (val_a, val_b) {
857                (Some(a), Some(b)) => {
858                    if a != b {
859                        return false;
860                    }
861                }
862                (None, None) => {}
863                _ => return false,
864            }
865        }
866
867        true
868    }
869}

871/// Vanilla unbreaking formula: `1 / (unbreaking_level + 1)` chance to consume durability.
872fn should_consume_durability(unbreaking_level: i32) -> bool {
873    if unbreaking_level <= 0 {
874        return true;
875    }
876    // TODO: Armor uses a different formula: `3 / (unbreaking_level + 3)`
877    rand::rng().random_range(0..unbreaking_level + 1) == 0
878}

947impl ToNbtTag for ItemStack {
948    /// Converts this item stack to an NBT tag for persistent storage.
949    ///
950    /// Format (matching vanilla Minecraft):
951    /// ```text
952    /// {
953    ///     id: "minecraft:stone",
954    ///     count: 64,
955    ///     components: { ... }  // Only present if patch is non-empty
956    /// }
957    /// ```
958    fn to_nbt_tag(self) -> simdnbt::owned::NbtTag {
959        self.to_nbt_tag_ref()
960    }
961}
962
963impl ItemStack {
964    /// Converts this item stack to an NBT tag for persistent storage without consuming it.
965    #[must_use]
966    pub fn to_nbt_tag_ref(&self) -> simdnbt::owned::NbtTag {
967        if self.is_empty() {
968            // Empty stacks are represented as an empty compound
969            return simdnbt::owned::NbtTag::Compound(NbtCompound::new());
970        }
971
972        let mut compound = NbtCompound::new();
973
974        // id: The item identifier
975        compound.insert("id", self.item.key.to_string());
976
977        // count: The stack count (vanilla uses Int for NBT storage)
978        compound.insert("count", self.count);
979
980        // components: The component patch (only if non-empty)
981        if !self.patch.is_empty() {
982            compound.insert("components", self.patch.to_nbt_tag_ref());
983        }
984
985        simdnbt::owned::NbtTag::Compound(compound)
986    }
987}

989impl FromNbtTag for ItemStack {
990    /// Parses an item stack from an NBT tag.
991    ///
992    /// Accepts the vanilla format:
993    /// ```text
994    /// {
995    ///     id: "minecraft:stone",
996    ///     count: 64,
997    ///     components: { ... }
998    /// }
999    /// ```
1000    fn from_nbt_tag(tag: BorrowedNbtTag) -> Option<Self> {
1001        let compound = tag.compound()?;
1002
1003        // Get the item ID
1004        let id_str = compound.get("id")?.string()?.to_str();
1005        let id = id_str.parse::<Identifier>().ok()?;
1006
1007        // Look up the item in the registry
1008        let item = REGISTRY.items.by_key(&id)?;
1009
1010        // Get the count (default to 1 if not present)
1011        let count = compound.get("count").and_then(|t| t.int()).unwrap_or(1);
1012
1013        // Parse components if present
1014        let patch = compound
1015            .get("components")
1016            .and_then(DataComponentPatch::from_nbt_tag)
1017            .unwrap_or_default();
1018
1019        Some(Self { item, count, patch })
1020    }
1021}

1024    /// Parses an `ItemStack` from a borrowed `NbtCompoundView`.
1025    ///
1026    /// This is useful for loading items from disk where we have borrowed NBT data
1027    /// and want to avoid the overhead of converting to an owned tag first.

1029    pub fn from_borrowed_compound(compound: &NbtCompoundView<'_, '_>) -> Option<Self> {
