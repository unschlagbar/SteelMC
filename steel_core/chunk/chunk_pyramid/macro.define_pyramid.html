            $($status:ident => {
                $(requirements: [$( ($req_status:ident, $req_radius:expr) ),* $(,)?] ,)?
                $(block_state_write_radius: $bswr:expr ,)?
                task: $task:expr $(,)?
            }),* $(,)?
        };
    ) => { ... };
    (@bswr) => { ... };
    (@bswr $bswr:expr) => { ... };
}
